41  /**
42   * This controller returns a profile for the authenticated user
43   * (identifier + attributes), found with the access token (CAS granting
44   * ticket).
45   *
46   * @author Jerome Leleu
47   * @since 3.5.0
48   */
49  public final class OAuth20ProfileController extends AbstractController {
50  
51      private static Logger LOGGER = LoggerFactory.getLogger(OAuth20ProfileController.class);
52  
53      private static final String ID = "id";
54  
55      private static final String ATTRIBUTES = "attributes";
56  
57      private final TicketRegistry ticketRegistry;
58  
59      private final JsonFactory jsonFactory = new JsonFactory(new ObjectMapper());
60  
61      public OAuth20ProfileController(final TicketRegistry ticketRegistry) {
62          this.ticketRegistry = ticketRegistry;
63      }
64  
65      @Override
66      protected ModelAndView handleRequestInternal(final HttpServletRequest request, final HttpServletResponse response)
67              throws Exception {
68  
69          final String accessToken = request.getParameter(OAuthConstants.ACCESS_TOKEN);
70          LOGGER.debug("{} : {}", OAuthConstants.ACCESS_TOKEN, accessToken);
71  
72          final JsonGenerator jsonGenerator = this.jsonFactory.createJsonGenerator(response.getWriter());
73  
74          try {
75              response.setContentType("application/json");
76  
77              // accessToken is required
78              if (StringUtils.isBlank(accessToken)) {
79                  LOGGER.error("Missing {}", OAuthConstants.ACCESS_TOKEN);
80                  jsonGenerator.writeStartObject();
81                  jsonGenerator.writeStringField("error", OAuthConstants.MISSING_ACCESS_TOKEN);
82                  jsonGenerator.writeEndObject();
83                  return null;
84              }
85              // get ticket granting ticket
86              final TicketGrantingTicket ticketGrantingTicket = (TicketGrantingTicket) this.ticketRegistry
87                      .getTicket(accessToken);
88              if (ticketGrantingTicket == null || ticketGrantingTicket.isExpired()) {
89                  LOGGER.error("expired accessToken : {}", accessToken);
90                  jsonGenerator.writeStartObject();
91                  jsonGenerator.writeStringField("error", OAuthConstants.EXPIRED_ACCESS_TOKEN);
92                  jsonGenerator.writeEndObject();
93                  return null;
94              }
95              // generate profile : identifier + attributes
96              final Principal principal = ticketGrantingTicket.getAuthentication().getPrincipal();
97              jsonGenerator.writeStartObject();
98              jsonGenerator.writeStringField(ID, principal.getId());
99              jsonGenerator.writeArrayFieldStart(ATTRIBUTES);
100             final Map<String, Object> attributes = principal.getAttributes();
101             for (final String key : attributes.keySet()) {
102                 jsonGenerator.writeStartObject();
103                 jsonGenerator.writeObjectField(key, attributes.get(key));
104                 jsonGenerator.writeEndObject();
105             }
106             jsonGenerator.writeEndArray();
107             jsonGenerator.writeEndObject();
108             return null;
109         } finally {
110             IOUtils.closeQuietly(jsonGenerator);
111             response.flushBuffer();
112         }
113     }
