19  package org.jasig.cas.services;
20  
21  import javax.validation.constraints.NotNull;
22  import java.util.ArrayList;
23  import java.util.List;
24  
25  /**
26   * Default In Memory Service Registry Dao for test/demonstration purposes.
27   *
28   * @author Scott Battaglia
29   * @since 3.1
30   *
31   */
32  public final class InMemoryServiceRegistryDaoImpl implements ServiceRegistryDao {
33  
34      @NotNull
35      private List<RegisteredService> registeredServices = new ArrayList<RegisteredService>();
36  
37      public boolean delete(final RegisteredService registeredService) {
38          return this.registeredServices.remove(registeredService);
39      }
40  
41      public RegisteredService findServiceById(final long id) {
42          for (final RegisteredService r : this.registeredServices) {
43              if (r.getId() == id) {
44                  return r;
45              }
46          }
47  
48          return null;
49      }
50  
51      public List<RegisteredService> load() {
52          return this.registeredServices;
53      }
54  
55      public RegisteredService save(final RegisteredService registeredService) {
56          if (registeredService.getId() == RegisteredService.INITIAL_IDENTIFIER_VALUE) {
57              ((AbstractRegisteredService) registeredService).setId(findHighestId()+1);
58          }
59  
60          this.registeredServices.remove(registeredService);
61          this.registeredServices.add(registeredService);
62  
63          return registeredService;
64      }
65  
66      public void setRegisteredServices(final List<RegisteredService> registeredServices) {
67          this.registeredServices = registeredServices;
68      }
69  
70      /**
71       * This isn't super-fast but we don't expect thousands of services.
72       *
73       * @return the highest service id in the list of registered services
74       */
75      private long findHighestId() {
76          long id = 0;
77  
78          for (final RegisteredService r : this.registeredServices) {
79              if (r.getId() > id) {
80                  id = r.getId();
81              }
82          }
83  
84          return id;
85      }
86  }
