38  /**
39   * Wrapper implementation around a CentralAuthenticationService that does
40   * completes the marshalling of parameters from the web-service layer to the
41   * service layer. Typically the only thing that is done is to validate the
42   * parameters (as you would in the web tier) and then delegate to the service
43   * layer.
44   * <p>
45   * The following properties are required:
46   * </p>
47   * <ul>
48   * <li>centralAuthenticationService - the service layer we are delegating to.</li>
49   * </ul>
50   *
51   * @author Scott Battaglia
52  
53   * @since 3.0
54   */
55  public final class RemoteCentralAuthenticationService implements CentralAuthenticationService {
56  
57      /** The CORE to delegate to. */
58      @NotNull
59      private CentralAuthenticationService centralAuthenticationService;
60  
61      /** The validators to check the Credential. */
62      @NotNull
63      private Validator validator = Validation.buildDefaultValidatorFactory().getValidator();
64  
65      /**
66       * {@inheritDoc}
67       * @throws IllegalArgumentException if the Credentials are null or if given
68       * invalid credentials.
69       */
70      @Override
71      public String createTicketGrantingTicket(final Credential... credentials)
72              throws AuthenticationException, TicketException {
73  
74          Assert.notNull(credentials, "credentials cannot be null");
75          checkForErrors(credentials);
76  
77          return this.centralAuthenticationService.createTicketGrantingTicket(credentials);
78      }
79  
80      /**
81       * {@inheritDoc}
82       */
83      @Override
84      public String grantServiceTicket(final String ticketGrantingTicketId, final Service service)
85              throws TicketException {
86          return this.centralAuthenticationService.grantServiceTicket(ticketGrantingTicketId, service);
87      }
88  
89      /**
90       * {@inheritDoc}
91       * @throws IllegalArgumentException if given invalid credentials
92       */
93      @Override
94      public String grantServiceTicket(
95              final String ticketGrantingTicketId, final Service service, final Credential... credentials)
96              throws AuthenticationException, TicketException {
97  
98          checkForErrors(credentials);
99  
100         return this.centralAuthenticationService.grantServiceTicket(ticketGrantingTicketId, service, credentials);
101     }
102 
103     /**
104      * {@inheritDoc}
105      */
106     @Override
107     public Assertion validateServiceTicket(final String serviceTicketId, final Service service) throws TicketException {
108         return this.centralAuthenticationService.validateServiceTicket(serviceTicketId, service);
109     }
110 
111     /**
112      * {@inheritDoc}
113      * <p>Destroy a TicketGrantingTicket and perform back channel logout. This has the effect of invalidating any
114      * Ticket that was derived from the TicketGrantingTicket being destroyed. May throw an
115      * {@link IllegalArgumentException} if the TicketGrantingTicket ID is null.
116      *
117      * @param ticketGrantingTicketId the id of the ticket we want to destroy
118      * @return the logout requests.
119      */
120     @Override
121     public List<LogoutRequest> destroyTicketGrantingTicket(final String ticketGrantingTicketId) {
122         return this.centralAuthenticationService.destroyTicketGrantingTicket(ticketGrantingTicketId);
123     }
124 
125     /**
126      * {@inheritDoc}
127      * @throws IllegalArgumentException if the credentials are invalid.
128      */
129     @Override
130     public String delegateTicketGrantingTicket(final String serviceTicketId, final Credential... credentials)
131             throws AuthenticationException, TicketException {
132 
133         checkForErrors(credentials);
134 
135         return this.centralAuthenticationService.delegateTicketGrantingTicket(serviceTicketId, credentials);
136     }
137 
138     private void checkForErrors(final Credential... credentials) {
139         if (credentials == null) {
140             return;
141         }
142 
143         for (final Credential c : credentials) {
144             final Set<ConstraintViolation<Credential>> errors = this.validator.validate(c);
145             if (!errors.isEmpty()) {
146                 throw new IllegalArgumentException("Error validating credentials: " + errors.toString());
147             }
148         }
149     }
150 
151     /**
152      * Set the CentralAuthenticationService.
153      *
154      * @param centralAuthenticationService The CentralAuthenticationService to
155      * set.
156      */
157     public void setCentralAuthenticationService(
158         final CentralAuthenticationService centralAuthenticationService) {
159         this.centralAuthenticationService = centralAuthenticationService;
160     }
161 
162     /**
163      * Set the list of validators.
164      *
165      * @param validator The array of validators to use.
166      */
167     public void setValidator(final Validator validator) {
168         this.validator = validator;
169     }
