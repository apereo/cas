85      @Override
86      public void afterPropertiesSet() throws Exception {
87          if (System.getProperty(SYS_PROP_LOGIN_CONF) != null) {
88              logger.warn("found login config in system property, may overide : "
89                      + System.getProperty(SYS_PROP_LOGIN_CONF));
90          }
91  
92          URL url = getClass().getResource(
93                  this.loginConf == null ? DEFAULT_LOGIN_CONFIG : this.loginConf);
94          if (url != null) {
95              this.loginConf = url.toExternalForm();
96          }
97          if (this.loginConf != null) {
98              System.setProperty(SYS_PROP_LOGIN_CONF, this.loginConf);
99          } else {
100             url = getClass().getResource("/jcifs/http/login.conf");
101             if (url != null) {
102                 System.setProperty(SYS_PROP_LOGIN_CONF, url.toExternalForm());
103             }
104         }
105         logger.debug("configured login configuration path : "
106                 + System.getProperty(SYS_PROP_LOGIN_CONF));
107     }
