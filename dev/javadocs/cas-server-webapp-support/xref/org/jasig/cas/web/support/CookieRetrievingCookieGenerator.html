19  package org.jasig.cas.web.support;
20  
21  import javax.servlet.http.Cookie;
22  import javax.servlet.http.HttpServletRequest;
23  import javax.servlet.http.HttpServletResponse;
24  
25  import org.jasig.cas.authentication.RememberMeCredential;
26  import org.springframework.util.StringUtils;
27  import org.springframework.web.util.CookieGenerator;
28  
29  /**
30   * Extends CookieGenerator to allow you to retrieve a value from a request.
31   * <p>
32   * Also has support for RememberMe Services
33   *
34   * @author Scott Battaglia
35   * @since 3.1
36   *
37   */
38  public class CookieRetrievingCookieGenerator extends CookieGenerator {
39  
40      /** The maximum age the cookie should be remembered for.
41       * The default is three months (7889231 in seconds, according to Google) */
42      private int rememberMeMaxAge = 7889231;
43  
44      public void addCookie(final HttpServletRequest request, final HttpServletResponse response, final String cookieValue) {
45  
46          if (!StringUtils.hasText(request.getParameter(RememberMeCredential.REQUEST_PARAMETER_REMEMBER_ME))) {
47              super.addCookie(response, cookieValue);
48          } else {
49              final Cookie cookie = createCookie(cookieValue);
50              cookie.setMaxAge(this.rememberMeMaxAge);
51              if (isCookieSecure()) {
52                  cookie.setSecure(true);
53              }
54              response.addCookie(cookie);
55          }
56      }
57  
58      public String retrieveCookieValue(final HttpServletRequest request) {
59          final Cookie cookie = org.springframework.web.util.WebUtils.getCookie(
60                  request, getCookieName());
61  
62          return cookie == null ? null : cookie.getValue();
63      }
64  
65      public void setRememberMeMaxAge(final int maxAge) {
66          this.rememberMeMaxAge = maxAge;
67      }
68  }
