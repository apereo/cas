34  /**
35   * Class to represent that this service wants to use SAML. We use this in
36   * combination with the CentralAuthenticationServiceImpl to choose the right
37   * UniqueTicketIdGenerator.
38   *
39   * @author Scott Battaglia
40   * @since 3.1
41   */
42  public final class SamlService extends AbstractWebApplicationService {
43  
44      private static final Logger LOGGER = LoggerFactory.getLogger(SamlService.class);
45  
46      /** Constant representing service. */
47      private static final String CONST_PARAM_SERVICE = "TARGET";
48  
49      /** Constant representing artifact. */
50      private static final String CONST_PARAM_TICKET = "SAMLart";
51  
52      private static final String CONST_START_ARTIFACT_XML_TAG_NO_NAMESPACE = "<AssertionArtifact>";
53  
54      private static final String CONST_END_ARTIFACT_XML_TAG_NO_NAMESPACE = "</AssertionArtifact>";
55  
56      private static final String CONST_START_ARTIFACT_XML_TAG = "<samlp:AssertionArtifact>";
57  
58      private static final String CONST_END_ARTIFACT_XML_TAG = "</samlp:AssertionArtifact>";
59  
60      private String requestId;
61  
62      /**
63       * Unique Id for serialization.
64       */
65      private static final long serialVersionUID = -6867572626767140223L;

67      protected SamlService(final String id) {
68          super(id, id, null);
69      }

71      protected SamlService(final String id, final String originalUrl,
72              final String artifactId, final String requestId) {
73          super(id, originalUrl, artifactId);
74          this.requestId = requestId;

78       * {@inheritDoc}

90      public static SamlService createServiceFrom(
91              final HttpServletRequest request) {
92          final String service = request.getParameter(CONST_PARAM_SERVICE);
93          final String artifactId;
94          final String requestBody = getRequestBody(request);
95          final String requestId;
96  
97          if (!StringUtils.hasText(service) && !StringUtils.hasText(requestBody)) {
98              return null;
99          }
100 
101         final String id = cleanupUrl(service);
102 
103         if (StringUtils.hasText(requestBody)) {
104 
105             final String tagStart;
106             final String tagEnd;
107             if (requestBody.contains(CONST_START_ARTIFACT_XML_TAG)) {
108                 tagStart = CONST_START_ARTIFACT_XML_TAG;
109                 tagEnd = CONST_END_ARTIFACT_XML_TAG;
110             } else {
111                 tagStart = CONST_START_ARTIFACT_XML_TAG_NO_NAMESPACE;
112                 tagEnd = CONST_END_ARTIFACT_XML_TAG_NO_NAMESPACE;
113             }
114             final int startTagLocation = requestBody.indexOf(tagStart);
115             final int artifactStartLocation = startTagLocation + tagStart.length();
116             final int endTagLocation = requestBody.indexOf(tagEnd);
117 
118             artifactId = requestBody.substring(artifactStartLocation, endTagLocation).trim();
119 
120             // is there a request id?
121             requestId = extractRequestId(requestBody);
122         } else {
123             artifactId = null;
124             requestId = null;
125         }
126 
127         LOGGER.debug("Attempted to extract Request from HttpServletRequest. Results:");
128         LOGGER.debug(String.format("Request Body: %s", requestBody));
129         LOGGER.debug(String.format("Extracted ArtifactId: %s", artifactId));
130         LOGGER.debug(String.format("Extracted Request Id: %s", requestId));
131 
132         return new SamlService(id, service, artifactId, requestId);
133     }
134 
135     @Override
136     public Response getResponse(final String ticketId) {
137         final Map<String, String> parameters = new HashMap<String, String>();
138 
139         parameters.put(CONST_PARAM_TICKET, ticketId);
140         parameters.put(CONST_PARAM_SERVICE, getOriginalUrl());
141 
142         return Response.getRedirectResponse(getOriginalUrl(), parameters);
143     }
144 
145     protected static String extractRequestId(final String requestBody) {
146         if (!requestBody.contains("RequestID")) {
147             return null;
148         }
149 
150         try {
151             final int position = requestBody.indexOf("RequestID=\"") + 11;
152             final int nextPosition = requestBody.indexOf("\"", position);
153 
154             return requestBody.substring(position,  nextPosition);
155         } catch (final Exception e) {
156             LOGGER.debug("Exception parsing RequestID from request.", e);
157             return null;
158         }
159     }
160 
161     protected static String getRequestBody(final HttpServletRequest request) {
162         final StringBuilder builder = new StringBuilder();
163         try {
164             final BufferedReader reader = request.getReader();
165 
166             String line;
167             while ((line = reader.readLine()) != null) {
168                 builder.append(line);
169             }
170             return builder.toString();
171         } catch (final Exception e) {
172             return null;
173         }
174     }
