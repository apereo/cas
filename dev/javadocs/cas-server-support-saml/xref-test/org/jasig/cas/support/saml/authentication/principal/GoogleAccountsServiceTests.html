90      protected static String encodeMessage(final String xmlString) throws IOException {
91          byte[] xmlBytes = xmlString.getBytes("UTF-8");
92          ByteArrayOutputStream byteOutputStream = new ByteArrayOutputStream();
93          DeflaterOutputStream deflaterOutputStream = new DeflaterOutputStream(
94                  byteOutputStream);
95          deflaterOutputStream.write(xmlBytes, 0, xmlBytes.length);
96          deflaterOutputStream.close();
97  
98          // next, base64 encode it
99          Base64 base64Encoder = new Base64();
100         byte[] base64EncodedByteArray = base64Encoder.encode(byteOutputStream
101                 .toByteArray());
102         return new String(base64EncodedByteArray);
103     }
