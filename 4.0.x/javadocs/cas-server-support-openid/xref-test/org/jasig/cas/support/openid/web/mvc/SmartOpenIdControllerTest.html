35  /**
36   * Test case of the Smart OpenId Controller.
37   * @author Frederic Esnault
38   */
39  public class SmartOpenIdControllerTest {
40      private final MockHttpServletRequest request = new MockHttpServletRequest();
41      private final HttpServletResponse response = new MockHttpServletResponse();
42      private ServerManager manager;
43      private final SmartOpenIdController smartOpenIdController = new SmartOpenIdController();
44  
45      @Before
46      public void setUp() {
47          manager = new ServerManager();
48          manager.setOPEndpointUrl("https://localshot:8443/cas/login");
49          manager.setEnforceRpId(false);
50          smartOpenIdController.setServerManager(manager);
51      }
52  
53      @Test
54      public void testCanHandle() {
55          request.addParameter("openid.mode", "associate");
56          boolean canHandle = smartOpenIdController.canHandle(request, response);
57          request.removeParameter("openid.mode");
58          assertEquals(true, canHandle);
59      }
60  
61      @Test
62      public void testCannotHandle() {
63          request.addParameter("openid.mode", "anythingElse");
64          boolean canHandle = smartOpenIdController.canHandle(request, response);
65          request.removeParameter("openid.mode");
66          assertEquals(false, canHandle);
67      }
68  
69      @Test
70      public void testGetAssociationResponse() {
71          request.addParameter("openid.mode", "associate");
72          request.addParameter("openid.session_type", "DH-SHA1");
73          request.addParameter("openid.assoc_type", "HMAC-SHA1");
74          request.addParameter("openid.dh_consumer_public",
75                  "NzKoFMyrzFn/5iJFPdX6MVvNA/BChV1/sJdnYbupDn7ptn+cerwEzyFfWFx25KsoLSkxQCaSMmYtc1GPy/2GI1BSKSDhpdJmDBb"
76                  + "QRa/9Gs+giV/5fHcz/mHz8sREc7RTGI+0Ka9230arwrWt0fnoaJLRKEGUsmFR71rCo4EUOew=");
77          Map<String, String> assocResponse = smartOpenIdController.getAssociationResponse(request);
78          assertTrue(assocResponse.containsKey("assoc_handle"));
79          assertTrue(assocResponse.containsKey("expires_in"));
80          assertTrue(assocResponse.containsKey("dh_server_public"));
81          assertTrue(assocResponse.containsKey("enc_mac_key"));
82          request.removeParameter("openid.mode");
83      }
