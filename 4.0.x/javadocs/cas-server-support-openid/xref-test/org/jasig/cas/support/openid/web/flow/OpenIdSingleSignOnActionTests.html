73      @Before
74      public void setUp() throws Exception {
75          this.ticketRegistry = new DefaultTicketRegistry();
76          final OpenIdCredentialsAuthenticationHandler handler = new OpenIdCredentialsAuthenticationHandler();
77          handler.setTicketRegistry(this.ticketRegistry);
78          this.authenticationManager = new PolicyBasedAuthenticationManager(
79                  Collections.<AuthenticationHandler, PrincipalResolver>singletonMap(
80                          handler,
81                          new OpenIdPrincipalResolver()));
82  
83          final Map<String, UniqueTicketIdGenerator> generator = new HashMap<String, UniqueTicketIdGenerator>();
84          generator.put(OpenIdService.class.getName(), new DefaultUniqueTicketIdGenerator());
85  
86          impl = new CentralAuthenticationServiceImpl(this.ticketRegistry, null, this.authenticationManager,
87                  new DefaultUniqueTicketIdGenerator(), generator, new NeverExpiresExpirationPolicy(),
88                  new NeverExpiresExpirationPolicy(),
89                  new DefaultServicesManagerImpl(new InMemoryServiceRegistryDaoImpl()), mock(LogoutManager.class));
90  
91          this.action = new OpenIdSingleSignOnAction();
92          this.action.setCentralAuthenticationService(this.impl);
93          this.action.setExtractor(new DefaultOpenIdUserNameExtractor());
94          this.action.afterPropertiesSet();
95      }
