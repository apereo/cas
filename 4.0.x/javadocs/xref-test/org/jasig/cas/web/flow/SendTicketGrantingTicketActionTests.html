35  public class SendTicketGrantingTicketActionTests extends AbstractCentralAuthenticationServiceTest {
36      private SendTicketGrantingTicketAction action;
37  
38      private CookieRetrievingCookieGenerator ticketGrantingTicketCookieGenerator;
39  
40      private MockRequestContext context;
41  
42      @Before
43      public void onSetUp() throws Exception {
44          this.action = new SendTicketGrantingTicketAction();
45  
46          this.ticketGrantingTicketCookieGenerator = new CookieRetrievingCookieGenerator();
47  
48          this.ticketGrantingTicketCookieGenerator.setCookieName("TGT");
49  
50          this.action.setCentralAuthenticationService(getCentralAuthenticationService());
51  
52          this.action.setTicketGrantingTicketCookieGenerator(this.ticketGrantingTicketCookieGenerator);
53  
54          this.action.afterPropertiesSet();
55  
56          this.context = new MockRequestContext();
57      }
58  
59      @Test
60      public void testNoTgtToSet() throws Exception {
61          this.context.setExternalContext(new ServletExternalContext(new MockServletContext(),
62                  new MockHttpServletRequest(), new MockHttpServletResponse()));
63  
64          assertEquals("success", this.action.execute(this.context).getId());
65      }
66  
67      @Test
68      public void testTgtToSet() throws Exception {
69          final MockHttpServletResponse response = new MockHttpServletResponse();
70          final String TICKET_VALUE = "test";
71  
72          WebUtils.putTicketGrantingTicketInRequestScope(this.context, TICKET_VALUE);
73          this.context.setExternalContext(new ServletExternalContext(new MockServletContext(), new MockHttpServletRequest(), response));
74  
75          assertEquals("success", this.action.execute(this.context).getId());
76          assertEquals(TICKET_VALUE, response.getCookies()[0].getValue());
77      }
78  
79      @Test
80      public void testTgtToSetRemovingOldTgt() throws Exception {
81          final MockHttpServletResponse response = new MockHttpServletResponse();
82          final MockHttpServletRequest request = new MockHttpServletRequest();
83          final String TICKET_VALUE = "test";
84          request.setCookies(new Cookie[] {new Cookie("TGT", "test5")});
85          WebUtils.putTicketGrantingTicketInRequestScope(this.context, TICKET_VALUE);
86          this.context.setExternalContext(new ServletExternalContext(new MockServletContext(), request, response));
87  
88          assertEquals("success", this.action.execute(this.context).getId());
89          assertEquals(TICKET_VALUE, response.getCookies()[0].getValue());
90      }
