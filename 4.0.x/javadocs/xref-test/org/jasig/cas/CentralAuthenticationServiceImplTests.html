375     /**
376      * This test simulates :
377      * - a first authentication for a default service
378      * - a second authentication with the renew parameter and the same service (and same credentials)
379      * - a validation of the second ticket.
380      * 
381      * When supplemental authentications were returned with the chained authentications, the validation specification
382      * failed as it only expects one authentication. Thus supplemental authentications should not be returned in the
383      * chained authentications. Both concepts are orthogonal.
384      *  
385      * @throws TicketException
386      * @throws AuthenticationException
387      */
388     @Test
389     public void authenticateTwiceWithRenew() throws TicketException, AuthenticationException {
390         final CentralAuthenticationService cas = getCentralAuthenticationService();
391         final Service svc = TestUtils.getService("testDefault");
392         final UsernamePasswordCredential goodCredential = TestUtils.getCredentialsWithSameUsernameAndPassword();
393         final String tgtId = cas.createTicketGrantingTicket(goodCredential);
394         cas.grantServiceTicket(tgtId, svc);
395         // simulate renew with new good same credentials
396         final String st2Id = cas.grantServiceTicket(tgtId, svc, goodCredential);
397         final Assertion assertion = cas.validateServiceTicket(st2Id, svc);
398         final ValidationSpecification validationSpecification = new Cas20WithoutProxyingValidationSpecification();
399         assertTrue(validationSpecification.isSatisfiedBy(assertion));
400     }

402     /**
403      * This test checks that the TGT destruction happens properly for a remote registry.
404      * It previously failed when the deletion happens before the ticket was marked expired because an update was necessary for that.
405      *
406      * @throws AuthenticationException
407      * @throws TicketException
408      */
409     @Test
410     public void testDestroyRemoteRegistry() throws TicketException, AuthenticationException {
411         final MockOnlyOneTicketRegistry registry = new MockOnlyOneTicketRegistry();
412         final TicketGrantingTicketImpl tgt = new TicketGrantingTicketImpl("TGT-1", mock(Authentication.class),
413                 mock(ExpirationPolicy.class));
414         final MockExpireUpdateTicketLogoutManager logoutManager = new MockExpireUpdateTicketLogoutManager(registry);
415         // consider authentication has happened and the TGT is in the registry
416         registry.addTicket(tgt);
417         // create a new CASimpl
418         final CentralAuthenticationServiceImpl cas = new CentralAuthenticationServiceImpl(registry,  null,  null, null, null, null, null,
419                 null, logoutManager);
420         // destroy to mark expired and then delete : the opposite would fail with a "No ticket to update" error from the registry
421         cas.destroyTicketGrantingTicket(tgt.getId());
422     }
