19  package org.jasig.cas.support.saml.web.support;
20  
21  import java.security.PrivateKey;
22  import java.security.PublicKey;
23  
24  import javax.servlet.http.HttpServletRequest;
25  import javax.validation.constraints.NotNull;
26  
27  import org.jasig.cas.authentication.principal.WebApplicationService;
28  import org.jasig.cas.support.saml.authentication.principal.GoogleAccountsService;
29  import org.jasig.cas.web.support.AbstractArgumentExtractor;
30  
31  /**
32   * Constructs a GoogleAccounts compatible service and provides the public and
33   * private keys.
34   *
35   * @author Scott Battaglia
36   * @since 3.1
37   */
38  public final class GoogleAccountsArgumentExtractor extends AbstractArgumentExtractor {
39  
40      @NotNull
41      private PublicKey publicKey;
42  
43      @NotNull
44      private PrivateKey privateKey;
45  
46      private String alternateUsername;
47  
48      @Override
49      public WebApplicationService extractServiceInternal(final HttpServletRequest request) {
50          return GoogleAccountsService.createServiceFrom(request,
51                  this.privateKey, this.publicKey, this.alternateUsername);
52      }
53  
54      public void setPrivateKey(final PrivateKey privateKey) {
55          this.privateKey = privateKey;
56      }
57  
58      public void setPublicKey(final PublicKey publicKey) {
59          this.publicKey = publicKey;
60      }
61  
62      /**
63       * Sets an alternate username to send to Google (i.e. fully qualified email address).  Relies on an appropriate
64       * attribute available for the user.
65       * <p>
66       * Note that this is optional and the default is to use the normal identifier.
67       *
68       * @param alternateUsername the alternate username.  This is OPTIONAL.
69       */
70      public void setAlternateUsername(final String alternateUsername) {
71          this.alternateUsername = alternateUsername;
72      }
73  }
