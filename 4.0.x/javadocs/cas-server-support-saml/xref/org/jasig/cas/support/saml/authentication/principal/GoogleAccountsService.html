128     public static GoogleAccountsService createServiceFrom(
129             final HttpServletRequest request, final PrivateKey privateKey,
130             final PublicKey publicKey, final String alternateUserName) {
131         final String relayState = request.getParameter(CONST_RELAY_STATE);
132 
133         final String xmlRequest = decodeAuthnRequestXML(request
134                 .getParameter(CONST_PARAM_SERVICE));
135 
136         if (!StringUtils.hasText(xmlRequest)) {
137             return null;
138         }
139 
140         final Document document = SamlUtils
141                 .constructDocumentFromXmlString(xmlRequest);
142 
143         if (document == null) {
144             return null;
145         }
146 
147         final String assertionConsumerServiceUrl = document.getRootElement().getAttributeValue("AssertionConsumerServiceURL");
148         final String requestId = document.getRootElement().getAttributeValue("ID");
149 
150         return new GoogleAccountsService(assertionConsumerServiceUrl,
151                 relayState, requestId, privateKey, publicKey, alternateUserName);
152     }

176     private String constructSamlResponse() {
177         String samlResponse = TEMPLATE_SAML_RESPONSE;
178 
179         final Calendar c = Calendar.getInstance();
180         c.setTime(new Date());
181         c.add(Calendar.YEAR, 1);
182 
183         final String userId;
184 
185         if (this.alternateUserName == null) {
186             userId = getPrincipal().getId();
187         } else {
188             final String attributeValue = (String) getPrincipal().getAttributes().get(this.alternateUserName);
189             if (attributeValue == null) {
190                 userId = getPrincipal().getId();
191             } else {
192                 userId = attributeValue;
193             }
194         }
195 
196         samlResponse = samlResponse.replace("<USERNAME_STRING>", userId);
197         samlResponse = samlResponse.replace("<RESPONSE_ID>", createID());
198         samlResponse = samlResponse.replace("<ISSUE_INSTANT>", SamlDateUtils
199                 .getCurrentDateAndTime());
200         samlResponse = samlResponse.replace("<AUTHN_INSTANT>", SamlDateUtils
201                 .getCurrentDateAndTime());
202         samlResponse = samlResponse.replaceAll("<NOT_ON_OR_AFTER>", SamlDateUtils
203                 .getFormattedDateAndTime(c.getTime()));
204         samlResponse = samlResponse.replace("<ASSERTION_ID>", createID());
205         samlResponse = samlResponse.replaceAll("<ACS_URL>", getId());
206         samlResponse = samlResponse.replace("<REQUEST_ID>", this.requestId);
207 
208         return samlResponse;
209     }

211     private static String createID() {
212         final byte[] bytes = new byte[20]; // 160 bits
213         RANDOM_GENERATOR.nextBytes(bytes);
214 
215         final char[] chars = new char[40];
216 
217         for (int i = 0; i < bytes.length; i++) {
218             int left = bytes[i] >> 4 & 0x0f;
219             int right = bytes[i] & 0x0f;
220             chars[i * 2] = CHAR_MAPPINGS[left];
221             chars[i * 2 + 1] = CHAR_MAPPINGS[right];
222         }
223 
224         return String.valueOf(chars);
225     }

227     private static String decodeAuthnRequestXML(
228             final String encodedRequestXmlString) {
229         if (encodedRequestXmlString == null) {
230             return null;
231         }
232 
233         final byte[] decodedBytes = base64Decode(encodedRequestXmlString);
234 
235         if (decodedBytes == null) {
236             return null;
237         }
238 
239         final String inflated = inflate(decodedBytes);
240 
241         if (inflated != null) {
242             return inflated;
243         }
244 
245         return zlibDeflate(decodedBytes);
246     }

248     private static String zlibDeflate(final byte[] bytes) {
249         final ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
250         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
251         final InflaterInputStream iis = new InflaterInputStream(bais);
252         final byte[] buf = new byte[1024];
253 
254         try {
255             int count = iis.read(buf);
256             while (count != -1) {
257                 baos.write(buf, 0, count);
258                 count = iis.read(buf);
259             }
260             return new String(baos.toByteArray());
261         } catch (final Exception e) {
262             return null;
263         } finally {
264             IOUtils.closeQuietly(iis);
265         }
266     }

268     private static byte[] base64Decode(final String xml) {
269         try {
270             final byte[] xmlBytes = xml.getBytes("UTF-8");
271             return Base64.decodeBase64(xmlBytes);
272         } catch (final Exception e) {
273             return null;
274         }
275     }

277     private static String inflate(final byte[] bytes) {
278         final Inflater inflater = new Inflater(true);
279         final byte[] xmlMessageBytes = new byte[10000];
280 
281         final byte[] extendedBytes = new byte[bytes.length + 1];
282         System.arraycopy(bytes, 0, extendedBytes, 0, bytes.length);
283         extendedBytes[bytes.length] = 0;
284 
285         inflater.setInput(extendedBytes);
286 
287         try {
288             final int resultLength = inflater.inflate(xmlMessageBytes);
289             inflater.end();
290 
291             if (!inflater.finished()) {
292                 throw new RuntimeException("buffer not large enough.");
293             }
294 
295             inflater.end();
296             return new String(xmlMessageBytes, 0, resultLength, "UTF-8");
297         } catch (final DataFormatException e) {
298             return null;
299         } catch (final UnsupportedEncodingException e) {
300             throw new RuntimeException("Cannot find encoding: UTF-8", e);
301         }
302     }
