19  package org.jasig.cas.authentication.principal;

21  import java.net.URLEncoder;
22  import java.util.Map;
23  import java.util.regex.Matcher;
24  import java.util.regex.Pattern;

26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;

29  /**
30   * Encapsulates a Response to send back for a particular service.
31   *
32   * @author Scott Battaglia
33   * @author Arnaud Lesueur
34   * @since 3.1
35   */
36  public final class Response {
37      /** Pattern to detect unprintable ASCII characters. */
38      private static final Pattern NON_PRINTABLE =
39          Pattern.compile("[\\x00-\\x1F\\x7F]+");
40  
41      /** Log instance. */
42      protected static final Logger LOGGER = LoggerFactory.getLogger(Response.class);
43  
44      /** An enumeration of different response types. **/
45      public static enum ResponseType {
46          POST, REDIRECT
47      }
48  
49      private final ResponseType responseType;
50  
51      private final String url;
52  
53      private final Map<String, String> attributes;
54  
55      protected Response(final ResponseType responseType, final String url, final Map<String, String> attributes) {
56          this.responseType = responseType;
57          this.url = url;
58          this.attributes = attributes;
59      }
60  
61      public static Response getPostResponse(final String url, final Map<String, String> attributes) {
62          return new Response(ResponseType.POST, url, attributes);
63      }
64  
65      public static Response getRedirectResponse(final String url, final Map<String, String> parameters) {
66          final StringBuilder builder = new StringBuilder(parameters.size() * 40 + 100);
67          boolean isFirst = true;
68          final String[] fragmentSplit = sanitizeUrl(url).split("#");
69  
70          builder.append(fragmentSplit[0]);
71  
72          for (final Map.Entry<String, String> entry : parameters.entrySet()) {
73              if (entry.getValue() != null) {
74                  if (isFirst) {
75                      builder.append(url.contains("?") ? "&" : "?");
76                      isFirst = false;
77                  } else {
78                      builder.append("&");
79                  }
80                  builder.append(entry.getKey());
81                  builder.append("=");
82  
83                  try {
84                      builder.append(URLEncoder.encode(entry.getValue(), "UTF-8"));
85                  } catch (final Exception e) {
86                      builder.append(entry.getValue());
87                  }
88              }
89          }
90  
91          if (fragmentSplit.length > 1) {
92              builder.append("#");
93              builder.append(fragmentSplit[1]);
94          }
95  
96          return new Response(ResponseType.REDIRECT, builder.toString(), parameters);
97      }
98  
99      public Map<String, String> getAttributes() {
100         return this.attributes;
101     }
102 
103     public ResponseType getResponseType() {
104         return this.responseType;
105     }
106 
107     public String getUrl() {
108         return this.url;
109     }

111     /**
112      * Sanitize a URL provided by a relying party by normalizing non-printable
113      * ASCII character sequences into spaces.  This functionality protects
114      * against CRLF attacks and other similar attacks using invisible characters
115      * that could be abused to trick user agents.
116      *
117      * @param  url  URL to sanitize.
118      *
119      * @return  Sanitized URL string.
120      */
121     private static String sanitizeUrl(final String url) {
122         final Matcher m = NON_PRINTABLE.matcher(url);
123         final StringBuffer sb = new StringBuffer(url.length());
124         boolean hasNonPrintable = false;
125         while (m.find()) {
126             m.appendReplacement(sb, " ");
127             hasNonPrintable = true;
128         }
129         m.appendTail(sb);
130         if (hasNonPrintable) {
131             LOGGER.warn("The following redirect URL has been sanitized and may be sign of attack:\n{}", url);
132         }
133         return sb.toString();
134     }
