19  package org.jasig.cas.ticket.support;
20  
21  import org.jasig.cas.ticket.ExpirationPolicy;
22  import org.jasig.cas.ticket.TicketState;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  import org.springframework.beans.factory.InitializingBean;
26  import org.springframework.util.Assert;
27  
28  import java.io.Serializable;
29  import java.util.concurrent.TimeUnit;
30  
31  /**
32   * Provides the Ticket Granting Ticket expiration policy.  Ticket Granting Tickets
33   * can be used any number of times, have a fixed lifetime, and an idle timeout.
34   *
35   * @author William G. Thompson, Jr.
36  
37   * @since 3.4.10
38   */
39  public final class TicketGrantingTicketExpirationPolicy implements ExpirationPolicy, InitializingBean, Serializable {
40  
41      private static final Logger LOGGER = LoggerFactory.getLogger(TicketGrantingTicketExpirationPolicy.class);
42  
43      /** Serialization support. */
44      private static final long serialVersionUID = 7670537200691354820L;
45  
46      /** Maximum time this ticket is valid.  */
47      private long maxTimeToLiveInMilliSeconds;
48  
49      /** Time to kill in milliseconds. */
50      private long timeToKillInMilliSeconds;
51  
52      public void setMaxTimeToLiveInMilliSeconds(final long maxTimeToLiveInMilliSeconds){
53          this.maxTimeToLiveInMilliSeconds = maxTimeToLiveInMilliSeconds;
54      }
55  
56      public void setTimeToKillInMilliSeconds(final long timeToKillInMilliSeconds) {
57          this.timeToKillInMilliSeconds = timeToKillInMilliSeconds;
58      }
59  
60      /**
61       * Convenient virtual property setter to set time in seconds.
62       * @param maxTimeToLiveInSeconds max number of seconds for the tickets to stay alive
63       **/
64      public void setMaxTimeToLiveInSeconds(final long maxTimeToLiveInSeconds){
65          if(this.maxTimeToLiveInMilliSeconds == 0L) {
66              this.maxTimeToLiveInMilliSeconds = TimeUnit.SECONDS.toMillis(maxTimeToLiveInSeconds);
67          }
68      }
69  
70      /**
71       * @param timeToKillInSeconds time for the ticket to stay active in seconds
72       * Convenient virtual property setter to set time in seconds.
73       **/
74      public void setTimeToKillInSeconds(final long timeToKillInSeconds) {
75          if(this.timeToKillInMilliSeconds == 0L) {
76              this.timeToKillInMilliSeconds = TimeUnit.SECONDS.toMillis(timeToKillInSeconds);
77          }
78      }
79  
80      public void afterPropertiesSet() throws Exception {
81          Assert.isTrue((maxTimeToLiveInMilliSeconds >= timeToKillInMilliSeconds),
82                  "maxTimeToLiveInMilliSeconds must be greater than or equal to timeToKillInMilliSeconds.");
83      }
84  
85      public boolean isExpired(final TicketState ticketState) {
86          // Ticket has been used, check maxTimeToLive (hard window)
87          if ((System.currentTimeMillis() - ticketState.getCreationTime() >= maxTimeToLiveInMilliSeconds)) {
88              LOGGER.debug("Ticket is expired because the time since creation is greater than maxTimeToLiveInMilliSeconds");
89              return true;
90          }
91  
92          // Ticket is within hard window, check timeToKill (sliding window)
93          if ((System.currentTimeMillis() - ticketState.getLastTimeUsed() >= timeToKillInMilliSeconds)) {
94              LOGGER.debug("Ticket is expired because the time since last use is greater than timeToKillInMilliseconds");
95              return true;
96          }
97  
98          return false;
99      }
100
