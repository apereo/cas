38  /**
39   * Concrete implementation of a TicketGrantingTicket. A TicketGrantingTicket is
40   * the global identifier of a principal into the system. It grants the Principal
41   * single-sign on access to any service that opts into single-sign on.
42   * Expiration of a TicketGrantingTicket is controlled by the ExpirationPolicy
43   * specified as object creation.

45   * @author Scott Battaglia
46   * @since 3.0
47   */

120     /**
121      * {@inheritDoc}
122      * <p>The state of the ticket is affected by this operation and the
123      * ticket will be considered used. The state update subsequently may
124      * impact the ticket expiration policy in that, depending on the policy
125      * configuration, the ticket may be considered expired.
126      */
127     @Override
128     public synchronized ServiceTicket grantServiceTicket(final String id,
129         final Service service, final ExpirationPolicy expirationPolicy,
130         final boolean credentialsProvided) {
131         final ServiceTicket serviceTicket = new ServiceTicketImpl(id, this,
132             service, this.getCountOfUses() == 0 || credentialsProvided,
133             expirationPolicy);
134 
135         updateState();
136 
137         final List<Authentication> authentications = getChainedAuthentications();
138         service.setPrincipal(authentications.get(authentications.size()-1).getPrincipal());
139 
140         this.services.put(id, service);
141 
142         return serviceTicket;
143     }

145     /**
146      * Gets an immutable map of service ticket and services accessed by this ticket-granting ticket.
147      *
148      * @return an immutable map of service ticket and services accessed by this ticket-granting ticket.
149     */
150     @Override
151     public synchronized Map<String, Service> getServices() {
152         final Map<String, Service> map = new HashMap<String, Service>(services.size());
153         for (final String ticket : services.keySet()) {
154             map.put(ticket, services.get(ticket));
155         }
156         return Collections.unmodifiableMap(map);
157     }

213     public List<Authentication> getChainedAuthentications() {
214         final List<Authentication> list = new ArrayList<Authentication>();
215 
216         list.add(getAuthentication());
217 
218         if (getGrantingTicket() == null) {
219             return Collections.unmodifiableList(list);
220         }
221 
222         list.addAll(getGrantingTicket().getChainedAuthentications());
223         return Collections.unmodifiableList(list);
224     }
