19  package org.jasig.cas.monitor;
20  
21  import javax.validation.constraints.NotNull;
22  
23  /**
24   * Monitors the status of a {@link org.jasig.cas.ticket.registry.TicketRegistry}
25   * that supports the {@link TicketRegistryState} interface for exposing internal
26   * state information used in status reports.
27   *
28   * @author Marvin S. Addison
29   * @since 3.5.0
30   */
31  public class SessionMonitor implements Monitor<SessionStatus> {
32      /** Ticket registry instance that exposes state info. */
33      @NotNull
34      private TicketRegistryState registryState;
35  
36      /** Threshold above which warnings are issued for session count. */
37      private int sessionCountWarnThreshold = -1;
38  
39      /** Threshold above which warnings are issued for service ticket count. */
40      private int serviceTicketCountWarnThreshold = -1;
41  
42  
43      /**
44       * Sets the ticket registry that exposes state information that may be queried by this monitor.
45       * @param state the ticket registry state instance
46       */
47      public void setTicketRegistry(final TicketRegistryState state) {
48          this.registryState = state;
49      }
50  
51  
52      /**
53       * Sets the threshold above which warnings are issued for session counts in excess of value.
54       *
55       * @param threshold Warn threshold if non-negative value, otherwise warnings are disabled.
56       */
57      public void setSessionCountWarnThreshold(final int threshold) {
58          this.sessionCountWarnThreshold = threshold;
59      }
60  
61  
62      /**
63       * Sets the threshold above which warnings are issued for service ticket counts in excess of value.
64       *
65       * @param threshold Warn threshold if non-negative value, otherwise warnings are disabled.
66       */
67      public void setServiceTicketCountWarnThreshold(final int threshold) {
68          this.serviceTicketCountWarnThreshold = threshold;
69      }
70  
71  
72      /** {@inheritDoc} */
73      @Override
74      public String getName() {
75          return SessionMonitor.class.getSimpleName();
76      }
77  
78  
79      /** {@inheritDoc} */
80      @Override
81      public SessionStatus observe() {
82          try {
83              final int sessionCount = this.registryState.sessionCount();
84              final int ticketCount = this.registryState.serviceTicketCount();
85  
86              if (sessionCount == Integer.MIN_VALUE || ticketCount == Integer.MIN_VALUE) {
87                  return new SessionStatus(StatusCode.UNKNOWN,
88                                           String.format("Ticket registry %s reports unknown session and/or ticket counts.",
89                                           this.registryState.getClass().getName()),
90                                           sessionCount, ticketCount);
91              }
92  
93              final StringBuilder msg = new StringBuilder();
94              StatusCode code = StatusCode.OK;
95              if (this.sessionCountWarnThreshold > -1 && sessionCount > this.sessionCountWarnThreshold) {
96                  code = StatusCode.WARN;
97                  msg.append(String.format(
98                          "Session count (%s) is above threshold %s. ", sessionCount, this.sessionCountWarnThreshold));
99              } else {
100                 msg.append(sessionCount).append(" sessions. ");
101             }
102             if (this.serviceTicketCountWarnThreshold > -1 && ticketCount > this.serviceTicketCountWarnThreshold) {
103                 code = StatusCode.WARN;
104                 msg.append(String.format(
105                         "Service ticket count (%s) is above threshold %s.",
106                         ticketCount,
107                         this.serviceTicketCountWarnThreshold));
108             } else {
109                 msg.append(ticketCount).append(" service tickets.");
110             }
111             return new SessionStatus(code, msg.toString(), sessionCount, ticketCount);
112         } catch (final Exception e) {
113             return new SessionStatus(StatusCode.ERROR, e.getMessage());
114         }
115     }
116 }
