19  package org.jasig.cas.monitor;
20  
21  import java.util.Collection;
22  import java.util.Collections;
23  import java.util.LinkedHashMap;
24  import java.util.Map;
25  import javax.validation.constraints.NotNull;
26  
27  /**
28   * Simple health check monitor that reports the overall health as the greatest reported
29   * {@link StatusCode} of an arbitrary number of individual checks.
30   *
31   * @author Marvin S. Addison
32   * @since 3.5.0
33   */
34  public class HealthCheckMonitor implements Monitor<HealthStatus> {
35      /** Individual monitors that comprise health check. */
36      @NotNull
37      private Collection<Monitor> monitors = Collections.emptySet();
38  
39  
40      /**
41       * Sets the monitors that comprise the health check.
42       *
43       * @param monitors Collection of monitors responsible for observing various aspects of CAS.
44       */
45      public void setMonitors(final Collection<Monitor> monitors) {
46          this.monitors = monitors;
47      }
48  
49      /** {@inheritDoc} */
50      @Override
51      public String getName() {
52          return HealthCheckMonitor.class.getSimpleName();
53      }
54  
55      /** {@inheritDoc} */
56      @Override
57      public HealthStatus observe() {
58          final Map<String, Status> results = new LinkedHashMap<String, Status>(this.monitors.size());
59          StatusCode code = StatusCode.UNKNOWN;
60          Status result;
61          for (final Monitor monitor : this.monitors) {
62              try {
63                  result = monitor.observe();
64                  if (result.getCode().value() > code.value()) {
65                      code = result.getCode();
66                  }
67              } catch (final Exception e) {
68                  code = StatusCode.ERROR;
69                  result = new Status(code, e.getClass().getSimpleName() + ": " + e.getMessage());
70              }
71              results.put(monitor.getName(), result);
72          }
73  
74          return new HealthStatus(code, results);
75      }
76  }
