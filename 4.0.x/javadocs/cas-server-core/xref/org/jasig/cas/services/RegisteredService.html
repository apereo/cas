19  package org.jasig.cas.services;
20  
21  import java.io.Serializable;
22  import java.util.List;
23  import java.util.Set;
24  
25  import org.jasig.cas.authentication.principal.Service;
26  
27  /**
28   * Interface for a service that can be registered by the Services Management
29   * interface.
30   *
31   * @author Scott Battaglia
32   * @since 3.1
33   */
34  public interface RegisteredService extends Cloneable, Serializable {
35  
36      /** Initial ID value of newly created (but not persisted) registered service. **/
37      long INITIAL_IDENTIFIER_VALUE = Long.MAX_VALUE;
38  
39      /**
40       * Is this application currently allowed to use CAS?
41       *
42       * @return true if it can use CAS, false otherwise.
43       */
44      boolean isEnabled();
45  
46      /**
47       * Determines whether the service is allowed anonymous or privileged access
48       * to user information. Anonymous access should not return any identifying
49       * information such as user id.
50       *
51       * @return if we should use a pseudo random identifier instead of their real id
52       */
53      boolean isAnonymousAccess();
54  
55      /**
56       * Sets whether we should bother to read the attribute list or not.
57       *
58       * @return true if we should read it, false otherwise.
59       */
60      boolean isIgnoreAttributes();
61  
62      /**
63       * Returns the list of allowed attributes.
64       *
65       * @return the list of attributes
66       */
67      List<String> getAllowedAttributes();
68  
69      /**
70       * Is this application allowed to take part in the proxying capabilities of
71       * CAS?
72       *
73       * @return true if it can, false otherwise.
74       */
75      boolean isAllowedToProxy();
76  
77      /**
78       * The unique identifier for this service.
79       *
80       * @return the unique identifier for this service.
81       */
82      String getServiceId();
83  
84      /**
85       * The numeric identifier for this service. Implementations
86       * are expected to initialize the id with the value of {@link #INITIAL_IDENTIFIER_VALUE}.
87       * @return the numeric identifier for this service.
88       */
89      long getId();
90  
91      /**
92       * Returns the name of the service.
93       *
94       * @return the name of the service.
95       */
96      String getName();
97  
98      /**
99       * Returns a short theme name. Services do not need to have unique theme
100      * names.
101      *
102      * @return the theme name associated with this service.
103      */
104     String getTheme();
105 
106     /**
107      * Does this application participate in the SSO session?
108      *
109      * @return true if it does, false otherwise.
110      */
111     boolean isSsoEnabled();
112 
113     /**
114      * Returns the description of the service.
115      *
116      * @return the description of the service.
117      */
118     String getDescription();
119 
120     /**
121      * Gets the relative evaluation order of this service when determining
122      * matches.
123      * @return Evaluation order relative to other registered services.
124      * Services with lower values will be evaluated for a match before others.
125      */
126     int getEvaluationOrder();
127 
128     /**
129      * Sets the relative evaluation order of this service when determining
130      * matches.
131      * @param evaluationOrder the service evaluation order
132      */
133     void setEvaluationOrder(final int evaluationOrder);
134 
135     /**
136      * Get the name of the attribute this service prefers to consume as username.
137      *
138      * @return Either of the following values:
139      * <ul>
140      *  <li><code>String</code> representing the name of the attribute to consume as username</li>
141      *  <li><code>null</code> indicating the default username</li>
142      * </ul>
143      */
144     String getUsernameAttribute();
145 
146     /**
147      * Gets the set of handler names that must successfully authenticate credentials in order to access the service.
148      * An empty set indicates that there are no requirements on particular authentication handlers; any will suffice.
149      *
150      * @return Non-null set of required handler names.
151      */
152     Set<String> getRequiredHandlers();
153 
154     /**
155      * Returns whether the service matches the registered service.
156      * <p>Note, as of 3.1.2, matches are case insensitive.
157      *
158      * @param service the service to match.
159      * @return true if they match, false otherwise.
160      */
161     boolean matches(final Service service);
162 
163     RegisteredService clone() throws CloneNotSupportedException;
164 
165     /**
166      * An instance of the attribute filter that imposes validation rules over
167      * the attribute release policy.
168      * @return An instance of an attribute filter for this service
169      */
170     RegisteredServiceAttributeFilter getAttributeFilter();
171 
172     /**
173      * Returns the logout type of the service.
174      *
175      * @return the logout type of the service.
176      */
177     LogoutType getLogoutType();
178 }
