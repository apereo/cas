19  package org.jasig.cas.monitor;
20  
21  import javax.validation.constraints.NotNull;
22  
23  import net.sf.ehcache.Cache;
24  
25  /**
26   * Monitors a {@link net.sf.ehcache.Cache} instance.
27   * The accuracy of statistics is governed by the value of {@link Cache#getStatistics()}.
28   *
29   * <p>NOTE: computation of highly accurate statistics is expensive.</p>
30   *
31   * @author Marvin S. Addison
32   * @since 3.5.1
33   */
34  public class EhCacheMonitor extends AbstractCacheMonitor {
35  
36      @NotNull
37      private final Cache cache;
38  
39      public EhCacheMonitor(final Cache cache) {
40          this.cache = cache;
41      }
42  
43      @Override
44      protected CacheStatistics[] getStatistics() {
45          return new EhCacheStatistics[] {new EhCacheStatistics(cache)};
46      }
47  }
