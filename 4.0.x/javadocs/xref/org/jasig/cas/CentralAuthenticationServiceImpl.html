70  /**

72   * central, organizing component of CAS's internal implementation.
73   * <p>
74   * This class is threadsafe.
75   * <p>

90   * <li> <code>serviceTicketExpirationPolicy</code> - The expiration policy for

198     /**
199      * {@inheritDoc}
200      * Destroy a TicketGrantingTicket and perform back channel logout. This has the effect of invalidating any
201      * Ticket that was derived from the TicketGrantingTicket being destroyed. May throw an
202      * {@link IllegalArgumentException} if the TicketGrantingTicket ID is null.
203      *
204      * @param ticketGrantingTicketId the id of the ticket we want to destroy
205      * @return the logout requests.
206      */
207     @Audit(
208             action="TICKET_GRANTING_TICKET_DESTROYED",
209             actionResolverName="DESTROY_TICKET_GRANTING_TICKET_RESOLVER",
210             resourceResolverName="DESTROY_TICKET_GRANTING_TICKET_RESOURCE_RESOLVER")
211     @Profiled(tag = "DESTROY_TICKET_GRANTING_TICKET", logFailuresSeparately = false)
212     @Transactional(readOnly = false)
213     @Override
214     public List<LogoutRequest> destroyTicketGrantingTicket(final String ticketGrantingTicketId) {
215         Assert.notNull(ticketGrantingTicketId);
216 
217         logger.debug("Removing ticket [{}] from registry.", ticketGrantingTicketId);
218         final TicketGrantingTicket ticket = this.ticketRegistry.getTicket(ticketGrantingTicketId,
219                 TicketGrantingTicket.class);
220 
221         if (ticket == null) {
222             logger.debug("TicketGrantingTicket [{}] cannot be found in the ticket registry.", ticketGrantingTicketId);
223             return Collections.emptyList();
224         }
225 
226         logger.debug("Ticket found. Processing logout requests and then deleting the ticket...");
227         final List<LogoutRequest> logoutRequests = logoutManager.performLogout(ticket);
228         this.ticketRegistry.deleteTicket(ticketGrantingTicketId);
229 
230         return logoutRequests;
231     }

233     /**
234      * @throws IllegalArgumentException if ticketGrantingTicketId or service are null.
235      */
236     @Audit(
237         action="SERVICE_TICKET",
238         actionResolverName="GRANT_SERVICE_TICKET_RESOLVER",
239         resourceResolverName="GRANT_SERVICE_TICKET_RESOURCE_RESOLVER")
240     @Profiled(tag="GRANT_SERVICE_TICKET", logFailuresSeparately = false)
241     @Transactional(readOnly = false)
242     public String grantServiceTicket(
243             final String ticketGrantingTicketId, final Service service, final Credential... credentials)
244             throws AuthenticationException, TicketException {
245         Assert.notNull(ticketGrantingTicketId, "ticketGrantingticketId cannot be null");
246         Assert.notNull(service, "service cannot be null");
247 
248         final TicketGrantingTicket ticketGrantingTicket = this.ticketRegistry.getTicket(ticketGrantingTicketId, TicketGrantingTicket.class);
249 
250         if (ticketGrantingTicket == null) {
251             logger.debug("TicketGrantingTicket [{}] cannot be found in the ticket registry.", ticketGrantingTicketId);
252             throw new InvalidTicketException(ticketGrantingTicketId);
253         }
254 
255         synchronized (ticketGrantingTicket) {
256             if (ticketGrantingTicket.isExpired()) {
257                 this.ticketRegistry.deleteTicket(ticketGrantingTicketId);
258                 logger.debug("TicketGrantingTicket[{}] has expired and is now deleted from the ticket registry.", ticketGrantingTicketId);
259                 throw new InvalidTicketException(ticketGrantingTicketId);
260             }
261         }
262 
263         final RegisteredService registeredService = this.servicesManager.findServiceBy(service);
264 
265         verifyRegisteredServiceProperties(registeredService, service);
266         
267         if (!registeredService.isSsoEnabled() && credentials == null
268             && ticketGrantingTicket.getCountOfUses() > 0) {
269             logger.warn("ServiceManagement: Service [{}] is not allowed to use SSO.", service.getId());
270             throw new UnauthorizedSsoServiceException();
271         }
272 
273         //CAS-1019
274         final List<Authentication> authns = ticketGrantingTicket.getChainedAuthentications();
275         if(authns.size() > 1) {
276             if (!registeredService.isAllowedToProxy()) {
277                 final String message = String.
278                         format("ServiceManagement: Proxy attempt by service [%s] (registered service [%s]) is not allowed.",
279                         service.getId(), registeredService.toString());
280                 logger.warn(message);
281                 throw new UnauthorizedProxyingException(message);
282             }
283         }
284 
285         if (credentials != null) {
286             final Authentication current = this.authenticationManager.authenticate(credentials);
287             final Authentication original = ticketGrantingTicket.getAuthentication();
288             if (!current.getPrincipal().equals(original.getPrincipal())) {
289                 throw new MixedPrincipalException(current, current.getPrincipal(), original.getPrincipal());
290             }
291             ticketGrantingTicket.getSupplementalAuthentications().add(current);
292         }
293 
294         // Perform security policy check by getting the authentication that satisfies the configured policy
295         // This throws if no suitable policy is found
296         getAuthenticationSatisfiedByPolicy(ticketGrantingTicket, new ServiceContext(service, registeredService));
297 
298         final String uniqueTicketIdGenKey = service.getClass().getName();
299         if (!this.uniqueTicketIdGeneratorsForService.containsKey(uniqueTicketIdGenKey)) {
300             logger.warn("Cannot create service ticket because the key [{}] for service [{}] is not linked to a ticket id generator",
301                     uniqueTicketIdGenKey, service.getId());
302             throw new UnauthorizedSsoServiceException();
303         }
304         
305         final UniqueTicketIdGenerator serviceTicketUniqueTicketIdGenerator =
306                 this.uniqueTicketIdGeneratorsForService.get(uniqueTicketIdGenKey);
307 
308         final String generatedServiceTicketId = serviceTicketUniqueTicketIdGenerator.getNewTicketId(ServiceTicket.PREFIX);
309         logger.debug("Generated service ticket id [{}] for ticket granting ticket [{}]",
310                 generatedServiceTicketId, ticketGrantingTicket.getId());
311         
312         final ServiceTicket serviceTicket = ticketGrantingTicket.grantServiceTicket(generatedServiceTicketId, service,
313                 this.serviceTicketExpirationPolicy, credentials != null);
314 
315         this.serviceTicketRegistry.addTicket(serviceTicket);
316 
317         if (logger.isInfoEnabled()) {
318             final List<Authentication> authentications = serviceTicket.getGrantingTicket().getChainedAuthentications();
319             final String formatString = "Granted %s ticket [%s] for service [%s] for user [%s]";
320             final String type;
321             final String principalId = authentications.get(authentications.size() - 1).getPrincipal().getId();
322 
323             if (authentications.size() == 1) {
324                 type = "service";
325             } else {
326                 type = "proxy";
327             }
328 
329             logger.info(String.format(formatString, type, serviceTicket.getId(), service.getId(), principalId));
330         }
331 
332         return serviceTicket.getId();
333     }

350     /**
351      * @throws IllegalArgumentException if the ServiceTicketId or the
352      * Credential are null.
353      */
354     @Audit(
355         action="PROXY_GRANTING_TICKET",
356         actionResolverName="GRANT_PROXY_GRANTING_TICKET_RESOLVER",
357         resourceResolverName="GRANT_PROXY_GRANTING_TICKET_RESOURCE_RESOLVER")
358     @Profiled(tag="GRANT_PROXY_GRANTING_TICKET", logFailuresSeparately = false)
359     @Transactional(readOnly = false)
360     public String delegateTicketGrantingTicket(final String serviceTicketId, final Credential... credentials)
361             throws AuthenticationException, TicketException {
362 
363         Assert.notNull(serviceTicketId, "serviceTicketId cannot be null");
364         Assert.notNull(credentials, "credentials cannot be null");
365 
366         final ServiceTicket serviceTicket =  this.serviceTicketRegistry.getTicket(serviceTicketId, ServiceTicket.class);
367 
368         if (serviceTicket == null || serviceTicket.isExpired()) {
369             logger.debug("ServiceTicket [{}] has expired or cannot be found in the ticket registry", serviceTicketId);
370             throw new InvalidTicketException(serviceTicketId);
371         }
372 
373         final RegisteredService registeredService = this.servicesManager
374                 .findServiceBy(serviceTicket.getService());
375 
376         verifyRegisteredServiceProperties(registeredService, serviceTicket.getService());
377         
378         if (!registeredService.isAllowedToProxy()) {
379             logger.warn("ServiceManagement: Service [{}] attempted to proxy, but is not allowed.", serviceTicket.getService().getId());
380             throw new UnauthorizedProxyingException();
381         }
382 
383         final Authentication authentication = this.authenticationManager.authenticate(credentials);
384 
385         final TicketGrantingTicket ticketGrantingTicket = serviceTicket
386                 .grantTicketGrantingTicket(
387                         this.ticketGrantingTicketUniqueTicketIdGenerator
388                                 .getNewTicketId(TicketGrantingTicket.PREFIX),
389                         authentication, this.ticketGrantingTicketExpirationPolicy);
390 
391         this.ticketRegistry.addTicket(ticketGrantingTicket);
392 
393         return ticketGrantingTicket.getId();
394     }

396     /**
397      * @throws IllegalArgumentException if the ServiceTicketId or the Service
398      * are null.
399      */
400     @Audit(
401         action="SERVICE_TICKET_VALIDATE",
402         actionResolverName="VALIDATE_SERVICE_TICKET_RESOLVER",
403         resourceResolverName="VALIDATE_SERVICE_TICKET_RESOURCE_RESOLVER")
404     @Profiled(tag="VALIDATE_SERVICE_TICKET", logFailuresSeparately = false)
405     @Transactional(readOnly = false)
406     public Assertion validateServiceTicket(final String serviceTicketId, final Service service) throws TicketException {
407         Assert.notNull(serviceTicketId, "serviceTicketId cannot be null");
408         Assert.notNull(service, "service cannot be null");
409  
410         final ServiceTicket serviceTicket =  this.serviceTicketRegistry.getTicket(serviceTicketId, ServiceTicket.class);
411 
412         if (serviceTicket == null) {
413             logger.info("ServiceTicket [{}] does not exist.", serviceTicketId);
414             throw new InvalidTicketException(serviceTicketId);
415         }
416 
417         final RegisteredService registeredService = this.servicesManager.findServiceBy(service);
418 
419         verifyRegisteredServiceProperties(registeredService, serviceTicket.getService());
420         
421         try {
422             synchronized (serviceTicket) {
423                 if (serviceTicket.isExpired()) {
424                     logger.info("ServiceTicket [{}] has expired.", serviceTicketId);
425                     throw new InvalidTicketException(serviceTicketId);
426                 }
427 
428                 if (!serviceTicket.isValidFor(service)) {
429                     logger.error("ServiceTicket [{}] with service [{}] does not match supplied service [{}]",
430                             serviceTicketId, serviceTicket.getService().getId(), service);
431                     throw new TicketValidationException(serviceTicket.getService());
432                 }
433             }
434 
435             final TicketGrantingTicket root = serviceTicket.getGrantingTicket().getRoot();
436             final Authentication authentication = getAuthenticationSatisfiedByPolicy(
437                     root, new ServiceContext(serviceTicket.getService(), registeredService));
438             final Principal principal = authentication.getPrincipal();
439 
440             Map<String, Object> attributesToRelease = this.defaultAttributeFilter.filter(principal.getId(),
441                     principal.getAttributes(), registeredService);
442             if (registeredService.getAttributeFilter() != null) {
443                 attributesToRelease = registeredService.getAttributeFilter().filter(principal.getId(),
444                         attributesToRelease, registeredService);
445             }
446 
447             final String principalId = determinePrincipalIdForRegisteredService(principal, registeredService, serviceTicket);
448             final Principal modifiedPrincipal = new SimplePrincipal(principalId, attributesToRelease);
449             final AuthenticationBuilder builder = AuthenticationBuilder.newInstance(authentication);
450             builder.setPrincipal(modifiedPrincipal);
451 
452             return new ImmutableAssertion(
453                     builder.build(),
454                     serviceTicket.getGrantingTicket().getChainedAuthentications(),
455                     serviceTicket.getService(),
456                     serviceTicket.isFromNewLogin());
457         } finally {
458             if (serviceTicket.isExpired()) {
459                 this.serviceTicketRegistry.deleteTicket(serviceTicketId);
460             }
461         }
462     }

464     /**
465      * Determines the principal id to use for a {@link RegisteredService} using the following rules:
466      *
467      * <ul>
468      *  <li> If the service is marked to allow anonymous access, a persistent id is returned. </li>
469      *  <li> If the {@link org.jasig.cas.services.RegisteredService#getUsernameAttribute()} is blank, then the default
470      *  principal id is returned.</li>
471      *  <li>If the username attribute is available as part of the principal's attributes,
472      *  the corresponding attribute value will be returned.
473      *  </li>
474      *   <li>Otherwise, the default principal's id is returned as the username attribute
475      *   with an additional warning.</li>
476      * </ul>
477      *
478      * @param principal The principal object to be validated and constructed
479      * @param registeredService Requesting service for which a principal is being validated.
480      * @param serviceTicket An instance of the service ticket used for validation
481      *
482      * @return The principal id to use for the requesting registered service
483      */
484     private String determinePrincipalIdForRegisteredService(final Principal principal,
485                                                             final RegisteredService registeredService,
486                                                             final ServiceTicket serviceTicket) {
487         String principalId = null;
488         final String serviceUsernameAttribute = registeredService.getUsernameAttribute();
489 
490         if (registeredService.isAnonymousAccess()) {
491             principalId = this.persistentIdGenerator.generate(principal, serviceTicket.getService());
492         } else if (StringUtils.isBlank(serviceUsernameAttribute)) {
493             principalId = principal.getId();
494         } else {
495             if (principal.getAttributes().containsKey(serviceUsernameAttribute)) {
496                 principalId = principal.getAttributes().get(serviceUsernameAttribute).toString();
497             } else {
498                 principalId = principal.getId();
499                 final Object[] errorLogParameters = new Object[] {
500                         principalId,
501                         registeredService.getUsernameAttribute(),
502                         principal.getAttributes(),
503                         registeredService.getServiceId(),
504                         principalId };
505                 logger.warn("Principal [{}] did not have attribute [{}] among attributes [{}] so CAS cannot "
506                         + "provide on the validation response the user attribute the registered service [{}] expects. "
507                         + "CAS will instead return the default username attribute [{}]", errorLogParameters);
508             }
509 
510         }
511 
512         logger.debug("Principal id to return for service [{}] is [{}]. The default principal id is [{}].",
513                 new Object[]{registeredService.getName(), principal.getId(), principalId});
514         return principalId;
515     }

565     private Authentication getAuthenticationSatisfiedByPolicy(
566             final TicketGrantingTicket ticket, final ServiceContext context) throws TicketException {
567 
568         final ContextualAuthenticationPolicy<ServiceContext> policy =
569                 serviceContextAuthenticationPolicyFactory.createPolicy(context);
570         if (policy.isSatisfiedBy(ticket.getAuthentication())) {
571             return ticket.getAuthentication();
572         }
573         for (final Authentication auth : ticket.getSupplementalAuthentications()) {
574             if (policy.isSatisfiedBy(auth)) {
575                 return auth;
576             }
577         }
578         throw new UnsatisfiedAuthenticationPolicyException(policy);
579     }
