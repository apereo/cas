20  package org.jasig.cas.authentication.support;
21  
22  import java.util.List;
23  
24  import javax.validation.constraints.NotNull;
25  
26  import org.jasig.cas.Message;
27  import org.ldaptive.LdapAttribute;
28  import org.ldaptive.auth.AccountState;
29  import org.ldaptive.auth.AuthenticationResponse;
30  
31  /**
32   * The component supports both opt-in and opt-out warnings on a per-user basis using a simple algorithm of three
33   * variables:
34   * <ol>
35   *     <li>{@link #setWarningAttributeName(String) warningAttributeName}</li>
36   *     <li>{@link #setWarningAttributeValue(String)} warningAttributeValue}</li>
37   *     <li>{@link #setDisplayWarningOnMatch(boolean) displayWarningOnMatch}</li>
38   * </ol>
39   * The first two parameters define an attribute on the user entry to match on, and the third parameter determines
40   * whether password expiration warnings should be displayed on match.
41   * <p>
42   * Deployers MUST configure LDAP components to provide <code>warningAttributeName</code> in the set of attributes
43   * returned from the LDAP query for user details.
44   *
45   * @author Marvin S. Addison
46   * @since 4.0.0
47   */
48  public class OptionalWarningAccountStateHandler extends DefaultAccountStateHandler {
49  
50      /** Name of user attribute that describes whether or not to display expiration warnings. */
51      @NotNull
52      private String warningAttributeName;
53  
54      /** Attribute value to match. */
55      @NotNull
56      private String warningAttributeValue;
57  
58      /** True to opt into password expiration warnings on match, false to opt out on match. */
59      private boolean displayWarningOnMatch = true;
60  
61  
62      /**
63       * Sets the user attribute used to determine whether to display password expiration warnings.
64       *
65       * @param warningAttributeName Attribute on authenticated user entry.
66       */
67      public void setWarningAttributeName(final String warningAttributeName) {
68          this.warningAttributeName = warningAttributeName;
69      }
70  
71      /**
72       * Sets the value of {@link #warningAttributeName} used as basis of comparison.
73       *
74       * @param warningAttributeValue Value to match against.
75       */
76      public void setWarningAttributeValue(final String warningAttributeValue) {
77          this.warningAttributeValue = warningAttributeValue;
78      }
79  
80      /**
81       * Determines whether password expiration warnings are opt-in or opt-out.
82       *
83       * @param displayWarningOnMatch True to opt into password expiration warnings on match, false to opt out on match.
84       *                              Default is true.
85       */
86      public void setDisplayWarningOnMatch(final boolean displayWarningOnMatch) {
87          this.displayWarningOnMatch = displayWarningOnMatch;
88      }
89  
90      @Override
91      protected void handleWarning(
92              final AccountState.Warning warning,
93              final AuthenticationResponse response,
94              final LdapPasswordPolicyConfiguration configuration,
95              final List<Message> messages) {
96  
97          final LdapAttribute attribute = response.getLdapEntry().getAttribute(this.warningAttributeName);
98          boolean matches = false;
99          if (attribute != null) {
100             logger.debug("Found warning attribute {} with value {}", attribute.getName(), attribute.getStringValue());
101             matches = this.warningAttributeValue.equals(attribute.getStringValue());
102         }
103         logger.debug("matches={}, displayWarningOnMatch={}", matches, displayWarningOnMatch);
104         if (displayWarningOnMatch == matches) {
105             super.handleWarning(warning, response, configuration, messages);
106         }
107     }
108 }
