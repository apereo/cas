19  package org.jasig.cas.adaptors.x509.authentication.principal;
20  
21  import java.security.cert.X509Certificate;
22  
23  import org.jasig.cas.adaptors.x509.util.CertUtils;
24  import org.jasig.cas.authentication.AbstractCredential;
25  
26  /**
27   * An X.509 certificate credential.
28   *
29   * @author Scott Battaglia
30   * @author Marvin S. Addison
31   * @since 3.0.4
32   *
33   */
34  public final class X509CertificateCredential extends AbstractCredential {
35  
36      /** Unique Id for serialization. */
37      private static final long serialVersionUID = 631753409512746474L;
38  
39      /** The collection of certificates sent with the request. */
40      private final X509Certificate[] certificates;
41  
42      /** The certificate that we actually use. */
43      private X509Certificate certificate;
44  
45      public X509CertificateCredential(final X509Certificate[] certificates) {
46          this.certificates = certificates;
47      }
48  
49      public X509Certificate[] getCertificates() {
50          return this.certificates;
51      }
52  
53      public void setCertificate(final X509Certificate certificate) {
54          this.certificate = certificate;
55      }
56  
57      public X509Certificate getCertificate() {
58          return this.certificate;
59      }
60  
61      @Override
62      public String getId() {
63          X509Certificate cert = null;
64          if (this.certificate != null) {
65              cert = this.certificate;
66          } else if (this.certificates != null && this.certificates.length > 0) {
67              cert = this.certificates[0];
68          }
69  
70          if (cert != null) {
71              return CertUtils.toString(cert);
72          }
73          return UNKNOWN_ID;
74      }
75  }
