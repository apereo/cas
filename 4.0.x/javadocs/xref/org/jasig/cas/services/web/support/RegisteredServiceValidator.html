19  package org.jasig.cas.services.web.support;
20  
21  import java.util.Set;
22  
23  import org.apache.commons.lang.StringUtils;
24  import org.jasig.cas.services.RegisteredService;
25  import org.jasig.cas.services.ServicesManager;
26  import org.jasig.services.persondir.IPersonAttributeDao;
27  import org.springframework.validation.Errors;
28  import org.springframework.validation.Validator;
29  
30  import javax.validation.constraints.Min;
31  import javax.validation.constraints.NotNull;
32  
33  /**
34   * RegisteredServiceValidator ensures that a new RegisteredService does not have
35   * a conflicting Service Id with another service already in the registry.
36   *
37   * @author Scott Battaglia
38   * @since 3.1
39   */
40  public final class RegisteredServiceValidator implements Validator {
41  
42      /** Default length, which matches what is in the view. */
43      private static final int DEFAULT_MAX_DESCRIPTION_LENGTH = 300;
44  
45      /** {@link ServicesManager} to look up services. */
46      @NotNull
47      private ServicesManager servicesManager;
48  
49      /** The maximum length of the description we will accept. */
50      @Min(0)
51      private int maxDescriptionLength = DEFAULT_MAX_DESCRIPTION_LENGTH;
52  
53      /** {@link IPersonAttributeDao} to manage person attributes. */
54      @NotNull
55      private IPersonAttributeDao personAttributeDao;
56  
57      /**
58       * {@inheritDoc}
59       * Supports {@link RegisteredService} objects.
60       *
61       * @see org.springframework.validation.Validator#supports(java.lang.Class)
62       */
63      @Override
64      public boolean supports(final Class<?> clazz) {
65          return RegisteredService.class.isAssignableFrom(clazz);
66      }
67  
68      @Override
69      public void validate(final Object o, final Errors errors) {
70          final RegisteredService r = (RegisteredService) o;
71  
72          if (r.getServiceId() != null) {
73              for (final RegisteredService service : this.servicesManager.getAllServices()) {
74                  if (r.getServiceId().equals(service.getServiceId())
75                          && r.getId() != service.getId()) {
76                      errors.rejectValue("serviceId",
77                              "registeredService.serviceId.exists", null);
78                      break;
79                  }
80              }
81          }
82  
83          if (r.getDescription() != null
84                  && r.getDescription().length() > this.maxDescriptionLength) {
85              errors.rejectValue("description",
86                      "registeredService.description.length", null);
87          }
88  
89          if (!StringUtils.isBlank(r.getUsernameAttribute()) && !r.isAnonymousAccess()) {
90              if (!r.isIgnoreAttributes() && !r.getAllowedAttributes().contains(r.getUsernameAttribute())) {
91                  errors.rejectValue("usernameAttribute", "registeredService.usernameAttribute.notAvailable",
92                          "This attribute is not available for this service.");
93              } else {
94                  Set<String> availableAttributes = this.personAttributeDao.getPossibleUserAttributeNames();
95                  if (availableAttributes != null) {
96                      if (!availableAttributes.contains(r.getUsernameAttribute())) {
97                          errors.rejectValue("usernameAttribute", "registeredService.usernameAttribute.notAvailable",
98                                  "This attribute is not available from configured user attribute sources.");
99                      }
100                 }
101             }
102         }
103     }
104 
105     public void setServicesManager(final ServicesManager serviceRegistry) {
106         this.servicesManager = serviceRegistry;
107     }
108 
109     public void setMaxDescriptionLength(final int maxLength) {
110         this.maxDescriptionLength = maxLength;
111     }
112 
113     public void setPersonAttributeDao(final IPersonAttributeDao personAttributeDao) {
114         this.personAttributeDao = personAttributeDao;
115     }
116 }
