20  package org.jasig.cas.web;
21  
22  import org.springframework.web.servlet.ModelAndView;
23  import org.springframework.web.servlet.mvc.AbstractController;
24  
25  import javax.servlet.http.HttpServletRequest;
26  import javax.servlet.http.HttpServletResponse;
27  import javax.validation.constraints.NotNull;
28  import java.util.List;
29  
30  /**
31   * Delegating controller.
32   * Tries to find a controller among its delegates, that can handle the current request.
33   * If none is found, an error is generated.
34   * @author Frederic Esnault
35   * @since 3.5
36   */
37  public class DelegatingController extends AbstractController {
38      private List<DelegateController> delegates;
39      /** View if Service Ticket Validation Fails. */
40      private static final String DEFAULT_ERROR_VIEW_NAME = "casServiceFailureView";
41  
42      /** The view to redirect if no delegate can handle the request. */
43      @NotNull
44      private String failureView = DEFAULT_ERROR_VIEW_NAME;
45  
46      /**
47       * Handles the request.
48       * Ask all delegates if they can handle the current request.
49       * The first to answer true is elected as the delegate that will process the request.
50       * If no controller answers true, we redirect to the error page.
51       * @param request the request to handle
52       * @param response the response to write to
53       * @return the model and view object
54       * @throws Exception if an error occurs during request handling
55       */
56      @Override
57      protected final ModelAndView handleRequestInternal(final HttpServletRequest request, final HttpServletResponse response)
58                                                      throws Exception {
59          for (DelegateController delegate : delegates) {
60              if (delegate.canHandle(request, response)) {
61                  return delegate.handleRequest(request, response);
62              }
63          }
64          return generateErrorView("INVALID_REQUEST", "INVALID_REQUEST", null);
65      }
66  
67      private ModelAndView generateErrorView(final String code, final String description, final Object[] args) {
68          final ModelAndView modelAndView = new ModelAndView(this.failureView);
69          final String convertedDescription = getMessageSourceAccessor().getMessage(description, args, description);
70          modelAndView.addObject("code", code);
71          modelAndView.addObject("description", convertedDescription);
72  
73          return modelAndView;
74      }
75  
76      /**
77       * @param delegates the delegate controllers to set
78       */
79      @NotNull
80      public void setDelegates(final List<DelegateController> delegates) {
81          this.delegates = delegates;
82      }
83  
84      /**
85       * @param failureView The failureView to set.
86       */
87      public final void setFailureView(final String failureView) {
88          this.failureView = failureView;
89      }
90  }
