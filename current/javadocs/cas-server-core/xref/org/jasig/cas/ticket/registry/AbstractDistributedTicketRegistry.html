31  /**
32   * Abstract Implementation that handles some of the commonalities between
33   * distributed ticket registries.
34   *
35   * @author Scott Battaglia
36  
37   * @since 3.1
38   */
39  public abstract class AbstractDistributedTicketRegistry extends AbstractTicketRegistry {
40  
41      protected abstract void updateTicket(final Ticket ticket);
42  
43      protected abstract boolean needsCallback();
44  
45      protected final Ticket getProxiedTicketInstance(final Ticket ticket) {
46          if (ticket == null) {
47              return null;
48          }
49  
50          if (ticket instanceof TicketGrantingTicket) {
51              return new TicketGrantingTicketDelegator(this, (TicketGrantingTicket) ticket, needsCallback());
52          }
53  
54          return new ServiceTicketDelegator(this, (ServiceTicket) ticket, needsCallback());
55      }
56  
57      private static class TicketDelegator<T extends Ticket> implements Ticket {
58  
59          private static final long serialVersionUID = 1780193477774123440L;
60  
61          private final AbstractDistributedTicketRegistry ticketRegistry;
62  
63          private final T ticket;
64  
65          private final boolean callback;
66  
67          protected TicketDelegator(final AbstractDistributedTicketRegistry ticketRegistry,
68                  final T ticket, final boolean callback) {
69              this.ticketRegistry = ticketRegistry;
70              this.ticket = ticket;
71              this.callback = callback;
72          }
73  
74          protected void updateTicket() {
75              this.ticketRegistry.updateTicket(this.ticket);
76          }
77  
78          protected T getTicket() {
79              return this.ticket;
80          }
81  
82          public final String getId() {
83              return this.ticket.getId();
84          }
85  
86          public final boolean isExpired() {
87              if (!callback) {
88                  return this.ticket.isExpired();
89              }
90  
91              final TicketGrantingTicket t = getGrantingTicket();
92  
93              return this.ticket.isExpired() || (t != null && t.isExpired());
94          }
95  
96          public final TicketGrantingTicket getGrantingTicket() {
97              final TicketGrantingTicket old = this.ticket.getGrantingTicket();
98  
99              if (old == null || !callback) {
100                 return old;
101             }
102 
103             return this.ticketRegistry.getTicket(old.getId(), Ticket.class);
104         }
105 
106         public final long getCreationTime() {
107             return this.ticket.getCreationTime();
108         }
109 
110         public final int getCountOfUses() {
111             return this.ticket.getCountOfUses();
112         }
113 
114         @Override
115         public int hashCode() {
116             return this.ticket.hashCode();
117         }
118 
119         @Override
120         public boolean equals(final Object o) {
121             return this.ticket.equals(o);
122         }
123     }

143         }
144 
145         @Override
146         public boolean isValidFor(final Service service) {
147             final boolean b = this.getTicket().isValidFor(service);
148             updateTicket();
149             return b;
150         }
151 
152         @Override

154                 final Authentication authentication, final ExpirationPolicy expirationPolicy) {
155             final TicketGrantingTicket t = this.getTicket().grantTicketGrantingTicket(id,
156                     authentication, expirationPolicy);
157             updateTicket();

159         }
