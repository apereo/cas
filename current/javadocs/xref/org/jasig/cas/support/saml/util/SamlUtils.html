74      public static String signSamlResponse(final String samlResponse,
75              final PrivateKey privateKey, final PublicKey publicKey) {
76          final Document doc = constructDocumentFromXmlString(samlResponse);
77  
78          if (doc != null) {
79              final Element signedElement = signSamlElement(doc.getRootElement(),
80                      privateKey, publicKey);
81              doc.setRootElement((Element) signedElement.detach());
82              return new XMLOutputter().outputString(doc);
83          }
84          throw new RuntimeException("Error signing SAML Response: Null document");
85      }

87      public static Document constructDocumentFromXmlString(final String xmlString) {
88          try {
89              final SAXBuilder builder = new SAXBuilder();
90              builder.setFeature("http://xml.org/sax/features/external-general-entities", false);
91              builder.setFeature("http://apache.org/xml/features/disallow-doctype-decl", true);
92              return builder
93                      .build(new ByteArrayInputStream(xmlString.getBytes()));
94          } catch (final Exception e) {
95              return null;
96          }
97      }

99      private static Element signSamlElement(final Element element, final PrivateKey privKey,
100             final PublicKey pubKey) {
101         try {
102             final String providerName = System.getProperty("jsr105Provider",
103                     JSR_105_PROVIDER);
104             final XMLSignatureFactory sigFactory = XMLSignatureFactory
105                     .getInstance("DOM", (Provider) Class.forName(providerName)
106                             .newInstance());
107 
108             final List envelopedTransform = Collections
109                     .singletonList(sigFactory.newTransform(Transform.ENVELOPED,
110                             (TransformParameterSpec) null));
111 
112             final Reference ref = sigFactory.newReference("", sigFactory
113                     .newDigestMethod(DigestMethod.SHA1, null), envelopedTransform,
114                     null, null);
115 
116             // Create the SignatureMethod based on the type of key
117             SignatureMethod signatureMethod;
118             if (pubKey instanceof DSAPublicKey) {
119                 signatureMethod = sigFactory.newSignatureMethod(
120                         SignatureMethod.DSA_SHA1, null);
121             } else if (pubKey instanceof RSAPublicKey) {
122                 signatureMethod = sigFactory.newSignatureMethod(
123                         SignatureMethod.RSA_SHA1, null);
124             } else {
125                 throw new RuntimeException(
126                         "Error signing SAML element: Unsupported type of key");
127             }
128 
129             final CanonicalizationMethod canonicalizationMethod = sigFactory
130                     .newCanonicalizationMethod(
131                             CanonicalizationMethod.INCLUSIVE_WITH_COMMENTS,
132                             (C14NMethodParameterSpec) null);
133 
134             // Create the SignedInfo
135             final SignedInfo signedInfo = sigFactory.newSignedInfo(
136                     canonicalizationMethod, signatureMethod, Collections
137                     .singletonList(ref));
138 
139             // Create a KeyValue containing the DSA or RSA PublicKey
140             final KeyInfoFactory keyInfoFactory = sigFactory
141                     .getKeyInfoFactory();
142             final KeyValue keyValuePair = keyInfoFactory.newKeyValue(pubKey);
143 
144             // Create a KeyInfo and add the KeyValue to it
145             final KeyInfo keyInfo = keyInfoFactory.newKeyInfo(Collections
146                     .singletonList(keyValuePair));
147             // Convert the JDOM document to w3c (Java XML signature API requires
148             // w3c
149             // representation)
150             org.w3c.dom.Element w3cElement = toDom(element);
151 
152             // Create a DOMSignContext and specify the DSA/RSA PrivateKey and
153             // location of the resulting XMLSignature's parent element
154             DOMSignContext dsc = new DOMSignContext(privKey, w3cElement);
155 
156             org.w3c.dom.Node xmlSigInsertionPoint = getXmlSignatureInsertLocation(w3cElement);
157             dsc.setNextSibling(xmlSigInsertionPoint);
158 
159             // Marshal, generate (and sign) the enveloped signature
160             XMLSignature signature = sigFactory.newXMLSignature(signedInfo,
161                     keyInfo);
162             signature.sign(dsc);
163 
164             return toJdom(w3cElement);
165 
166         } catch (final Exception e) {
167             throw new RuntimeException("Error signing SAML element: "
168                     + e.getMessage(), e);
169         }
170     }

172     private static Node getXmlSignatureInsertLocation(final org.w3c.dom.Element elem) {
173         org.w3c.dom.Node insertLocation = null;
174         org.w3c.dom.NodeList nodeList = elem.getElementsByTagNameNS(
175                 SAML_PROTOCOL_NS_URI_V20, "Extensions");
176         if (nodeList.getLength() != 0) {
177             insertLocation = nodeList.item(nodeList.getLength() - 1);
178         } else {
179             nodeList = elem.getElementsByTagNameNS(SAML_PROTOCOL_NS_URI_V20,
180                     "Status");
181             insertLocation = nodeList.item(nodeList.getLength() - 1);
182         }
183         return insertLocation;
184     }

186     private static org.w3c.dom.Element toDom(final Element element) {
187         return toDom(element.getDocument()).getDocumentElement();
188     }
189 
190     private static org.w3c.dom.Document toDom(final Document doc) {
191         try {
192             final XMLOutputter xmlOutputter = new XMLOutputter();
193             final StringWriter elemStrWriter = new StringWriter();
194             xmlOutputter.output(doc, elemStrWriter);
195             final byte[] xmlBytes = elemStrWriter.toString().getBytes();
196             final DocumentBuilderFactory dbf = DocumentBuilderFactory
197                     .newInstance();
198             dbf.setNamespaceAware(true);
199             return dbf.newDocumentBuilder().parse(
200                     new ByteArrayInputStream(xmlBytes));
201         } catch (final Exception e) {
202             return null;
203         }
204     }
205 
206     private static Element toJdom(final org.w3c.dom.Element e) {
207         return  new DOMBuilder().build(e);
208     }
