20  package org.jasig.cas.support.openid.web.mvc;
21  
22  import org.jasig.cas.web.DelegateController;
23  import org.openid4java.message.Message;
24  import org.openid4java.message.ParameterList;
25  import org.openid4java.server.ServerManager;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  import org.springframework.web.servlet.ModelAndView;
29  
30  import javax.servlet.http.HttpServletRequest;
31  import javax.servlet.http.HttpServletResponse;
32  import javax.validation.constraints.NotNull;
33  import java.io.Serializable;
34  import java.util.HashMap;
35  import java.util.Map;
36  
37  /**
38   * Generates an association to an openid association request.
39   * @author Frederic Esnault
40   * @since 3.5
41   */
42  public class SmartOpenIdController extends DelegateController implements Serializable {
43  
44      private static final long serialVersionUID = -594058549445950430L;
45  
46      private final Logger logger = LoggerFactory.getLogger(SmartOpenIdController.class);
47  
48      private ServerManager serverManager;
49  
50      /** View if association Fails. */
51      private static final String DEFAULT_ASSOCIATION_FAILURE_VIEW_NAME = "casOpenIdAssociationFailureView";
52  
53      /** View if association Succeeds. */
54      private static final String DEFAULT_ASSOCIATION_SUCCESS_VIEW_NAME = "casOpenIdAssociationSuccessView";
55  
56      /** The view to redirect to on a successful validation. */
57      @NotNull
58      private String successView = DEFAULT_ASSOCIATION_SUCCESS_VIEW_NAME;
59  
60      /** The view to redirect to on a validation failure. Not used for now,
61       *  the succes view may return failed association attemps. No need for another view. */
62      @NotNull
63      private String failureView = DEFAULT_ASSOCIATION_FAILURE_VIEW_NAME;
64  
65      public Map<String, String> getAssociationResponse(final HttpServletRequest request) {
66          ParameterList parameters = new ParameterList(request.getParameterMap());
67  
68          final String mode = parameters.hasParameter("openid.mode")
69                  ? parameters.getParameterValue("openid.mode")
70                  : null;
71  
72          Message response = null;
73          if (mode != null && mode.equals("associate")) {
74              response = serverManager.associationResponse(parameters);
75          }
76          final Map<String, String> responseParams = new HashMap<String, String>();
77          if (response != null) {
78              responseParams.putAll(response.getParameterMap());
79          }
80  
81          return responseParams;
82  
83      }
84  
85      @Override
86      protected ModelAndView handleRequestInternal(final HttpServletRequest request, final HttpServletResponse response)
87              throws Exception {
88          final Map<String, String> parameters = new HashMap<String, String>();
89          parameters.putAll(getAssociationResponse(request));
90          return new ModelAndView(successView, "parameters", parameters);
91      }
92  
93      @Override
94      public boolean canHandle(final HttpServletRequest request, final HttpServletResponse response) {
95          String openIdMode = request.getParameter("openid.mode");
96          if (openIdMode != null && openIdMode.equals("associate")) {
97              logger.info("Handling request. openid.mode : {}", openIdMode);
98              return true;
99          }
100         logger.info("Cannot handle request. openid.mode : {}", openIdMode);
101         return false;
102     }
103 
104     public void setSuccessView(final String successView) {
105         this.successView = successView;
106     }
107 
108     public void setFailureView(final String failureView) {
109         this.failureView = failureView;
110     }
111 
112     @NotNull
113     public void setServerManager(final ServerManager serverManager) {
114         this.serverManager = serverManager;
115     }
116 }
