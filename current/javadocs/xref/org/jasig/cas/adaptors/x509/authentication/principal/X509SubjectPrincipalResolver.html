33  /**
34   * Credential to principal resolver that extracts one or more attribute values
35   * from the certificate subject DN and combines them with intervening delimiters.
36   *
37   * @author Marvin S. Addison
38   * @since 3.4.4
39   *
40   */
41  public class X509SubjectPrincipalResolver extends AbstractX509PrincipalResolver {
42  
43      /** Pattern used to extract attribute names from descriptor. */
44      private static final Pattern ATTR_PATTERN = Pattern.compile("\\$(\\w+)");
45  
46      /** Descriptor representing an abstract format of the principal to be resolved.*/
47      @NotNull
48      private String descriptor;
49  
50      /**
51       * Sets the descriptor that describes for format of the principal ID to
52       * create from X.509 subject DN attributes.  The descriptor is made up of
53       * common X.509 attribute names prefixed by "$", which are replaced by
54       * attribute values extracted from DN attribute values.
55       * <p>
56       * EXAMPLE:
57       * <p>
58       * <pre>
59       * <bean class="org.jasig.cas.adaptors.x509.authentication.principal.X509SubjectPrincipalResolver"
60       *   p:descriptor="$UID@$DC.$DC" />
61       * </pre>
62       * <p>
63       * The above bean when applied to a certificate with the DN<br>
64       * <strong>DC=edu, DC=vt/UID=jacky, CN=Jascarnella Ellagwonto</strong><br>
65       * produces the principal <strong>jacky@vt.edu</strong>.
66       *
67       * @param s Descriptor string where attribute names are prefixed with "$"
68       * to identify replacement by real attribute values from the subject DN.
69       * Valid attributes include common X.509 DN attributes such as the following:
70       * <ul>
71       * <li>C</li>
72       * <li>CN</li>
73       * <li>DC</li>
74       * <li>EMAILADDRESS</li>
75       * <li>L</li>
76       * <li>O</li>
77       * <li>OU</li>
78       * <li>SERIALNUMBER</li>
79       * <li>ST</li>
80       * <li>UID</li>
81       * <li>UNIQUEIDENTIFIER</li>
82       * </ul>
83       * For a complete list of supported attributes, see
84       * {@link edu.vt.middleware.crypt.x509.types.AttributeType}.
85       *
86       */
87      public void setDescriptor(final String s) {
88          this.descriptor = s;
89      }
90  
91      /**
92       * Replaces placeholders in the descriptor with values extracted from attribute
93       * values in relative distinguished name components of the DN.
94       *
95       * @param certificate X.509 certificate credential.
96       * @return Resolved principal ID.
97       * @see AbstractX509PrincipalResolver#resolvePrincipalInternal(java.security.cert.X509Certificate)
98       */
99      @Override
100     protected String resolvePrincipalInternal(final X509Certificate certificate) {
101         logger.debug("Resolving principal for {}", certificate);
102         final StringBuffer sb = new StringBuffer();
103         final Matcher m = ATTR_PATTERN.matcher(this.descriptor);
104         final Map<String, AttributeContext> attrMap = new HashMap<String, AttributeContext>();
105         String name;
106         String[] values;
107         AttributeContext context;
108         while (m.find()) {
109             name = m.group(1);
110             if (!attrMap.containsKey(name)) {
111                 values = DNUtils.getAttributeValues(
112                         certificate.getSubjectX500Principal(),
113                         AttributeType.fromName(name));
114                 attrMap.put(name, new AttributeContext(name, values));
115             }
116             context = attrMap.get(name);
117             m.appendReplacement(sb, context.nextValue());
118         }
119         m.appendTail(sb);
120         return sb.toString();
121     }
122 
123     private static final class AttributeContext {
124         private int currentIndex;
125         private String name;
126         private final String[] values;
127 
128         public AttributeContext(final String name, final String[] values) {
129             this.values = values;
130         }
131 
132         public String nextValue() {
133             if (this.currentIndex == this.values.length) {
134                 throw new IllegalStateException("No values remaining for attribute " + this.name);
135             }
136             return this.values[this.currentIndex++];
137         }
138     }
