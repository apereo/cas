97      @Override
98      public LdapEntry mapFromRegisteredService(final String dn, final RegisteredService svc) {
99  
100 
101         if (svc.getId() == RegisteredService.INITIAL_IDENTIFIER_VALUE) {
102             ((AbstractRegisteredService) svc).setId(System.nanoTime());
103         }
104         final String newDn = getDnForRegisteredService(dn, svc);
105         LOGGER.debug("Creating entry {}", newDn);
106 
107         final Collection<LdapAttribute> attrs = new ArrayList<LdapAttribute>();
108         attrs.add(new LdapAttribute(this.idAttribute, String.valueOf(svc.getId())));
109         attrs.add(new LdapAttribute(this.serviceIdAttribute, svc.getServiceId()));
110         attrs.add(new LdapAttribute(this.serviceNameAttribute, svc.getName()));
111         attrs.add(new LdapAttribute(this.serviceDescriptionAttribute, svc.getDescription()));
112         attrs.add(new LdapAttribute(this.serviceEnabledAttribute, Boolean.toString(svc.isEnabled()).toUpperCase()));
113         attrs.add(new LdapAttribute(this.serviceAllowedToProxyAttribute, Boolean.toString(svc.isAllowedToProxy()).toUpperCase()));
114         attrs.add(new LdapAttribute(this.serviceAnonymousAccessAttribute, Boolean.toString(svc.isAnonymousAccess()).toUpperCase()));
115         attrs.add(new LdapAttribute(this.serviceSsoEnabledAttribute, Boolean.toString(svc.isSsoEnabled()).toUpperCase()));
116         attrs.add(new LdapAttribute(this.ignoreAttributesAttribute, Boolean.toString(svc.isAnonymousAccess()).toUpperCase()));
117         attrs.add(new LdapAttribute(this.evaluationOrderAttribute, String.valueOf(svc.getEvaluationOrder())));
118         attrs.add(new LdapAttribute(this.serviceThemeAttribute, svc.getTheme()));
119         attrs.add(new LdapAttribute(this.usernameAttribute, svc.getUsernameAttribute()));
120 
121         if (svc.getAllowedAttributes().size() > 0) {
122             attrs.add(new LdapAttribute(this.serviceAllowedAttributesAttribute, svc.getAllowedAttributes().toArray(new String[] {})));
123         }
124 
125         if (svc.getRequiredHandlers().size() > 0) {
126             attrs.add(new LdapAttribute(this.requiredHandlersAttribute, svc.getRequiredHandlers().toArray(new String[] {})));
127         }
128 
129         attrs.add(new LdapAttribute(LdapUtils.OBJECTCLASS_ATTRIBUTE, this.objectClass));
130 
131         return new LdapEntry(newDn, attrs);
132     }

134     @Override
135     public RegisteredService mapToRegisteredService(final LdapEntry entry) {
136 
137         final LdapAttribute attr = entry.getAttribute(this.serviceIdAttribute);
138 
139         if (attr != null) {
140             final AbstractRegisteredService s = getRegisteredService(attr.getStringValue());
141 
142             if (s != null) {
143                 s.setId(LdapUtils.getLong(entry, this.idAttribute, Long.valueOf(entry.getDn().hashCode())));
144 
145                 s.setServiceId(LdapUtils.getString(entry, this.serviceIdAttribute));
146                 s.setName(LdapUtils.getString(entry, this.serviceNameAttribute));
147                 s.setDescription(LdapUtils.getString(entry, this.serviceDescriptionAttribute));
148                 s.setEnabled(LdapUtils.getBoolean(entry, this.serviceEnabledAttribute));
149                 s.setTheme(LdapUtils.getString(entry, this.serviceThemeAttribute));
150                 s.setEvaluationOrder(LdapUtils.getLong(entry, this.evaluationOrderAttribute).intValue());
151                 s.setUsernameAttribute(LdapUtils.getString(entry, this.usernameAttribute));
152                 s.setAllowedToProxy(LdapUtils.getBoolean(entry, this.serviceAllowedToProxyAttribute));
153                 s.setAnonymousAccess(LdapUtils.getBoolean(entry, this.serviceAnonymousAccessAttribute));
154                 s.setSsoEnabled(LdapUtils.getBoolean(entry, this.serviceSsoEnabledAttribute));
155                 s.setAllowedAttributes(new ArrayList<String>(getMultiValuedAttributeValues(entry, this.serviceAllowedAttributesAttribute)));
156                 s.setIgnoreAttributes(LdapUtils.getBoolean(entry, this.ignoreAttributesAttribute));
157                 s.setRequiredHandlers(new HashSet<String>(getMultiValuedAttributeValues(entry, this.requiredHandlersAttribute)));
158             }
159             return s;
160         }
161         return null;
162     }

238     private boolean isValidRegexPattern(final String pattern) {
239         try {
240             Pattern.compile(pattern);
241         } catch (final PatternSyntaxException e) {
242             LOGGER.debug("Failed to identify [{}] as a regular expression", pattern);
243             return false;
244         }
245         return true;
246     }
247 
248     private Collection<String> getMultiValuedAttributeValues(@NotNull final LdapEntry entry, @NotNull final String attrName) {
249         final LdapAttribute attrs = entry.getAttribute(attrName);
250         if (attrs != null) {
251             return attrs.getStringValues();
252         }
253         return Collections.emptyList();
254     }
255 
256     private AbstractRegisteredService getRegisteredService(@NotNull final String id) {
257         if (isValidRegexPattern(id)) {
258             return new RegexRegisteredService();
259         }
260 
261         if (new AntPathMatcher().isPattern(id)) {
262             return new RegisteredServiceImpl();
263         }
264         return null;
265     }
