36  import org.slf4j.Logger;
37  import org.slf4j.LoggerFactory;
38  import org.springframework.beans.factory.annotation.Autowired;

40  /**
41   * Implementation of a Restlet resource for creating Service Tickets from a
42   * TicketGrantingTicket, as well as deleting a TicketGrantingTicket.
43   *
44   * @author Scott Battaglia
45   * @since 3.3
46   *
47   */
48  public final class TicketGrantingTicketResource extends ServerResource {
49      private static final Logger LOGGER = LoggerFactory.getLogger(TicketGrantingTicketResource.class);
50  
51      @Autowired
52      private CentralAuthenticationService centralAuthenticationService;
53  
54      private String ticketGrantingTicketId;
55  
56      public void init(final Context context, final Request request, final Response response) {
57          super.init(context, request, response);
58          this.ticketGrantingTicketId = (String) request.getAttributes().get("ticketGrantingTicketId");
59          this.setNegotiated(false);
60          this.getVariants().add(new Variant(MediaType.APPLICATION_WWW_FORM));
61      }
62  
63      @Delete
64      public void removeRepresentations() {
65          this.centralAuthenticationService.destroyTicketGrantingTicket(this.ticketGrantingTicketId);
66          getResponse().setStatus(Status.SUCCESS_OK);
67      }
68  
69      @Post
70      public void acceptRepresentation(final Representation entity) {
71          final Form form = new Form(entity);
72          final String serviceUrl = form.getFirstValue("service");
73          try {
74              final String serviceTicketId = this.centralAuthenticationService.grantServiceTicket(
75                      this.ticketGrantingTicketId,
76                      new SimpleWebApplicationServiceImpl(serviceUrl));
77              getResponse().setEntity(serviceTicketId, MediaType.TEXT_PLAIN);
78          } catch (final InvalidTicketException e) {
79              getResponse().setStatus(Status.CLIENT_ERROR_NOT_FOUND, "TicketGrantingTicket could not be found.");
80          } catch (final Exception e) {
81              LOGGER.error(e.getMessage(), e);
82              getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
83          }
84      }
85  }
