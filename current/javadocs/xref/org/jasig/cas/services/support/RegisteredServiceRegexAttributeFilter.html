19  package org.jasig.cas.services.support;

21  import java.util.Collection;
22  import java.util.HashMap;
23  import java.util.Map;
24  import java.util.Vector;
25  import java.util.regex.Pattern;
26  
27  import javax.validation.constraints.NotNull;
28  
29  import org.jasig.cas.services.RegisteredService;
30  import org.jasig.cas.services.RegisteredServiceAttributeFilter;
31  import org.slf4j.Logger;
32  import org.slf4j.LoggerFactory;

34  /**
35   * The regex filter that is responsible to make sure only attributes that match a certain regex pattern
36   * registered service are released.
37   *
38   * @author Misagh Moayyed
39   * @since 4.0.0
40   */
41  public class RegisteredServiceRegexAttributeFilter implements RegisteredServiceAttributeFilter {
42      private final Logger logger = LoggerFactory.getLogger(this.getClass());
43  
44      public RegisteredServiceRegexAttributeFilter(final String regex) {
45          this.pattern = Pattern.compile(regex);
46      }
47  
48      @NotNull
49      private Pattern pattern;
50  
51      /**
52       * {@inheritDoc}
53       *
54       * Given attribute values may be an extension of {@link Collection}, {@link Map} or an array.
55       * <ul>
56       * <li>The filtering operation is non-recursive. </li>
57       * <li>Multi-valued attributes such as those of type {@link Collection} and
58       * {@link Map} are expected to allow casting to <code>Map&lt;String, String&gt;</code>
59       * or <code>Collection&lt;String&gt;</code>.
60       * Values that are of type array are expected to allow casting to <code>String[]</code>.
61       * </li>
62       * <li>Multi-valued attributes are always put back into the final released collection of
63       * attributes as <code>String[]</code>.</li>
64       * <li>If the final filtered collection is empty, it will not be put into the collection of attributes.</li>
65       * </ul>
66       */
67      @Override
68      @SuppressWarnings("unchecked")
69      public Map<String, Object> filter(final String principalId, final Map<String, Object> givenAttributes,
70              final RegisteredService registeredService) {
71          final Map<String, Object> attributesToRelease = new HashMap<String, Object>();
72          for (final String attributeName : givenAttributes.keySet()) {
73              final Object attributeValue = givenAttributes.get(attributeName);
74  
75              logger.debug("Received attribute [{}] with value [{}]", attributeName, attributeValue);
76              if (attributeValue != null) {
77                  if (attributeValue instanceof Collection) {
78                      final String[] filteredAttributes = filterArrayAttributes(
79                              ((Collection<String>) attributeValue).toArray(new String[] {}), attributeName);
80                      if (filteredAttributes.length > 0) {
81                          attributesToRelease.put(attributeName, filteredAttributes);
82                      }
83                  } else if (attributeValue.getClass().isArray()) {
84                      final String[] filteredAttributes = filterArrayAttributes((String[]) attributeValue, attributeName);
85                      if (filteredAttributes.length > 0) {
86                          attributesToRelease.put(attributeName, filteredAttributes);
87                      }
88                  } else if (attributeValue instanceof Map) {
89                      final Map<String, String> filteredAttributes = filterMapAttributes((Map<String, String>) attributeValue);
90                      if (filteredAttributes.size() > 0) {
91                          attributesToRelease.put(attributeName, filteredAttributes);
92                      }
93                  } else if (patternMatchesAttributeValue(attributeValue.toString())) {
94                      logReleasedAttributeEntry(attributeName, attributeValue.toString());
95                      attributesToRelease.put(attributeName, attributeValue);
96                  }
97              }
98          }
99  
100         logger.debug("Received {} attributes. Filtered and released {}", givenAttributes.size(),
101                 attributesToRelease.size());
102         return attributesToRelease;
103     }
104 
105     private Map<String, String> filterMapAttributes(final Map<String, String> valuesToFilter) {
106         final Map<String, String> attributesToFilter = new HashMap<String, String>(valuesToFilter.size());
107         for (final String attributeName : valuesToFilter.keySet()) {
108             final String attributeValue = valuesToFilter.get(attributeName);
109             if (patternMatchesAttributeValue(attributeValue)) {
110                 logReleasedAttributeEntry(attributeName, attributeValue);
111                 attributesToFilter.put(attributeName, valuesToFilter.get(attributeName));
112             }
113         }
114         return attributesToFilter;
115     }
116 
117     private boolean patternMatchesAttributeValue(final String value) {
118         return this.pattern.matcher(value).matches();
119     }
120 
121     private String[] filterArrayAttributes(final String[] valuesToFilter, final String attributeName) {
122         final Vector<String> vector = new Vector<String>(valuesToFilter.length);
123         for (final String attributeValue : valuesToFilter) {
124             if (patternMatchesAttributeValue(attributeValue)) {
125                 logReleasedAttributeEntry(attributeName, attributeValue);
126                 vector.add(attributeValue);
127             }
128         }
129         return vector.toArray(new String[] {});
130     }
131 
132     private void logReleasedAttributeEntry(final String attributeName, final String attributeValue) {
133         logger.debug("The attribute value [{}] for attribute name {} matches the pattern {}. Releasing attribute...",
134                 attributeValue, attributeName, this.pattern.pattern());
135     }
