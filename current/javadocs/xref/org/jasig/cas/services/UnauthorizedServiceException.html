19  package org.jasig.cas.services;
20  
21  /**
22   * Exception that is thrown when an Unauthorized Service attempts to use CAS.
23   *
24   * @author Scott Battaglia
25   * @since 3.0
26   */
27  public class UnauthorizedServiceException extends RuntimeException {
28  
29      /** The Unique ID for serialization. */
30      private static final long serialVersionUID = 3905807495715960369L;
31      
32      /** Error code that indicates the service is unauthorized for use. **/
33      public static final String CODE_UNAUTHZ_SERVICE = "screen.service.error.message";
34      
35      /** Exception object that indicates the service manager is empty with no service definitions. **/
36      public static final String CODE_EMPTY_SVC_MGMR = "screen.service.empty.error.message";
37          
38      private String code = null;
39      
40      /**
41       * Construct the exception object with the associated error code.
42       * @param message the error message
43       */
44      public UnauthorizedServiceException(final String message) {
45          super(message);
46      }
47      
48      /**
49       * Constructs an UnauthorizedServiceException with a custom message and the
50       * root cause of this exception.
51       *
52       * @param message an explanatory message. Maybe null or blank.
53       * @param code the error code mapped to the messaged bundle.
54       */
55      public UnauthorizedServiceException(final String code, final String message) {
56          this(message);
57          this.code = code;
58      }
59      /**
60       * Constructs an UnauthorizedServiceException with a custom message and the
61       * root cause of this exception.
62       *
63       * @param message an explanatory message.
64       * @param cause the root cause of the exception.
65       */
66      public UnauthorizedServiceException(final String message, final Throwable cause) {
67          super(message, cause);
68      }
69      
70      /**
71       * The error code associated with this exception.
72       * @return the error code.
73       */
74      public final String getCode() {
75          return this.code;
76      }
77  }
