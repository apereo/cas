45  /**
46   * Process the /validate , /serviceValidate , and /proxyValidate URL requests.
47   * <p>
48   * Obtain the Service Ticket and Service information and present them to the CAS
49   * validation services. Receive back an Assertion containing the user Principal
50   * and (possibly) a chain of Proxy Principals. Store the Assertion in the Model
51   * and chain to a View to generate the appropriate response (CAS 1, CAS 2 XML,
52   * SAML, ...).

99      /**
100      * Overrideable method to determine which credentials to use to grant a
101      * proxy granting ticket. Default is to use the pgtUrl.
102      *
103      * @param request the HttpServletRequest object.
104      * @return the credentials or null if there was an error or no credentials
105      * provided.
106      */
107     protected Credential getServiceCredentialsFromRequest(final HttpServletRequest request) {
108         final String pgtUrl = request.getParameter(PARAMETER_PROXY_CALLBACK_URL);
109         if (StringUtils.hasText(pgtUrl)) {
110             try {
111                 return new HttpBasedServiceCredential(new URL(pgtUrl));
112             } catch (final Exception e) {
113                 logger.error("Error constructing pgtUrl", e);
114             }
115         }
116 
117         return null;
118     }

124     @Override
125     protected final ModelAndView handleRequestInternal(final HttpServletRequest request, final HttpServletResponse response)
126             throws Exception {
127         final WebApplicationService service = this.argumentExtractor.extractService(request);
128         final String serviceTicketId = service != null ? service.getArtifactId() : null;
129 
130         if (service == null || serviceTicketId == null) {
131             logger.debug("Could not identify service and/or service ticket. Service: {}, Service ticket id: {}", service, serviceTicketId);
132             return generateErrorView("INVALID_REQUEST", "INVALID_REQUEST", null);
133         }
134 
135         try {
136             final Credential serviceCredential = getServiceCredentialsFromRequest(request);
137             String proxyGrantingTicketId = null;
138             
139             if (serviceCredential != null) {
140                 try {
141                     proxyGrantingTicketId = this.centralAuthenticationService.delegateTicketGrantingTicket(serviceTicketId,
142                                 serviceCredential);
143                 } catch (final AuthenticationException e) {
144                     logger.info("Failed to authenticate service credential {}", serviceCredential);
145                 } catch (final TicketException e) {
146                     logger.error("Failed to create proxy granting ticket for {}", serviceCredential, e);
147                 }
148                 
149                 if (StringUtils.isEmpty(proxyGrantingTicketId)) {
150                     return generateErrorView("INVALID_PROXY_CALLBACK", "INVALID_PROXY_CALLBACK",
151                             new Object[] {serviceCredential.getId()});
152                 }
153             }
154 
155             final Assertion assertion = this.centralAuthenticationService.validateServiceTicket(serviceTicketId, service);
156 
157             final ValidationSpecification validationSpecification = this.getCommandClass();
158             final ServletRequestDataBinder binder = new ServletRequestDataBinder(validationSpecification, "validationSpecification");
159             initBinder(request, binder);
160             binder.bind(request);
161 
162             if (!validationSpecification.isSatisfiedBy(assertion)) {
163                 logger.debug("Service ticket [{}] does not satisfy validation specification.", serviceTicketId);
164                 return generateErrorView("INVALID_TICKET", "INVALID_TICKET_SPEC", null);
165             }
166 
167             String proxyIou = null;
168             if (serviceCredential != null && proxyGrantingTicketId != null && this.proxyHandler.canHandle(serviceCredential)) {
169                 proxyIou = this.proxyHandler.handle(serviceCredential, proxyGrantingTicketId);
170                 if (StringUtils.isEmpty(proxyIou)) {
171                     return generateErrorView("INVALID_PROXY_CALLBACK", "INVALID_PROXY_CALLBACK",
172                             new Object[] {serviceCredential.getId()});
173                 }
174             }
175 
176             onSuccessfulValidation(serviceTicketId, assertion);
177             logger.debug("Successfully validated service ticket {} for service [{}]", serviceTicketId, service.getId());
178             return generateSuccessView(assertion, proxyIou);
179         } catch (final TicketValidationException e) {
180             return generateErrorView(e.getCode(), e.getCode(),
181                     new Object[] {serviceTicketId, e.getOriginalService().getId(), service.getId()});
182         } catch (final TicketException te) {
183             return generateErrorView(te.getCode(), te.getCode(),
184                 new Object[] {serviceTicketId});
185         } catch (final UnauthorizedProxyingException e) {
186             return generateErrorView(e.getMessage(), e.getMessage(), new Object[] {service.getId()});
187         } catch (final UnauthorizedServiceException e) {
188             return generateErrorView(e.getMessage(), e.getMessage(), null);
189         }
190     }
191 
192     protected void onSuccessfulValidation(final String serviceTicketId, final Assertion assertion) {
193         // template method with nothing to do.
194     }

196     private ModelAndView generateErrorView(final String code, final String description, final Object[] args) {
197         final ModelAndView modelAndView = new ModelAndView(this.failureView);
198         final String convertedDescription = getMessageSourceAccessor().getMessage(description, args, description);
199         modelAndView.addObject("code", code);
200         modelAndView.addObject("description", convertedDescription);
201 
202         return modelAndView;
203     }
204     
205     private ModelAndView generateSuccessView(final Assertion assertion, final String proxyIou) {
206         final ModelAndView success = new ModelAndView(this.successView);
207         success.addObject(MODEL_ASSERTION, assertion);
208         success.addObject(MODEL_PROXY_GRANTING_TICKET_IOU, proxyIou);
209         return success;
210     }
