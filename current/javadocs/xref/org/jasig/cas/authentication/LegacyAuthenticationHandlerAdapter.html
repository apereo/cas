19  package org.jasig.cas.authentication;
20  
21  import java.security.GeneralSecurityException;
22  
23  import javax.security.auth.login.FailedLoginException;
24  import javax.validation.constraints.NotNull;
25  
26  import org.jasig.cas.authentication.handler.AuthenticationException;
27  import org.jasig.cas.authentication.handler.NamedAuthenticationHandler;
28  import org.jasig.cas.authentication.principal.UsernamePasswordCredentials;
29  
30  /**
31   * Adapts a CAS 3.x {@link org.jasig.cas.authentication.handler.AuthenticationHandler} onto a CAS 4.x
32   * {@link AuthenticationHandler}.
33   *
34   * @author Marvin S. Addison
35   * @since 4.0
36   */
37  public class LegacyAuthenticationHandlerAdapter implements AuthenticationHandler {
38  
39      /** Wrapped CAS 3.0 authentication handler. */
40      @NotNull
41      private final org.jasig.cas.authentication.handler.AuthenticationHandler legacyHandler;
42  
43      /** Adapts CAS 4.0 credentials onto CAS 3.0 credentials. */
44      @NotNull
45      private final CredentialsAdapter credentialsAdapter;
46  
47  
48      /**
49       * Creates a new instance that adapts the given legacy authentication handler.
50       *
51       * @param legacy CAS 3.0 authentication handler.
52       */
53      public LegacyAuthenticationHandlerAdapter(final org.jasig.cas.authentication.handler.AuthenticationHandler legacy) {
54          if (!legacy.supports(new UsernamePasswordCredentials())) {
55              throw new IllegalArgumentException(
56                      "Cannot infer credential conversion strategy - specify CredentialsAdapter explicitly.");
57          }
58          this.legacyHandler = legacy;
59          this.credentialsAdapter = new UsernamePasswordCredentialsAdapter();
60      }
61  
62      /**
63       * Creates a new instance that adapts the given legacy authentication handler.
64       * Use this form for a handler that supports a credential type other than username/password credentials.
65       *
66       * @param legacy CAS 3.0 authentication handler.
67       * @param adapter Adapts CAS 4.0 credential onto 3.0 credential.
68       */
69      public LegacyAuthenticationHandlerAdapter(
70              final org.jasig.cas.authentication.handler.AuthenticationHandler legacy,
71              final CredentialsAdapter adapter) {
72          this.legacyHandler = legacy;
73          this.credentialsAdapter = adapter;
74      }
75  
76      @Override
77      public HandlerResult authenticate(final Credential credential) throws GeneralSecurityException, PreventedException {
78          try {
79              if (this.legacyHandler.authenticate(credentialsAdapter.convert(credential))) {
80                  final CredentialMetaData md;
81                  if (credential instanceof CredentialMetaData) {
82                      md = (CredentialMetaData) credential;
83                  } else {
84                      md = new BasicCredentialMetaData(credential);
85                  }
86                  return new HandlerResult(this, md);
87              } else {
88                  throw new FailedLoginException(
89                          String.format("%s failed to authenticate %s", this.getName(), credential));
90              }
91          } catch (final AuthenticationException e) {
92              throw new GeneralSecurityException(
93                      String.format("%s failed to authenticate %s", this.getName(), credential), e);
94          }
95      }
96  
97      @Override
98      public boolean supports(final Credential credential) {
99          return this.legacyHandler.supports(credentialsAdapter.convert(credential));
100     }
101 
102     @Override
103     public String getName() {
104         if (this.legacyHandler instanceof NamedAuthenticationHandler) {
105             return ((NamedAuthenticationHandler) this.legacyHandler).getName();
106         } else {
107             return this.legacyHandler.getClass().getSimpleName();
108         }
109     }
110 }
