19  package org.jasig.cas.authentication;
20  
21  import javax.validation.constraints.NotNull;
22  
23  /**
24   * Authentication security policy that is satisfied iff a specified authentication handler successfully authenticates
25   * at least one credential.
26   *
27   * @author Marvin S. Addison
28   */
29  public class RequiredHandlerAuthenticationPolicy implements AuthenticationPolicy {
30  
31      /** Authentication handler name that is required to satisfy policy. */
32      @NotNull
33      private final String requiredHandlerName;
34  
35      /** Flag to try all credentials before policy is satisfied. */
36      private boolean tryAll = false;
37  
38      public RequiredHandlerAuthenticationPolicy(final String requiredHandlerName) {
39          this.requiredHandlerName = requiredHandlerName;
40      }
41  
42      /**
43       * Sets the flag to try all credentials before the policy is satisfied.
44       * This flag is disabled by default such that the policy is satisfied immediately upon the first
45       * credential that is successfully authenticated by the required handler.
46       *
47       * @param tryAll True to force all credentials to be authenticated, false otherwise.
48       */
49      public void setTryAll(final boolean tryAll) {
50          this.tryAll = tryAll;
51      }
52  
53      @Override
54      public boolean isSatisfiedBy(final Authentication authn) {
55          boolean credsOk = true;
56          if (this.tryAll) {
57              credsOk = authn.getCredentials().size() == authn.getSuccesses().size() + authn.getFailures().size();
58          }
59          return credsOk && authn.getSuccesses().containsKey(this.requiredHandlerName);
60      }
61  }
