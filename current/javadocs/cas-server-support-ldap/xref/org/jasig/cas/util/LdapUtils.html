19  package org.jasig.cas.util;

21  import org.apache.commons.lang.math.NumberUtils;
22  import org.ldaptive.Connection;
23  import org.ldaptive.LdapAttribute;
24  import org.ldaptive.LdapEntry;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  /**
29   * Utilities related to LDAP functions.
30   *
31   * @author Scott Battaglia
32   * @author Misagh Moayyed
33   * @since 3.0
34   */
35  public final class LdapUtils {
36  
37      public static final String OBJECTCLASS_ATTRIBUTE = "objectClass";
38  
39      private static final Logger LOGGER = LoggerFactory.getLogger(LdapUtils.class);
40  
41      private LdapUtils() {
42          // private constructor so that no one can instantiate.
43      }
44  
45      /**
46       * Close the given context and ignore any thrown exception. This is useful
47       * for typical finally blocks in manual Ldap statements.
48       *
49       * @param context the Ldap connection to close
50       */
51      public static void closeConnection(final Connection context) {
52          if (context != null && context.isOpen()) {
53              try {
54                  context.close();
55              } catch (final Exception ex) {
56                  LOGGER.warn("Could not close ldap connection", ex);
57              }
58          }
59      }
60  
61      /**
62       * Reads a Boolean value from the LdapEntry.
63       *
64       * @param ctx       the ldap entry
65       * @param attribute the attribute name
66       * @return <code>true</code> if the attribute's value matches (case-insensitive) <code>"true"</code>, otherwise false
67       */
68      public static Boolean getBoolean(final LdapEntry ctx, final String attribute) {
69          return getBoolean(ctx, attribute, false);
70      }
71  
72      /**
73       * Reads a Boolean value from the LdapEntry.
74       *
75       * @param ctx       the ldap entry
76       * @param attribute the attribute name
77       * @param nullValue the value which should be returning in case of a null value
78       * @return <code>true</code> if the attribute's value matches (case-insensitive) <code>"true"</code>, otherwise false
79       */
80      public static Boolean getBoolean(final LdapEntry ctx, final String attribute, final Boolean nullValue) {
81          final String v = getString(ctx, attribute, nullValue.toString());
82          if (v != null) {
83              return v.equalsIgnoreCase(Boolean.TRUE.toString());
84          }
85          return nullValue;
86      }
87  
88      /**
89       * Reads a Long value from the LdapEntry.
90       *
91       * @param ctx       the ldap entry
92       * @param attribute the attribute name
93       */
94      public static Long getLong(final LdapEntry ctx, final String attribute) {
95          return getLong(ctx, attribute, Long.MIN_VALUE);
96      }
97  
98      /**
99       * Reads a Long value from the LdapEntry.
100      *
101      * @param ctx       the ldap entry
102      * @param attribute the attribute name
103      * @param nullValue the value which should be returning in case of a null value
104      */
105     public static Long getLong(final LdapEntry ctx, final String attribute, final Long nullValue) {
106         final String v = getString(ctx, attribute, nullValue.toString());
107         if (v != null && NumberUtils.isNumber(v)) {
108             return Long.valueOf(v);
109         }
110         return nullValue;
111     }
112 
113     /**
114      * Reads a String value from the LdapEntry.
115      *
116      * @param ctx       the ldap entry
117      * @param attribute the attribute name
118      */
119     public static String getString(final LdapEntry ctx, final String attribute) {
120         return getString(ctx, attribute, null);
121     }
122 
123     /**
124      * Reads a String value from the LdapEntry.
125      *
126      * @param ctx       the ldap entry
127      * @param attribute the attribute name
128      * @param nullValue the value which should be returning in case of a null value
129      */
130     public static String getString(final LdapEntry ctx, final String attribute, final String nullValue) {
131         final LdapAttribute attr = ctx.getAttribute(attribute);
132         if (attr == null) {
133             return nullValue;
134         }
135 
136         final String v = attr.getStringValue();
137         if (v != null) {
138             return v;
139         }
140         return nullValue;
141     }
