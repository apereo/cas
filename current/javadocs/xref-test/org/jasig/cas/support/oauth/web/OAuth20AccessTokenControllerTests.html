239     @Test
240     public void testOK() throws Exception {
241         final MockHttpServletRequest mockRequest = new MockHttpServletRequest("GET", CONTEXT
242                 + OAuthConstants.ACCESS_TOKEN_URL);
243         mockRequest.setParameter(OAuthConstants.CLIENT_ID, CLIENT_ID);
244         mockRequest.setParameter(OAuthConstants.REDIRECT_URI, REDIRECT_URI);
245         mockRequest.setParameter(OAuthConstants.CLIENT_SECRET, CLIENT_SECRET);
246         mockRequest.setParameter(OAuthConstants.CODE, CODE);
247         final MockHttpServletResponse mockResponse = new MockHttpServletResponse();
248         final ServicesManager servicesManager = mock(ServicesManager.class);
249         final List<RegisteredService> services = new ArrayList<RegisteredService>();
250         services.add(getRegisteredService(REDIRECT_URI, CLIENT_SECRET));
251         when(servicesManager.getAllServices()).thenReturn(services);
252         final TicketRegistry ticketRegistry = mock(TicketRegistry.class);
253         final ServiceTicket serviceTicket = mock(ServiceTicket.class);
254         final TicketGrantingTicket ticketGrantingTicket = mock(TicketGrantingTicket.class);
255         // 10 seconds
256         final int TIME_BEFORE = 10;
257         when(ticketGrantingTicket.getCreationTime()).thenReturn(System.currentTimeMillis() - TIME_BEFORE * 1000);
258         when(ticketGrantingTicket.getId()).thenReturn(TGT_ID);
259         when(serviceTicket.isExpired()).thenReturn(false);
260         when(serviceTicket.getId()).thenReturn(CODE);
261         when(serviceTicket.getGrantingTicket()).thenReturn(ticketGrantingTicket);
262         when(ticketRegistry.getTicket(CODE)).thenReturn(serviceTicket);
263         final OAuth20WrapperController oauth20WrapperController = new OAuth20WrapperController();
264         oauth20WrapperController.setServicesManager(servicesManager);
265         oauth20WrapperController.setTicketRegistry(ticketRegistry);
266         oauth20WrapperController.setTimeout(TIMEOUT);
267         oauth20WrapperController.afterPropertiesSet();
268         oauth20WrapperController.handleRequest(mockRequest, mockResponse);
269         verify(ticketRegistry).deleteTicket(CODE);
270         assertEquals("text/plain", mockResponse.getContentType());
271         assertEquals(200, mockResponse.getStatus());
272         final String body = mockResponse.getContentAsString();
273         assertTrue(body.startsWith(OAuthConstants.ACCESS_TOKEN + "=" + TGT_ID + "&" + OAuthConstants.EXPIRES + "="));
274         // delta = 2 seconds
275         final int DELTA = 2;
276         final int timeLeft = Integer.parseInt(StringUtils.substringAfter(body, "&" + OAuthConstants.EXPIRES + "="));
277         assertTrue(timeLeft >= TIMEOUT - TIME_BEFORE - DELTA);
278         assertTrue(timeLeft <= TIMEOUT - TIME_BEFORE + DELTA);
279     }

282         final OAuthRegisteredService registeredServiceImpl = new OAuthRegisteredService();
