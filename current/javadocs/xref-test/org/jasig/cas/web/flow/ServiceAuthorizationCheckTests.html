19  package org.jasig.cas.web.flow;
20  
21  import java.util.ArrayList;
22  import java.util.List;
23  
24  import org.jasig.cas.authentication.principal.WebApplicationService;
25  import org.jasig.cas.services.RegisteredService;
26  import org.jasig.cas.services.RegisteredServiceImpl;
27  import org.jasig.cas.services.ServicesManager;
28  import org.jasig.cas.services.UnauthorizedServiceException;
29  import org.junit.Before;
30  import org.junit.Test;
31  import org.springframework.webflow.execution.Event;
32  import org.springframework.webflow.test.MockRequestContext;
33  
34  import static org.junit.Assert.*;
35  import static org.mockito.Mockito.*;
36  
37  /**
38   * Mockito based tests for @{link ServiceAuthorizationCheck}
39   *
40   * @author Dmitriy Kopylenko
41   */
42  public class ServiceAuthorizationCheckTests {
43  
44      private ServiceAuthorizationCheck serviceAuthorizationCheck;
45  
46      private WebApplicationService authorizedService = mock(WebApplicationService.class);
47  
48      private WebApplicationService unauthorizedService = mock(WebApplicationService.class);
49  
50      private WebApplicationService undefinedService = mock(WebApplicationService.class);
51  
52      private ServicesManager servicesManager = mock(ServicesManager.class);
53  
54  
55      @Before
56      public void setUpMocks() {
57          RegisteredServiceImpl authorizedRegisteredService = new RegisteredServiceImpl();
58          RegisteredServiceImpl unauthorizedRegisteredService = new RegisteredServiceImpl();
59          unauthorizedRegisteredService.setEnabled(false);
60  
61          List<RegisteredService> list = new ArrayList<RegisteredService>();
62          list.add(authorizedRegisteredService);
63          list.add(unauthorizedRegisteredService);
64          
65          when(this.servicesManager.findServiceBy(this.authorizedService)).thenReturn(authorizedRegisteredService);
66          when(this.servicesManager.findServiceBy(this.unauthorizedService)).thenReturn(unauthorizedRegisteredService);
67          when(this.servicesManager.findServiceBy(this.undefinedService)).thenReturn(null);
68          
69          when(this.servicesManager.getAllServices()).thenReturn(list);
70          
71          this.serviceAuthorizationCheck = new ServiceAuthorizationCheck(this.servicesManager);
72      }
73  
74      @Test
75      public void noServiceProvided() throws Exception {
76          MockRequestContext mockRequestContext = new MockRequestContext();
77          Event event = this.serviceAuthorizationCheck.doExecute(mockRequestContext);
78          assertEquals("success", event.getId());
79  
80      }
81  
82      @Test
83      public void authorizedServiceProvided() throws Exception {
84          MockRequestContext mockRequestContext = new MockRequestContext();
85          mockRequestContext.getFlowScope().put("service", this.authorizedService);
86          Event event = this.serviceAuthorizationCheck.doExecute(mockRequestContext);
87          assertEquals("success", event.getId());
88      }
89  
90      @Test(expected=UnauthorizedServiceException.class)
91      public void unauthorizedServiceProvided() throws Exception {
92          MockRequestContext mockRequestContext = new MockRequestContext();
93          mockRequestContext.getFlowScope().put("service", this.unauthorizedService);
94  
95          this.serviceAuthorizationCheck.doExecute(mockRequestContext);
96          fail("Should have thrown UnauthorizedServiceException");
97      }
98  
99      @Test(expected=UnauthorizedServiceException.class)
100     public void serviceThatIsNotRegisteredProvided() throws Exception {
101         MockRequestContext mockRequestContext = new MockRequestContext();
102         mockRequestContext.getFlowScope().put("service", this.undefinedService);
103         this.serviceAuthorizationCheck.doExecute(mockRequestContext);
104         fail("Should have thrown UnauthorizedServiceException");
105     }
106 }
