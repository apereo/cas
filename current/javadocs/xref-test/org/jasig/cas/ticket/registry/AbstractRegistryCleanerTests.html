35  public abstract class AbstractRegistryCleanerTests {
36  
37      private RegistryCleaner registryCleaner;
38  
39      private TicketRegistry ticketRegistry;
40  
41      @Before
42      public void setUp() throws Exception {
43          this.ticketRegistry = this.getNewTicketRegistry();
44          this.registryCleaner = this.getNewRegistryCleaner(this.ticketRegistry);
45      }
46  
47      public abstract RegistryCleaner getNewRegistryCleaner(TicketRegistry newTicketRegistry);
48  
49      public abstract TicketRegistry getNewTicketRegistry();
50  
51      @Test
52      public void testCleanEmptyTicketRegistry() {
53          this.registryCleaner.clean();
54          assertTrue(this.ticketRegistry.getTickets().isEmpty());
55      }
56  
57      @Test
58      public void testCleanRegistryOfExpiredTicketsAllExpired() {
59          populateRegistryWithExpiredTickets();
60          this.registryCleaner.clean();
61          assertTrue(this.ticketRegistry.getTickets().isEmpty());
62      }
63  
64      @Test
65      public void testCleanRegistryOneNonExpired() {
66          populateRegistryWithExpiredTickets();
67          TicketGrantingTicket ticket = new TicketGrantingTicketImpl("testNoExpire", TestUtils.getAuthentication(),
68                  new NeverExpiresExpirationPolicy());
69          this.ticketRegistry.addTicket(ticket);
70  
71          this.registryCleaner.clean();
72  
73          assertEquals(this.ticketRegistry.getTickets().size(), 1);
74      }
75  
76      private void populateRegistryWithExpiredTickets() {
77          for (int i = 0; i < 10; i++) {
78              TicketGrantingTicket ticket = new TicketGrantingTicketImpl("test" + i, TestUtils.getAuthentication(),
79                      new NeverExpiresExpirationPolicy());
80              ticket.markTicketExpired();
81              this.ticketRegistry.addTicket(ticket);
82          }
83      }
