150     /**
151      * Is remember me authentication?
152      * looks at the authentication object to find {@link RememberMeCredential#AUTHENTICATION_ATTRIBUTE_REMEMBER_ME}
153      * and expects the assertion to also note a new login session.
154      * @param model the model
155      * @return true if remember-me, false if otherwise.
156      */
157     protected final boolean isRememberMeAuthentication(final Map<String, Object> model) {
158         final Map<String, Object> authnAttributes = getAuthenticationAttributesAsMultiValuedAttributes(model);
159         final Collection authnMethod = (Collection) authnAttributes.get(RememberMeCredential.AUTHENTICATION_ATTRIBUTE_REMEMBER_ME);
160         return authnMethod != null && authnMethod.contains(Boolean.TRUE) && isAssertionBackedByNewLogin(model);
161     }

171         return getAssertionFrom(model).isFromNewLogin();

174     /**
175      * Convert attribute values to multi valued objects.
176      *
177      * @param attributes the attributes
178      * @return the map of attributes to return
179      */
180     private static Map<String, Object> convertAttributeValuesToMultiValuedObjects(final Map<String, Object> attributes) {
181         final Map<String, Object> attributesToReturn = new HashMap<>();
182         final Set<Map.Entry<String, Object>> entries = attributes.entrySet();
183         for (final Map.Entry<String, Object> entry : entries) {
184             final Object value = entry.getValue();
185             if (value instanceof Collection || value instanceof Map || value instanceof Object[]
186                     || value instanceof Iterator || value instanceof Enumeration) {
187                 attributesToReturn.put(entry.getKey(), value);
188             } else {
189                 attributesToReturn.put(entry.getKey(), Collections.singleton(value));
190             }
191         }
192         return attributesToReturn;
193     }

216     /**
217      * Gets chained authentications.
218      *
219      * @param model the model
220      * @return the chained authentications
221      */
222     protected final Collection<Authentication> getChainedAuthentications(final Map<String, Object> model) {
223         final List<Authentication> chainedAuthenticationsToReturn = new ArrayList<>();
224 
225         final Assertion assertion = getAssertionFrom(model);
226         final List<Authentication> chainedAuthentications = assertion.getChainedAuthentications();
227 
228         /**
229          * Note that the last index in the list always describes the primary authentication
230          * event. All others in the chain should denote proxies. Per the CAS protocol,
231          * when authentication has proceeded through multiple proxies,
232          * the order in which the proxies were traversed MUST be reflected in the response.
233          * The most recently-visited proxy MUST be the first proxy listed, and all the
234          * other proxies MUST be shifted down as new proxies are added. I
235          */
236         final int numberAuthenticationsExceptPrimary = chainedAuthentications.size() - 1;
237         for (int i = 0; i < numberAuthenticationsExceptPrimary; i++) {
238             chainedAuthenticationsToReturn.add(chainedAuthentications.get(i));
239         }
240         return chainedAuthenticationsToReturn;
241     }

243     /**
244      * Decide if credential password should be released as attribute.
245      * The credential must have been cached as an authentication attribute
246      * and the attribute release policy must be allowed to release the
247      * attribute.
248      *
249      * @param attributes the attributes
250      * @param model the model
251      * @param service the service
252      */
253     protected void decideIfCredentialPasswordShouldBeReleasedAsAttribute(final Map<String, Object> attributes,
254                                                                          final Map<String, Object> model,
255                                                                          final RegisteredService service) {
256         decideAttributeReleaseBasedOnServiceAttributePolicy(attributes,
257                 getAuthenticationAttribute(model, CasViewConstants.MODEL_ATTRIBUTE_NAME_PRINCIPAL_CREDENTIAL),
258                 CasViewConstants.MODEL_ATTRIBUTE_NAME_PRINCIPAL_CREDENTIAL,
259                 service, service.getAttributeReleasePolicy().isAuthorizedToReleaseCredentialPassword());
260     }

263      * Decide if PGT should be released as attribute.
264      * The PGT must have been cached as an authentication attribute
265      * and the attribute release policy must be allowed to release the

268      * @param attributes the attributes

272     protected void decideIfProxyGrantingTicketShouldBeReleasedAsAttribute(final Map<String, Object> attributes,

278                 service, service.getAttributeReleasePolicy().isAuthorizedToReleaseProxyGrantingTicket());

281     /**
282      * Decide attribute release based on service attribute policy.
283      *
284      * @param attributes the attributes
285      * @param attributeValue the attribute value
286      * @param attributeName the attribute name
287      * @param service the service
288      * @param doesAttributePolicyAllow does attribute policy allow release of this attribute?
289      */
290     protected void decideAttributeReleaseBasedOnServiceAttributePolicy(final Map<String, Object> attributes,
291                                                                        final String attributeValue,
292                                                                        final String attributeName,
293                                                                        final RegisteredService service,
294                                                                        final boolean doesAttributePolicyAllow) {
295         if (StringUtils.isNotBlank(attributeValue)) {
296             logger.debug("Obtained [{}] as an authentication attribute", attributeName);
297 
298             if (doesAttributePolicyAllow) {
299                 logger.debug("Obtained [{}] is passed to the CAS validation payload", attributeName);
300                 attributes.put(attributeName, Collections.singleton(attributeValue));
301             } else {
302                 logger.debug("Attribute release policy for [{}] does not authorize the release of [{}]",
303                         service.getServiceId(), attributeName);
304                 attributes.remove(attributeName);
305             }
306         } else {
307             logger.trace("[{}] is not available and will not be released to the validation response.", attributeName);
308         }
309     }
