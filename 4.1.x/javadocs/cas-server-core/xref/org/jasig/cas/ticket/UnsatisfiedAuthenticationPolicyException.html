19  package org.jasig.cas.ticket;
20  
21  import org.jasig.cas.authentication.ContextualAuthenticationPolicy;
22  import org.springframework.util.Assert;
23  
24  /**
25   * Error condition arising at ticket creation or validation time when a ticketing operation relying on authentication
26   * cannot proceed due to unsatisfied authentication security policy.
27   *
28   * @author Marvin S. Addison
29   * @since 4.0.0
30   */
31  public class UnsatisfiedAuthenticationPolicyException extends TicketException {
32  
33      /** Serializable ID for unique id. */
34      private static final long serialVersionUID = -827432780367197133L;
35  
36      /** Code description. */
37      private static final String CODE = "UNSATISFIED_AUTHN_POLICY";
38  
39      /** Unfulfilled policy that caused this exception. */
40      private final ContextualAuthenticationPolicy<?> policy;
41  
42      /**
43       * Creates a new instance with no cause.
44       *
45       * @param policy Non-null unfulfilled security policy that caused exception.
46       */
47      public UnsatisfiedAuthenticationPolicyException(final ContextualAuthenticationPolicy<?> policy) {
48          super(CODE);
49          Assert.notNull(policy, "ContextualAuthenticationPolicy cannot be null");
50          this.policy = policy;
51      }
52  
53      /**
54       * Gets the unsatisfied policy that caused this exception.
55       *
56       * @return Non-null unsatisfied policy cause.
57       */
58      public ContextualAuthenticationPolicy<?> getPolicy() {
59          return policy;
60      }
61  }
