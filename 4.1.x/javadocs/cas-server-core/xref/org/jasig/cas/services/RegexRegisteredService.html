19  package org.jasig.cas.services;
20  
21  import org.jasig.cas.authentication.principal.Service;
22  
23  import javax.persistence.DiscriminatorValue;
24  import javax.persistence.Entity;
25  
26  import java.util.regex.Pattern;
27  
28  /**
29   * Mutable registered service that uses Java regular expressions for service matching.
30   * Matching is case insensitive, and is successful, if, and only if, the entire region
31   * sequence matches the pattern.
32   *
33   * @author Marvin S. Addison
34   * @author Misagh Moayyed
35   * @since 3.4
36   */
37  @Entity
38  @DiscriminatorValue("regex")
39  public class RegexRegisteredService extends AbstractRegisteredService {
40  
41      private static final long serialVersionUID = -8258660210826975771L;
42  
43      private transient Pattern servicePattern;
44  
45      public void setServiceId(final String id) {
46          serviceId = id;
47      }
48      
49      @Override
50      public boolean matches(final Service service) {
51          if (servicePattern == null) {
52              servicePattern = createPattern(serviceId);
53          }
54          return service != null && servicePattern.matcher(service.getId()).matches();
55      }
56  
57      @Override
58      protected AbstractRegisteredService newInstance() {
59          return new RegexRegisteredService();
60      }
61  
62      /**
63       * Creates the pattern. Matching is by default
64       * case insensitive.
65       *
66       * @param pattern the pattern, may not be null.
67       * @return the pattern
68       */
69      private Pattern createPattern(final String pattern) {
70          if (pattern == null) {
71              throw new IllegalArgumentException("Pattern cannot be null.");
72          }
73          
74          return Pattern.compile(pattern, Pattern.CASE_INSENSITIVE);
75      }
76  }
