39  /**
40   * This is {@link DefaultRegisteredServiceAccessStrategy}
41   * that allows the following rules:
42   *
43   * <ul>
44   *     <li>A service may be disallowed to use CAS for authentication</li>
45   *     <li>A service may be disallowed to take part in CAS single sign-on such that
46   *     presentation of credentials would always be required.</li>
47   *     <li>A service may be prohibited from receiving a service ticket
48   *     if the existing principal attributes don't contain the required attributes
49   *     that otherwise grant access to the service.</li>
50   * </ul>
51   *
52   * @author Misagh Moayyed mmoayyed@unicon.net
53   * @since 4.1
54   */
55  public class DefaultRegisteredServiceAccessStrategy implements RegisteredServiceAccessStrategy {

192     /**
193      * Defines the required attribute names and values that
194      * must be available to the principal before the flow
195      * can proceed to the next step. Every attribute in
196      * the map can be linked to multiple values.
197      *
198      * @param requiredAttributes the required attributes
199      */
200     public final void setRequiredAttributes(final Map<String, Set<String>> requiredAttributes) {
201         this.requiredAttributes = requiredAttributes;
202     }

204     /**
205      * {@inheritDoc}
206      *
207      * Verify presence of service required attributes.
208      * <ul>
209      *     <li>If no required attributes are specified, authz is granted.</li>
210      *     <li>If ALL required attributes must be present, and the principal contains all and there is
211      *     at least one attribute value that matches the required, authz is granted.</li>
212      *     <li>If ALL required attributes don't have to be present, and there is at least
213      *     one principal attribute present whose value matches the required, authz is granted.</li>
214      *     <li>Otherwise, access is denied</li>
215      * </ul>
216      * Note that comparison of principal/required attributes is case-sensitive. Exact matches are required
217      * for any individual attribute value.
218      */
219     @Override
220     public boolean doPrincipalAttributesAllowServiceAccess(final Map<String, Object> principalAttributes) {
221         if (this.requiredAttributes.isEmpty()) {
222             logger.debug("No required attributes are specified");
223             return true;
224         }
225         if (principalAttributes.isEmpty()) {
226             logger.debug("No principal attributes are found to satisfy attribute requirements");
227             return false;
228         }
229 
230         if (principalAttributes.size() < this.requiredAttributes.size()) {
231             logger.debug("The size of the principal attributes that are [{}] does not match requirements, "
232                     + "which means the principal is not carrying enough data to grant authorization",
233                     principalAttributes);
234             return false;
235         }
236 
237         final Map<String, Set<String>> requiredAttrs = this.getRequiredAttributes();
238         logger.debug("These required attributes [{}] are examined against [{}] before service can proceed.",
239                 requiredAttrs, principalAttributes);
240 
241         final Sets.SetView<String> difference = Sets.intersection(requiredAttrs.keySet(), principalAttributes.keySet());
242         final Set<String> copy = difference.immutableCopy();
243 
244         if (this.requireAllAttributes && copy.size() < this.requiredAttributes.size()) {
245             logger.debug("Not all required attributes are available to the principal");
246             return false;
247         }
248 
249         for (final String key : copy) {
250             final Set<String> requiredValues = this.requiredAttributes.get(key);
251             final Set<String> availableValues;
252 
253             final Object objVal = principalAttributes.get(key);
254             if (objVal instanceof Collection) {
255                 final Collection valCol = (Collection) objVal;
256                 availableValues = Sets.newHashSet(valCol.iterator());
257             } else {
258                 availableValues = Sets.newHashSet(objVal.toString());
259             }
260 
261             final Set<?> differenceInValues;
262             final Pattern pattern = RegexUtils.concatenate(requiredValues, this.caseInsensitive);
263             if (pattern != null) {
264                 differenceInValues = Sets.filter(availableValues, Predicates.contains(pattern));
265             } else {
266                 differenceInValues = Sets.intersection(availableValues, requiredValues);
267             }
268 
269             if (!differenceInValues.isEmpty()) {
270                 logger.info("Principal is authorized to access the service");
271                 return true;
272             }
273         }
274         logger.info("Principal is denied access as the required attributes for the registered service are missing");
275         return false;
276     }

278     @Override
279     public boolean isServiceAccessAllowedForSso() {
280         if (!this.ssoEnabled) {
281             logger.trace("Service is not authorized to participate in SSO.");
282         }
283         return this.ssoEnabled;
284     }

286     @Override
287     public boolean isServiceAccessAllowed() {
288         if (!this.enabled) {
289             logger.trace("Service is not enabled in service registry.");
290         }
291 
292         final DateTime now = DateTime.now();
293 
294         if (this.startingDateTime != null) {
295             final DateTime st = DateTime.parse(this.startingDateTime);
296 
297             if (now.isBefore(st)) {
298                 logger.warn("Service access not allowed because it starts at {}. Now is {}",
299                         this.startingDateTime, now);
300                 return false;
301             }
302         }
303 
304         if (this.endingDateTime != null) {
305             final DateTime et = DateTime.parse(this.endingDateTime);
306             if  (now.isAfter(et)) {
307                 logger.warn("Service access not allowed because it ended at {}. Now is {}",
308                         this.endingDateTime, now);
309                 return false;
310             }
311         }
312 
313         return this.enabled;
314     }

317     @Override
318     public boolean equals(final Object obj) {
319         if (obj == null) {
320             return false;
321         }
322         if (obj == this) {
323             return true;
324         }
325         if (obj.getClass() != getClass()) {
326             return false;
327         }
328         final DefaultRegisteredServiceAccessStrategy rhs = (DefaultRegisteredServiceAccessStrategy) obj;
329         return new EqualsBuilder()
330                 .append(this.enabled, rhs.enabled)
331                 .append(this.ssoEnabled, rhs.ssoEnabled)
332                 .append(this.requireAllAttributes, rhs.requireAllAttributes)
333                 .append(this.requiredAttributes, rhs.requiredAttributes)
334                 .append(this.startingDateTime, rhs.startingDateTime)
335                 .append(this.endingDateTime, rhs.endingDateTime)
336                 .append(this.unauthorizedRedirectUrl, rhs.unauthorizedRedirectUrl)
337                 .append(this.caseInsensitive, rhs.caseInsensitive)
338                 .isEquals();
339     }
