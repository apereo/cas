52  /**
53   * Implementationn of CAS {@link HttpClient}
54   * which delegates requests to a {@link #httpClient} instance.
55   *
56   * @author Jerome Leleu
57   * @author Scott Battaglia
58   * @author Misagh Moayyed
59   * @since 3.1
60   */
61  final class SimpleHttpClient implements HttpClient, Serializable, DisposableBean {
62  
63      /** Unique Id for serialization. */
64      private static final long serialVersionUID = -4949380008568071855L;
65  
66      private static final Logger LOGGER = LoggerFactory.getLogger(SimpleHttpClient.class);
67  
68      /** the acceptable codes supported by this client. */
69      private final List<Integer> acceptableCodes;
70  
71      /** the HTTP client for this client. */
72      private final CloseableHttpClient httpClient;
73  
74      /** the request executor service for this client. */
75      private final FutureRequestExecutionService requestExecutorService;
76  
77      /**
78       * Instantiates a new Simple HTTP client, based on the provided inputs.
79       *
80       * @param acceptableCodes the acceptable codes of the client
81       * @param httpClient the HTTP client used by the client
82       * @param requestExecutorService the request executor service used by the client
83       */
84      public SimpleHttpClient(final List<Integer> acceptableCodes, final CloseableHttpClient httpClient,
85              final FutureRequestExecutionService requestExecutorService) {
86          this.acceptableCodes = ImmutableList.copyOf(acceptableCodes);
87          this.httpClient = httpClient;
88          this.requestExecutorService = requestExecutorService;
89      }
90  
91      @Override
92      public boolean sendMessageToEndPoint(final HttpMessage message) {
93          Assert.notNull(this.httpClient);
94  
95          try {
96              final HttpPost request = new HttpPost(message.getUrl().toURI());
97              request.addHeader("Content-Type", message.getContentType());
98              
99              final StringEntity entity = new StringEntity(message.getMessage(), ContentType.create(message.getContentType()));
100             request.setEntity(entity);
101 
102             final HttpRequestFutureTask<String> task = this.requestExecutorService.execute(request,
103                     HttpClientContext.create(), new BasicResponseHandler());
104 
105             if (message.isAsynchronous()) {
106                 return true;
107             }
108             
109             return StringUtils.isNotBlank(task.get());
110         } catch (final RejectedExecutionException e) {
111             LOGGER.warn(e.getMessage(), e);
112             return false;
113         } catch (final Exception e) {
114             LOGGER.trace(e.getMessage(), e);
115             return false;
116         }
117     }
118         
119     @Override
120     public boolean isValidEndPoint(final String url) {
121         try {
122             final URL u = new URL(url);
123             return isValidEndPoint(u);
124         } catch (final MalformedURLException e) {
125             LOGGER.error(e.getMessage(), e);
126             return false;
127         }
128     }
129 
130     @Override
131     public boolean isValidEndPoint(final URL url) {
132         Assert.notNull(this.httpClient);
133 
134         HttpEntity entity = null;
135 
136         try (final CloseableHttpResponse response = this.httpClient.execute(new HttpGet(url.toURI()))) {
137             final int responseCode = response.getStatusLine().getStatusCode();
138 
139             for (final int acceptableCode : this.acceptableCodes) {
140                 if (responseCode == acceptableCode) {
141                     LOGGER.debug("Response code from server matched {}.", responseCode);
142                     return true;
143                 }
144             }
145 
146             LOGGER.debug("Response code did not match any of the acceptable response codes. Code returned was {}",
147                     responseCode);
148 
149             if (responseCode == HttpStatus.SC_INTERNAL_SERVER_ERROR) {
150                 final String value = response.getStatusLine().getReasonPhrase();
151                 LOGGER.error("There was an error contacting the endpoint: {}; The error was:\n{}", url.toExternalForm(),
152                         value);
153             }
154 
155             entity = response.getEntity();
156         } catch (final Exception e) {
157             LOGGER.error(e.getMessage(), e);
158         } finally {
159             EntityUtils.consumeQuietly(entity);
160         }
161         return false;
162     }

164     /**
165      * Shutdown the executor service and close the http client.
166      * @throws Exception if the executor cannot properly shut down
167      */
168     public void destroy() throws Exception {
169         IOUtils.closeQuietly(this.requestExecutorService);
170     }
