19  package org.jasig.cas.util;
20  
21  import org.apache.commons.lang3.StringUtils;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  import java.util.Collection;
26  import java.util.regex.Pattern;
27  import java.util.regex.PatternSyntaxException;
28  
29  /**
30   * Utility class to assist with resource operations.
31   *
32   * @author Misagh Moayyed
33   * @since 4.1.4
34   */
35  public final class RegexUtils {
36      private static final Logger LOGGER = LoggerFactory.getLogger(RegexUtils.class);
37  
38      /**
39       * Instantiates a new Regex utils.
40       */
41      private RegexUtils() {}
42  
43      /**
44       * Check to see if the specified pattern is a valid regular expression.
45       *
46       * @param pattern the pattern
47       * @return whether this is a valid regex or not
48       */
49      public static boolean isValidRegex(final String pattern) {
50          try {
51              LOGGER.debug("Pattern {} is a valid regex.", Pattern.compile(pattern).pattern());
52              return true;
53          } catch (final PatternSyntaxException exception) {
54              return false;
55          }
56      }
57  
58      /**
59       * Concatenate all elements in the given collection to form a regex pattern.
60       *
61       * @param requiredValues  the required values
62       * @param caseInsensitive the case insensitive
63       * @return the pattern
64       */
65      public static Pattern concatenate(final Collection<String> requiredValues, final boolean caseInsensitive) {
66          final StringBuilder builder = new StringBuilder(requiredValues.size());
67          for (final String requiredValue : requiredValues) {
68              builder.append('(').append(requiredValue).append(")|");
69          }
70          final String pattern = StringUtils.removeEnd(builder.toString(), "|");
71          if (isValidRegex(pattern)) {
72              return Pattern.compile(pattern, caseInsensitive ? Pattern.CASE_INSENSITIVE : 0);
73          }
74          return null;
75      }
76  }
