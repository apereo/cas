19  package org.jasig.cas.authentication.principal;
20  
21  import org.jasig.cas.authentication.Credential;
22  import org.jasig.services.persondir.IPersonAttributeDao;
23  import org.jasig.services.persondir.IPersonAttributes;
24  import org.jasig.services.persondir.support.StubPersonAttributeDao;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import javax.validation.constraints.NotNull;
29  import java.util.HashMap;
30  import java.util.List;
31  import java.util.Map;
32  
33  /**
34   * Resolves principals by querying a data source using the Jasig
35   * <a href="http://developer.jasig.org/projects/person-directory/1.5.0-SNAPSHOT/apidocs/">Person Directory API</a>.
36   * The {@link org.jasig.cas.authentication.principal.Principal#getAttributes()} are populated by the results of the
37   * query and the principal ID may optionally be set by proving an attribute whose first non-null value is used;
38   * otherwise the credential ID is used for the principal ID.
39   *
40   * @author Marvin S. Addison
41   * @since 4.0.0
42   *
43   */
44  public class PersonDirectoryPrincipalResolver implements PrincipalResolver {
45  
46      /** Log instance. */
47      protected final Logger logger = LoggerFactory.getLogger(this.getClass());
48  
49      private boolean returnNullIfNoAttributes;
50  
51      /** Repository of principal attributes to be retrieved. */
52      @NotNull
53      private IPersonAttributeDao attributeRepository = new StubPersonAttributeDao(new HashMap<String, List<Object>>());
54  
55      /** Factory to create the principal type. **/
56      @NotNull
57      private PrincipalFactory principalFactory = new DefaultPrincipalFactory();
58  
59      /** Optional principal attribute name. */
60      private String principalAttributeName;
61  
62      @Override
63      public boolean supports(final Credential credential) {
64          return true;
65      }
66  
67      @Override
68      public final Principal resolve(final Credential credential) {
69          logger.debug("Attempting to resolve a principal...");
70  
71          String principalId = extractPrincipalId(credential);
72  
73          if (principalId == null) {
74              logger.debug("Got null for extracted principal ID; returning null.");
75              return null;
76          }
77  
78          logger.debug("Creating SimplePrincipal for [{}]", principalId);
79  
80          final IPersonAttributes personAttributes = this.attributeRepository.getPerson(principalId);
81          final Map<String, List<Object>> attributes;
82  
83          if (personAttributes == null) {
84              attributes = null;
85          } else {
86              attributes = personAttributes.getAttributes();
87          }
88  
89          if (attributes == null & !this.returnNullIfNoAttributes) {
90              return this.principalFactory.createPrincipal(principalId);
91          }
92  
93          if (attributes == null) {
94              return null;
95          }
96  
97          final Map<String, Object> convertedAttributes = new HashMap<>();
98          for (final Map.Entry<String, List<Object>> entry : attributes.entrySet()) {
99              final String key = entry.getKey();
100             final List<Object> values = entry.getValue();
101             if (key.equalsIgnoreCase(this.principalAttributeName)) {
102                 if (values.isEmpty()) {
103                     logger.debug("{} is empty, using {} for principal", this.principalAttributeName, principalId);
104                 } else {
105                     principalId = values.get(0).toString();
106                     logger.debug(
107                             "Found principal attribute value {}; removing {} from attribute map.",
108                             principalId,
109                             this.principalAttributeName);
110                 }
111             } else {
112                 convertedAttributes.put(key, values.size() == 1 ? values.get(0) : values);
113             }
114         }
115         return this.principalFactory.createPrincipal(principalId, convertedAttributes);
116     }
117 
118     public final void setAttributeRepository(final IPersonAttributeDao attributeRepository) {
119         this.attributeRepository = attributeRepository;
120     }
121 
122     public void setReturnNullIfNoAttributes(final boolean returnNullIfNoAttributes) {
123         this.returnNullIfNoAttributes = returnNullIfNoAttributes;
124     }
125 
126     /**
127      * Sets the name of the attribute whose first non-null value should be used for the principal ID.
128      *
129      * @param attribute Name of attribute containing principal ID.
130      */
131     public void setPrincipalAttributeName(final String attribute) {
132         this.principalAttributeName = attribute;
133     }
134 
135     /**
136      * Sets principal factory to create principal objects.
137      *
138      * @param principalFactory the principal factory
139      */
140     public void setPrincipalFactory(final PrincipalFactory principalFactory) {
141         this.principalFactory = principalFactory;
142     }
143 
144     /**
145      * Extracts the id of the user from the provided credential. This method should be overridded by subclasses to
146      * achieve more sophisticated strategies for producing a principal ID from a credential.
147      *
148      * @param credential the credential provided by the user.
149      * @return the username, or null if it could not be resolved.
150      */
151     protected String extractPrincipalId(final Credential credential) {
152         return credential.getId();
153     }
154 }
