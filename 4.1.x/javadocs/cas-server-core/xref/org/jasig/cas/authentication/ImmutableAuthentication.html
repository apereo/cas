22  import org.apache.commons.lang3.builder.HashCodeBuilder;

24  import org.joda.time.DateTime;
25  import org.springframework.util.Assert;

27  import java.util.Collections;
28  import java.util.Date;
29  import java.util.List;

32  /**
33   * Immutable authentication event whose attributes may not change after creation.
34   * This class is designed for serialization and is suitable for long-term storage.
35   *
36   * @author Dmitriy Kopylenko
37   * @author Scott Battaglia
38   * @author Marvin S. Addison
39   *
40   * @since 3.0.0
41   */
42  public final class ImmutableAuthentication implements Authentication {
43  
44      /** UID for serializing. */
45      private static final long serialVersionUID = 3206127526058061391L;
46  
47      /** Authentication date stamp. */
48      private final long authenticationDate;
49  
50      /** List of metadata about credentials presented at authentication. */
51      private final List<CredentialMetaData> credentials;
52  
53      /** Authenticated principal. */
54      private final Principal principal;
55  
56      /** Authentication metadata attributes. */
57      private final Map<String, Object> attributes;
58  
59      /** Map of handler name to handler authentication success event. */
60      private final Map<String, HandlerResult> successes;
61  
62      /** Map of handler name to handler authentication failure cause. */
63      private final Map<String, Class<? extends Exception>> failures;
64  
65      /** No-arg constructor for serialization support. */
66      private ImmutableAuthentication() {
67          this.authenticationDate = 0;
68          this.credentials = null;
69          this.principal = null;
70          this.attributes = null;
71          this.successes = null;
72          this.failures = null;
73      }
74  
75      /**
76       * Creates a new instance with the given data.
77       *
78       * @param date Non-null authentication date.
79       * @param credentials Non-null list of credential metadata containing at least one entry.
80       * @param principal Non-null authenticated principal.
81       * @param attributes Nullable map of authentication metadata.
82       * @param successes Non-null map of authentication successes containing at least one entry.
83       * @param failures Nullable map of authentication failures.
84       */
85      public ImmutableAuthentication(
86              final DateTime date,
87              final List<CredentialMetaData> credentials,
88              final Principal principal,
89              final Map<String, Object> attributes,
90              final Map<String, HandlerResult> successes,
91              final Map<String, Class<? extends Exception>> failures) {
92  
93          Assert.notNull(date, "Date cannot be null");
94          Assert.notNull(credentials, "Credential cannot be null");
95          Assert.notNull(principal, "Principal cannot be null");
96          Assert.notNull(successes, "Successes cannot be null");
97          Assert.notEmpty(credentials, "Credential cannot be empty");
98          Assert.notEmpty(successes, "Successes cannot be empty");
99  
100         this.authenticationDate = date.toDate().getTime();
101         this.credentials = credentials;
102         this.principal = principal;
103         this.attributes = attributes.isEmpty() ? null : attributes;
104         this.successes = successes;
105         this.failures = failures.isEmpty() ? null : failures;
106     }
107 
108     @Override
109     public Principal getPrincipal() {
110         return this.principal;
111     }
112 
113     public Date getAuthenticationDate() {
114         return new ImmutableDate(this.authenticationDate);
115     }
116 
117     @Override
118     public Map<String, Object> getAttributes() {
119         return wrap(this.attributes);
120     }
121 
122     @Override
123     public List<CredentialMetaData> getCredentials() {
124         return Collections.unmodifiableList(this.credentials);
125     }
126 
127     @Override
128     public Map<String, HandlerResult> getSuccesses() {
129         return Collections.unmodifiableMap(this.successes);
130     }
131 
132     @Override
133     public Map<String, Class<? extends Exception>> getFailures() {
134         return wrap(this.failures);
135     }
136 
137     @Override
138     public int hashCode() {
139         final HashCodeBuilder builder = new HashCodeBuilder(97, 31);
140         builder.append(this.principal);
141         builder.append(this.authenticationDate);
142         builder.append(this.attributes);
143         builder.append(this.credentials);
144         builder.append(this.successes);
145         builder.append(this.failures);
146         return builder.toHashCode();
147     }
148 
149     @Override
150     public boolean equals(final Object obj) {
151         if (!(obj instanceof Authentication)) {
152             return false;
153         }
154         if (obj == this) {
155             return true;
156         }
157         final Authentication other = (Authentication) obj;
158         final EqualsBuilder builder = new EqualsBuilder();
159         builder.append(this.principal, other.getPrincipal());
160         builder.append(this.credentials, other.getCredentials());
161         builder.append(this.successes, other.getSuccesses());
162         builder.append(this.authenticationDate, other.getAuthenticationDate().getTime());
163         builder.append(wrap(this.attributes), other.getAttributes());
164         builder.append(wrap(this.failures), other.getFailures());
165         return builder.isEquals();
166     }
167 
168     /**
169      * Wraps a possibly null map in an immutable wrapper.
170      *
171      * @param <K> the key type
172      * @param <V> the value type
173      * @param source Nullable map to wrap.
174      * @return {@link Collections#unmodifiableMap(java.util.Map)} if given map is not null, otherwise
175      * {@link java.util.Collections#emptyMap()}.
176      */
177     private static <K, V> Map<K, V> wrap(final Map<K, V> source) {
178         if (source != null) {
179             return Collections.unmodifiableMap(source);
180         }
181         return Collections.emptyMap();
182     }
183 
184     /**
185      * Immutable date implementation that throws {@link UnsupportedOperationException} for setter methods.
186      */
187     private static final class ImmutableDate extends Date {
188 
189         private static final long serialVersionUID = 6275827030191703183L;
190 
191         /** No-arg constructor for serialization support. */
192         private ImmutableDate() {}
193 
194         /**
195          * Creates a new instance with the given epoch time in milliseconds.
196          *
197          * @param instant Milliseconds since the Unix epoch.
198          */
199         ImmutableDate(final long instant) {
200             super(instant);
201         }
202 
203         @Override
204         public void setYear(final int year) {
205             throw new UnsupportedOperationException();
206         }
207 
208         @Override
209         public void setDate(final int date) {
210             throw new UnsupportedOperationException();
211         }
212 
213         @Override
214         public void setHours(final int hours) {
215             throw new UnsupportedOperationException();
216         }
217 
218         @Override
219         public void setMinutes(final int minutes) {
220             throw new UnsupportedOperationException();
221         }
222 
223         @Override
224         public void setSeconds(final int seconds) {
225             throw new UnsupportedOperationException();
226         }
227 
228         @Override
229         public void setTime(final long time) {
230             throw new UnsupportedOperationException();
231         }
232     }
