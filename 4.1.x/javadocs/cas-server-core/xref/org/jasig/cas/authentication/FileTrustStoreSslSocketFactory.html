47  /**
48   * The SSL socket factory that loads the SSL context from a custom
49   * truststore file strictly used ssl handshakes for proxy authentication. 
50   * @author Misagh Moayyed
51   * @since 4.1.0
52   */
53  public final class FileTrustStoreSslSocketFactory extends SSLConnectionSocketFactory {
54  
55      private static final Logger LOGGER = LoggerFactory.getLogger(FileTrustStoreSslSocketFactory.class);
56  
57      /**
58       * Instantiates a new trusted proxy authentication trust store ssl socket factory.
59       * Defaults to <code>TLSv1</code> and {@link SSLConnectionSocketFactory#BROWSER_COMPATIBLE_HOSTNAME_VERIFIER}
60       * for the supported protocols and hostname verification.
61       * @param trustStoreFile the trust store file
62       * @param trustStorePassword the trust store password
63       */
64      public FileTrustStoreSslSocketFactory(final File trustStoreFile, final String trustStorePassword) {
65          this(trustStoreFile, trustStorePassword, KeyStore.getDefaultType());
66      }
67  
68  
69      /**
70       * Instantiates a new trusted proxy authentication trust store ssl socket factory.
71       * @param trustStoreFile the trust store file
72       * @param trustStorePassword the trust store password
73       * @param trustStoreType the trust store type
74       */
75      public FileTrustStoreSslSocketFactory(final File trustStoreFile, final String trustStorePassword,
76                                            final String trustStoreType) {
77          super(getTrustedSslContext(trustStoreFile, trustStorePassword, trustStoreType));
78      }
79  
80      /**
81       * Gets the trusted ssl context.
82       *
83       * @param trustStoreFile the trust store file
84       * @param trustStorePassword the trust store password
85       * @param trustStoreType the trust store type
86       * @return the trusted ssl context
87       */
88      private static SSLContext getTrustedSslContext(final File trustStoreFile, final String trustStorePassword,
89                                              final String trustStoreType) {
90          try {
91  
92              if (!trustStoreFile.exists() || !trustStoreFile.canRead()) {
93                  throw new FileNotFoundException("Truststore file cannot be located at " + trustStoreFile.getCanonicalPath());
94              }
95  
96              final KeyStore casTrustStore = KeyStore.getInstance(trustStoreType);
97              final char[] trustStorePasswordCharArray = trustStorePassword.toCharArray();
98  
99              try (final FileInputStream casStream = new FileInputStream(trustStoreFile)) {
100                 casTrustStore.load(casStream, trustStorePasswordCharArray);
101             }
102 
103             final String defaultAlgorithm = KeyManagerFactory.getDefaultAlgorithm();
104             final X509KeyManager customKeyManager = getKeyManager("PKIX", casTrustStore, trustStorePasswordCharArray);
105             final X509KeyManager jvmKeyManager = getKeyManager(defaultAlgorithm, null, null);
106             final X509TrustManager customTrustManager = getTrustManager("PKIX", casTrustStore);
107             final X509TrustManager jvmTrustManager = getTrustManager(defaultAlgorithm, null);
108 
109             final KeyManager[] keyManagers = {
110                     new CompositeX509KeyManager(Arrays.asList(jvmKeyManager, customKeyManager))
111             };
112             final TrustManager[] trustManagers = {
113                     new CompositeX509TrustManager(Arrays.asList(jvmTrustManager, customTrustManager))
114             };
115 
116             final SSLContext context = SSLContexts.custom().useSSL().build();
117             context.init(keyManagers, trustManagers, null);
118             return context;
119 
120         } catch (final Exception e) {
121             LOGGER.error(e.getMessage(), e);
122             throw new RuntimeException(e);
123         }
124     }
125 
126     /**
127      * Gets key manager.
128      *
129      * @param algorithm the algorithm
130      * @param keystore the keystore
131      * @param password the password
132      * @return the key manager
133      * @throws Exception the exception
134      */
135     private static X509KeyManager getKeyManager(final String algorithm, final KeyStore keystore,
136                                                 final char[] password) throws Exception {
137         final KeyManagerFactory factory = KeyManagerFactory.getInstance(algorithm);
138         factory.init(keystore, password);
139         return (X509KeyManager) factory.getKeyManagers()[0];
140     }
141 
142     /**
143      * Gets trust manager.
144      *
145      * @param algorithm the algorithm
146      * @param keystore the keystore
147      * @return the trust manager
148      * @throws Exception the exception
149      */
150     private static X509TrustManager getTrustManager(final String algorithm,
151                                                     final KeyStore keystore) throws Exception{
152         final TrustManagerFactory factory = TrustManagerFactory.getInstance(algorithm);
153         factory.init(keystore);
154         return (X509TrustManager) factory.getTrustManagers()[0];
155     }

157     private static class DoesNotTrustStrategy implements TrustStrategy {
158         @Override
159         public boolean isTrusted(final X509Certificate[] x509Certificates, final String s) throws CertificateException {
160             return false;
161         }
162     }

164     private static class CompositeX509KeyManager implements X509KeyManager {
165 
166         private final List<X509KeyManager> keyManagers;
167 
168         /**
169          * Represents an ordered list of {@link X509KeyManager}s with most-preferred managers first.
170          * @param keyManagers list of key managers
171          */
172         CompositeX509KeyManager(final List<X509KeyManager> keyManagers) {
173             this.keyManagers = keyManagers;
174         }
175 
176         @Override
177         public String chooseClientAlias(final String[] keyType, final Principal[] issuers, final Socket socket) {
178             for (final X509KeyManager keyManager : keyManagers) {
179                 final String alias = keyManager.chooseClientAlias(keyType, issuers, socket);
180                 if (alias != null) {
181                     return alias;
182                 }
183             }
184             return null;
185         }
186 
187 
188         @Override
189         public String chooseServerAlias(final String keyType, final Principal[] issuers, final Socket socket) {
190             for (final X509KeyManager keyManager : keyManagers) {
191                 final String alias = keyManager.chooseServerAlias(keyType, issuers, socket);
192                 if (alias != null) {
193                     return alias;
194                 }
195             }
196             return null;
197         }
198 
199 
200         @Override
201         public PrivateKey getPrivateKey(final String alias) {
202             for (final X509KeyManager keyManager : keyManagers) {
203                 final PrivateKey privateKey = keyManager.getPrivateKey(alias);
204                 if (privateKey != null) {
205                     return privateKey;
206                 }
207             }
208             return null;
209         }
210 
211 
212         @Override
213         public X509Certificate[] getCertificateChain(final String alias) {
214             for (final X509KeyManager keyManager : keyManagers) {
215                 final X509Certificate[] chain = keyManager.getCertificateChain(alias);
216                 if (chain != null && chain.length > 0) {
217                     return chain;
218                 }
219             }
220             return null;
221         }
222 
223         @Override
224         public String[] getClientAliases(final String keyType, final Principal[] issuers) {
225             final List<String> aliases = new ArrayList<>();
226             for (final X509KeyManager keyManager : keyManagers) {
227                 final List<String> list = Arrays.asList(keyManager.getClientAliases(keyType, issuers));
228                 aliases.addAll(list);
229             }
230             return aliases.toArray(new String[] {});
231         }
232 
233         @Override
234         public  String[] getServerAliases(final String keyType, final Principal[] issuers) {
235             final List<String> aliases = new ArrayList<>();
236             for (final X509KeyManager keyManager : keyManagers) {
237                 final List<String> list = Arrays.asList(keyManager.getServerAliases(keyType, issuers));
238                 aliases.addAll(list);
239             }
240             return aliases.toArray(new String[] {});
241         }
242 
243     }

245     /**
246      * Represents an ordered list of {@link X509TrustManager}s with additive trust. If any one of the
247      * composed managers trusts a certificate chain, then it is trusted by the composite manager.
248      */
249     private static class CompositeX509TrustManager implements X509TrustManager {
250 
251         private static final Logger LOGGER = LoggerFactory.getLogger(CompositeX509TrustManager.class);
252 
253         private final List<X509TrustManager> trustManagers;
254 
255         /**
256          * Instantiates a new Composite x 509 trust manager.
257          *
258          * @param trustManagers the trust managers
259          */
260         CompositeX509TrustManager(final List<X509TrustManager> trustManagers) {
261             this.trustManagers = trustManagers;
262         }
263 
264         @Override
265         public void checkClientTrusted(final X509Certificate[] chain, final String authType) throws CertificateException {
266             for (final X509TrustManager trustManager : trustManagers) {
267                 try {
268                     trustManager.checkClientTrusted(chain, authType);
269                     return;
270                 } catch (final CertificateException e) {
271                     LOGGER.debug(e.getMessage(), e);
272                 }
273             }
274             throw new CertificateException("None of the TrustManagers trust this certificate chain");
275         }
276 
277         @Override
278         public void checkServerTrusted(final X509Certificate[] chain, final String authType) throws CertificateException {
279             for (final X509TrustManager trustManager : trustManagers) {
280                 try {
281                     trustManager.checkServerTrusted(chain, authType);
282                     return;
283                 } catch (final CertificateException e) {
284                     LOGGER.debug(e.getMessage(), e);
285                 }
286             }
287             throw new CertificateException("None of the TrustManagers trust this certificate chain");
288         }
289 
290         @Override
291         public X509Certificate[] getAcceptedIssuers() {
292             final List<X509Certificate> certificates = new ArrayList<>();
293             for (final X509TrustManager trustManager : trustManagers) {
294                 final List<X509Certificate> list = Arrays.asList(trustManager.getAcceptedIssuers());
295                 certificates.addAll(list);
296             }
297             return certificates.toArray(new X509Certificate[] {});
298         }
299     }
