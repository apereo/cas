20  package org.jasig.cas.authentication.support;
21  
22  import org.apache.commons.lang3.StringUtils;
23  import org.jasig.cas.services.RegisteredService;
24  import org.jasig.cas.services.ServicesManager;
25  import org.jasig.cas.util.services.RegisteredServiceCipherExecutor;
26  import org.jasig.cas.web.view.CasViewConstants;
27  import java.util.Map;
28  
29  /**
30   * The default implementation of the attribute
31   * encoder that will use a per-service key-pair
32   * to encrypt the credential password and PGT
33   * when available. All other attributes remain in
34   * place.
35   *
36   * @author Misagh Moayyed
37   * @since 4.1
38   */
39  public class DefaultCasAttributeEncoder extends AbstractCasAttributeEncoder {
40  
41      /**
42       * Instantiates a new Default cas attribute encoder.
43       *
44       * @param servicesManager the services manager
45       */
46      public DefaultCasAttributeEncoder(final ServicesManager servicesManager) {
47          super(servicesManager);
48      }
49  
50      /**
51       * Instantiates a new Default cas attribute encoder.
52       *
53       * @param servicesManager the services manager
54       * @param cipherExecutor the cipher executor
55       */
56      public DefaultCasAttributeEncoder(final ServicesManager servicesManager,
57                                        final RegisteredServiceCipherExecutor cipherExecutor) {
58          super(servicesManager, cipherExecutor);
59      }
60  
61      /**
62       * Encode and encrypt credential password using the public key
63       * supplied by the service. The result is base64 encoded
64       * and put into the attributes collection again, overwriting
65       * the previous value.
66       *
67       * @param attributes the attributes
68       * @param cachedAttributesToEncode the cached attributes to encode
69       * @param cipher the cipher
70       * @param registeredService the registered service
71       */
72      protected final void encodeAndEncryptCredentialPassword(final Map<String, Object> attributes,
73                                                        final Map<String, String> cachedAttributesToEncode,
74                                                        final RegisteredServiceCipherExecutor cipher,
75                                                        final RegisteredService registeredService) {
76          encryptAndEncodeAndPutIntoAttributesMap(attributes, cachedAttributesToEncode,
77                  CasViewConstants.MODEL_ATTRIBUTE_NAME_PRINCIPAL_CREDENTIAL,
78                  cipher, registeredService);
79      }
80  
81      /**
82       * Encode and encrypt pgt.
83       *
84       * @param attributes the attributes
85       * @param cachedAttributesToEncode the cached attributes to encode
86       * @param cipher the cipher
87       * @param registeredService the registered service
88       */
89      protected final void encodeAndEncryptProxyGrantingTicket(final Map<String, Object> attributes,
90                                                         final Map<String, String> cachedAttributesToEncode,
91                                                         final RegisteredServiceCipherExecutor cipher,
92                                                         final RegisteredService registeredService) {
93          encryptAndEncodeAndPutIntoAttributesMap(attributes, cachedAttributesToEncode,
94                  CasViewConstants.MODEL_ATTRIBUTE_NAME_PROXY_GRANTING_TICKET, cipher, registeredService);
95      }
96  
97      /**
98       * Encrypt, encode and put the attribute into attributes map.
99       *
100      * @param attributes the attributes
101      * @param cachedAttributesToEncode the cached attributes to encode
102      * @param cachedAttributeName the cached attribute name
103      * @param cipher the cipher
104      * @param registeredService the registered service
105      */
106     protected final void encryptAndEncodeAndPutIntoAttributesMap(final Map<String, Object> attributes,
107                                                            final Map<String, String> cachedAttributesToEncode,
108                                                            final String cachedAttributeName,
109                                                            final RegisteredServiceCipherExecutor cipher,
110                                                            final RegisteredService registeredService) {
111         final String cachedAttribute = cachedAttributesToEncode.remove(cachedAttributeName);
112         if (StringUtils.isNotBlank(cachedAttribute)) {
113             logger.debug("Retrieved [{}] as a cached model attribute...", cachedAttributeName);
114             final String encodedValue = cipher.encode(cachedAttribute, registeredService);
115             if (StringUtils.isNotBlank(encodedValue)) {
116                 attributes.put(cachedAttributeName, encodedValue);
117                 logger.debug("Encrypted and encoded [{}] as an attribute to [{}].",
118                         cachedAttributeName, encodedValue);
119             }
120         } else {
121             logger.debug("[{}] is not available as a cached model attribute to encrypt...", cachedAttributeName);
122         }
123     }
124 
125     @Override
126     protected void encodeAttributesInternal(final Map<String, Object> attributes,
127                                             final Map<String, String> cachedAttributesToEncode,
128                                             final RegisteredServiceCipherExecutor cipher,
129                                             final RegisteredService registeredService) {
130         encodeAndEncryptCredentialPassword(attributes, cachedAttributesToEncode, cipher, registeredService);
131         encodeAndEncryptProxyGrantingTicket(attributes, cachedAttributesToEncode, cipher, registeredService);
132     }
133 }
