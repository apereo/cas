19  package org.jasig.cas.monitor;
20  
21  import javax.validation.constraints.NotNull;
22  import java.util.Collection;
23  import java.util.Collections;
24  import java.util.LinkedHashMap;
25  import java.util.Map;
26  
27  /**
28   * Simple health check monitor that reports the overall health as the greatest reported
29   * {@link StatusCode} of an arbitrary number of individual checks.
30   *
31   * @author Marvin S. Addison
32   * @since 3.5.0
33   */
34  public class HealthCheckMonitor implements Monitor<HealthStatus> {
35      /** Individual monitors that comprise health check. */
36      @NotNull
37      private Collection<Monitor> monitors = Collections.emptySet();
38  
39  
40      /**
41       * Sets the monitors that comprise the health check.
42       *
43       * @param monitors Collection of monitors responsible for observing various aspects of CAS.
44       */
45      public void setMonitors(final Collection<Monitor> monitors) {
46          this.monitors = monitors;
47      }
48  
49      /**
50       * {@inheritDoc}
51       **/
52      @Override
53      public String getName() {
54          return HealthCheckMonitor.class.getSimpleName();
55      }
56  
57      /**
58       * {@inheritDoc}
59       **/
60      @Override
61      public HealthStatus observe() {
62          final Map<String, Status> results = new LinkedHashMap<>(this.monitors.size());
63          StatusCode code = StatusCode.UNKNOWN;
64          Status result;
65          for (final Monitor monitor : this.monitors) {
66              try {
67                  result = monitor.observe();
68                  final StatusCode resCode = result.getCode();
69                  if (resCode.value() > code.value()) {
70                      code = resCode;
71                  }
72              } catch (final Exception e) {
73                  code = StatusCode.ERROR;
74                  result = new Status(code, e.getClass().getSimpleName() + ": " + e.getMessage());
75              }
76              results.put(monitor.getName(), result);
77          }
78  
79          return new HealthStatus(code, results);
80      }
81  }
