19  package org.jasig.cas.authentication.principal;
20  
21  import org.jasig.cas.authentication.Credential;
22  import org.junit.Test;
23  import org.mockito.ArgumentMatcher;
24  
25  import java.util.Arrays;
26  import java.util.Collections;
27  
28  import static org.junit.Assert.*;
29  import static org.mockito.Mockito.*;
30  
31  /**
32   * Unit test for {@link ChainingPrincipalResolver}.
33   *
34   * @author Marvin S. Addison
35   * @since 4.0.0
36   */
37  public class ChainingPrincipalResolverTest {
38  
39      private final PrincipalFactory principalFactory = new DefaultPrincipalFactory();
40  
41      @Test
42      public void examineSupports() throws Exception {
43          final Credential credential = mock(Credential.class);
44          when(credential.getId()).thenReturn("a");
45  
46          final PrincipalResolver resolver1 = mock(PrincipalResolver.class);
47          when(resolver1.supports(eq(credential))).thenReturn(true);
48  
49          final PrincipalResolver resolver2 = mock(PrincipalResolver.class);
50          when(resolver2.supports(eq(credential))).thenReturn(false);
51  
52          final ChainingPrincipalResolver resolver = new ChainingPrincipalResolver();
53          resolver.setChain(Arrays.asList(resolver1, resolver2));
54          assertTrue(resolver.supports(credential));
55      }
56  
57      @Test
58      public void examineResolve() throws Exception {
59          final Credential credential = mock(Credential.class);
60          when(credential.getId()).thenReturn("input");
61  
62          final PrincipalResolver resolver1 = mock(PrincipalResolver.class);
63          when(resolver1.supports(eq(credential))).thenReturn(true);
64          when(resolver1.resolve((eq(credential)))).thenReturn(principalFactory.createPrincipal("output"));
65  
66          final PrincipalResolver resolver2 = mock(PrincipalResolver.class);
67          when(resolver2.supports(any(Credential.class))).thenReturn(false);
68          when(resolver2.resolve(argThat(new ArgumentMatcher<Credential>() {
69              @Override
70              public boolean matches(final Object o) {
71                  return "output".equals(((Credential) o).getId());
72              }
73          }))).thenReturn(principalFactory.createPrincipal("final", Collections.<String, Object>singletonMap("mail", "final@example.com")));
74  
75          final ChainingPrincipalResolver resolver = new ChainingPrincipalResolver();
76          resolver.setChain(Arrays.asList(resolver1, resolver2));
77          final Principal principal = resolver.resolve(credential);
78          assertEquals("final", principal.getId());
79          assertEquals("final@example.com", principal.getAttributes().get("mail"));
80      }
81  
82  }
