19  package org.jasig.cas.authentication.principal;
20  
21  
22  import static org.junit.Assert.*;
23  
24  import java.util.HashMap;
25  import java.util.Map;
26  
27  import org.junit.Test;
28  
29  /**
30   * @author Scott Battaglia
31   * @since 3.4.4
32   */
33  public class ResponseTests {
34  
35      @Test
36      public void verifyConstructionWithoutFragmentAndNoQueryString() {
37          final String url = "http://localhost:8080/foo";
38          final Map<String, String> attributes = new HashMap<>();
39          attributes.put("ticket", "foobar");
40          final Response response = DefaultResponse.getRedirectResponse(url, attributes);
41          assertEquals(url + "?ticket=foobar", response.getUrl());
42      }
43  
44      @Test
45      public void verifyConstructionWithoutFragmentButHasQueryString() {
46          final String url = "http://localhost:8080/foo?test=boo";
47          final Map<String, String> attributes = new HashMap<>();
48          attributes.put("ticket", "foobar");
49          final Response response = DefaultResponse.getRedirectResponse(url, attributes);
50          assertEquals(url + "&ticket=foobar", response.getUrl());
51      }
52  
53      @Test
54      public void verifyConstructionWithFragmentAndQueryString() {
55          final String url = "http://localhost:8080/foo?test=boo#hello";
56          final Map<String, String> attributes = new HashMap<>();
57          attributes.put("ticket", "foobar");
58          final Response response = DefaultResponse.getRedirectResponse(url, attributes);
59          assertEquals("http://localhost:8080/foo?test=boo&ticket=foobar#hello", response.getUrl());
60      }
61  
62      @Test
63      public void verifyConstructionWithFragmentAndNoQueryString() {
64          final String url = "http://localhost:8080/foo#hello";
65          final Map<String, String> attributes = new HashMap<>();
66          attributes.put("ticket", "foobar");
67          final Response response = DefaultResponse.getRedirectResponse(url, attributes);
68          assertEquals("http://localhost:8080/foo?ticket=foobar#hello", response.getUrl());
69  
70      }
71  
72      @Test
73      public void verifyUrlSanitization() {
74          final String url = "https://www.example.com\r\nLocation: javascript:\r\n\r\n<script>alert(document.cookie)</script>";
75          final Map<String, String> attributes = new HashMap<>();
76          attributes.put("ticket", "ST-12345");
77          final Response response = DefaultResponse.getRedirectResponse(url, attributes);
78          assertEquals("https://www.example.com Location: javascript: <script>alert(document.cookie)</script>?ticket=ST-12345",
79                  response.getUrl());
80      }
81  
82      @Test
83      public void verifyUrlWithUnicode() {
84          final String url = "https://www.example.com/πολιτικῶν";
85          final Map<String, String> attributes = new HashMap<>();
86          attributes.put("ticket", "ST-12345");
87          final Response response = DefaultResponse.getRedirectResponse(url, attributes);
88          assertEquals("https://www.example.com/πολιτικῶν?ticket=ST-12345", response.getUrl());
89      }
90  }
