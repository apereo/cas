27  import org.junit.Before;
28  import org.junit.Test;
29  import org.springframework.mock.web.MockHttpServletRequest;
30  import org.springframework.mock.web.MockHttpServletResponse;
31  import org.springframework.mock.web.MockServletContext;
32  import org.springframework.webflow.context.servlet.ServletExternalContext;
33  import org.springframework.webflow.test.MockRequestContext;
34  
35  import javax.servlet.http.Cookie;
36  
37  import static org.junit.Assert.*;
38  import static org.mockito.Mockito.*;
39  
40  /**
41   * @author Marvin S. Addison
42   * @since 3.4.0
43   */
44  public class SendTicketGrantingTicketActionTests extends AbstractCentralAuthenticationServiceTest {
45      private SendTicketGrantingTicketAction action;
46      private CookieRetrievingCookieGenerator ticketGrantingTicketCookieGenerator;
47      private MockRequestContext context;
48  
49      @Before
50      public void onSetUp() throws Exception {
51  
52          this.ticketGrantingTicketCookieGenerator = new CookieRetrievingCookieGenerator();
53          ticketGrantingTicketCookieGenerator.setCookieName("TGT");
54  
55          this.action = new SendTicketGrantingTicketAction(ticketGrantingTicketCookieGenerator,
56                  getCentralAuthenticationService(), getServicesManager());
57          this.action.setServicesManager(getServicesManager());
58          this.action.setCreateSsoSessionCookieOnRenewAuthentications(true);
59          this.action.afterPropertiesSet();
60  
61          this.context = new MockRequestContext();
62      }
63  
64      @Test
65      public void verifyNoTgtToSet() throws Exception {
66          this.context.setExternalContext(new ServletExternalContext(new MockServletContext(),
67                  new MockHttpServletRequest(), new MockHttpServletResponse()));
68  
69          assertEquals("success", this.action.execute(this.context).getId());
70      }
71  
72      @Test
73      public void verifyTgtToSet() throws Exception {
74          final MockHttpServletResponse response = new MockHttpServletResponse();
75          final MockHttpServletRequest request = new MockHttpServletRequest();
76          final TicketGrantingTicket tgt = mock(TicketGrantingTicket.class);
77          when(tgt.getId()).thenReturn("test");
78  
79          WebUtils.putTicketGrantingTicketInScopes(this.context, tgt);
80          this.context.setExternalContext(new ServletExternalContext(new MockServletContext(),
81                  request, response));
82  
83          assertEquals("success", this.action.execute(this.context).getId());
84          request.setCookies(response.getCookies());
85          assertEquals(tgt.getId(), this.ticketGrantingTicketCookieGenerator.retrieveCookieValue(request));
86      }
87  
88      @Test
89      public void verifyTgtToSetRemovingOldTgt() throws Exception {
90          final MockHttpServletResponse response = new MockHttpServletResponse();
91          final MockHttpServletRequest request = new MockHttpServletRequest();
92  
93          final TicketGrantingTicket tgt = mock(TicketGrantingTicket.class);
94          when(tgt.getId()).thenReturn("test");
95  
96          request.setCookies(new Cookie("TGT", "test5"));
97          WebUtils.putTicketGrantingTicketInScopes(this.context, tgt);
98          this.context.setExternalContext(new ServletExternalContext(new MockServletContext(), request, response));
99  
100         assertEquals("success", this.action.execute(this.context).getId());
101         request.setCookies(response.getCookies());
102         assertEquals(tgt.getId(), this.ticketGrantingTicketCookieGenerator.retrieveCookieValue(request));
103     }
104 
105     @Test
106     public void verifySsoSessionCookieOnRenewAsParameter() throws Exception {
107         final MockHttpServletResponse response = new MockHttpServletResponse();
108         final MockHttpServletRequest request = new MockHttpServletRequest();
109         request.addParameter(CasProtocolConstants.PARAMETER_RENEW, "true");
110 
111         final TicketGrantingTicket tgt = mock(TicketGrantingTicket.class);
112         when(tgt.getId()).thenReturn("test");
113         request.setCookies(new Cookie("TGT", "test5"));
114         WebUtils.putTicketGrantingTicketInScopes(this.context, tgt);
115         this.context.setExternalContext(new ServletExternalContext(new MockServletContext(), request, response));
116 
117         this.action.setCreateSsoSessionCookieOnRenewAuthentications(false);
118         assertEquals("success", this.action.execute(this.context).getId());
119         assertEquals(0, response.getCookies().length);
120     }
121 
122     @Test
123     public void verifySsoSessionCookieOnServiceSsoDisallowed() throws Exception {
124         final MockHttpServletResponse response = new MockHttpServletResponse();
125         final MockHttpServletRequest request = new MockHttpServletRequest();
126 
127         final WebApplicationService svc = mock(WebApplicationService.class);
128         when(svc.getId()).thenReturn("TestSsoFalse");
129 
130         final TicketGrantingTicket tgt = mock(TicketGrantingTicket.class);
131         when(tgt.getId()).thenReturn("test");
132         request.setCookies(new Cookie("TGT", "test5"));
133         WebUtils.putTicketGrantingTicketInScopes(this.context, tgt);
134         this.context.setExternalContext(new ServletExternalContext(new MockServletContext(), request, response));
135         this.context.getFlowScope().put("service", svc);
136         this.action.setCreateSsoSessionCookieOnRenewAuthentications(false);
137         assertEquals("success", this.action.execute(this.context).getId());
138         assertEquals(0, response.getCookies().length);
139     }
