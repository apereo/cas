19  package org.jasig.cas.util;
20  
21  import org.jasig.cas.authentication.FileTrustStoreSslSocketFactory;
22  import org.jasig.cas.util.http.HttpClient;
23  import org.jasig.cas.util.http.SimpleHttpClientFactoryBean;
24  import org.junit.Before;
25  import org.junit.Ignore;
26  import org.junit.Test;
27  import org.springframework.core.io.ClassPathResource;
28  
29  import static org.junit.Assert.assertTrue;
30  
31  /**
32   * @author Misagh Moayyed
33   * @since 4.1.0
34   */
35  public class TrustedProxyAuthenticationTrustStoreSslSocketFactoryTests {
36      private static final ClassPathResource TRUST_STORE = new ClassPathResource("truststore.jks");
37      private static final String TRUST_STORE_PSW = "changeit";
38  
39      private HttpClient client;
40  
41      @Before
42      public void prepareHttpClient() throws Exception {
43          final FileTrustStoreSslSocketFactory sslFactory = new FileTrustStoreSslSocketFactory(
44                  TRUST_STORE.getFile(), TRUST_STORE_PSW);
45  
46          final SimpleHttpClientFactoryBean clientFactory = new SimpleHttpClientFactoryBean();
47          clientFactory.setSslSocketFactory(sslFactory);
48          this.client = clientFactory.getObject();
49      }
50  
51      @Ignore
52      public void verifySuccessfulConnection() {
53          final boolean valid = client.isValidEndPoint("https://www.github.com");
54          assertTrue(valid);
55      }
56  
57      @Test
58      public void verifySuccessfulConnectionWithCustomSSLCert() {
59          final boolean valid = client.isValidEndPoint("https://www.cacert.org");
60          assertTrue(valid);
61      }
62  
63  }
