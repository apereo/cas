46  /**
47   * Unit tests for {@link org.jasig.cas.support.rest.TicketsResource}.
48   *
49   * @author Dmitriy Kopylenko
50   * @since 4.0.0
51   */
52  @RunWith(MockitoJUnitRunner.class)
53  public class TicketsResourceTests {
54  
55      @Mock
56      private CentralAuthenticationService casMock;
57  
58      @InjectMocks
59      private TicketsResource ticketsResourceUnderTest;
60  
61      private MockMvc mockMvc;
62  
63      @Before
64      public void setup() {
65          this.mockMvc = MockMvcBuilders.standaloneSetup(this.ticketsResourceUnderTest)
66                  .defaultRequest(get("/")
67                  .contextPath("/cas")
68                  .servletPath("/v1")
69                  .contentType(MediaType.APPLICATION_FORM_URLENCODED))
70                  .build();
71      }
72  
73      @Test
74      public void normalCreationOfTGT() throws Throwable {
75          final String expectedReturnEntityBody = "<!DOCTYPE HTML PUBLIC \\\"-//IETF//DTD HTML 2.0//EN\\\">"
76                  + "<html><head><title>201 Created</title></head><body><h1>TGT Created</h1>"
77                  + "<form action=\"http://localhost/cas/v1/tickets/TGT-1\" "
78                  + "method=\"POST\">Service:<input type=\"text\" name=\"service\" value=\"\">"
79                  + "<br><input type=\"submit\" value=\"Submit\"></form></body></html>";
80  
81          configureCasMockToCreateValidTGT();
82  
83          this.mockMvc.perform(post("/cas/v1/tickets")
84                  .param("username", "test")
85                  .param("password", "test"))
86                  .andExpect(status().isCreated())
87                  .andExpect(header().string("Location", "http://localhost/cas/v1/tickets/TGT-1"))
88                  .andExpect(content().contentType(MediaType.TEXT_HTML))
89                  .andExpect(content().string(expectedReturnEntityBody));
90      }
91  
92      @Test
93      public void creationOfTGTWithAuthenticationException() throws Throwable {
94          configureCasMockTGTCreationToThrowAuthenticationException();
95  
96          this.mockMvc.perform(post("/cas/v1/tickets")
97                  .param("username", "test")
98                  .param("password", "test"))
99                  .andExpect(status().is4xxClientError())
100                 .andExpect(content().string("1 errors, 0 successes"));
101     }
102 
103     @Test
104     public void normalCreationOfST() throws Throwable {
105         configureCasMockToCreateValidST();
106 
107         this.mockMvc.perform(post("/cas/v1/tickets/TGT-1")
108                 .param("service", "https://www.google.com"))
109                 .andExpect(status().isOk())
110                 .andExpect(content().contentType("text/plain;charset=ISO-8859-1"))
111                 .andExpect(content().string("ST-1"));
112     }
113 
114     @Test
115     public void creationOfSTWithInvalidTicketException() throws Throwable {
116         configureCasMockSTCreationToThrow(new InvalidTicketException("TGT-1"));
117 
118         this.mockMvc.perform(post("/cas/v1/tickets/TGT-1")
119                 .param("service", "https://www.google.com"))
120                 .andExpect(status().isNotFound())
121                 .andExpect(content().string("TicketGrantingTicket could not be found"));
122     }
123 
124     @Test
125     public void creationOfSTWithGeneralException() throws Throwable {
126         configureCasMockSTCreationToThrow(new RuntimeException("Other exception"));
127 
128         this.mockMvc.perform(post("/cas/v1/tickets/TGT-1")
129                 .param("service", "https://www.google.com"))
130                 .andExpect(status().is4xxClientError())
131                 .andExpect(content().string("Other exception"));
132     }
133 
134     @Test
135     public void deletionOfTGT() throws Throwable {
136         this.mockMvc.perform(delete("/cas/v1/tickets/TGT-1"))
137                 .andExpect(status().isOk());
138     }
139 
140     private void configureCasMockToCreateValidTGT() throws Throwable {
141         final TicketGrantingTicket tgt = mock(TicketGrantingTicket.class);
142         when(tgt.getId()).thenReturn("TGT-1");
143         when(this.casMock.createTicketGrantingTicket(any(Credential.class))).thenReturn(tgt);
144     }
145 
146     private void configureCasMockTGTCreationToThrowAuthenticationException() throws Throwable {
147         final Map<String, Class<? extends Exception>> handlerErrors = new HashMap<>(1);
148         handlerErrors.put("TestCaseAuthenticationHander", LoginException.class);
149         when(this.casMock.createTicketGrantingTicket(any(Credential.class))).thenThrow(new AuthenticationException(handlerErrors));
150     }
151 
152     private void configureCasMockSTCreationToThrow(final Throwable e) throws Throwable {
153         when(this.casMock.grantServiceTicket(anyString(), any(Service.class))).thenThrow(e);
154     }
155 
156     private void configureCasMockToCreateValidST() throws Throwable {
157         final ServiceTicket st = mock(ServiceTicket.class);
158         when(st.getId()).thenReturn("ST-1");
159         when(this.casMock.grantServiceTicket(anyString(), any(Service.class))).thenReturn(st);
160     }
161 }
