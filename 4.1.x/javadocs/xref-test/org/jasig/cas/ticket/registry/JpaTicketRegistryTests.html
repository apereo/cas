111     @Test
112     public void verifyTicketCreationAndDeletion() throws Exception {
113         final TicketGrantingTicket newTgt = newTGT();
114         addTicketInTransaction(newTgt);
115         final TicketGrantingTicket tgtFromDb = (TicketGrantingTicket) getTicketInTransaction(newTgt.getId());
116         assertNotNull(tgtFromDb);
117         assertEquals(newTgt.getId(), tgtFromDb.getId());
118         final ServiceTicket newSt = grantServiceTicketInTransaction(tgtFromDb);
119         final ServiceTicket stFromDb = (ServiceTicket) getTicketInTransaction(newSt.getId());
120         assertNotNull(stFromDb);
121         assertEquals(newSt.getId(), stFromDb.getId());
122         deleteTicketInTransaction(newTgt.getId());
123         assertNull(getTicketInTransaction(newTgt.getId()));
124         assertNull(getTicketInTransaction(newSt.getId()));
125     }
126 
127     @Test
128     @IfProfileValue(name="cas.jpa.concurrent", value="true")
129     public void verifyConcurrentServiceTicketGeneration() throws Exception {
130         final TicketGrantingTicket newTgt = newTGT();
131         addTicketInTransaction(newTgt);
132         final ExecutorService executor = Executors.newFixedThreadPool(CONCURRENT_SIZE);
133         try {
134             final List<ServiceTicketGenerator> generators = new ArrayList<>(CONCURRENT_SIZE);
135             for (int i = 0; i < CONCURRENT_SIZE; i++) {
136                 generators.add(new ServiceTicketGenerator(newTgt.getId(), this.jpaTicketRegistry, this.txManager));
137             }
138             final List<Future<String>> results = executor.invokeAll(generators);
139             for (final Future<String> result : results) {
140                 assertNotNull(result.get());
141             }
142         } catch (final Exception e) {
143             logger.debug("testConcurrentServiceTicketGeneration produced an error", e);
144             fail("testConcurrentServiceTicketGeneration failed.");
145         } finally {
146             executor.shutdownNow();
147         }
148     }

151     static TicketGrantingTicket newTGT() {
152         final Principal principal = new DefaultPrincipalFactory().createPrincipal(
153                 "bob", Collections.singletonMap("displayName", (Object) "Bob"));
154         return new TicketGrantingTicketImpl(
155                 ID_GENERATOR.getNewTicketId("TGT"),
156                 TestUtils.getAuthentication(principal),
157                 EXP_POLICY_TGT);
158     }
159 
160     static ServiceTicket newST(final TicketGrantingTicket parent) {
161        return parent.grantServiceTicket(
162                ID_GENERATOR.getNewTicketId("ST"),
163                new MockService("https://service.example.com"),
164                EXP_POLICY_ST,
165                false);
166     }

168     void addTicketInTransaction(final Ticket ticket) {
169         new TransactionTemplate(txManager).execute(new TransactionCallback<Void>() {
170             public Void doInTransaction(final TransactionStatus status) {
171                 jpaTicketRegistry.addTicket(ticket);
172                 return null;
173             }
174         });
175     }
176 
177     void deleteTicketInTransaction(final String ticketId) {
178         new TransactionTemplate(txManager).execute(new TransactionCallback<Void>() {
179             public Void doInTransaction(final TransactionStatus status) {
180                 jpaTicketRegistry.deleteTicket(ticketId);
181                 return null;
182             }
183         });
184     }
185 
186     Ticket getTicketInTransaction(final String ticketId) {
187         return new TransactionTemplate(txManager).execute(new TransactionCallback<Ticket>() {
188             public Ticket doInTransaction(final TransactionStatus status) {
189                 return jpaTicketRegistry.getTicket(ticketId);
190             }
191         });
192     }
193 
194     ServiceTicket grantServiceTicketInTransaction(final TicketGrantingTicket parent) {
195         return new TransactionTemplate(txManager).execute(new TransactionCallback<ServiceTicket>() {
196             public ServiceTicket doInTransaction(final TransactionStatus status) {
197                 final ServiceTicket st = newST(parent);
198                 jpaTicketRegistry.addTicket(st);
199                 return st;
200             }
201         });
202     }
203 
204     private static class ServiceTicketGenerator implements Callable<String> {
205         private PlatformTransactionManager txManager;
206         private final String parentTgtId;
207         private final JpaTicketRegistry jpaTicketRegistry;
208 
209         public ServiceTicketGenerator(final String tgtId, final JpaTicketRegistry jpaTicketRegistry,
210                                       final PlatformTransactionManager txManager) {
211             parentTgtId = tgtId;
212             this.jpaTicketRegistry = jpaTicketRegistry;
213             this.txManager = txManager;
214         }
215 
216         /**
217      * {@inheritDoc}
218      */
219         @Override
220         public String call() throws Exception {
221             return new TransactionTemplate(txManager).execute(new TransactionCallback<String>() {
222                 public String doInTransaction(final TransactionStatus status) {
223                     // Querying for the TGT prior to updating it as done in
224                     // CentralAuthenticationServiceImpl#grantServiceTicket(String, Service, Credential)
225                     final ServiceTicket st = newST((TicketGrantingTicket) jpaTicketRegistry.getTicket(parentTgtId));
226                     jpaTicketRegistry.addTicket(st);
227                     return st.getId();
228                 }
229             });
230         }
