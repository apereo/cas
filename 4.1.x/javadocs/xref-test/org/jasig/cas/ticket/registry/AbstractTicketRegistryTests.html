189     @Test
190     public void verifyGetTicketsFromRegistryEqualToTicketsAdded() {
191         final Collection<Ticket> tickets = new ArrayList<>();
192 
193         for (int i = 0; i < TICKETS_IN_REGISTRY; i++) {
194             final TicketGrantingTicket ticketGrantingTicket = new TicketGrantingTicketImpl("TEST" + i,
195                     TestUtils.getAuthentication(), new NeverExpiresExpirationPolicy());
196             final ServiceTicket st = ticketGrantingTicket.grantServiceTicket("tests" + i, TestUtils.getService(),
197                     new NeverExpiresExpirationPolicy(), false);
198             tickets.add(ticketGrantingTicket);
199             tickets.add(st);
200             this.ticketRegistry.addTicket(ticketGrantingTicket);
201             this.ticketRegistry.addTicket(st);
202         }
203 
204         try {
205             final Collection<Ticket> ticketRegistryTickets = this.ticketRegistry.getTickets();
206             assertEquals("The size of the registry is not the same as the collection.", ticketRegistryTickets.size(),
207                     tickets.size());
208 
209             for (final Ticket ticket : tickets) {
210                 if (!ticketRegistryTickets.contains(ticket)) {
211                     fail("Ticket was added to registry but was not found in retrieval of collection of all tickets.");
212                 }
213             }
214         } catch (final Exception e) {
215             fail("Caught an exception. But no exception should have been thrown.");
216         }
217     }

219     @Test
220     public void verifyDeleteTicketWithChildren() {
221         try {
222             this.ticketRegistry.addTicket(new TicketGrantingTicketImpl(
223                     "TGT", TestUtils.getAuthentication(), new NeverExpiresExpirationPolicy()));
224             final TicketGrantingTicket tgt = this.ticketRegistry.getTicket(
225                     "TGT", TicketGrantingTicket.class);
226 
227             final Service service = TestUtils.getService("TGT_DELETE_TEST");
228 
229             final ServiceTicket st1 = tgt.grantServiceTicket(
230                     "ST1", service, new NeverExpiresExpirationPolicy(), true);
231             final ServiceTicket st2 = tgt.grantServiceTicket(
232                     "ST2", service, new NeverExpiresExpirationPolicy(), true);
233             final ServiceTicket st3 = tgt.grantServiceTicket(
234                     "ST3", service, new NeverExpiresExpirationPolicy(), true);
235 
236             this.ticketRegistry.addTicket(st1);
237             this.ticketRegistry.addTicket(st2);
238             this.ticketRegistry.addTicket(st3);
239 
240             assertNotNull(this.ticketRegistry.getTicket("TGT", TicketGrantingTicket.class));
241             assertNotNull(this.ticketRegistry.getTicket("ST1", ServiceTicket.class));
242             assertNotNull(this.ticketRegistry.getTicket("ST2", ServiceTicket.class));
243             assertNotNull(this.ticketRegistry.getTicket("ST3", ServiceTicket.class));
244 
245             this.ticketRegistry.deleteTicket(tgt.getId());
246 
247             assertNull(this.ticketRegistry.getTicket("TGT", TicketGrantingTicket.class));
248             assertNull(this.ticketRegistry.getTicket("ST1", ServiceTicket.class));
249             assertNull(this.ticketRegistry.getTicket("ST2", ServiceTicket.class));
250             assertNull(this.ticketRegistry.getTicket("ST3", ServiceTicket.class));
251         } catch (final Exception e) {
252             fail("Caught an exception. But no exception should have been thrown.");
253         }
254     }
