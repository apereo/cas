27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  import java.io.BufferedReader;
30  import java.io.IOException;
31  import java.io.InputStream;
32  import java.io.InputStreamReader;
33  import java.io.StringReader;
34  import java.util.ArrayList;
35  import java.util.Collection;

37  /**
38   * Utility class used by all tests that provision and deprovision LDAP test data.
39   *
40   * @author Marvin S. Addison
41   * @since 4.0.0
42   */
43  public final class LdapTestUtils {
44  
45      /** Placeholder for base DN in LDIF files. */
46      private static final String BASE_DN_PLACEHOLDER = "${ldapBaseDn}";
47  
48      /** System-wide newline character string. */
49      private static final String NEWLINE = System.getProperty("line.separator");
50  
51      private static final Logger LOGGER = LoggerFactory.getLogger(LdapTestUtils.class);
52  
53      /** Private constructor of utility class. */
54      private LdapTestUtils() {
55      }
56  
57      /**
58       * Reads an LDIF into a collection of LDAP entries. The components performs a simple property
59       * replacement in the LDIF data where <pre>${ldapBaseDn}</pre> is replaced with the environment-specific base
60       * DN.
61       *
62       * @param ldif LDIF resource, typically a file on filesystem or classpath.
63       * @param baseDn The directory branch where the entry resides.
64       *
65       * @return LDAP entries contained in the LDIF.
66       *
67       * @throws IOException On IO errors reading LDIF.
68       */
69      public static Collection<LdapEntry> readLdif(final InputStream ldif, final String baseDn) throws IOException {
70          final StringBuilder builder = new StringBuilder();
71          try (final BufferedReader reader = new BufferedReader(new InputStreamReader(ldif))) {
72              String line;
73              while ((line = reader.readLine()) != null) {
74                  if (line.contains(BASE_DN_PLACEHOLDER)) {
75                      builder.append(line.replace(BASE_DN_PLACEHOLDER, baseDn));
76                  } else {
77                      builder.append(line);
78                  }
79                  builder.append(NEWLINE);
80              }
81          }
82          return new LdifReader(new StringReader(builder.toString())).read().getEntries();
83      }
84  
85      /**
86       * Creates the given LDAP entries.
87       *
88       * @param connection Open LDAP connection used to connect to directory.
89       * @param entries Collection of LDAP entries.
90       *
91       * @throws Exception On LDAP errors.
92       */
93      public static void createLdapEntries(final LDAPConnection connection, final Collection<LdapEntry> entries) throws Exception {
94          for (final LdapEntry entry : entries) {
95              final Collection<Attribute> attrs = new ArrayList<>(entry.getAttributeNames().length);
96              for (final LdapAttribute a : entry.getAttributes()) {
97                  attrs.add(new Attribute(a.getName(), a.getStringValues()));
98              }
99              connection.add(new AddRequest(entry.getDn(), attrs));
100         }
101     }
102 }
