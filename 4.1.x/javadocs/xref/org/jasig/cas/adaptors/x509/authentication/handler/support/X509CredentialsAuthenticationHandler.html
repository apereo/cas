40  /**
41   * Authentication Handler that accepts X509 Certificates, determines their
42   * validity and ensures that they were issued by a trusted issuer. (targeted at
43   * X509v3) Optionally checks KeyUsage extension in the user certificate
44   * (container should do that too). Note that this handler trusts the servlet
45   * container to do some initial checks like path validation. Deployers can
46   * supply an optional pattern to match subject dns against to further restrict
47   * certificates in case they are not using their own issuer. It's also possible
48   * to specify a maximum pathLength for the SUPPLIED certificates. (note that
49   * this does not include a pathLength check for the root certificate)
50   * [PathLength is 0 for the CA certificate that issues the end-user certificate]
51   *
52   * @author Scott Battaglia
53   * @author Jan Van der Velpen
54   * @since 3.0.0.4
55   */
56  public class X509CredentialsAuthenticationHandler extends AbstractPreAndPostProcessingAuthenticationHandler {
57  
58      /** Default setting to limit the number of intermediate certificates. */
59      private static final int DEFAULT_MAXPATHLENGTH = 1;
60  
61      /** Default setting whether to allow unspecified number of intermediate certificates. */
62      private static final boolean DEFAULT_MAXPATHLENGTH_ALLOW_UNSPECIFIED = false;
63  
64      /** Default setting to check keyUsage extension. */
65      private static final boolean DEFAULT_CHECK_KEYUSAGE = false;
66  
67      /**
68       * Default setting to force require "KeyUsage" extension.
69       */
70      private static final boolean DEFAULT_REQUIRE_KEYUSAGE = false;
71  
72      /** Default subject pattern match. */
73      private static final Pattern DEFAULT_SUBJECT_DN_PATTERN = Pattern.compile(".*");
74  
75      /** OID for KeyUsage X.509v3 extension field. */
76      private static final String KEY_USAGE_OID = "2.5.29.15";
77  
78      /** Instance of Logging. */
79      private final Logger logger = LoggerFactory.getLogger(getClass());
80  
81      /** The compiled pattern supplied by the deployer. */
82      @NotNull
83      private Pattern regExTrustedIssuerDnPattern;
84  
85      /**
86       * Deployer supplied setting for maximum pathLength in a SUPPLIED
87       * certificate.
88       */
89      private int maxPathLength = DEFAULT_MAXPATHLENGTH;
90  
91      /**
92       * Deployer supplied setting to allow unlimited pathLength in a SUPPLIED
93       * certificate.
94       */
95      private boolean maxPathLengthAllowUnspecified = DEFAULT_MAXPATHLENGTH_ALLOW_UNSPECIFIED;
96  
97      /** Deployer supplied setting to check the KeyUsage extension. */
98      private boolean checkKeyUsage = DEFAULT_CHECK_KEYUSAGE;
99  
100     /**
101      * Deployer supplied setting to force require the correct KeyUsage
102      * extension.
103      */
104     private boolean requireKeyUsage = DEFAULT_REQUIRE_KEYUSAGE;
105 
106     /** The compiled pattern for trusted DN's supplied by the deployer. */
107     @NotNull
108     private Pattern regExSubjectDnPattern = DEFAULT_SUBJECT_DN_PATTERN;
109 
110     /** Certificate revocation checker component. */
111     @NotNull
112     private RevocationChecker revocationChecker = new NoOpRevocationChecker();
113 
114 
115     @Override
116     public boolean supports(final Credential credential) {
117         return credential != null
118                 && X509CertificateCredential.class.isAssignableFrom(credential
119                         .getClass());
120     }
121 
122     /**
123      * {@inheritDoc}
124      */
125     @Override
126     protected final HandlerResult doAuthentication(final Credential credential) throws GeneralSecurityException, PreventedException {
127 
128         final X509CertificateCredential x509Credential = (X509CertificateCredential) credential;
129         final X509Certificate[] certificates = x509Credential.getCertificates();
130 
131         X509Certificate clientCert = null;
132         boolean hasTrustedIssuer = false;
133         for (int i = certificates.length - 1; i >= 0; i--) {
134             final X509Certificate certificate = certificates[i];
135             logger.debug("Evaluating {}", CertUtils.toString(certificate));
136 
137             validate(certificate);
138 
139             if (!hasTrustedIssuer) {
140                 hasTrustedIssuer = isCertificateFromTrustedIssuer(certificate);
141             }
142 
143             // getBasicConstraints returns pathLenContraint which is generally
144             // >=0 when this is a CA cert and -1 when it's not
145             final int pathLength = certificate.getBasicConstraints();
146             if (pathLength < 0) {
147                 logger.debug("Found valid client certificate");
148                 clientCert = certificate;
149             } else {
150                 logger.debug("Found valid CA certificate");
151             }
152         }
153         if (hasTrustedIssuer && clientCert != null) {
154             x509Credential.setCertificate(clientCert);
155             return new DefaultHandlerResult(this, x509Credential, this.principalFactory.createPrincipal(x509Credential.getId()));
156         }
157         throw new FailedLoginException();
158     }

196     /**
197      * Sets the component responsible for evaluating certificate revocation status for client
198      * certificates presented to handler. The default checker is a NO-OP implementation
199      * for backward compatibility with previous versions that do not perform revocation
200      * checking.
201      *
202      * @param checker Revocation checker component.
203      */
204     public void setRevocationChecker(final RevocationChecker checker) {
205         this.revocationChecker = checker;
206     }
207 
208     /**
209      * Validate the X509Certificate received.
210      *
211      * @param cert the cert
212      * @throws GeneralSecurityException the general security exception
213      */
214     private void validate(final X509Certificate cert) throws GeneralSecurityException {
215         cert.checkValidity();
216         this.revocationChecker.check(cert);
217 
218         final int pathLength = cert.getBasicConstraints();
219         if (pathLength < 0) {
220             if (!isCertificateAllowed(cert)) {
221                 throw new FailedLoginException(
222                         "Certificate subject does not match pattern " + this.regExSubjectDnPattern.pattern());
223             }
224             if (this.checkKeyUsage && !isValidKeyUsage(cert)) {
225                 throw new FailedLoginException(
226                         "Certificate keyUsage constraint forbids SSL client authentication.");
227             }
228         } else {
229             // Check pathLength for CA cert
230             if (pathLength == Integer.MAX_VALUE && !this.maxPathLengthAllowUnspecified) {
231                 throw new FailedLoginException("Unlimited certificate path length not allowed by configuration.");
232             } else if (pathLength > this.maxPathLength && pathLength < Integer.MAX_VALUE) {
233                 throw new FailedLoginException(String.format(
234                         "Certificate path length %s exceeds maximum value %s.", pathLength, this.maxPathLength));
235             }
236         }
237     }
238 
239     /**
240      * Checks if is valid key usage. <p>
241      * KeyUsage ::= BIT STRING { digitalSignature (0), nonRepudiation (1),
242      * keyEncipherment (2), dataEncipherment (3), keyAgreement (4),
243      * keyCertSign (5), cRLSign (6), encipherOnly (7), decipherOnly (8) }
244      *          
245      * @param certificate the certificate
246      * @return true, if  valid key usage
247      */
248     private boolean isValidKeyUsage(final X509Certificate certificate) {
249         logger.debug("Checking certificate keyUsage extension");
250         final boolean[] keyUsage = certificate.getKeyUsage();
251         if (keyUsage == null) {
252             logger.warn("Configuration specifies checkKeyUsage but keyUsage extension not found in certificate.");
253             return !this.requireKeyUsage;
254         }
255 
256         final boolean valid;
257         if (isCritical(certificate, KEY_USAGE_OID) || this.requireKeyUsage) {
258             logger.debug("KeyUsage extension is marked critical or required by configuration.");
259             valid = keyUsage[0];
260         } else {
261             logger.debug(
262                "KeyUsage digitalSignature=%s, Returning true since keyUsage validation not required by configuration.");
263             valid = true;
264         }
265         return valid;
266     }
267 
268     /**
269      * Checks if critical extension oids contain the extension oid.
270      *
271      * @param certificate the certificate
272      * @param extensionOid the extension oid
273      * @return true, if  critical
274      */
275     private boolean isCritical(final X509Certificate certificate, final String extensionOid) {
276         final Set<String> criticalOids = certificate.getCriticalExtensionOIDs();
277 
278         if (criticalOids == null || criticalOids.isEmpty()) {
279             return false;
280         }
281 
282         return criticalOids.contains(extensionOid);
283     }
284 
285     /**
286      * Checks if is certificate allowed based no the pattern given.
287      *
288      * @param cert the cert
289      * @return true, if  certificate allowed
290      */
291     private boolean isCertificateAllowed(final X509Certificate cert) {
292         return doesNameMatchPattern(cert.getSubjectDN(), this.regExSubjectDnPattern);
293     }
294 
295     /**
296      * Checks if is certificate from trusted issuer based on the regex pattern.
297      *
298      * @param cert the cert
299      * @return true, if  certificate from trusted issuer
300      */
301     private boolean isCertificateFromTrustedIssuer(final X509Certificate cert) {
302         return doesNameMatchPattern(cert.getIssuerDN(), this.regExTrustedIssuerDnPattern);
303     }
304 
305     /**
306      * Does principal name match pattern?
307      *
308      * @param principal the principal
309      * @param pattern the pattern
310      * @return true, if successful
311      */
312     private boolean doesNameMatchPattern(final Principal principal,
313             final Pattern pattern) {
314         final String name = principal.getName();
315         final boolean result = pattern.matcher(name).matches();
316         logger.debug(String.format("%s matches %s == %s", pattern.pattern(), name, result));
317         return result;
318     }
