118     /**
119      * From registered service to a service bean.
120      *
121      * @param svc the svc
122      * @return the registered service bean
123      */
124     public static RegisteredServiceEditBean fromRegisteredService(final RegisteredService svc) {
125         final RegisteredServiceEditBean serviceBean = new RegisteredServiceEditBean();
126         final ServiceData bean = serviceBean.getServiceData();
127 
128         bean.setAssignedId(svc.getId());
129         bean.setServiceId(svc.getServiceId());
130         bean.setName(svc.getName());
131         bean.setDescription(svc.getDescription());
132         if (svc.getLogo() != null) {
133             bean.setLogoUrl(svc.getLogo().toExternalForm());
134         }
135         bean.setRequiredHandlers(svc.getRequiredHandlers());
136 
137         configureAccessStrategy(svc, bean);
138 
139         if (svc instanceof OAuthRegisteredCallbackAuthorizeService) {
140             bean.setType(RegisteredServiceTypeEditBean.OAUTH_CALLBACK_AUTHZ.toString());
141         }
142 
143         if (svc instanceof OAuthRegisteredService) {
144             bean.setType(RegisteredServiceTypeEditBean.OAUTH.toString());
145             final OAuthRegisteredService oauth = (OAuthRegisteredService) svc;
146             final RegisteredServiceOAuthTypeEditBean oauthBean = bean.getOauth();
147             oauthBean.setBypass(oauth.isBypassApprovalPrompt());
148             oauthBean.setClientId(oauth.getClientId());
149             oauthBean.setClientSecret(oauth.getClientSecret());
150         }
151 
152         bean.setTheme(svc.getTheme());
153         bean.setEvalOrder(svc.getEvaluationOrder());
154         final LogoutType logoutType = svc.getLogoutType();
155         switch (logoutType) {
156             case BACK_CHANNEL:
157                 bean.setLogoutType(RegisteredServiceLogoutTypeEditBean.BACK.toString());
158                 break;
159             case FRONT_CHANNEL:
160                 bean.setLogoutType(RegisteredServiceLogoutTypeEditBean.FRONT.toString());
161                 break;
162             default:
163                 bean.setLogoutType(RegisteredServiceLogoutTypeEditBean.NONE.toString());
164                 break;
165         }
166         final URL url = svc.getLogoutUrl();
167         if (url != null) {
168             bean.setLogoutUrl(url.toExternalForm());
169         }
170         final RegisteredServiceUsernameAttributeProvider provider = svc.getUsernameAttributeProvider();
171         final RegisteredServiceUsernameAttributeProviderEditBean uBean = bean.getUserAttrProvider();
172 
173         configureUsernameAttributeProvider(provider, uBean);
174 
175         final RegisteredServicePublicKey key = svc.getPublicKey();
176         final RegisteredServicePublicKeyEditBean pBean = bean.getPublicKey();
177         if (key != null) {
178             pBean.setAlgorithm(key.getAlgorithm());
179             pBean.setLocation(key.getLocation());
180         }
181         configureProxyPolicy(svc, bean);
182         configureAttributeReleasePolicy(svc, bean);
183         return serviceBean;
184     }

186     /**
187      * Configure access strategy.
188      *
189      * @param svc the svc
190      * @param bean the bean
191      */
192     private static void configureAccessStrategy(final RegisteredService svc, final ServiceData bean) {
193         final RegisteredServiceAccessStrategy accessStrategy = svc.getAccessStrategy();
194         final RegisteredServiceSupportAccessEditBean accessBean = bean.getSupportAccess();
195         accessBean.setCasEnabled(accessStrategy.isServiceAccessAllowed());
196         accessBean.setSsoEnabled(accessStrategy.isServiceAccessAllowedForSso());
197 
198         if (accessStrategy instanceof DefaultRegisteredServiceAccessStrategy) {
199             final DefaultRegisteredServiceAccessStrategy def = (DefaultRegisteredServiceAccessStrategy) accessStrategy;
200             accessBean.setRequireAll(def.isRequireAllAttributes());
201             accessBean.setRequiredAttr(def.getRequiredAttributes());
202         }
203     }

205     /**
206      * Configure attribute release policy.
207      *
208      * @param svc the svc
209      * @param bean the bean
210      */
211     private static void configureAttributeReleasePolicy(final RegisteredService svc, final ServiceData bean) {
212         final AbstractAttributeReleasePolicy attrPolicy = (AbstractAttributeReleasePolicy) svc.getAttributeReleasePolicy();
213         if (attrPolicy != null) {
214             final RegisteredServiceAttributeReleasePolicyEditBean attrPolicyBean = bean.getAttrRelease();
215 
216             attrPolicyBean.setReleasePassword(attrPolicy.isAuthorizedToReleaseCredentialPassword());
217             attrPolicyBean.setReleaseTicket(attrPolicy.isAuthorizedToReleaseProxyGrantingTicket());
218 
219             final AttributeFilter filter = attrPolicy.getAttributeFilter();
220             if (filter != null) {
221                 if (filter instanceof RegisteredServiceRegexAttributeFilter) {
222                     final RegisteredServiceRegexAttributeFilter regex =
223                             (RegisteredServiceRegexAttributeFilter) filter;
224                     attrPolicyBean.setAttrFilter(regex.getPattern().pattern());
225                 }
226             }
227 
228             final PrincipalAttributesRepository pr = attrPolicy.getPrincipalAttributesRepository();
229             if (pr instanceof DefaultPrincipalAttributesRepository) {
230                 attrPolicyBean.setAttrOption(
231                         RegisteredServiceAttributeReleasePolicyEditBean.Types.DEFAULT.toString());
232             } else if (pr instanceof CachingPrincipalAttributesRepository) {
233                 attrPolicyBean.setAttrOption(
234                         RegisteredServiceAttributeReleasePolicyEditBean.Types.CACHED.toString());
235                 final CachingPrincipalAttributesRepository cc = (CachingPrincipalAttributesRepository) pr;
236                 final Duration duration = cc.getDuration();
237                 attrPolicyBean.setCachedExpiration(duration.getDurationAmount());
238                 attrPolicyBean.setCachedTimeUnit(duration.getTimeUnit().name());
239 
240                 final IAttributeMerger merger = cc.getMergingStrategy();
241 
242                 if (merger != null) {
243                     if (merger instanceof NoncollidingAttributeAdder) {
244                         attrPolicyBean.setMergingStrategy(
245                                 RegisteredServiceAttributeReleasePolicyEditBean.AttributeMergerTypes.ADD.toString());
246                     } else if (merger instanceof MultivaluedAttributeMerger) {
247                         attrPolicyBean.setMergingStrategy(
248                                 RegisteredServiceAttributeReleasePolicyEditBean.AttributeMergerTypes.MULTIVALUED.toString());
249                     } else if (merger instanceof ReplacingAttributeAdder) {
250                         attrPolicyBean.setMergingStrategy(
251                                 RegisteredServiceAttributeReleasePolicyEditBean.AttributeMergerTypes.REPLACE.toString());
252                     }
253                 }
254             }
255             final RegisteredServiceAttributeReleasePolicyStrategyEditBean sBean = attrPolicyBean.getAttrPolicy();
256             if (attrPolicy instanceof ReturnAllAttributeReleasePolicy) {
257                 sBean.setType(AbstractRegisteredServiceAttributeReleasePolicyStrategyBean.Types.ALL.toString());
258             } else if (attrPolicy instanceof ReturnAllowedAttributeReleasePolicy) {
259                 final ReturnAllowedAttributeReleasePolicy attrPolicyAllowed = (ReturnAllowedAttributeReleasePolicy) attrPolicy;
260                 sBean.setType(AbstractRegisteredServiceAttributeReleasePolicyStrategyBean.Types.ALLOWED.toString());
261                 sBean.setAttributes(attrPolicyAllowed.getAllowedAttributes());
262             } else if (attrPolicy instanceof ReturnMappedAttributeReleasePolicy) {
263                 final ReturnMappedAttributeReleasePolicy attrPolicyAllowed = (ReturnMappedAttributeReleasePolicy) attrPolicy;
264                 sBean.setType(AbstractRegisteredServiceAttributeReleasePolicyStrategyBean.Types.MAPPED.toString());
265                 sBean.setAttributes(attrPolicyAllowed.getAllowedAttributes());
266             }
267         }
268     }

270     /**
271      * Configure proxy policy.
272      *
273      * @param svc the svc
274      * @param bean the bean
275      */
276     private static void configureProxyPolicy(final RegisteredService svc, final ServiceData bean) {
277         final RegisteredServiceProxyPolicy policy = svc.getProxyPolicy();
278         final RegisteredServiceProxyPolicyBean cBean = bean.getProxyPolicy();
279         if (policy == null || policy instanceof RefuseRegisteredServiceProxyPolicy) {
280             cBean.setType(RegisteredServiceProxyPolicyBean.Types.REFUSE.toString());
281         } else if (policy instanceof RegexMatchingRegisteredServiceProxyPolicy) {
282             final RegexMatchingRegisteredServiceProxyPolicy regex =
283                     (RegexMatchingRegisteredServiceProxyPolicy) policy;
284             cBean.setType(RegisteredServiceProxyPolicyBean.Types.REGEX.toString());
285             cBean.setValue(regex.getPattern().toString());
286         }
287     }

289     /**
290      * Configure username attribute provider.
291      *
292      * @param provider the provider
293      * @param uBean the u bean
294      */
295     private static void configureUsernameAttributeProvider(final RegisteredServiceUsernameAttributeProvider provider,
296                                                            final RegisteredServiceUsernameAttributeProviderEditBean uBean) {
297         if (provider instanceof DefaultRegisteredServiceUsernameProvider) {
298             uBean.setType(RegisteredServiceUsernameAttributeProviderEditBean.Types.DEFAULT.toString());
299         } else if (provider instanceof AnonymousRegisteredServiceUsernameAttributeProvider) {
300             final AnonymousRegisteredServiceUsernameAttributeProvider anonymous =
301                     (AnonymousRegisteredServiceUsernameAttributeProvider) provider;
302             uBean.setType(RegisteredServiceUsernameAttributeProviderEditBean.Types.ANONYMOUS.toString());
303             final PersistentIdGenerator generator = anonymous.getPersistentIdGenerator();
304             if (generator instanceof ShibbolethCompatiblePersistentIdGenerator) {
305                 final ShibbolethCompatiblePersistentIdGenerator sh =
306                         (ShibbolethCompatiblePersistentIdGenerator) generator;
307 
308                 String salt = new String(sh.getSalt(), Charset.defaultCharset());
309                 if (Base64.isBase64(salt)) {
310                     salt = new String(Base64.decodeBase64(salt));
311                 }
312 
313                 uBean.setValue(salt);
314             }
315         } else if (provider instanceof PrincipalAttributeRegisteredServiceUsernameProvider) {
316             final PrincipalAttributeRegisteredServiceUsernameProvider p =
317                     (PrincipalAttributeRegisteredServiceUsernameProvider) provider;
318             uBean.setType(RegisteredServiceUsernameAttributeProviderEditBean.Types.ATTRIBUTE.toString());
319             uBean.setValue(p.getUsernameAttribute());
320         }
321     }

498         /**
499          * To registered service.
500          *
501          * @param dao the dao
502          * @return the registered service
503          */
504         public RegisteredService toRegisteredService(final IPersonAttributeDao dao) {
505             try {
506                 final AbstractRegisteredService regSvc;
507 
508                 if (StringUtils.equalsIgnoreCase(this.type,
509                         RegisteredServiceTypeEditBean.OAUTH_CALLBACK_AUTHZ.toString())) {
510                     regSvc = new OAuthRegisteredCallbackAuthorizeService();
511                 } else if  (StringUtils.equalsIgnoreCase(this.type,
512                         RegisteredServiceTypeEditBean.OAUTH.toString())) {
513                     regSvc = new OAuthRegisteredService();
514 
515                     final RegisteredServiceOAuthTypeEditBean oauthBean = this.oauth;
516                     ((OAuthRegisteredService) regSvc).setClientId(oauthBean.getClientId());
517                     ((OAuthRegisteredService) regSvc).setClientSecret(oauthBean.getClientSecret());
518                     ((OAuthRegisteredService) regSvc).setBypassApprovalPrompt(oauthBean.isBypass());
519                 } else {
520                     regSvc = determineServiceTypeByPattern(this.serviceId);
521                 }
522 
523                 if (this.assignedId <=0) {
524                     regSvc.setId(RegisteredService.INITIAL_IDENTIFIER_VALUE);
525                 } else {
526                     regSvc.setId(this.assignedId);
527                 }
528 
529                 regSvc.setServiceId(this.serviceId);
530                 regSvc.setName(this.name);
531                 regSvc.setDescription(this.description);
532 
533                 if (StringUtils.isNotBlank(this.logoUrl)) {
534                     regSvc.setLogo(new URL(this.logoUrl));
535                 }
536                 regSvc.setTheme(this.theme);
537                 regSvc.setEvaluationOrder(this.evalOrder);
538                 regSvc.setRequiredHandlers(this.requiredHandlers);
539 
540                 if (StringUtils.equalsIgnoreCase(this.logoutType,
541                         RegisteredServiceLogoutTypeEditBean.BACK.toString())) {
542                     regSvc.setLogoutType(LogoutType.BACK_CHANNEL);
543                 } else if (StringUtils.equalsIgnoreCase(this.logoutType,
544                         RegisteredServiceLogoutTypeEditBean.FRONT.toString())) {
545                     regSvc.setLogoutType(LogoutType.FRONT_CHANNEL);
546                 } else {
547                     regSvc.setLogoutType(LogoutType.NONE);
548                 }
549 
550                 if (StringUtils.isNotBlank(this.logoutUrl)) {
551                     regSvc.setLogoutUrl(new URL(this.logoutUrl));
552                 }
553 
554                 final RegisteredServiceAccessStrategy accessStrategy = regSvc.getAccessStrategy();
555 
556                 ((DefaultRegisteredServiceAccessStrategy) accessStrategy)
557                         .setEnabled(this.supportAccess.isCasEnabled());
558                 ((DefaultRegisteredServiceAccessStrategy) accessStrategy)
559                         .setSsoEnabled(this.supportAccess.isSsoEnabled());
560                 ((DefaultRegisteredServiceAccessStrategy) accessStrategy)
561                         .setRequireAllAttributes(this.supportAccess.isRequireAll());
562 
563                 final Map<String, Set<String>> requiredAttrs = this.supportAccess.getRequiredAttr();
564                 final Set<Map.Entry<String, Set<String>>> entries = requiredAttrs.entrySet();
565                 final Iterator<Map.Entry<String, Set<String>>> it = entries.iterator();
566                 while (it.hasNext()) {
567                     final Map.Entry<String, Set<String>> entry = it.next();
568                     if (entry.getValue().isEmpty()) {
569                         it.remove();
570                     }
571                 }
572                 ((DefaultRegisteredServiceAccessStrategy) accessStrategy)
573                         .setRequiredAttributes(requiredAttrs);
574 
575                 final String proxyType = this.proxyPolicy.getType();
576                 if (StringUtils.equalsIgnoreCase(proxyType,
577                         RegisteredServiceProxyPolicyBean.Types.REFUSE.toString())) {
578                     regSvc.setProxyPolicy(new RefuseRegisteredServiceProxyPolicy());
579                 } else if (StringUtils.equalsIgnoreCase(proxyType,
580                         RegisteredServiceProxyPolicyBean.Types.REGEX.toString())) {
581                     final String value = this.proxyPolicy.getValue();
582                     if (StringUtils.isNotBlank(value) && isValidRegex(value)) {
583                         regSvc.setProxyPolicy(new RegexMatchingRegisteredServiceProxyPolicy(value));
584                     } else {
585                         throw new IllegalArgumentException("Invalid regex pattern specified for proxy policy: " + value);
586                     }
587                 }
588 
589                 final String uidType = this.userAttrProvider.getType();
590                 if (StringUtils.equalsIgnoreCase(uidType,
591                         RegisteredServiceUsernameAttributeProviderEditBean.Types.DEFAULT.toString())) {
592                     regSvc.setUsernameAttributeProvider(new DefaultRegisteredServiceUsernameProvider());
593                 } else if (StringUtils.equalsIgnoreCase(uidType,
594                         RegisteredServiceUsernameAttributeProviderEditBean.Types.ANONYMOUS.toString())) {
595                     final String salt = this.userAttrProvider.getValue();
596                     if (StringUtils.isNotBlank(salt)) {
597 
598                         final ShibbolethCompatiblePersistentIdGenerator generator =
599                                 new ShibbolethCompatiblePersistentIdGenerator(salt);
600                         regSvc.setUsernameAttributeProvider(
601                                 new AnonymousRegisteredServiceUsernameAttributeProvider(generator));
602                     } else {
603                         throw new IllegalArgumentException("Invalid sale value for anonymous ids " + salt);
604                     }
605                 } else if (StringUtils.equalsIgnoreCase(uidType,
606                         RegisteredServiceUsernameAttributeProviderEditBean.Types.ATTRIBUTE.toString())) {
607                     final String attr = this.userAttrProvider.getValue();
608 
609                     if (StringUtils.isNotBlank(attr)) {
610                         regSvc.setUsernameAttributeProvider(
611                                 new PrincipalAttributeRegisteredServiceUsernameProvider(attr));
612                     } else {
613                         throw new IllegalArgumentException("Invalid attribute specified for username");
614                     }
615                 }
616 
617                 if (this.publicKey != null && this.publicKey.isValid()) {
618                     final RegisteredServicePublicKey publicKey = new RegisteredServicePublicKeyImpl(
619                             this.publicKey.getLocation(), this.publicKey.getAlgorithm());
620                     regSvc.setPublicKey(publicKey);
621                 }
622 
623                 final RegisteredServiceAttributeReleasePolicyStrategyEditBean policyBean =
624                         this.attrRelease.getAttrPolicy();
625                 final String policyType = policyBean.getType();
626 
627                 AbstractAttributeReleasePolicy policy = null;
628                 if (StringUtils.equalsIgnoreCase(policyType,
629                         AbstractRegisteredServiceAttributeReleasePolicyStrategyBean.Types.ALL.toString())) {
630                     policy = new ReturnAllAttributeReleasePolicy();
631                 } else if (StringUtils.equalsIgnoreCase(policyType,
632                         AbstractRegisteredServiceAttributeReleasePolicyStrategyBean.Types.ALLOWED.toString())) {
633                     policy = new ReturnAllowedAttributeReleasePolicy((List) policyBean.getAttributes());
634                 } else if (StringUtils.equalsIgnoreCase(policyType,
635                         AbstractRegisteredServiceAttributeReleasePolicyStrategyBean.Types.MAPPED.toString())) {
636                     policy = new ReturnMappedAttributeReleasePolicy((Map) policyBean.getAttributes());
637                 } else {
638                     policy = new ReturnAllowedAttributeReleasePolicy();
639                 }
640 
641                 final String filter = this.attrRelease.getAttrFilter();
642                 if (StringUtils.isNotBlank(filter)) {
643                     policy.setAttributeFilter(new RegisteredServiceRegexAttributeFilter(filter));
644                 }
645 
646                 policy.setAuthorizedToReleaseCredentialPassword(this.attrRelease.isReleasePassword());
647                 policy.setAuthorizedToReleaseProxyGrantingTicket(this.attrRelease.isReleaseTicket());
648 
649                 final String attrType = this.attrRelease.getAttrOption();
650                 if (StringUtils.equalsIgnoreCase(attrType,
651                         RegisteredServiceAttributeReleasePolicyEditBean.Types.CACHED.toString())) {
652                     policy.setPrincipalAttributesRepository(new CachingPrincipalAttributesRepository(dao,
653                             TimeUnit.valueOf(this.attrRelease.getCachedTimeUnit().toUpperCase()),
654                             this.attrRelease.getCachedExpiration()));
655                 } else if (StringUtils.equalsIgnoreCase(attrType,
656                         RegisteredServiceAttributeReleasePolicyEditBean.Types.DEFAULT.toString())) {
657                     policy.setPrincipalAttributesRepository(new DefaultPrincipalAttributesRepository());
658                 }
659                 regSvc.setAttributeReleasePolicy(policy);
660 
661                 return regSvc;
662             } catch (final Exception e) {
663                 throw new RuntimeException(e);
664             }
665         }
666 
667         /**
668          * Determine service type by pattern.
669          *
670          * @param serviceId the service id
671          * @return the abstract registered service
672          */
673         private AbstractRegisteredService determineServiceTypeByPattern(final String serviceId) {
674             try {
675                 Pattern.compile(serviceId);
676                 LOGGER.debug("Service id {} is a valid regex.", serviceId);
677                 return new RegexRegisteredService();
678             } catch (final PatternSyntaxException exception) {
679                 LOGGER.debug("Service id {} is not a valid regex. Checking ant patterns...", serviceId);
680                 if (new AntPathMatcher().isPattern(serviceId)) {
681                     LOGGER.debug("Service id {} is a valid ant pattern.", serviceId);
682                     return new RegisteredServiceImpl();
683                 }
684             }
685             throw new RuntimeException("Service id " + serviceId + " cannot be resolve to a service type");
686         }

688         /**
689          * Determine service type by pattern.
690          *
691          * @param pattern the pattern
692          * @return the abstract registered service
693          */
694         private boolean isValidRegex(final String pattern) {
695             try {
696                 Pattern.compile(serviceId);
697                 LOGGER.debug("Pattern is a valid regex.", pattern);
698                 return true;
699             } catch (final PatternSyntaxException exception) {
700                 return false;
701             }
702         }
