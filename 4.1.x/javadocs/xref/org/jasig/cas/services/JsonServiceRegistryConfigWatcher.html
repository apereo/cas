20  package org.jasig.cas.services;
21  
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  import java.io.File;
26  import java.io.IOException;
27  import java.nio.file.FileSystems;
28  import java.nio.file.Path;
29  import java.nio.file.WatchEvent;
30  import java.nio.file.WatchKey;
31  import java.nio.file.WatchService;
32  import java.util.Arrays;
33  import java.util.concurrent.atomic.AtomicBoolean;
34  import java.util.concurrent.locks.Lock;
35  import java.util.concurrent.locks.ReadWriteLock;
36  import java.util.concurrent.locks.ReentrantReadWriteLock;
37  
38  import static java.nio.file.StandardWatchEventKinds.*;
39  
40  /**
41   * This is {@link JsonServiceRegistryConfigWatcher} that watches the json config directory
42   * for changes and promptly attempts to reload the CAS service registry configuration.
43   *
44   * @author Misagh Moayyed
45   * @since 4.1.0
46   */
47  class JsonServiceRegistryConfigWatcher implements Runnable {
48      private static final Logger LOGGER = LoggerFactory.getLogger(JsonServiceRegistryConfigWatcher.class);
49  
50      private final AtomicBoolean running = new AtomicBoolean(false);
51      private final ReadWriteLock lock = new ReentrantReadWriteLock();
52      private final Lock readLock = this.lock.readLock();
53  
54      private final WatchService watcher;
55  
56      private final JsonServiceRegistryDao serviceRegistryDao;
57  
58      /**
59       * Instantiates a new Json service registry config watcher.
60       *
61       * @param serviceRegistryDao the registry to callback
62       */
63      public JsonServiceRegistryConfigWatcher(final JsonServiceRegistryDao serviceRegistryDao) {
64          try {
65              this.serviceRegistryDao = serviceRegistryDao;
66              this.watcher = FileSystems.getDefault().newWatchService();
67              final WatchEvent.Kind[] kinds = (WatchEvent.Kind[])
68                      Arrays.asList(ENTRY_CREATE, ENTRY_DELETE, ENTRY_MODIFY).toArray();
69              this.serviceRegistryDao.getServiceRegistryDirectory().register(this.watcher, kinds);
70          } catch (final IOException e) {
71              throw new RuntimeException(e);
72          }
73      }
74  
75      @Override
76      public void run() {
77          if (running.compareAndSet(false, true)) {
78              while (running.get()) {
79                  // wait for key to be signaled
80                  WatchKey key = null;
81                  try {
82                      key = watcher.take();
83                      handleEvent(key);
84                  } catch (final InterruptedException e) {
85                      return;
86                  } finally {
87                      /*
88                          Reset the key -- this step is critical to receive
89                          further watch events. If the key is no longer valid, the directory
90                          is inaccessible so exit the loop.
91                       */
92                      final boolean valid = (key != null && key.reset());
93                      if (!valid) {
94                          LOGGER.warn("Directory key is no longer valid. Quitting watcher service");
95                          break;
96                      }
97                  }
98              }
99          }
100 
101     }
102 
103     /**
104      * Handle event.
105      *
106      * @param key the key
107      */
108     private void handleEvent(final WatchKey key) {
109         this.readLock.lock();
110         try {
111             for (final WatchEvent<?> event : key.pollEvents()) {
112                 if (event.count() <= 1) {
113                     final WatchEvent.Kind kind = event.kind();
114 
115                     //The filename is the context of the event.
116                     final WatchEvent<Path> ev = (WatchEvent<Path>) event;
117                     final Path filename = ev.context();
118 
119                     final Path parent = (Path) key.watchable();
120                     final Path fullPath = parent.resolve(filename);
121                     final File file = fullPath.toFile();
122 
123                     LOGGER.trace("Detected event [{}] on file [{}]. Loading change...", kind, file);
124                     if (kind.name().equals(ENTRY_CREATE.name()) && file.exists()) {
125                         handleCreateEvent(file);
126                     } else if (kind.name().equals(ENTRY_DELETE.name())) {
127                         handleDeleteEvent();
128                     } else if (kind.name().equals(ENTRY_MODIFY.name()) && file.exists()) {
129                         handleModifyEvent(file);
130                     }
131                 }
132 
133             }
134         } finally {
135             this.readLock.unlock();
136         }
137     }
138 
139     /**
140      * Handle modify event.
141      *
142      * @param file the file
143      */
144     private void handleModifyEvent(final File file) {
145     /*
146         load the entry and save it back to the map
147         without any warnings on duplicate ids.
148      */
149         final RegisteredService newService = this.serviceRegistryDao.loadRegisteredServiceFromFile(file);
150         if (newService == null) {
151             LOGGER.warn("New service definition could not be loaded from [{}]", file.getAbsolutePath());
152         } else {
153             final RegisteredService oldService = this.serviceRegistryDao.findServiceById(newService.getId());
154 
155             if (!newService.equals(oldService)) {
156                 this.serviceRegistryDao.updateRegisteredService(newService);
157                 this.serviceRegistryDao.refreshServicesManager();
158             } else {
159                 LOGGER.debug("Service [{}] loaded from [{}] is idential to the existing entry. "
160                             + "Services manager will not reload", newService.getId(),
161                         file.getName());
162             }
163         }
164     }
165 
166     /**
167      * Handle delete event.
168      */
169     private void handleDeleteEvent() {
170         this.serviceRegistryDao.load();
171         this.serviceRegistryDao.refreshServicesManager();
172     }
173 
174     /**
175      * Handle create event.
176      *
177      * @param file the file
178      */
179     private void handleCreateEvent(final File file) {
180         //load the entry and add it to the map
181         final RegisteredService service = this.serviceRegistryDao.loadRegisteredServiceFromFile(file);
182         if (service == null) {
183             LOGGER.warn("No service definition was loaded from [{}]", file);
184             return;
185         }
186         if (this.serviceRegistryDao.findServiceById(service.getId()) != null) {
187             LOGGER.warn("Found a service definition [{}] with a duplicate id [{}] in [{}]. "
188                             + "This will overwrite previous service definitions and is likely a "
189                             + "configuration problem. Make sure all services have a unique id and try again.",
190                     service.getServiceId(), service.getId(), file.getAbsolutePath());
191 
192         }
193         this.serviceRegistryDao.updateRegisteredService(service);
194         this.serviceRegistryDao.refreshServicesManager();
195     }
196 }
