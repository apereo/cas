23  import org.apache.commons.lang3.builder.ToStringBuilder;
24  import org.jasig.cas.support.spnego.util.ReverseDNSRunnable;
25  import org.jasig.cas.web.support.WebUtils;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  import org.springframework.webflow.action.AbstractAction;
29  import org.springframework.webflow.execution.Event;
30  import org.springframework.webflow.execution.RequestContext;
31  
32  import javax.servlet.http.HttpServletRequest;
33  import javax.validation.constraints.NotNull;
34  import java.util.regex.Matcher;
35  import java.util.regex.Pattern;

37  /**
38   * Abstract class for defining a simple binary filter to determine whether a
39   * given client system should be prompted for SPNEGO / KRB / NTLM credentials.
40   *
41   * Envisioned implementations would include LDAP and DNS based determinations,
42   * but of course others may have value as well for local architectures.
43   *
44   * @author Sean Baker sean.baker@usuhs.edu
45   * @author Misagh Moayyed
46   * @since 4.1
47   */
48  public class BaseSpnegoKnownClientSystemsFilterAction extends AbstractAction {
49      private static final int DEFAULT_TIMEOUT = 2000;
50  
51      /** Logger instance. **/
52      protected final Logger logger = LoggerFactory.getLogger(this.getClass());
53  
54      /** Pattern of ip addresses to check. **/
55      private Pattern ipsToCheckPattern;
56  
57      /** Alternative remote host attribute. **/
58      private String alternativeRemoteHostAttribute;
59  
60      /** Timeout for DNS Requests. **/
61      private long timeout = DEFAULT_TIMEOUT;
62      
63      /**
64       * Instantiates a new Base.
65       */
66      public BaseSpnegoKnownClientSystemsFilterAction() {}
67  
68      /**
69       * Instantiates a new Base.
70       *
71       * @param ipsToCheckPattern the ips to check pattern
72       */
73      public BaseSpnegoKnownClientSystemsFilterAction(final String ipsToCheckPattern) {
74          setIpsToCheckPattern(ipsToCheckPattern);
75      }
76  
77      /**
78       * Instantiates a new Base.
79       *
80       * @param ipsToCheckPattern the ips to check pattern
81       * @param alternativeRemoteHostAttribute the alternative remote host attribute
82       */
83      public BaseSpnegoKnownClientSystemsFilterAction(final String ipsToCheckPattern,
84                                                      final String alternativeRemoteHostAttribute) {
85          setIpsToCheckPattern(ipsToCheckPattern);
86          this.alternativeRemoteHostAttribute = alternativeRemoteHostAttribute;
87      }
88  
89      /**
90       * Instantiates a new Base.
91       *
92       * @param ipsToCheckPattern the ips to check pattern
93       * @param alternativeRemoteHostAttribute the alternative remote host attribute
94       */
95      public BaseSpnegoKnownClientSystemsFilterAction(final Pattern ipsToCheckPattern,
96                                                      final String alternativeRemoteHostAttribute) {
97          this.ipsToCheckPattern = ipsToCheckPattern;
98          this.alternativeRemoteHostAttribute = alternativeRemoteHostAttribute;
99      }
100 
101     /**
102      * {@inheritDoc}
103      * Gets the remote ip from the request, and invokes spnego if it isn't filtered.
104      *
105      * @param context the request context
106      * @return {@link #yes()} if spnego should be invoked and ip isn't filtered,
107      * {@link #no()} otherwise.
108      */
109     @Override
110     protected final Event doExecute(final RequestContext context) {
111         final String remoteIp = getRemoteIp(context);
112         logger.debug("Current user IP {}", remoteIp);
113         return shouldDoSpnego(remoteIp) ? yes() : no();
114     }
115 
116     /**
117      * Default implementation -- simply check the IP filter.
118      * @param remoteIp the remote ip
119      * @return true boolean
120      */
121     protected boolean shouldDoSpnego(final String remoteIp) {
122         return ipPatternCanBeChecked(remoteIp) && ipPatternMatches(remoteIp);
123     }
124 
125     /**
126      * Base class definition for whether the IP should be checked or not; overridable.
127      * @param remoteIp the remote ip
128      * @return whether or not the IP can / should be matched against the pattern
129      */
130     protected boolean ipPatternCanBeChecked(final String remoteIp) {
131         return (this.ipsToCheckPattern != null && StringUtils.isNotBlank(remoteIp));
132     }
133 
134     /**
135      * Simple pattern match to determine whether an IP should be checked.
136      * Could stand to be extended to support "real" IP addresses and patterns, but
137      * for the local / first implementation regex made more sense.
138      * @param remoteIp the remote ip
139      * @return whether the remote ip received should be queried
140      */
141     protected boolean ipPatternMatches(final String remoteIp) {
142         final Matcher matcher = this.ipsToCheckPattern.matcher(remoteIp);
143         if (matcher.find()) {
144             logger.debug("Remote IP address {} should be checked based on the defined pattern {}",
145                     remoteIp, this.ipsToCheckPattern.pattern());
146             return true;
147         }
148         logger.debug("No pattern or remote IP defined, or pattern does not match remote IP [{}]",
149                 remoteIp);
150         return false;
151     }
152 
153     /**
154      * Pulls the remote IP from the current HttpServletRequest, or grabs the value
155      * for the specified alternative attribute (say, for proxied requests).  Falls
156      * back to providing the "normal" remote address if no value can be retrieved
157      * from the specified alternative header value.
158      * @param context the context
159      * @return the remote ip
160      */
161     private String getRemoteIp(@NotNull final RequestContext context) {
162         final HttpServletRequest request = WebUtils.getHttpServletRequest(context);
163         String userAddress = request.getRemoteAddr();
164         logger.debug("Remote Address = {}", userAddress);
165 
166         if (StringUtils.isNotBlank(this.alternativeRemoteHostAttribute)) {
167 
168             userAddress = request.getHeader(this.alternativeRemoteHostAttribute);
169             logger.debug("Header Attribute [{}] = [{}]", this.alternativeRemoteHostAttribute, userAddress);
170 
171             if (StringUtils.isBlank(userAddress)) {
172                 userAddress = request.getRemoteAddr();
173                 logger.warn("No value could be retrieved from the header [{}]. Falling back to [{}].",
174                         this.alternativeRemoteHostAttribute, userAddress);
175             }
176         }
177         return userAddress;
178     }
179 
180     /**
181      * Alternative header to be used for retrieving the remote system IP address.
182      * @param alternativeRemoteHostAttribute the alternative remote host attribute
183      */
184     public final void setAlternativeRemoteHostAttribute(@NotNull final String alternativeRemoteHostAttribute) {
185         this.alternativeRemoteHostAttribute = alternativeRemoteHostAttribute;
186     }
187 
188     /**
189      * Regular expression string to define IPs which should be considered.
190      * @param ipsToCheckPattern the ips to check as a regex pattern
191      */
192     public final void setIpsToCheckPattern(@NotNull final String ipsToCheckPattern) {
193         this.ipsToCheckPattern = Pattern.compile(ipsToCheckPattern);
194     }
195 
196 
197     @Override
198     public final String toString() {
199         return new ToStringBuilder(this)
200                 .append("ipsToCheckPattern", this.ipsToCheckPattern)
201                 .append("alternativeRemoteHostAttribute", this.alternativeRemoteHostAttribute)
202                 .append("timeout", this.timeout)
203                 .toString();
204     }
205 
206     /**
207      * Set timeout (ms) for DNS requests; valuable for heterogeneous environments employing
208      * fall-through authentication mechanisms.
209      * @param timeout # of milliseconds to wait for a DNS request to return
210      */
211     public final void setTimeout(final long timeout) {
212         this.timeout = timeout;
213     }
214 
215     /**
216      * Convenience method to perform a reverse DNS lookup. Threads the request
217      * through a custom Runnable class in order to prevent inordinately long
218      * user waits while performing reverse lookup.
219      * @param remoteIp the remote ip
220      * @return the remote host name
221      */
222     protected String getRemoteHostName(final String remoteIp) {
223         final ReverseDNSRunnable revDNS = new ReverseDNSRunnable(remoteIp);
224 
225         final Thread t = new Thread(revDNS);
226         t.start();
227 
228         try {
229             t.join(this.timeout);
230         } catch (final InterruptedException e) {
231             logger.debug("Threaded lookup failed.  Defaulting to IP {}.", remoteIp, e);
232         }
233 
234         final String remoteHostName = revDNS.get();
235         logger.debug("Found remote host name {}.", remoteHostName);
236 
237         return StringUtils.isNotEmpty(remoteHostName) ? remoteHostName : remoteIp;
238     }
