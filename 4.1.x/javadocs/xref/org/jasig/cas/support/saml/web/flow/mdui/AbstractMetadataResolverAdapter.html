49  /**
50   * This is {@link AbstractMetadataResolverAdapter} that encapsulates
51   * commons between static and dynamic resolvers.
52   *
53   * @author Misagh Moayyed
54   * @since 4.1.0
55   */
56  public abstract class AbstractMetadataResolverAdapter implements MetadataResolverAdapter {
57      /** Logger instance. */
58      protected final Logger logger = LoggerFactory.getLogger(this.getClass());
59  
60      /** Metadata resources along with filters to perform validation. */
61      @NotNull
62      protected final Map<Resource, MetadataFilterChain> metadataResources;
63  
64      /** Whether the metadata resolver should require valid metadata. Default is true. */
65      protected boolean requireValidMetadata = true;
66  
67      /** The openSAML config bean. **/
68      @Autowired
69      @NotNull
70      protected OpenSamlConfigBean configBean;
71  
72      private ChainingMetadataResolver metadataResolver;
73  
74      private final Object lock = new Object();
75  
76      /**
77       * Instantiates a new abstract metadata resolver adapter.
78       */
79      public AbstractMetadataResolverAdapter() {
80          this.metadataResources = new HashMap<>();
81      }
82  
83      /**
84       * Instantiates a new static metadata resolver adapter.
85       *
86       * @param metadataResources the metadata resources
87       */
88      public AbstractMetadataResolverAdapter(final Map<Resource, MetadataFilterChain> metadataResources) {
89          this.metadataResources = metadataResources;
90      }
91  
92      public void setRequireValidMetadata(final boolean requireValidMetadata) {
93          this.requireValidMetadata = requireValidMetadata;
94      }
95  
96      /**
97       * Retrieve the remote source's input stream to parse data.
98       * @param resource the resource
99       * @param entityId the entity id
100      * @return the input stream
101      * @throws IOException if stream cannot be read
102      */
103     protected InputStream getResourceInputStream(final Resource resource, final String entityId) throws IOException {
104         logger.debug("Locating metadata resource from input stream.");
105         if (!resource.exists() || !resource.isReadable()) {
106             throw new FileNotFoundException("Resource does not exist or is unreadable");
107         }
108         return resource.getInputStream();
109     }
110 
111     @Override
112     public EntityDescriptor getEntityDescriptorForEntityId(final String entityId) {
113         try {
114             final CriteriaSet criterions = new CriteriaSet(new EntityIdCriterion(entityId));
115             if (this.metadataResolver != null) {
116                 return metadataResolver.resolveSingle(criterions);
117             }
118         } catch (final Exception ex) {
119             throw new RuntimeException(ex.getMessage(), ex);
120         }
121         return null;
122 
123     }
124 
125     /**
126      * Build metadata resolver aggregate.
127      *
128      */
129     protected final void buildMetadataResolverAggregate() {
130         buildMetadataResolverAggregate(null);
131     }
132 
133     /**
134      * Build metadata resolver aggregate. Loops through metadata resources
135      * and attempts to resolve the metadata.
136      * @param entityId the entity id
137      */
138     public final void buildMetadataResolverAggregate(final String entityId) {
139         try {
140             final Set<Map.Entry<Resource, MetadataFilterChain>> entries = metadataResources.entrySet();
141             for (final Map.Entry<Resource, MetadataFilterChain> entry : entries) {
142                 final Resource resource = entry.getKey();
143                 logger.debug("Loading [{}]", resource.getFilename());
144                 loadMetadataFromResource(entry.getValue(), resource, entityId);
145             }
146         } catch (final Exception ex) {
147             throw new RuntimeException(ex.getMessage(), ex);
148         }
149     }
150 
151     /**
152      * Load metadata from resource.
153      *
154      * @param metadataFilter the metadata filter
155      * @param resource the resource
156      * @param entityId the entity id
157      */
158     private void loadMetadataFromResource(final MetadataFilter metadataFilter,
159                                           final Resource resource, final String entityId) {
160 
161         try (final InputStream in = getResourceInputStream(resource, entityId)) {
162             logger.debug("Parsing [{}]", resource.getFilename());
163             final Document document = this.configBean.getParserPool().parse(in);
164 
165             final List<MetadataResolver> resolvers = buildSingleMetadataResolver(metadataFilter, resource, document);
166             this.metadataResolver = new ChainingMetadataResolver();
167             synchronized (this.lock) {
168                 this.metadataResolver.setId(ChainingMetadataResolver.class.getCanonicalName());
169                 this.metadataResolver.setResolvers(resolvers);
170                 logger.info("Collected metadata from [{}] resource(s). Initializing aggregate resolver...",
171                         resolvers.size());
172                 this.metadataResolver.initialize();
173                 logger.info("Metadata aggregate initialized successfully.", resolvers.size());
174             }
175         } catch (final Exception e) {
176             logger.warn("Could not retrieve input stream from resource. Moving on...", e);
177         }
178     }
179 
180     /**
181      * Build single metadata resolver.
182      *
183      * @param metadataFilterChain the metadata filters chained together
184      * @param resource the resource
185      * @param document the xml document to parse
186      * @return list of resolved metadata from resources.
187      * @throws IOException the iO exception
188      */
189     private List<MetadataResolver> buildSingleMetadataResolver(final MetadataFilter metadataFilterChain,
190                                              final Resource resource, final Document document) throws IOException {
191         final List<MetadataResolver> resolvers = new ArrayList<>();
192         final Element metadataRoot = document.getDocumentElement();
193         final DOMMetadataResolver metadataProvider = new DOMMetadataResolver(metadataRoot);
194 
195         metadataProvider.setParserPool(this.configBean.getParserPool());
196         metadataProvider.setFailFastInitialization(true);
197         metadataProvider.setRequireValidMetadata(this.requireValidMetadata);
198         metadataProvider.setId(metadataProvider.getClass().getCanonicalName());
199         if (metadataFilterChain != null) {
200             metadataProvider.setMetadataFilter(metadataFilterChain);
201         }
202         logger.debug("Initializing metadata resolver for [{}]", resource.getURL());
203 
204         try {
205             metadataProvider.initialize();
206         } catch (final ComponentInitializationException ex) {
207             logger.warn("Could not initialize metadata resolver. Resource will be ignored", ex);
208         }
209         resolvers.add(metadataProvider);
210         return resolvers;
211     }
