57   * This is the response builder for Saml1 Protocol.

59   * @author Misagh Moayyed mmoayyed@unicon.net

62  public final class Saml10ObjectBuilder extends AbstractSamlObjectBuilder {
63  
64      private static final String CONFIRMATION_METHOD = "urn:oasis:names:tc:SAML:1.0:cm:artifact";
65  
66      /**
67       * Create a new SAML response object.
68       * @param id the id
69       * @param issueInstant the issue instant
70       * @param recipient the recipient
71       * @param service the service
72       * @return the response
73       */
74      public Response newResponse(final String id, final DateTime issueInstant,
75                                           final String recipient, final WebApplicationService service) {
76  
77          final Response samlResponse = newSamlObject(Response.class);
78          samlResponse.setID(id);
79          samlResponse.setIssueInstant(issueInstant);
80          samlResponse.setVersion(SAMLVersion.VERSION_11);
81          samlResponse.setInResponseTo(recipient);
82          if (service instanceof SamlService) {
83              final SamlService samlService = (SamlService) service;
84  
85              final String requestId = samlService.getRequestID();
86              if (StringUtils.isNotBlank(requestId)) {
87                  samlResponse.setInResponseTo(requestId);
88              }
89          }
90          return samlResponse;
91      }
92  
93      /**
94       * Create a new SAML1 response object.
95       *
96       * @param authnStatement the authn statement
97       * @param issuer the issuer
98       * @param issuedAt the issued at
99       * @param id the id
100      * @return the assertion
101      */
102     public Assertion newAssertion(final AuthenticationStatement authnStatement, final String issuer,
103                                         final DateTime issuedAt, final String id) {
104         final Assertion assertion = newSamlObject(Assertion.class);
105 
106         assertion.setID(id);
107         assertion.setIssueInstant(issuedAt);
108         assertion.setIssuer(issuer);
109         assertion.getAuthenticationStatements().add(authnStatement);
110         return assertion;
111     }
112 
113     /**
114      * New conditions element.
115      *
116      * @param issuedAt the issued at
117      * @param audienceUri the service id
118      * @param issueLength the issue length
119      * @return the conditions
120      */
121     public Conditions newConditions(final DateTime issuedAt, final String audienceUri, final long issueLength) {
122         final Conditions conditions = newSamlObject(Conditions.class);
123         conditions.setNotBefore(issuedAt);
124         conditions.setNotOnOrAfter(issuedAt.plus(issueLength));
125         final AudienceRestrictionCondition audienceRestriction = newSamlObject(AudienceRestrictionCondition.class);
126         final Audience audience = newSamlObject(Audience.class);
127         audience.setUri(audienceUri);
128         audienceRestriction.getAudiences().add(audience);
129         conditions.getAudienceRestrictionConditions().add(audienceRestriction);
130         return conditions;
131     }
132 
133     /**
134      * Create a new SAML status object.
135      *
136      * @param codeValue the code value
137      * @param statusMessage the status message
138      * @return the status
139      */
140     public Status newStatus(final QName codeValue, final String statusMessage) {
141         final Status status = newSamlObject(Status.class);
142         final StatusCode code = newSamlObject(StatusCode.class);
143         code.setValue(codeValue);
144         status.setStatusCode(code);
145         if (statusMessage != null) {
146             final StatusMessage message = newSamlObject(StatusMessage.class);
147             message.setMessage(statusMessage);
148             status.setStatusMessage(message);
149         }
150         return status;
151     }

153     /**
154      * New authentication statement.
155      *
156      * @param authenticationDate the authentication date
157      * @param authenticationMethod the authentication method
158      * @param subjectId the subject id
159      * @return the authentication statement
160      */
161     public AuthenticationStatement newAuthenticationStatement(final Date authenticationDate,
162                                                               final String authenticationMethod,
163                                                               final String subjectId) {
164 
165         final AuthenticationStatement authnStatement = newSamlObject(AuthenticationStatement.class);
166         authnStatement.setAuthenticationInstant(new DateTime(authenticationDate));
167         authnStatement.setAuthenticationMethod(
168                 authenticationMethod != null
169                         ? authenticationMethod
170                         : SamlAuthenticationMetaDataPopulator.AUTHN_METHOD_UNSPECIFIED);
171         authnStatement.setSubject(newSubject(subjectId));
172         return authnStatement;
173     }

175     /**
176      * New subject element that uses the confirmation method
177      * {@link #CONFIRMATION_METHOD}.
178      *
179      * @param identifier the identifier
180      * @return the subject
181      */
182     public Subject newSubject(final String identifier) {
183         return newSubject(identifier, CONFIRMATION_METHOD);
184     }
185 
186     /**
187      * New subject element with given confirmation method.
188      *
189      * @param identifier the identifier
190      * @param confirmationMethod the confirmation method
191      * @return the subject
192      */
193     public Subject newSubject(final String identifier, final String confirmationMethod) {
194         final SubjectConfirmation confirmation = newSamlObject(SubjectConfirmation.class);
195         final ConfirmationMethod method = newSamlObject(ConfirmationMethod.class);
196         method.setConfirmationMethod(confirmationMethod);
197         confirmation.getConfirmationMethods().add(method);
198         final NameIdentifier nameIdentifier = newSamlObject(NameIdentifier.class);
199         nameIdentifier.setNameIdentifier(identifier);
200         final Subject subject = newSamlObject(Subject.class);
201         subject.setNameIdentifier(nameIdentifier);
202         subject.setSubjectConfirmation(confirmation);
203         return subject;
204     }

206     /**
207      * New attribute statement.
208      *
209      * @param subject the subject
210      * @param attributes the attributes
211      * @param attributeNamespace the attribute namespace
212      * @return the attribute statement
213      */
214     public AttributeStatement newAttributeStatement(final Subject subject,
215                                                     final Map<String, Object> attributes,
216                                                     final String attributeNamespace) {
217 
218         final AttributeStatement attrStatement = newSamlObject(AttributeStatement.class);
219         attrStatement.setSubject(subject);
220         for (final Map.Entry<String, Object> e : attributes.entrySet()) {
221             if (e.getValue() instanceof Collection<?> && ((Collection<?>) e.getValue()).isEmpty()) {
222                 //don't add the attribute, it causes a org.opensaml.MalformedException
223                 logger.info("Skipping attribute {} because it does not have any values.", e.getKey());
224                 continue;
225             }
226             final Attribute attribute = newSamlObject(Attribute.class);
227             attribute.setAttributeName(e.getKey());
228             attribute.setAttributeNamespace(attributeNamespace);
229             if (e.getValue() instanceof Collection<?>) {
230                 final Collection<?> c = (Collection<?>) e.getValue();
231                 for (final Object value : c) {
232                     attribute.getAttributeValues().add(newAttributeValue(value, AttributeValue.DEFAULT_ELEMENT_NAME));
233                 }
234             } else {
235                 attribute.getAttributeValues().add(newAttributeValue(e.getValue(), AttributeValue.DEFAULT_ELEMENT_NAME));
236             }
237             attrStatement.getAttributes().add(attribute);
238         }
239 
240         return attrStatement;
241     }

243     /**
244      * Encode response and pass it onto the outbound transport.
245      * Uses {@link CasHTTPSOAP11Encoder} to handle encoding.
246      *
247      * @param httpResponse the http response
248      * @param httpRequest the http request
249      * @param samlMessage the saml response
250      * @throws Exception the exception in case encoding fails.
251      */
252     public void encodeSamlResponse(final HttpServletResponse httpResponse,
253                                    final HttpServletRequest httpRequest,
254                                    final Response samlMessage) throws Exception {
255 
256         final HTTPSOAP11Encoder encoder = new CasHTTPSOAP11Encoder();
257         final MessageContext<SAMLObject> context = new MessageContext();
258         context.setMessage(samlMessage);
259         encoder.setHttpServletResponse(httpResponse);
260         encoder.setMessageContext(context);
261         encoder.initialize();
262         encoder.prepareContext();
263         encoder.encode();
264     }
