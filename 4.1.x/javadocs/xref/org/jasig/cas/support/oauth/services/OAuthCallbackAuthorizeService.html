19  package org.jasig.cas.support.oauth.services;
20  
21  import org.apache.commons.lang3.builder.EqualsBuilder;
22  import org.apache.commons.lang3.builder.HashCodeBuilder;
23  import org.jasig.cas.services.RegexRegisteredService;
24  import org.jasig.cas.support.oauth.OAuthConstants;
25  
26  /**
27   * An extension of the {@link RegexRegisteredService} that attempts to enforce the
28   * correct url syntax for the OAuth callback authorize url. The url must end with
29   * {@link OAuthConstants#CALLBACK_AUTHORIZE_URL}.
30   * @author Misagh Moayyed
31   * @since 4.0.0
32   */
33  public final class OAuthCallbackAuthorizeService extends RegexRegisteredService {
34  
35      private static final long serialVersionUID = 1365893114273585648L;
36  
37      /**
38       * {@inheritDoc}.
39       * @throws IllegalArgumentException if the received url does not end with
40       *         {@link OAuthConstants#CALLBACK_AUTHORIZE_URL}
41       */
42      @Override
43      public void setServiceId(final String id) {
44          if (!id.endsWith(OAuthConstants.CALLBACK_AUTHORIZE_URL)) {
45              final String msg = String.format("OAuth callback authorize service id must end with [%s]",
46                      OAuthConstants.CALLBACK_AUTHORIZE_URL);
47              throw new IllegalArgumentException(msg);
48          }
49          super.setServiceId(id);
50      }
51  
52      @Override
53      public boolean equals(final Object obj) {
54          if (obj == null) {
55              return false;
56          }
57          if (obj == this) {
58              return true;
59          }
60          if (obj.getClass() != getClass()) {
61              return false;
62          }
63          final OAuthCallbackAuthorizeService rhs = (OAuthCallbackAuthorizeService) obj;
64          return new EqualsBuilder()
65                  .appendSuper(super.equals(obj))
66                  .isEquals();
67      }
68  
69      @Override
70      public int hashCode() {
71          return new HashCodeBuilder(13, 137)
72                  .appendSuper(super.hashCode())
73                  .toHashCode();
74      }
75  }
