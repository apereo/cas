19  package org.jasig.cas.web.flow;
20  
21  import org.jasig.cas.CentralAuthenticationService;
22  import org.jasig.cas.authentication.AuthenticationException;
23  import org.jasig.cas.authentication.Credential;
24  import org.jasig.cas.authentication.principal.Service;
25  import org.jasig.cas.ticket.TicketException;
26  import org.jasig.cas.ticket.InvalidTicketException;
27  import org.jasig.cas.ticket.ServiceTicket;
28  import org.jasig.cas.web.support.WebUtils;
29  import org.springframework.util.StringUtils;
30  import org.springframework.webflow.action.AbstractAction;
31  import org.springframework.webflow.execution.Event;
32  import org.springframework.webflow.execution.RequestContext;
33  
34  import javax.validation.constraints.NotNull;
35  
36  /**
37   * Action to generate a service ticket for a given Ticket Granting Ticket and
38   * Service.
39   *
40   * @author Scott Battaglia
41   * @since 3.0.0.4
42   */
43  public final class GenerateServiceTicketAction extends AbstractAction {
44  
45      /** Instance of CentralAuthenticationService. */
46      @NotNull
47      private CentralAuthenticationService centralAuthenticationService;
48  
49      @Override
50      protected Event doExecute(final RequestContext context) {
51          final Service service = WebUtils.getService(context);
52          final String ticketGrantingTicket = WebUtils.getTicketGrantingTicketId(context);
53  
54          try {
55              final Credential credential = WebUtils.getCredential(context);
56  
57              final ServiceTicket serviceTicketId = this.centralAuthenticationService
58                  .grantServiceTicket(ticketGrantingTicket, service, credential);
59              WebUtils.putServiceTicketInRequestScope(context, serviceTicketId);
60              return success();
61          } catch (final AuthenticationException e) {
62              logger.error("Could not verify credentials to grant service ticket", e);
63          } catch (final TicketException e) {
64              if (e instanceof InvalidTicketException) {
65                  this.centralAuthenticationService.destroyTicketGrantingTicket(ticketGrantingTicket);
66              }
67              if (isGatewayPresent(context)) {
68                  return result("gateway");
69              }
70          }
71  
72          return error();
73      }
74  
75      public void setCentralAuthenticationService(
76          final CentralAuthenticationService centralAuthenticationService) {
77          this.centralAuthenticationService = centralAuthenticationService;
78      }
79  
80      /**
81       * Checks if <code>gateway</code> is present in the request params.
82       *
83       * @param context the context
84       * @return true, if gateway present
85       */
86      protected boolean isGatewayPresent(final RequestContext context) {
87          return StringUtils.hasText(context.getExternalContext()
88              .getRequestParameterMap().get("gateway"));
89      }
90  }
