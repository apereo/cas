21  /**
22   * Stategy interface for pluggable authentication security policies.
23   *
24   * @author Marvin S. Addison
25   * @since 4.0.0
26   */
27  public interface AuthenticationPolicy {
28      /**
29       * Determines whether an authentication event isSatisfiedBy arbitrary security policy.
30       *
31       * @param authentication Authentication event to examine for compliance with security policy.
32       *
33       * @return True if authentication isSatisfiedBy security policy, false otherwise.
34       */
35      boolean isSatisfiedBy(Authentication authentication);
36  }
