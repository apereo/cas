19  package org.jasig.cas.authentication;
20  
21  import org.jasig.cas.authentication.principal.Principal;
22  import org.joda.time.DateTime;
23  import org.springframework.util.Assert;
24  
25  import java.util.ArrayList;
26  import java.util.Date;
27  import java.util.LinkedHashMap;
28  import java.util.List;
29  import java.util.Map;
30  
31  /**
32   * Constructs immutable {@link Authentication} objects using the builder pattern.
33   *
34   * @author Marvin S. Addison
35   * @since 4.0.0
36   */
37  public class DefaultAuthenticationBuilder implements AuthenticationBuilder {
38      /** Authenticated principal. */
39      private Principal principal;
40  
41      /** Credential metadata. */
42      private final List<CredentialMetaData> credentials = new ArrayList<>();
43  
44      /** Authentication metadata attributes. */
45      private final Map<String, Object> attributes = new LinkedHashMap<>();
46  
47      /** Map of handler names to authentication successes. */
48      private final Map<String, HandlerResult> successes = new LinkedHashMap<>();
49  
50      /** Map of handler names to authentication failures. */
51      private final Map<String, Class<? extends Exception>> failures = new LinkedHashMap<>();
52  
53      /** Authentication date. */
54      private DateTime authenticationDate;
55  
56      /**
57       * Creates a new instance using the current date for the authentication date.
58       */
59      public DefaultAuthenticationBuilder() {
60          authenticationDate = new DateTime();
61      }
62  
63      /**
64       * Creates a new instance using the current date for the authentication date and the given
65       * principal for the authenticated principal.
66       *
67       * @param p Authenticated principal.
68       */
69      public DefaultAuthenticationBuilder(final Principal p) {
70          this();
71          this.principal = p;
72      }
73  
74      /**
75       * Gets the authentication date.
76       *
77       * @return Authentication date.
78       */
79      public DateTime getAuthenticationDate() {
80          return this.authenticationDate == null ? null : new DateTime(this.authenticationDate);
81      }
82  
83      /**
84       * Sets the authentication date and returns this instance.
85       *
86       * @param d Authentication date.
87       *
88       * @return This builder instance.
89       */
90      @Override
91      public AuthenticationBuilder setAuthenticationDate(final Date d) {
92          this.authenticationDate = new DateTime(d);
93          return this;
94      }
95  
96      /**
97       * Gets the authenticated principal.
98       *
99       * @return Principal.
100      */
101     @Override
102     public Principal getPrincipal() {
103         return this.principal;
104     }
105 
106     /**
107      * Sets the principal returns this instance.
108      *
109      * @param p Authenticated principal.
110      *
111      * @return This builder instance.
112      */
113     @Override
114     public AuthenticationBuilder setPrincipal(final Principal p) {
115         this.principal = p;
116         return this;
117     }
118 
119     /**
120      * Gets the list of credentials that were attempted to be authenticated.
121      *
122      * @return Non-null list of credentials that attempted authentication.
123      */
124     public List<CredentialMetaData> getCredentials() {
125         return this.credentials;
126     }
127 
128     /**
129      * Sets the list of metadata about credentials presented for authentication.
130      *
131      * @param credentials Non-null list of credential metadata.
132      *
133      * @return This builder instance.
134      */
135     public AuthenticationBuilder setCredentials(final List<CredentialMetaData> credentials) {
136         Assert.notNull(credentials, "Credential cannot be null");
137         this.credentials.clear();
138         this.credentials.addAll(credentials);
139         return this;
140     }
141 
142     /**
143      * Adds metadata about a credential presented for authentication.
144      *
145      * @param credential Credential metadata.
146      *
147      * @return This builder instance.
148      */
149     @Override
150     public AuthenticationBuilder addCredential(final CredentialMetaData credential) {
151         this.credentials.add(credential);
152         return this;
153     }
154 
155     /**
156      * Gets the authentication attribute map.
157      *
158      * @return Non-null authentication attribute map.
159      */
160     public Map<String, Object> getAttributes() {
161         return this.attributes;
162     }
163 
164     /**
165      * Sets the authentication metadata attributes.
166      *
167      * @param attributes Non-null map of authentication metadata attributes.
168      *
169      * @return This builder instance.
170      */
171     @Override
172     public AuthenticationBuilder setAttributes(final Map<String, Object> attributes) {
173         Assert.notNull(attributes, "Attributes cannot be null");
174         this.attributes.clear();
175         for (final Map.Entry<String, Object> entry : attributes.entrySet()) {
176             this.attributes.put(entry.getKey(), entry.getValue());
177         }
178         return this;
179     }
180 
181     /**
182      * Adds an authentication metadata attribute key-value pair.
183      *
184      * @param key Authentication attribute key.
185      * @param value Authentication attribute value.
186      *
187      * @return This builder instance.
188      */
189     @Override
190     public AuthenticationBuilder addAttribute(final String key, final Object value) {
191         this.attributes.put(key, value);
192         return this;
193     }
194 
195     /**
196      * Gets the authentication success map.
197      *
198      * @return Non-null map of handler names to successful handler authentication results.
199      */
200     @Override
201     public Map<String, HandlerResult> getSuccesses() {
202         return this.successes;
203     }
204 
205     /**
206      * Sets the authentication handler success map.
207      *
208      * @param successes Non-null map of handler names to successful handler authentication results.
209      *
210      * @return This builder instance.
211      */
212     public AuthenticationBuilder setSuccesses(final Map<String, HandlerResult> successes) {
213         Assert.notNull(successes, "Successes cannot be null");
214         this.successes.clear();
215         for (final Map.Entry<String, HandlerResult> entry : successes.entrySet()) {
216             this.successes.put(entry.getKey(), entry.getValue());
217         }
218         return this;
219     }
220 
221     /**
222      * Adds an authentication success to the map of handler names to successful authentication handler results.
223      *
224      * @param key Authentication handler name.
225      * @param value Successful authentication handler result produced by handler of given name.
226      *
227      * @return This builder instance.
228      */
229     @Override
230     public AuthenticationBuilder addSuccess(final String key, final HandlerResult value) {
231         this.successes.put(key, value);
232         return this;
233     }
234 
235     /**
236      * Gets the authentication failure map.
237      *
238      * @return Non-null authentication failure map.
239      */
240     @Override
241     public Map<String, Class<? extends Exception>> getFailures() {
242         return this.failures;
243     }
244 
245     /**
246      * Sets the authentication handler failure map.
247      *
248      * @param failures Non-null map of handler name to authentication failures.
249      *
250      * @return This builder instance.
251      */
252     public AuthenticationBuilder setFailures(final Map<String, Class<? extends Exception>> failures) {
253         Assert.notNull(failures, "Failures cannot be null");
254         this.failures.clear();
255         for (final Map.Entry<String, Class<? extends Exception>> entry : failures.entrySet()) {
256             this.failures.put(entry.getKey(), entry.getValue());
257         }
258         return this;
259     }
260 
261     /**
262      * Adds an authentication failure to the map of handler names to the authentication handler failures.
263      *
264      * @param key Authentication handler name.
265      * @param value Exception raised on handler failure to authenticate credential.
266      *
267      * @return This builder instance.
268      */
269     @Override
270     public AuthenticationBuilder addFailure(final String key, final Class<? extends Exception> value) {
271         this.failures.put(key, value);
272         return this;
273     }
274 
275     /**
276      * Creates an immutable authentication instance from builder data.
277      *
278      * @return Immutable authentication.
279      */
280     @Override
281     public Authentication build() {
282         return new ImmutableAuthentication(
283                 this.authenticationDate,
284                 this.credentials,
285                 this.principal,
286                 this.attributes,
287                 this.successes,
288                 this.failures);
289     }
290 
291     /**
292      * Creates a new builder initialized with data from the given authentication source.
293      *
294      * @param source Authentication source.
295      *
296      * @return New builder instance initialized with all fields in the given authentication source.
297      */
298     public static AuthenticationBuilder newInstance(final Authentication source) {
299         final DefaultAuthenticationBuilder builder = new DefaultAuthenticationBuilder(source.getPrincipal());
300         builder.setAuthenticationDate(source.getAuthenticationDate());
301         builder.setCredentials(source.getCredentials());
302         builder.setSuccesses(source.getSuccesses());
303         builder.setFailures(source.getFailures());
304         builder.setAttributes(source.getAttributes());
305         return builder;
306     }
307 }
