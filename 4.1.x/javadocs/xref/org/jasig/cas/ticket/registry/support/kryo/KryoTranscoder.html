84  /**
85   * {@link net.spy.memcached.MemcachedClient} transcoder implementation based on Kryo fast serialization framework
86   * suited for efficient serialization of tickets.
87   *
88   * @author Marvin S. Addison
89   * @since 3.0.0
90   */
91  @SuppressWarnings("rawtypes")
92  public class KryoTranscoder implements Transcoder<Object> {
93  
94      /** Kryo serializer. */
95      private final Kryo kryo = new KryoReflectionFactorySupport();
96  
97      /** Logging instance. */
98      private final Logger logger = LoggerFactory.getLogger(getClass());
99  
100     /** Map of class to serializer that handles it. */
101     private Map<Class<?>, Serializer> serializerMap;
102 
103     /**
104      * Creates a Kryo-based transcoder.
105      */
106     public KryoTranscoder() {
107     }
108 
109     /**
110      * @deprecated
111      * Creates a Kryo-based transcoder.
112      *
113      * @param initialBufferSize Initial size for buffer holding encoded object data.
114      */
115     @Deprecated
116     public KryoTranscoder(final int initialBufferSize) {
117         logger.warn("It's no longer necessary to define the initialBufferSize. Use the empty constructor.");
118     }
119 
120     /**
121      * Sets a map of additional types that should be regisetered with Kryo,
122      * for example GoogleAccountsService and OpenIdService.
123      *
124      * @param map Map of class to the serializer instance that handles it.
125      */
126     public void setSerializerMap(final Map<Class<?>, Serializer> map) {
127         this.serializerMap = map;
128     }
129 
130     /**
131      * Initialize and register classes with kryo.
132      */
133     public void initialize() {
134         // Register types we know about and do not require external configuration
135         kryo.register(ArrayList.class);
136         kryo.register(BasicCredentialMetaData.class);
137         kryo.register(Class.class, new DefaultSerializers.ClassSerializer());
138         kryo.register(Date.class, new DefaultSerializers.DateSerializer());
139         kryo.register(HardTimeoutExpirationPolicy.class);
140         kryo.register(HashMap.class);
141         kryo.register(LinkedHashMap.class);
142         kryo.register(HashSet.class);
143         kryo.register(DefaultHandlerResult.class);
144         kryo.register(ImmutableAuthentication.class);
145         kryo.register(MultiTimeUseOrTimeoutExpirationPolicy.class);
146         kryo.register(NeverExpiresExpirationPolicy.class);
147         kryo.register(RememberMeDelegatingExpirationPolicy.class);
148         kryo.register(ServiceTicketImpl.class);
149         kryo.register(SimpleWebApplicationServiceImpl.class, new SimpleWebApplicationServiceSerializer());
150         kryo.register(ThrottledUseAndTimeoutExpirationPolicy.class);
151         kryo.register(TicketGrantingTicketExpirationPolicy.class);
152         kryo.register(TicketGrantingTicketImpl.class);
153         kryo.register(TimeoutExpirationPolicy.class);
154         kryo.register(UsernamePasswordCredential.class);
155         kryo.register(SimplePrincipal.class);
156         kryo.register(URL.class, new URLSerializer());
157         kryo.register(URI.class, new URISerializer());
158         kryo.register(Pattern.class, new RegexSerializer());
159         kryo.register(UUID.class, new UUIDSerializer());
160         kryo.register(EnumMap.class, new EnumMapSerializer());
161         kryo.register(EnumSet.class, new EnumSetSerializer());
162 
163         // we add these ones for tests only
164         kryo.register(RegisteredServiceImpl.class, new RegisteredServiceSerializer());
165         kryo.register(RegexRegisteredService.class, new RegisteredServiceSerializer());
166 
167         // new serializers to manage Joda dates and immutable collections
168         kryo.register(DateTime.class, new JodaDateTimeSerializer());
169         kryo.register(CasDelegatingLogger.class, new DefaultSerializers.VoidSerializer());
170 
171         // from the kryo-serializers library (https://github.com/magro/kryo-serializers)
172         UnmodifiableCollectionsSerializer.registerSerializers(kryo);
173         ImmutableListSerializer.registerSerializers(kryo);
174         ImmutableSetSerializer.registerSerializers(kryo);
175         ImmutableMapSerializer.registerSerializers(kryo);
176         ImmutableMultimapSerializer.registerSerializers(kryo);
177         
178         kryo.register(Collections.EMPTY_LIST.getClass(), new CollectionsEmptyListSerializer());
179         kryo.register(Collections.EMPTY_MAP.getClass(), new CollectionsEmptyMapSerializer());
180         kryo.register(Collections.EMPTY_SET.getClass(), new CollectionsEmptySetSerializer());
181 
182         // Register other types
183         if (serializerMap != null) {
184             for (final Map.Entry<Class<?>, Serializer> clazz : serializerMap.entrySet()) {
185                 kryo.register(clazz.getKey(), clazz.getValue());
186             }
187         }
188 
189         // don't reinit the registered classes after every write or read
190         kryo.setAutoReset(false);
191         // don't replace objects by references
192         kryo.setReferences(false);
193         // Catchall for any classes not explicitly registered
194         kryo.setRegistrationRequired(false);
195     }
196 
197     /**
198      * Asynchronous decoding is not supported.
199      *
200      * @param d Data to decode.
201      * @return False.
202      */
203     public boolean asyncDecode(final CachedData d) {
204         return false;
205     }
206 
207     @Override
208     public CachedData encode(final Object obj) {
209         final ByteArrayOutputStream byteStream = new ByteArrayOutputStream();
210         try (Output output = new Output(byteStream)) {
211             kryo.writeClassAndObject(output, obj);
212             output.flush();
213             final byte[] bytes = byteStream.toByteArray();
214             return new CachedData(0, bytes, bytes.length);
215         }
216     }
217 
218     @Override
219     public Object decode(final CachedData d) {
220         final byte[] bytes = d.getData();
221         try (Input input = new Input(new ByteArrayInputStream(bytes))) {
222             final Object obj =  kryo.readClassAndObject(input);
223             return obj;
224         }
225     }
226 
227     /**
228      * Maximum size of encoded data supported by this transcoder.
229      *
230      * @return  <code>net.spy.memcached.CachedData#MAX_SIZE</code>.
231      */
232     public int getMaxSize() {
233         return CachedData.MAX_SIZE;
234     }
235 
236     /**
237      * Gets the kryo object that provides encoding and decoding services for this instance.
238      *
239      * @return Underlying Kryo instance.
240      */
241     public Kryo getKryo() {
242         return kryo;
243     }
244 }
