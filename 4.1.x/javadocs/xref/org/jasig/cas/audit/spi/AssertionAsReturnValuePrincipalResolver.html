19  package org.jasig.cas.audit.spi;
20  
21  import org.aspectj.lang.JoinPoint;
22  import org.jasig.cas.authentication.Authentication;
23  import org.jasig.cas.validation.Assertion;
24  import org.jasig.inspektr.common.spi.PrincipalResolver;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  /**
29   * Inspektr's principal resolver that first checks if the return value of the target joinpoint is instance
30   * of {@link Assertion} and then retrieves the principal from there.
31   * <p/>
32   * If that is not the case, it delegates to the wrapped {@link TicketOrCredentialPrincipalResolver} which might return
33   * <code>org.apereo.inspektr.common.spi.PrincipalResolver.UNKNOWN_USER</code> if it's unable to resolve principal id.
34   *
35   * @author Dmitriy Kopylenko
36   * @since 4.1.9
37   */
38  public class AssertionAsReturnValuePrincipalResolver implements PrincipalResolver {
39  
40      private static final Logger LOGGER = LoggerFactory.getLogger(AssertionAsReturnValuePrincipalResolver.class);
41      
42      private TicketOrCredentialPrincipalResolver delegate;
43  
44      /**
45       * Instantiates a new Assertion as return value principal resolver.
46       *
47       * @param delegate the delegate
48       */
49      public AssertionAsReturnValuePrincipalResolver(final TicketOrCredentialPrincipalResolver delegate) {
50          this.delegate = delegate;
51      }
52  
53      @Override
54      public String resolveFrom(final JoinPoint auditTarget, final Object returnValue) {
55          LOGGER.debug("Trying to see if target's return value is instance of [Assertion]...");
56          if (returnValue instanceof Assertion) {
57              LOGGER.debug("Assertion instance is found in the return value. Resolving principal id from associated Authentication...");
58              final Authentication authentication = Assertion.class.cast(returnValue).getPrimaryAuthentication();
59              return this.delegate.getPrincipalIdProvider().getPrincipalIdFrom(authentication);
60          }
61          LOGGER.debug("Resolving principal from the delegate principal resolver: [{}]...", this.delegate);
62          final String principalIdResolvedByDelagate = this.delegate.resolveFrom(auditTarget, returnValue);
63          if (UNKNOWN_USER.equals(principalIdResolvedByDelagate)) {
64              LOGGER.debug("Meaningful principal id could not be resolved by [{}]. Returning [{}]...", this, principalIdResolvedByDelagate);
65          }
66          return principalIdResolvedByDelagate;
67      }
68  
69      @Override
70      public String resolveFrom(final JoinPoint auditTarget, final Exception exception) {
71          return this.delegate.resolveFrom(auditTarget, exception);
72      }
73  
74      @Override
75      public String resolve() {
76          return this.delegate.resolve();
77      }
78  }
