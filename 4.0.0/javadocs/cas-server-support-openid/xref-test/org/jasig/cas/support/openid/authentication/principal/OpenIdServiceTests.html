52      @Before
53      public void setUp() throws Exception {
54          request.addParameter("openid.identity", "http://openid.ja-sig.org/battags");
55          request.addParameter("openid.return_to", "http://www.ja-sig.org/?service=fa");
56          request.addParameter("openid.mode", "checkid_setup");
57          sharedAssociations = mock(ServerAssociationStore.class);
58          manager = new ServerManager();
59          manager.setOPEndpointUrl("https://localshot:8443/cas/login");
60          manager.setEnforceRpId(false);
61          manager.setSharedAssociations(sharedAssociations);
62          context = mock(ApplicationContext.class);
63          ApplicationContextProvider contextProvider = new ApplicationContextProvider();
64          contextProvider.setApplicationContext(context);
65          cas = mock(CentralAuthenticationService.class);
66      }
67  
68      @Test
69      public void testGetResponse() {
70          openIdService = OpenIdService.createServiceFrom(request);
71          when(context.getBean("serverManager")).thenReturn(manager);
72          when(context.getBean("centralAuthenticationService")).thenReturn(cas);
73          final Response response = this.openIdService.getResponse("test");
74          try {
75              verify(cas, never()).validateServiceTicket("test", openIdService);
76          } catch (final Exception e) {
77              LOGGER.debug("Exception during verification of service ticket", e);
78          }
79          assertNotNull(response);
80  
81          assertEquals("test", response.getAttributes().get("openid.assoc_handle"));
82          assertEquals("http://www.ja-sig.org/?service=fa", response.getAttributes().get("openid.return_to"));
83          assertEquals("http://openid.ja-sig.org/battags", response.getAttributes().get("openid.identity"));
84  
85          final Response response2 = this.openIdService.getResponse(null);
86          assertEquals("cancel", response2.getAttributes().get("openid.mode"));
87      }
