251     protected String constructHashedKey(final String key) {
252         if (key == null) {
253             return null;
254         }
255 
256         final MessageDigest messageDigest = getMessageDigest();
257         messageDigest.update(this.salt);
258         messageDigest.update(key.toLowerCase().getBytes());
259         final String hash = getFormattedText(messageDigest.digest());
260 
261         logger.debug(String.format("Generated hash of value [%s] for key [%s].", hash, key));
262         return hash;
263     }

265     protected String decrypt(final String value, final String hashedKey) {
266         if (value == null) {
267             return null;
268         }
269 
270         try {
271             final Cipher cipher = getCipherObject();
272             final byte[] ivCiphertext = decode(value.getBytes());
273             final int ivSize = byte2int(Arrays.copyOfRange(ivCiphertext, 0, INTEGER_LEN));
274             final byte[] ivValue = Arrays.copyOfRange(ivCiphertext, INTEGER_LEN, (INTEGER_LEN + ivSize));
275             final byte[] ciphertext = Arrays.copyOfRange(ivCiphertext, INTEGER_LEN + ivSize, ivCiphertext.length);
276             final IvParameterSpec ivSpec = new IvParameterSpec(ivValue);
277             
278             cipher.init(Cipher.DECRYPT_MODE, this.key, ivSpec);
279 
280             final byte[] plaintext = cipher.doFinal(ciphertext);
281 
282             return new String(plaintext);
283         } catch (final Exception e) {
284             throw new RuntimeException(e);
285         }
286     }

311     protected String encrypt(final String value, final String hashedKey) {
312         if (value == null) {
313             return null;
314         }
315 
316         try {
317             final Cipher cipher = getCipherObject();
318             final byte[] ivValue = generateIV(this.ivSize);
319             final IvParameterSpec ivSpec = new IvParameterSpec(ivValue);
320 
321             cipher.init(Cipher.ENCRYPT_MODE, this.key, ivSpec);
322             
323             final byte[] ciphertext = cipher.doFinal(value.getBytes());
324             final byte[] ivCiphertext = new byte[INTEGER_LEN + this.ivSize + ciphertext.length];
325 
326             System.arraycopy(int2byte(this.ivSize), 0, ivCiphertext, 0, INTEGER_LEN);
327             System.arraycopy(ivValue, 0, ivCiphertext, INTEGER_LEN, this.ivSize);
328             System.arraycopy(ciphertext, 0, ivCiphertext, INTEGER_LEN + this.ivSize, ciphertext.length);
329 
330             return new String(encode(ivCiphertext));
331         } catch(final Exception e) {
332             throw new RuntimeException(e);
333         }
334     }

352     /**
353      * Tries to clone the {@link MessageDigest} that was created during construction. If the clone fails
354      * that is remembered and from that point on new {@link MessageDigest} instances will be created on
355      * every call.
356      * <p>
357      * Adopted from the Spring EhCache Annotations project.
358      *
359      * @return Generates a {@link MessageDigest} to use
360      */
361     protected MessageDigest getMessageDigest() {
362         if (this.cloneNotSupported) {
363             final String algorithm = this.messageDigest.getAlgorithm();
364             try {
365                 return MessageDigest.getInstance(algorithm);
366             } catch (final NoSuchAlgorithmException e) {
367                 throw new IllegalStateException("MessageDigest algorithm '" + algorithm + "' was supported when "
368                         + this.getClass().getSimpleName()
369                         + " was created but is not now. This should not be possible.", e);
370             }
371         }
372 
373         try {
374             return (MessageDigest) this.messageDigest.clone();
375         } catch (final CloneNotSupportedException e) {
376             this.cloneNotSupported = true;
377             final String msg = String.format("Could not clone MessageDigest using algorithm '%s'. "
378                         + "MessageDigest.getInstance will be used from now on which will be much more expensive.",
379                         this.messageDigest.getAlgorithm());
380             logger.warn(msg, e);
381             return this.getMessageDigest();
382         }
383     }

385     /**
386      * Takes the raw bytes from the digest and formats them correct.
387      *
388      * @param bytes the raw bytes from the digest.
389      * @return the formatted bytes.
390      */
391     private static String getFormattedText(final byte[] bytes) {
392         final StringBuilder buf = new StringBuilder(bytes.length * 2);
393 
394         for (byte b : bytes) {
395             buf.append(HEX_DIGITS[b >> 4 & 0x0f]);
396             buf.append(HEX_DIGITS[b & 0x0f]);
397         }
398         return buf.toString();
399     }

405     private static Key getSecretKey(final String secretKeyAlgorithm, final String secretKey,
406             final String salt) throws Exception {
407 
408         SecretKeyFactory factory = SecretKeyFactory.getInstance(SECRET_KEY_FACTORY_ALGORITHM);
409         KeySpec spec = new PBEKeySpec(secretKey.toCharArray(), char2byte(salt), 65536, 128);
410         SecretKey tmp = factory.generateSecret(spec);
411         SecretKey secret = new SecretKeySpec(tmp.getEncoded(), secretKeyAlgorithm);
412 
413         return secret;
414     }
