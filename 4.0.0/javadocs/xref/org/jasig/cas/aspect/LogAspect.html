19  package org.jasig.cas.aspect;
20  
21  import org.aspectj.lang.JoinPoint;
22  import org.aspectj.lang.ProceedingJoinPoint;
23  import org.aspectj.lang.annotation.Around;
24  import org.aspectj.lang.annotation.Aspect;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import java.util.Arrays;
29  
30  
31  /**
32   * @since 3.3.6
33   */
34  @Aspect
35  public class LogAspect {
36  
37      @Around("(execution (public * org.jasig.cas..*.*(..))) && !(execution( * org.jasig.cas..*.set*(..)))")
38      public Object traceMethod(final ProceedingJoinPoint proceedingJoinPoint) throws Throwable {
39          Object returnVal = null;
40          final Logger logger = this.getLog(proceedingJoinPoint);
41          final String methodName = proceedingJoinPoint.getSignature().getName();
42  
43          try {
44              if (logger.isTraceEnabled()) {
45                  final Object[] args = proceedingJoinPoint.getArgs();
46                  final String arguments;
47                  if (args == null || args.length == 0) {
48                      arguments = "";
49                  } else {
50                      arguments = Arrays.deepToString(args);
51                  }
52                  logger.trace("Entering method [{}] with arguments [{}]", methodName, arguments);
53              }
54              returnVal = proceedingJoinPoint.proceed();
55              return returnVal;
56          } finally {
57              logger.trace("Leaving method [{}] with return value [{}].", methodName,
58                          (returnVal != null ? returnVal.toString() : "null"));
59          }
60      }
61  
62      protected Logger getLog(final JoinPoint joinPoint) {
63          final Object target = joinPoint.getTarget();
64  
65          if (target != null) {
66              return LoggerFactory.getLogger(target.getClass());
67          }
68  
69          return LoggerFactory.getLogger(getClass());
70      }
71  
72  }
