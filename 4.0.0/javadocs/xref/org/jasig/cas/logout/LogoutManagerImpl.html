40  /**
41   * This logout manager handles the Single Log Out process.
42   *
43   * @author Jerome Leleu
44   * @since 4.0.0
45   */
46  public final class LogoutManagerImpl implements LogoutManager {
47  
48      /** The logger. */
49      private static final Logger LOGGER = LoggerFactory.getLogger(LogoutManagerImpl.class);
50  
51      /** ASCII character set. */
52      private static final Charset ASCII = Charset.forName("ASCII");
53  
54      /** The services manager. */
55      @NotNull
56      private final ServicesManager servicesManager;
57  
58      /** An HTTP client. */
59      @NotNull
60      private final HttpClient httpClient;
61  
62      @NotNull
63      private final LogoutMessageCreator logoutMessageBuilder;
64      
65      /** Whether single sign out is disabled or not. */
66      private boolean disableSingleSignOut = false;
67  
68      /**
69       * Build the logout manager.
70       * @param servicesManager the services manager.
71       * @param httpClient an HTTP client.
72       * @param logoutMessageBuilder the builder to construct logout messages.
73       */
74      public LogoutManagerImpl(final ServicesManager servicesManager, final HttpClient httpClient,
75                               final LogoutMessageCreator logoutMessageBuilder) {
76          this.servicesManager = servicesManager;
77          this.httpClient = httpClient;
78          this.logoutMessageBuilder = logoutMessageBuilder;
79      }
80  
81      /**
82       * Perform a back channel logout for a given ticket granting ticket and returns all the logout requests.
83       *
84       * @param ticket a given ticket granting ticket.
85       * @return all logout requests.
86       */
87      @Override
88      public List<LogoutRequest> performLogout(final TicketGrantingTicket ticket) {
89          final Map<String, Service> services;
90          // synchronize the retrieval of the services and their cleaning for the TGT
91          // to avoid concurrent logout mess ups
92          synchronized (ticket) {
93              services = ticket.getServices();
94              ticket.removeAllServices();
95          }
96          ticket.markTicketExpired();
97  
98          final List<LogoutRequest> logoutRequests = new ArrayList<LogoutRequest>();
99          // if SLO is not disabled
100         if (!disableSingleSignOut) {
101             // through all services
102             for (final String ticketId : services.keySet()) {
103                 final Service service = services.get(ticketId);
104                 // it's a SingleLogoutService, else ignore
105                 if (service instanceof SingleLogoutService) {
106                     final SingleLogoutService singleLogoutService = (SingleLogoutService) service;
107                     // the logout has not performed already
108                     if (!singleLogoutService.isLoggedOutAlready()) {
109                         final LogoutRequest logoutRequest = new LogoutRequest(ticketId, singleLogoutService);
110                         // always add the logout request
111                         logoutRequests.add(logoutRequest);
112                         final RegisteredService registeredService = servicesManager.findServiceBy(service);
113                         // the service is no more defined, or the logout type is not defined or is back channel
114                         if (registeredService == null || registeredService.getLogoutType() == null
115                                 || registeredService.getLogoutType() == LogoutType.BACK_CHANNEL) {
116                             // perform back channel logout
117                             if (performBackChannelLogout(logoutRequest)) {
118                                 logoutRequest.setStatus(LogoutRequestStatus.SUCCESS);
119                             } else {
120                                 logoutRequest.setStatus(LogoutRequestStatus.FAILURE);
121                                 LOGGER.warn("Logout message not sent to [{}]; Continuing processing...",
122                                         singleLogoutService.getId());
123                             }
124                         }
125                     }
126                 }
127             }
128         }
129 
130         return logoutRequests;
131     }
132 
133     /**
134      * Log out of a service through back channel.
135      *
136      * @param request the logout request.
137      * @return if the logout has been performed.
138      */
139     private boolean performBackChannelLogout(final LogoutRequest request) {
140         final String logoutRequest = this.logoutMessageBuilder.create(request);
141         request.getService().setLoggedOutAlready(true);
142 
143         LOGGER.debug("Sending logout request for: [{}]", request.getService().getId());
144         return this.httpClient.sendMessageToEndPoint(request.getService().getOriginalUrl(), logoutRequest, true);
145     }
146 
147     /**
148      * Create a logout message for front channel logout.
149      *
150      * @param logoutRequest the logout request.
151      * @return a front SAML logout message.
152      */
153     public String createFrontChannelLogoutMessage(final LogoutRequest logoutRequest) {
154         final String logoutMessage = this.logoutMessageBuilder.create(logoutRequest);
155         final Deflater deflater = new Deflater();
156         deflater.setInput(logoutMessage.getBytes(ASCII));
157         deflater.finish();
158         final byte[] buffer = new byte[logoutMessage.length()];
159         final int resultSize = deflater.deflate(buffer);
160         final byte[] output = new byte[resultSize];
161         System.arraycopy(buffer, 0, output, 0, resultSize);
162         return Base64.encodeBase64String(output);
163     }

165     /**
166      * Set if the logout is disabled.
167      *
168      * @param disableSingleSignOut if the logout is disabled.
169      */
170     public void setDisableSingleSignOut(final boolean disableSingleSignOut) {
171         this.disableSingleSignOut = disableSingleSignOut;
172     }
