32  /**
33   * Immutable authentication event whose attributes may not change after creation.
34   * This class is designed for serialization and is suitable for long-term storage.
35   *
36   * @author Dmitriy Kopylenko
37   * @author Scott Battaglia
38   * @author Marvin S. Addison
39   *
40   * @since 3.0
41   */
42  public final class ImmutableAuthentication implements Authentication, Serializable {
43  
44      /** UID for serializing. */
45      private static final long serialVersionUID = 3206127526058061391L;
46  
47      /** Authentication date stamp. */
48      private final long authenticatedDate;
49  
50      /** List of metadata about credentials presented at authentication. */
51      private final List<CredentialMetaData> credentials;
52  
53      /** Authenticated principal. */
54      private final Principal principal;
55  
56      /** Authentication metadata attributes. */
57      private final Map<String, Object> attributes;
58  
59      /** Map of handler name to handler authentication success event. */
60      private final Map<String, HandlerResult> successes;
61  
62      /** Map of handler name to handler authentication failure cause. */
63      private final Map<String, Class<? extends Exception>> failures;
64  
65      /** No-arg constructor for serialization support. */
66      private ImmutableAuthentication() {
67          this.authenticatedDate = 0;
68          this.credentials = null;
69          this.principal = null;
70          this.attributes = null;
71          this.successes = null;
72          this.failures = null;
73      }

75      /**
76       * Creates a new instance with the given data.
77       *
78       * @param date Non-null authentication date.
79       * @param credentials Non-null list of credential metadata containing at least one entry.
80       * @param principal Non-null authenticated principal.
81       * @param attributes Nullable map of authentication metadata.
82       * @param successes Non-null map of authentication successes containing at least one entry.
83       * @param failures Nullable map of authentication failures.
84       */
85      public ImmutableAuthentication(
86              final Date date,
87              final List<CredentialMetaData> credentials,
88              final Principal principal,
89              final Map<String, Object> attributes,
90              final Map<String, HandlerResult> successes,
91              final Map<String, Class<? extends Exception>> failures) {
92  
93          Assert.notNull(date, "Date cannot be null");
94          Assert.notNull(credentials, "Credential cannot be null");
95          Assert.notNull(principal, "Principal cannot be null");
96          Assert.notNull(successes, "Successes cannot be null");
97          Assert.notEmpty(credentials, "Credential cannot be empty");
98          Assert.notEmpty(successes, "Successes cannot be empty");
99  
100         this.authenticatedDate = date.getTime();
101         this.credentials = credentials;
102         this.principal = principal;
103         this.attributes = attributes.isEmpty() ? null : attributes;
104         this.successes = successes;
105         this.failures = failures.isEmpty() ? null : failures;
106     }

151         if (!(obj instanceof Authentication)) {

158         final EqualsBuilder builder = new EqualsBuilder();
159         builder.append(this.principal, other.getPrincipal());
160         builder.append(this.credentials, other.getCredentials());
161         builder.append(this.successes, other.getSuccesses());
162         builder.append(this.authenticatedDate, other.getAuthenticatedDate().getTime());
163         builder.append(wrap(this.attributes), other.getAttributes());

168     /**
169      * Wraps a possibly null map in an immutable wrapper.
170      *
171      * @param source Nullable map to wrap.
172      *
173      * @return {@link Collections#unmodifiableMap(java.util.Map)} if given map is not null, otherwise
174      * {@link java.util.Collections#emptyMap()}.
175      */
176     private static <K, V> Map<K, V> wrap(final Map<K, V> source) {
177         if (source != null) {
178             return Collections.unmodifiableMap(source);
179         }
180         return Collections.emptyMap();
181     }
