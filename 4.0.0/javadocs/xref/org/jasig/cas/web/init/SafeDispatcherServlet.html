19  package org.jasig.cas.web.init;
20  
21  import java.io.IOException;
22  
23  import javax.servlet.ServletConfig;
24  import javax.servlet.ServletContext;
25  import javax.servlet.ServletException;
26  import javax.servlet.ServletRequest;
27  import javax.servlet.ServletResponse;
28  import javax.servlet.http.HttpServlet;
29  
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  import org.springframework.context.ApplicationContextException;
33  import org.springframework.web.servlet.DispatcherServlet;
34  
35  /**
36   * This servlet wraps the Spring DispatchServlet, catching any exceptions it
37   * throws on init() to guarantee that servlet initialization succeeds. This
38   * allows our application context to succeed in initializing so that we can
39   * display a friendly "CAS is not available" page to the deployer (an
40   * appropriate use of the page in development) or to the end user (an
41   * appropriate use of the page in production). The error page associated with
42   * this deployment failure is configured in the web.xml via the standard error
43   * handling mechanism.
44   * <p>
45   * If the underlying Spring DispatcherServlet failed to init(), this
46   * SafeDispatcherServlet will throw an
47   * <code>org.springframework.context.ApplicationContextException</code> on
48   * <code>service()</code>.
49   * <p>
50   * The exception thrown by the underlying Spring DispatcherServlet init() and
51   * caught in the SafeDispatcherServlet init() is exposed as a Servlet Context
52   * attribute under the key "exceptionCaughtByServlet".
53   *
54   * @author Andrew Petro
55   * @see DispatcherServlet
56   */
57  public final class SafeDispatcherServlet extends HttpServlet {
58  
59      /** Unique Id for serialization. */
60      private static final long serialVersionUID = 1L;
61  
62      /** Key under which we will store the exception in the ServletContext. */
63      public static final String CAUGHT_THROWABLE_KEY = "exceptionCaughtByServlet";
64  
65      /** Instance of Commons Logging. */
66      private static final Logger LOGGER = LoggerFactory.getLogger(SafeDispatcherServlet.class);
67  
68      /** The actual DispatcherServlet to which we will delegate to. */
69      private DispatcherServlet delegate = new DispatcherServlet();
70  
71      /** Boolean to determine if the application deployed successfully. */
72      private boolean initSuccess = true;
73  
74      public void init(final ServletConfig config) {
75          try {
76              this.delegate.init(config);
77  
78          } catch (final Throwable t) {
79              // let the service method know initialization failed.
80              this.initSuccess = false;
81  
82              /*
83               * no matter what went wrong, our role is to capture this error and
84               * prevent it from blocking initialization of the servlet. logging
85               * overkill so that our deployer will find a record of this problem
86               * even if unfamiliar with Commons Logging and properly configuring
87               * it.
88               */
89  
90              final String message = "SafeDispatcherServlet: \n"
91                  + "The Spring DispatcherServlet we wrap threw on init.\n"
92                  + "But for our having caught this error, the servlet would not have initialized.";
93  
94              // logger it via Commons Logging
95              LOGGER.error(message, t);
96  
97              // logger it to the ServletContext
98              ServletContext context = config.getServletContext();
99              context.log(message, t);
100 
101             /*
102              * record the error so that the application has access to later
103              * display a proper error message based on the exception.
104              */
105             context.setAttribute(CAUGHT_THROWABLE_KEY, t);
106 
107         }
108     }
109 
110     /**
111      * {@inheritDoc}
112      * @throws ApplicationContextException if the DispatcherServlet does not
113      * initialize properly, but the servlet attempts to process a request.
114      */
115     @Override
116     public void service(final ServletRequest req, final ServletResponse resp)
117         throws ServletException, IOException {
118         /*
119          * Since our container calls only this method and not any of the other
120          * HttpServlet runtime methods, such as doDelete(), etc., delegating
121          * this method is sufficient to delegate all of the methods in the
122          * HttpServlet API.
123          */
124         if (this.initSuccess) {
125             this.delegate.service(req, resp);
126         } else {
127             throw new ApplicationContextException("Unable to initialize application context.");
128         }
129     }
130 }
