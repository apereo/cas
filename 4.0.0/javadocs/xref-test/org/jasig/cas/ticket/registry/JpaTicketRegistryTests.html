115     @Test
116     public void testTicketCreationAndDeletion() throws Exception {
117         final TicketGrantingTicket newTgt = newTGT();
118         addTicketInTransaction(newTgt);
119         final TicketGrantingTicket tgtFromDb = (TicketGrantingTicket) getTicketInTransaction(newTgt.getId());
120         assertNotNull(tgtFromDb);
121         assertEquals(newTgt.getId(), tgtFromDb.getId());
122         final ServiceTicket newSt = grantServiceTicketInTransaction(tgtFromDb);
123         final ServiceTicket stFromDb = (ServiceTicket) getTicketInTransaction(newSt.getId());
124         assertNotNull(stFromDb);
125         assertEquals(newSt.getId(), stFromDb.getId());
126         deleteTicketInTransaction(newTgt.getId());
127         assertNull(getTicketInTransaction(newTgt.getId()));
128         assertNull(getTicketInTransaction(newSt.getId()));
129     }

131     @Test
132     @IfProfileValue(name="cas.jpa.concurrent", value="true")
133     public void testConcurrentServiceTicketGeneration() throws Exception {
134         final TicketGrantingTicket newTgt = newTGT();
135         addTicketInTransaction(newTgt);
136         final ExecutorService executor = Executors.newFixedThreadPool(CONCURRENT_SIZE);
137         try {
138             final List<ServiceTicketGenerator> generators = new ArrayList<ServiceTicketGenerator>(CONCURRENT_SIZE);
139             for (int i = 0; i < CONCURRENT_SIZE; i++) {
140                 generators.add(new ServiceTicketGenerator(newTgt.getId()));
141             }
142             final List<Future<String>> results = executor.invokeAll(generators);
143             for (Future<String> result : results) {
144                 assertNotNull(result.get());
145             }
146         } catch (final Exception e) {
147             logger.debug("testConcurrentServiceTicketGeneration produced an error", e);
148             fail("testConcurrentServiceTicketGeneration failed.");
149         } finally {
150             executor.shutdownNow();
151         }
152     }

198     ServiceTicket grantServiceTicketInTransaction(final TicketGrantingTicket parent) {
199         return new TransactionTemplate(txManager).execute(new TransactionCallback<ServiceTicket>() {
200             public ServiceTicket doInTransaction(final TransactionStatus status) {
201                 final ServiceTicket st = newST(parent);
202                 jpaTicketRegistry.addTicket(st);
203                 return st;

208     class ServiceTicketGenerator implements Callable<String> {
209 
210         private String parentTgtId;
211 
212         public ServiceTicketGenerator(final String tgtId) {
213             parentTgtId = tgtId;
214         }
215 
216         /** {@inheritDoc} */
217         @Override
218         public String call() throws Exception {
219             return new TransactionTemplate(txManager).execute(new TransactionCallback<String>() {
220                 public String doInTransaction(final TransactionStatus status) {
221                     // Querying for the TGT prior to updating it as done in
222                     // CentralAuthenticationServiceImpl#grantServiceTicket(String, Service, Credential)
223                     final ServiceTicket st = newST((TicketGrantingTicket) jpaTicketRegistry.getTicket(parentTgtId));
224                     jpaTicketRegistry.addTicket(st);
225                     return st.getId();
226                 }
227             });
228         }
