39      @Test

42          final MessageContext ctx = mock(MessageContext.class);
43          
44          final Map<String, Class<? extends Exception>> map = new HashMap<String, Class<? extends Exception>>();
45          map.put("notFound", AccountNotFoundException.class);
46          final String id = handler.handle(new AuthenticationException(map), ctx);

48      }
