35  /**
36   * Implementation of an AuthenticationHandler for SPNEGO supports. This Handler
37   * support both NTLM and Kerberos. NTLM is disabled by default.
38   *
39   * @author Arnaud Lesueur
40   * @author Marc-Antoine Garrigue
41   * @author Scott Battaglia
42   * @author Marvin S. Addison
43   *
44   * @since 3.1
45   */
46  public final class JCIFSSpnegoAuthenticationHandler extends AbstractPreAndPostProcessingAuthenticationHandler {
47  
48      private Authentication authentication;
49  
50      /**
51       * Principal contains the DomainName ? (true by default).
52       */
53      private boolean principalWithDomainName = true;
54  
55      /**
56       * Allow SPNEGO/NTLM Token as valid credentials. (false by default)
57       */
58      private boolean isNTLMallowed = false;
59  
60      @Override
61      protected HandlerResult doAuthentication(final Credential credential) throws GeneralSecurityException, PreventedException {
62          final SpnegoCredential spnegoCredential = (SpnegoCredential) credential;
63          Principal principal;
64          byte[] nextToken;
65          try {
66              // proceed authentication using jcifs
67              synchronized (this) {
68                  this.authentication.reset();
69                  this.authentication.process(spnegoCredential.getInitToken());
70                  principal = this.authentication.getPrincipal();
71                  nextToken = this.authentication.getNextToken();
72              }
73          } catch (final jcifs.spnego.AuthenticationException e) {
74              throw new FailedLoginException(e.getMessage());
75          }
76  
77          // evaluate jcifs response
78          if (nextToken != null) {
79              logger.debug("Setting nextToken in credential");
80              spnegoCredential.setNextToken(nextToken);
81          } else {
82              logger.debug("nextToken is null");
83          }
84  
85          boolean success = false;
86          if (principal != null) {
87              if (spnegoCredential.isNtlm()) {
88                  logger.debug("NTLM Credential is valid for user [{}]", principal.getName());
89                  spnegoCredential.setPrincipal(getSimplePrincipal(principal.getName(), true));
90                  success = this.isNTLMallowed;
91              }
92              // else => kerberos
93              logger.debug("Kerberos Credential is valid for user [{}]", principal.getName());
94              spnegoCredential.setPrincipal(getSimplePrincipal(principal.getName(), false));
95              success = true;
96          }
97  
98          if (!success) {
99              throw new FailedLoginException("Principal is null, the processing of the SPNEGO Token failed");
100         }
101         return new HandlerResult(this, new BasicCredentialMetaData(credential), spnegoCredential.getPrincipal());
102     }
103 
104     @Override
105     public boolean supports(final Credential credential) {
106         return credential instanceof SpnegoCredential;
107     }
108 
109     public void setAuthentication(final Authentication authentication) {
110         this.authentication = authentication;
111     }
112 
113     public void setPrincipalWithDomainName(final boolean principalWithDomainName) {
114         this.principalWithDomainName = principalWithDomainName;
115     }
116 
117     public void setNTLMallowed(final boolean isNTLMallowed) {
118         this.isNTLMallowed = isNTLMallowed;
119     }
120 
121     protected SimplePrincipal getSimplePrincipal(final String name, final boolean isNtlm) {
122         if (this.principalWithDomainName) {
123             return new SimplePrincipal(name);
124         }
125         if (isNtlm) {
126             return Pattern.matches("\\S+\\\\\\S+", name)
127                     ? new SimplePrincipal(name.split("\\\\")[1])
128                     : new SimplePrincipal(name);
129         }
130         return new SimplePrincipal(name.split("@")[0]);
131     }
