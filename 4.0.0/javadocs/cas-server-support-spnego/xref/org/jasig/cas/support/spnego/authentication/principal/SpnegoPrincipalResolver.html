19  package org.jasig.cas.support.spnego.authentication.principal;
20  
21  import java.util.Locale;
22  
23  import org.jasig.cas.authentication.principal.PersonDirectoryPrincipalResolver;
24  import org.jasig.cas.authentication.Credential;
25  
26  import javax.validation.constraints.NotNull;
27  
28  /**
29   * Implementation of a CredentialToPrincipalResolver that takes a
30   * SpnegoCredential and returns a SimplePrincipal.
31   *
32   * @author Arnaud Lesueur
33   * @author Marc-Antoine Garrigue
34   * @since 3.1
35   */
36  public final class SpnegoPrincipalResolver extends PersonDirectoryPrincipalResolver {
37  
38      /** Tranformation types. **/
39      public static enum Transform {NONE, UPPERCASE, LOWERCASE}
40  
41      @NotNull
42      private Transform transformPrincipalId = Transform.NONE;
43  
44      @Override
45      protected String extractPrincipalId(final Credential credential) {
46          final SpnegoCredential c = (SpnegoCredential) credential;
47  
48          switch (this.transformPrincipalId) {
49          case UPPERCASE:
50              return c.getPrincipal().getId().toUpperCase(Locale.ENGLISH);
51          case LOWERCASE:
52              return c.getPrincipal().getId().toLowerCase(Locale.ENGLISH);
53          default:
54              return c.getPrincipal().getId();
55          }
56      }
57  
58      @Override
59      public boolean supports(final Credential credential) {
60          return credential != null
61                  && SpnegoCredential.class.equals(credential.getClass());
62      }
63  
64      public void setTransformPrincipalId(final Transform transform) {
65          this.transformPrincipalId = transform;
66      }
67  }
