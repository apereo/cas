35  /**
36   * Key-value ticket registry implementation that stores tickets in memcached keyed on the ticket ID.
37   *
38   * @author Scott Battaglia
39   * @author Marvin S. Addison
40   * @since 3.3
41   */
42  public final class MemCacheTicketRegistry extends AbstractDistributedTicketRegistry implements DisposableBean {
43  
44      /** Memcached client. */
45      @NotNull
46      private final MemcachedClientIF client;
47  
48      /**
49       * TGT cache entry timeout in seconds.
50       */
51      @Min(0)
52      private final int tgtTimeout;
53  
54      /**
55       * ST cache entry timeout in seconds.
56       */
57      @Min(0)
58      private final int stTimeout;
59  
60  
61      /**
62       * Creates a new instance that stores tickets in the given memcached hosts.
63       *
64       * @param hostnames                   Array of memcached hosts where each element is of the form host:port.
65       * @param ticketGrantingTicketTimeOut TGT timeout in seconds.
66       * @param serviceTicketTimeOut        ST timeout in seconds.
67       */
68      public MemCacheTicketRegistry(final String[] hostnames, final int ticketGrantingTicketTimeOut,
69  final int serviceTicketTimeOut) {
70          try {
71              this.client = new MemcachedClient(AddrUtil.getAddresses(Arrays.asList(hostnames)));
72          } catch (final IOException e) {
73              throw new IllegalArgumentException("Invalid memcached host specification.", e);
74          }
75          this.tgtTimeout = ticketGrantingTicketTimeOut;
76          this.stTimeout = serviceTicketTimeOut;
77      }
78  
79      /**
80       * This alternative constructor takes time in milliseconds.
81       * It has the timeout parameters in order to create a unique method signature.
82       *
83       * @param ticketGrantingTicketTimeOut TGT timeout in milliseconds.
84       * @param serviceTicketTimeOut ST timeout in milliseconds.
85       * @param hostnames  Array of memcached hosts where each element is of the form host:port.
86       * @see MemCacheTicketRegistry#MemCacheTicketRegistry(String[], int, int)
87       * @deprecated This has been deprecated
88       */
89      @Deprecated
90      public MemCacheTicketRegistry(final long ticketGrantingTicketTimeOut, final long serviceTicketTimeOut,
91              final String[] hostnames) {
92          this(hostnames, (int) (ticketGrantingTicketTimeOut / 1000), (int) (serviceTicketTimeOut / 1000));
93      }
94  
95      /**
96       * Creates a new instance using the given memcached client instance, which is presumably configured via
97       * <code>net.spy.memcached.spring.MemcachedClientFactoryBean</code>.
98       *
99       * @param client                      Memcached client.
100      * @param ticketGrantingTicketTimeOut TGT timeout in seconds.
101      * @param serviceTicketTimeOut        ST timeout in seconds.
102      */
103     public MemCacheTicketRegistry(final MemcachedClientIF client, final int ticketGrantingTicketTimeOut,
104             final int serviceTicketTimeOut) {
105         this.tgtTimeout = ticketGrantingTicketTimeOut;
106         this.stTimeout = serviceTicketTimeOut;
107         this.client = client;
108     }
109 
110     protected void updateTicket(final Ticket ticket) {
111         logger.debug("Updating ticket {}", ticket);
112         try {
113             if (!this.client.replace(ticket.getId(), getTimeout(ticket), ticket).get()) {
114                 logger.error("Failed updating {}", ticket);
115             }
116         } catch (final InterruptedException e) {
117             logger.warn("Interrupted while waiting for response to async replace operation for ticket {}. "
118                         + "Cannot determine whether update was successful.", ticket);
119         } catch (final Exception e) {
120             logger.error("Failed updating {}", ticket, e);
121         }
122     }
123 
124     public void addTicket(final Ticket ticket) {
125         logger.debug("Adding ticket {}", ticket);
126         try {
127             if (!this.client.add(ticket.getId(), getTimeout(ticket), ticket).get()) {
128                 logger.error("Failed adding {}", ticket);
129             }
130         } catch (final InterruptedException e) {
131             logger.warn("Interrupted while waiting for response to async add operation for ticket {}."
132                     + "Cannot determine whether add was successful.", ticket);
133         } catch (final Exception e) {
134             logger.error("Failed adding {}", ticket, e);
135         }
136     }
137 
138     public boolean deleteTicket(final String ticketId) {
139         logger.debug("Deleting ticket {}", ticketId);
140         try {
141             return this.client.delete(ticketId).get();
142         } catch (final Exception e) {
143             logger.error("Failed deleting {}", ticketId, e);
144         }
145         return false;
146     }
147 
148     public Ticket getTicket(final String ticketId) {
149         try {
150             final Ticket t = (Ticket) this.client.get(ticketId);
151             if (t != null) {
152                 return getProxiedTicketInstance(t);
153             }
154         } catch (final Exception e) {
155             logger.error("Failed fetching {} ", ticketId, e);
156         }
157         return null;
158     }

160     /**
161      * {@inheritDoc}
162      * This operation is not supported.
163      *
164      * @throws UnsupportedOperationException if you try and call this operation.
165      */
166     @Override
167     public Collection<Ticket> getTickets() {
168         throw new UnsupportedOperationException("GetTickets not supported.");
169     }
170 
171     public void destroy() throws Exception {
172         this.client.shutdown();
173     }
174 
175     /**
176      * @param sync set to true, if updates to registry are to be synchronized
177      * @deprecated As of version 3.5, this operation has no effect since async writes can cause registry consistency issues.
178      */
179     @Deprecated
180     public void setSynchronizeUpdatesToRegistry(final boolean sync) {}
181 
182     @Override
183     protected boolean needsCallback() {
184         return true;
185     }
186 
187     private int getTimeout(final Ticket t) {
188         if (t instanceof TicketGrantingTicket) {
189             return this.tgtTimeout;
190         } else if (t instanceof ServiceTicket) {
191             return this.stTimeout;
192         }
193         throw new IllegalArgumentException("Invalid ticket type");
194     }
