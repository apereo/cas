36  /**
37   * <p>
38   * <a href="http://ehcache.org/">Ehcache</a> based distributed ticket registry.
39   * </p>
40   * <p>
41   * Use distinct caches for ticket granting tickets (TGT) and service tickets (ST) for:
42   * <ul>
43   * <li>Tuning : use cache level time to live with different values for TGT an ST.</li>
44   * <li>Monitoring : follow separately the number of TGT and ST.</li>
45   * </ul>
46   * </p>
47   *
48   * @author <a href="mailto:cleclerc@xebia.fr">Cyrille Le Clerc</a>
49   * @author Adam Rybicki
50   * @author Andrew Tillinghast
51   */
52  public final class EhCacheTicketRegistry extends AbstractDistributedTicketRegistry implements InitializingBean {
53  
54      private Cache serviceTicketsCache = null;
55  
56      private Cache ticketGrantingTicketsCache = null;
57  
58      /** @see #setSupportRegistryState(boolean)*/
59      private boolean supportRegistryState = true;
60  
61      public EhCacheTicketRegistry() {
62      }
63  
64      public EhCacheTicketRegistry(final Cache serviceTicketsCache, final Cache ticketGrantingTicketsCache) {
65          super();
66          setServiceTicketsCache(serviceTicketsCache);
67          setTicketGrantingTicketsCache(ticketGrantingTicketsCache);
68      }
69  
70      public EhCacheTicketRegistry(final Cache serviceTicketsCache, final Cache ticketGrantingTicketsCache,
71              final boolean supportRegistryState) {
72          this(serviceTicketsCache, ticketGrantingTicketsCache);
73          setSupportRegistryState(supportRegistryState);
74      }
75  
76      @Override
77      public void addTicket(final Ticket ticket) {
78          final Element element = new Element(ticket.getId(), ticket);
79          if (ticket instanceof ServiceTicket) {
80              logger.debug("Adding service ticket {} to the cache", ticket.getId(), this.serviceTicketsCache.getName());
81              this.serviceTicketsCache.put(element);
82          } else if (ticket instanceof TicketGrantingTicket) {
83              logger.debug("Adding ticket granting ticket {} to the cache {}", ticket.getId(),
84                      this.ticketGrantingTicketsCache.getName());
85              this.ticketGrantingTicketsCache.put(element);
86          } else {
87              throw new IllegalArgumentException("Invalid ticket type " + ticket);
88          }
89      }
90  
91      @Override
92      public boolean deleteTicket(final String ticketId) {
93          if (StringUtils.isBlank(ticketId)) {
94              return false;
95          }
96          return this.serviceTicketsCache.remove(ticketId) || this.ticketGrantingTicketsCache.remove(ticketId);
97      }
98  
99      @Override
100     public Ticket getTicket(final String ticketId) {
101         if (ticketId == null) {
102             return null;
103         }
104 
105         Element element = this.serviceTicketsCache.get(ticketId);
106         if (element == null) {
107             element = this.ticketGrantingTicketsCache.get(ticketId);
108         }
109         return element == null ? null : getProxiedTicketInstance((Ticket) element.getObjectValue());
110     }
111 
112     @Override
113     public Collection<Ticket> getTickets() {
114         final Collection<Element> serviceTickets = this.serviceTicketsCache.getAll(
115                 this.serviceTicketsCache.getKeysWithExpiryCheck()).values();
116         final Collection<Element> tgtTicketsTickets = this.ticketGrantingTicketsCache.getAll(
117                 this.ticketGrantingTicketsCache.getKeysWithExpiryCheck()).values();
118 
119         final Collection<Ticket> allTickets = new HashSet<Ticket>(serviceTickets.size() + tgtTicketsTickets.size());
120 
121         for (final Element ticket : serviceTickets) {
122             allTickets.add((Ticket) ticket.getObjectValue());
123         }
124 
125         for (final Element ticket : tgtTicketsTickets) {
126             allTickets.add((Ticket) ticket.getObjectValue());
127         }
128 
129         return allTickets;
130     }
131 
132     public void setServiceTicketsCache(final Cache serviceTicketsCache) {
133         this.serviceTicketsCache = serviceTicketsCache;
134     }
135 
136     public void setTicketGrantingTicketsCache(final Cache ticketGrantingTicketsCache) {
137         this.ticketGrantingTicketsCache = ticketGrantingTicketsCache;
138     }
139 
140     @Override
141     public String toString() {
142         return new ToStringCreator(this).append("ticketGrantingTicketsCache", this.ticketGrantingTicketsCache)
143                 .append("serviceTicketsCache", this.serviceTicketsCache).toString();
144     }
145 
146     @Override
147     protected void updateTicket(final Ticket ticket) {
148         addTicket(ticket);
149     }
150 
151     @Override
152     protected boolean needsCallback() {
153         return false;
154     }

156     /**
157      * Flag to indicate whether this registry instance should participate in reporting its state with
158      * default value set to <code>true</code>.
159      * Based on the <a href="http://ehcache.org/apidocs/net/sf/ehcache/Ehcache.html#getKeysWithExpiryCheck()">EhCache documentation</a>,
160      * determining the number of service tickets and the total session count from the cache can be considered
161      * an expensive operation with the time taken as O(n), where n is the number of elements in the cache.
162      *
163      * <p>Therefore, the flag provides a level of flexibility such that depending on the cache and environment
164      * settings, reporting statistics
165      * can be set to false and disabled.</p>
166      *
167      * @param supportRegistryState true, if the registry is to support registry state
168      * @see #sessionCount()
169      * @see #serviceTicketCount()
170      * @see org.jasig.cas.monitor.SessionMonitor
171      */
172     public void setSupportRegistryState(final boolean supportRegistryState) {
173         this.supportRegistryState = supportRegistryState;
174     }

176     @Override
177     public void afterPropertiesSet() throws Exception {
178         if (this.serviceTicketsCache == null || this.ticketGrantingTicketsCache == null) {
179             throw new BeanInstantiationException(this.getClass(),
180                     "Both serviceTicketsCache and ticketGrantingTicketsCache are required properties.");
181         }
182 
183         if (logger.isDebugEnabled()) {
184             CacheConfiguration config = this.serviceTicketsCache.getCacheConfiguration();
185             logger.debug("serviceTicketsCache.maxElementsInMemory={}", config.getMaxEntriesLocalHeap());
186             logger.debug("serviceTicketsCache.maxElementsOnDisk={}", config.getMaxElementsOnDisk());
187             logger.debug("serviceTicketsCache.isOverflowToDisk={}", config.isOverflowToDisk());
188             logger.debug("serviceTicketsCache.timeToLive={}", config.getTimeToLiveSeconds());
189             logger.debug("serviceTicketsCache.timeToIdle={}", config.getTimeToIdleSeconds());
190             logger.debug("serviceTicketsCache.cacheManager={}", this.serviceTicketsCache.getCacheManager().getName());
191 
192             config = this.ticketGrantingTicketsCache.getCacheConfiguration();
193             logger.debug("ticketGrantingTicketsCache.maxElementsInMemory={}", config.getMaxEntriesLocalHeap());
194             logger.debug("ticketGrantingTicketsCache.maxElementsOnDisk={}", config.getMaxElementsOnDisk());
195             logger.debug("ticketGrantingTicketsCache.isOverflowToDisk={}", config.isOverflowToDisk());
196             logger.debug("ticketGrantingTicketsCache.timeToLive={}", config.getTimeToLiveSeconds());
197             logger.debug("ticketGrantingTicketsCache.timeToIdle={}", config.getTimeToIdleSeconds());
198             logger.debug("ticketGrantingTicketsCache.cacheManager={}", this.ticketGrantingTicketsCache.getCacheManager()
199                     .getName());
200         }
201     }
