19  package org.jasig.cas.support.openid.web.mvc;
20  
21  import org.apache.commons.lang3.StringUtils;
22  import org.jasig.cas.web.DelegateController;
23  import org.openid4java.message.Message;
24  import org.openid4java.message.ParameterList;
25  import org.openid4java.server.ServerManager;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  import org.springframework.web.servlet.ModelAndView;
29  
30  import javax.servlet.http.HttpServletRequest;
31  import javax.servlet.http.HttpServletResponse;
32  import javax.validation.constraints.NotNull;
33  import java.io.Serializable;
34  import java.util.HashMap;
35  import java.util.Map;
36  
37  /**
38   * Generates an association to an openid association request.
39   * @author Frederic Esnault
40   * @since 3.5
41   */
42  public class SmartOpenIdController extends DelegateController implements Serializable {
43  
44      private static final long serialVersionUID = -594058549445950430L;
45  
46      /** View if association Fails. */
47      private static final String DEFAULT_ASSOCIATION_FAILURE_VIEW_NAME = "casOpenIdAssociationFailureView";
48  
49      /** View if association Succeeds. */
50      private static final String DEFAULT_ASSOCIATION_SUCCESS_VIEW_NAME = "casOpenIdAssociationSuccessView";
51  
52      private static final String ASSOCIATE = "associate";
53  
54      private final Logger logger = LoggerFactory.getLogger(SmartOpenIdController.class);
55  
56      private ServerManager serverManager;
57  
58      /** The view to redirect to on a successful validation. */
59      @NotNull
60      private String successView = DEFAULT_ASSOCIATION_SUCCESS_VIEW_NAME;
61  
62      /** The view to redirect to on a validation failure. Not used for now,
63       *  the succes view may return failed association attemps. No need for another view. */
64      @NotNull
65      private String failureView = DEFAULT_ASSOCIATION_FAILURE_VIEW_NAME;
66  
67      /**
68       * Gets the association response. Determines the mode first.
69       * If mode is set to associate, will set the response. Then
70       * builds the response parameters next and returns.
71       *
72       * @param request the request
73       * @return the association response
74       */
75      public Map<String, String> getAssociationResponse(final HttpServletRequest request) {
76          final ParameterList parameters = new ParameterList(request.getParameterMap());
77  
78          final String mode = parameters.hasParameter("openid.mode")
79                  ? parameters.getParameterValue("openid.mode")
80                  : null;
81  
82          Message response = null;
83  
84          if (StringUtils.equals(mode, ASSOCIATE)) {
85              response = serverManager.associationResponse(parameters);
86          }
87          final Map<String, String> responseParams = new HashMap<>();
88          if (response != null) {
89              responseParams.putAll(response.getParameterMap());
90          }
91  
92          return responseParams;
93  
94      }
95  
96      @Override
97      protected ModelAndView handleRequestInternal(final HttpServletRequest request, final HttpServletResponse response)
98              throws Exception {
99          final Map<String, String> parameters = new HashMap<>();
100         parameters.putAll(getAssociationResponse(request));
101         return new ModelAndView(successView, "parameters", parameters);
102     }
103 
104     @Override
105     public boolean canHandle(final HttpServletRequest request, final HttpServletResponse response) {
106         final String openIdMode = request.getParameter("openid.mode");
107         if (StringUtils.equals(openIdMode, ASSOCIATE)) {
108             logger.info("Handling request. openid.mode : {}", openIdMode);
109             return true;
110         }
111         logger.info("Cannot handle request. openid.mode : {}", openIdMode);
112         return false;
113     }
114 
115     public void setSuccessView(final String successView) {
116         this.successView = successView;
117     }
118 
119     public void setFailureView(final String failureView) {
120         this.failureView = failureView;
121     }
122 
123     @NotNull
124     public void setServerManager(final ServerManager serverManager) {
125         this.serverManager = serverManager;
126     }
127 }
