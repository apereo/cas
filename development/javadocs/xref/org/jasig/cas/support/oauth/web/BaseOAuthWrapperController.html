19  package org.jasig.cas.support.oauth.web;

21  import javax.servlet.http.HttpServletRequest;
22  import javax.servlet.http.HttpServletResponse;
23  import javax.validation.constraints.NotNull;
24  
25  import org.apache.commons.lang.StringUtils;
26  import org.jasig.cas.services.ServicesManager;
27  import org.jasig.cas.ticket.registry.TicketRegistry;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  import org.springframework.web.servlet.ModelAndView;
31  import org.springframework.web.servlet.mvc.AbstractController;

33  /**
34   * This controller is the base controller for wrapping OAuth protocol in CAS.
35   * It finds the right sub controller to call according to the url.
36   *
37   * @author Jerome Leleu
38   * @since 3.5.0
39   */
40  public abstract class BaseOAuthWrapperController extends AbstractController {
41  
42      protected final Logger logger = LoggerFactory.getLogger(BaseOAuthWrapperController.class);
43  
44      @NotNull
45      protected String loginUrl;
46  
47      @NotNull
48      protected ServicesManager servicesManager;
49  
50      @NotNull
51      protected TicketRegistry ticketRegistry;
52  
53      @NotNull
54      protected long timeout;
55  
56      @Override
57      protected ModelAndView handleRequestInternal(final HttpServletRequest request, final HttpServletResponse response)
58              throws Exception {
59  
60          final String method = getMethod(request);
61          logger.debug("method : {}", method);
62          return internalHandleRequest(method, request, response);
63      }
64  
65      protected abstract ModelAndView internalHandleRequest(String method, HttpServletRequest request,
66              HttpServletResponse response) throws Exception;
67  
68      /**
69       * Return the method to call according to the url.
70       *
71       * @param request the incoming http request
72       * @return the method to call according to the url
73       */
74      private String getMethod(final HttpServletRequest request) {
75          String method = request.getRequestURI();
76          if (method.indexOf("?") >= 0) {
77              method = StringUtils.substringBefore(method, "?");
78          }
79          final int pos = method.lastIndexOf("/");
80          if (pos >= 0) {
81              method = method.substring(pos + 1);
82          }
83          return method;
84      }
85  
86      public void setServicesManager(final ServicesManager servicesManager) {
87          this.servicesManager = servicesManager;
88      }
89  
90      public void setTicketRegistry(final TicketRegistry ticketRegistry) {
91          this.ticketRegistry = ticketRegistry;
92      }
93  
94      public void setLoginUrl(final String loginUrl) {
95          this.loginUrl = loginUrl;
96      }
97  
98      public void setTimeout(final long timeout) {
99          this.timeout = timeout;
100     }
