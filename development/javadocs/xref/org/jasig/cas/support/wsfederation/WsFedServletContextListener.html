20  package org.jasig.cas.support.wsfederation;
21  
22  
23  import org.jasig.cas.authentication.AuthenticationHandler;
24  import org.jasig.cas.authentication.principal.PrincipalResolver;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  import org.springframework.beans.BeansException;
28  import org.springframework.beans.factory.annotation.Autowired;
29  import org.springframework.beans.factory.annotation.Qualifier;
30  import org.springframework.context.ApplicationContext;
31  import org.springframework.context.ApplicationContextAware;
32  import org.springframework.stereotype.Component;
33  
34  import javax.servlet.ServletContextEvent;
35  import javax.servlet.ServletContextListener;
36  import javax.servlet.annotation.WebListener;
37  import java.util.Map;
38  
39  /**
40   * Initializes the CAS root servlet context to make sure
41   * ADFS validation can be activated and authentication handlers injected.
42   * @author Misagh Moayyed
43   * @since 4.2
44   */
45  @WebListener
46  @Component
47  public class WsFedServletContextListener implements ServletContextListener, ApplicationContextAware {
48  
49      private final Logger logger = LoggerFactory.getLogger(this.getClass());
50  
51      @Autowired
52      @Qualifier("adfsAuthNHandler")
53      private AuthenticationHandler adfsAuthNHandler;
54  
55      @Autowired
56      @Qualifier("adfsPrincipalResolver")
57      private PrincipalResolver adfsPrincipalResolver;
58  
59      @Override
60      public void contextInitialized(final ServletContextEvent sce) {
61          logger.info("Initializing WsFed servlet context...");
62      }
63  
64      @Override
65      public void contextDestroyed(final ServletContextEvent sce) {}
66  
67      @Override
68      public void setApplicationContext(final ApplicationContext applicationContext) throws BeansException {
69          try {
70              if (applicationContext.getParent() == null) {
71                  logger.info("Initializing WsFed root application context");
72  
73                  final Map<AuthenticationHandler, PrincipalResolver> authenticationHandlersResolvers =
74                          applicationContext.getBean("authenticationHandlersResolvers", Map.class);
75                  authenticationHandlersResolvers.put(adfsAuthNHandler, adfsPrincipalResolver);
76  
77                  logger.info("Initialized WsFed root application context successfully");
78              } else {
79                  logger.info("Initializing WsFed application context");
80  
81                  logger.info("Initialized WsFed application context successfully");
82              }
83          } catch (final Exception e) {
84              logger.error(e.getMessage(), e);
85          }
86      }
87  }
