77  /**
78   * An abstract builder to serve as the template handler
79   * for SAML1 and SAML2 responses.
80   *
81   * @author Misagh Moayyed mmoayyed@unicon.net
82   * @since 4.1

84  public abstract class AbstractSamlObjectBuilder {
85      /**
86       * The constant DEFAULT_ELEMENT_NAME_FIELD.
87       */
88      protected static final String DEFAULT_ELEMENT_NAME_FIELD = "DEFAULT_ELEMENT_NAME";
89  
90      /**
91       * The constant DEFAULT_ELEMENT_LOCAL_NAME_FIELD.
92       */
93      protected static final String DEFAULT_ELEMENT_LOCAL_NAME_FIELD = "DEFAULT_ELEMENT_LOCAL_NAME";
94  
95      private static final int RANDOM_ID_SIZE = 16;
96  
97      private  static final String SIGNATURE_FACTORY_PROVIDER_CLASS = "org.jcp.xml.dsig.internal.dom.XMLDSigRI";

102     /**
103      * Initialize and bootstrap opensaml.
104      * Check for prior OpenSAML initialization to prevent double init
105      * that would overwrite existing OpenSAML configuration.
106      */
107     static {
108         try {
109 
110             if (XMLObjectProviderRegistrySupport.getParserPool() == null) {
111                 InitializationService.initialize();
112             }
113         } catch (final InitializationException e) {
114             throw new IllegalStateException("Error initializing OpenSAML library.", e);
115         }
116     }

118     /**
119      * Create a new SAML object.
120      *
121      * @param <T> the generic type
122      * @param objectType the object type
123      * @return the t
124      */
125     public final <T extends SAMLObject> T newSamlObject(final Class<T> objectType) {
126         final QName qName = getSamlObjectQName(objectType);
127         final SAMLObjectBuilder<T> builder = (SAMLObjectBuilder<T>)
128                 XMLObjectProviderRegistrySupport.getBuilderFactory().getBuilder(qName);
129         if (builder == null) {
130             throw new IllegalStateException("No SAMLObjectBuilder registered for class " + objectType.getName());
131         }
132         return objectType.cast(builder.buildObject(qName));
133     }

135     /**
136      * Gets saml object QName.
137      *
138      * @param objectType the object type
139      * @return the saml object QName
140      * @throws RuntimeException the exception
141      */
142     public QName getSamlObjectQName(final Class objectType) throws RuntimeException {
143         try {
144             final Field f = objectType.getField(DEFAULT_ELEMENT_NAME_FIELD);
145             final QName qName = (QName) f.get(null);
146             return qName;
147         } catch (final NoSuchFieldException e) {
148             throw new IllegalStateException("Cannot find field " + objectType.getName() + '.' + DEFAULT_ELEMENT_NAME_FIELD);
149         } catch (final IllegalAccessException e) {
150             throw new IllegalStateException("Cannot access field " + objectType.getName() + '.' + DEFAULT_ELEMENT_NAME_FIELD);
151         }
152     }

154     /**
155      * Build the saml object based on its QName.
156      *
157      * @param objectType the object
158      * @param qName the QName
159      * @param <T> the object type
160      * @return the saml object
161      */
162     private <T extends SAMLObject> T newSamlObject(final Class<T> objectType, final QName qName) {
163         final SAMLObjectBuilder<T> builder = (SAMLObjectBuilder<T>) XMLObjectProviderRegistrySupport.getBuilderFactory().getBuilder(qName);
164         if (builder == null) {
165             throw new IllegalStateException("No SAMLObjectBuilder registered for class " + objectType.getName());
166         }
167         return objectType.cast(builder.buildObject());
168     }

170     /**
171      * New attribute value.
172      *
173      * @param value the value
174      * @param elementName the element name
175      * @return the xS string
176      */
177     protected final XSString newAttributeValue(final Object value, final QName elementName) {
178         final XSStringBuilder attrValueBuilder = new XSStringBuilder();
179         final XSString stringValue = attrValueBuilder.buildObject(elementName, XSString.TYPE_NAME);
180         if (value instanceof String) {
181             stringValue.setValue((String) value);
182         } else {
183             stringValue.setValue(value.toString());
184         }
185         return stringValue;
186     }

188     /**
189      * Generate a secure random id.
190      *
191      * @return the secure id string
192      */
193     public String generateSecureRandomId() {
194         try {
195             final SecureRandom random = SecureRandom.getInstance("SHA1PRNG");
196             final byte[] buf = new byte[RANDOM_ID_SIZE];
197             random.nextBytes(buf);
198             return "_".concat(new String(Hex.encode(buf)));
199         } catch (final Exception e) {
200             throw new IllegalStateException("Cannot create secure random ID generator for SAML message IDs.", e);
201         }
202     }

204     /**
205      * Marshal the saml xml object to raw xml.
206      *
207      * @param object the object
208      * @param writer the writer
209      * @return the xml string
210      */
211     public String marshalSamlXmlObject(final XMLObject object, final StringWriter writer)  {
212         try {
213             final MarshallerFactory marshallerFactory = XMLObjectProviderRegistrySupport.getMarshallerFactory();
214             final Marshaller marshaller = marshallerFactory.getMarshaller(object);
215             final Element element = marshaller.marshall(object);
216             element.setAttributeNS("http://www.w3.org/2000/xmlns/", "xmlns", SAMLConstants.SAML20_NS);
217             element.setAttributeNS("http://www.w3.org/2000/xmlns/", "xmlns:xenc", "http://www.w3.org/2001/04/xmlenc#");
218 
219             final TransformerFactory transFactory = TransformerFactory.newInstance();
220             final Transformer transformer = transFactory.newTransformer();
221             transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
222             transformer.setOutputProperty(OutputKeys.INDENT, "yes");
223             transformer.transform(new DOMSource(element), new StreamResult(writer));
224             return writer.toString();
225         } catch (final Exception e) {
226             throw new IllegalStateException("An error has occurred while marshalling SAML object to xml", e);
227         }
228     }

230     /**
231      * Sign SAML response.
232      *
233      * @param samlResponse the SAML response
234      * @param privateKey the private key
235      * @param publicKey the public key
236      * @return the response
237      */
238     public final String signSamlResponse(final String samlResponse,
239                                          final PrivateKey privateKey, final PublicKey publicKey) {
240         final Document doc = constructDocumentFromXml(samlResponse);
241 
242         if (doc != null) {
243             final org.jdom.Element signedElement = signSamlElement(doc.getRootElement(),
244                     privateKey, publicKey);
245             doc.setRootElement((org.jdom.Element) signedElement.detach());
246             return new XMLOutputter().outputString(doc);
247         }
248         throw new RuntimeException("Error signing SAML Response: Null document");
249     }

251     /**
252      * Construct document from xml string.
253      *
254      * @param xmlString the xml string
255      * @return the document
256      */
257     public static Document constructDocumentFromXml(final String xmlString) {
258         try {
259             final SAXBuilder builder = new SAXBuilder();
260             builder.setFeature("http://xml.org/sax/features/external-general-entities", false);
261             builder.setFeature("http://apache.org/xml/features/disallow-doctype-decl", true);
262             return builder
263                     .build(new ByteArrayInputStream(xmlString.getBytes(Charset.defaultCharset())));
264         } catch (final Exception e) {
265             return null;
266         }
267     }

269     /**
270      * Sign SAML element.
271      *
272      * @param element the element
273      * @param privKey the priv key
274      * @param pubKey the pub key
275      * @return the element
276      */
277     private org.jdom.Element signSamlElement(final org.jdom.Element element, final PrivateKey privKey,
278                                                     final PublicKey pubKey) {
279         try {
280             final String providerName = System.getProperty("jsr105Provider",
281                     SIGNATURE_FACTORY_PROVIDER_CLASS);
282 
283             final XMLSignatureFactory sigFactory = XMLSignatureFactory
284                     .getInstance("DOM", (Provider) Class.forName(providerName)
285                             .newInstance());
286 
287             final List<Transform> envelopedTransform = Collections
288                     .singletonList(sigFactory.newTransform(Transform.ENVELOPED,
289                             (TransformParameterSpec) null));
290 
291             final Reference ref = sigFactory.newReference("", sigFactory
292                             .newDigestMethod(DigestMethod.SHA1, null), envelopedTransform,
293                     null, null);
294 
295             // Create the SignatureMethod based on the type of key
296             final SignatureMethod signatureMethod;
297             if (pubKey instanceof DSAPublicKey) {
298                 signatureMethod = sigFactory.newSignatureMethod(
299                         SignatureMethod.DSA_SHA1, null);
300             } else if (pubKey instanceof RSAPublicKey) {
301                 signatureMethod = sigFactory.newSignatureMethod(
302                         SignatureMethod.RSA_SHA1, null);
303             } else {
304                 throw new RuntimeException("Error signing SAML element: Unsupported type of key");
305             }
306 
307             final CanonicalizationMethod canonicalizationMethod = sigFactory
308                     .newCanonicalizationMethod(
309                             CanonicalizationMethod.INCLUSIVE_WITH_COMMENTS,
310                             (C14NMethodParameterSpec) null);
311 
312             // Create the SignedInfo
313             final SignedInfo signedInfo = sigFactory.newSignedInfo(
314                     canonicalizationMethod, signatureMethod, Collections
315                             .singletonList(ref));
316 
317             // Create a KeyValue containing the DSA or RSA PublicKey
318             final KeyInfoFactory keyInfoFactory = sigFactory
319                     .getKeyInfoFactory();
320             final KeyValue keyValuePair = keyInfoFactory.newKeyValue(pubKey);
321 
322             // Create a KeyInfo and add the KeyValue to it
323             final KeyInfo keyInfo = keyInfoFactory.newKeyInfo(Collections
324                     .singletonList(keyValuePair));
325             // Convert the JDOM document to w3c (Java XML signature API requires
326             // w3c representation)
327             final org.w3c.dom.Element w3cElement = toDom(element);
328 
329             // Create a DOMSignContext and specify the DSA/RSA PrivateKey and
330             // location of the resulting XMLSignature's parent element
331             final DOMSignContext dsc = new DOMSignContext(privKey, w3cElement);
332 
333             final org.w3c.dom.Node xmlSigInsertionPoint = getXmlSignatureInsertLocation(w3cElement);
334             dsc.setNextSibling(xmlSigInsertionPoint);
335 
336             // Marshal, generate (and sign) the enveloped signature
337             final XMLSignature signature = sigFactory.newXMLSignature(signedInfo,
338                     keyInfo);
339             signature.sign(dsc);
340 
341             return toJdom(w3cElement);
342 
343         } catch (final Exception e) {
344             throw new RuntimeException("Error signing SAML element: "
345                     + e.getMessage(), e);
346         }
347     }

349     /**
350      * Gets the xml signature insert location.
351      *
352      * @param elem the elem
353      * @return the xml signature insert location
354      */
355     private static Node getXmlSignatureInsertLocation(final org.w3c.dom.Element elem) {
356         org.w3c.dom.Node insertLocation = null;
357         org.w3c.dom.NodeList nodeList = elem.getElementsByTagNameNS(
358                 SAMLConstants.SAML20P_NS, "Extensions");
359         if (nodeList.getLength() != 0) {
360             insertLocation = nodeList.item(nodeList.getLength() - 1);
361         } else {
362             nodeList = elem.getElementsByTagNameNS(SAMLConstants.SAML20P_NS, "Status");
363             insertLocation = nodeList.item(nodeList.getLength() - 1);
364         }
365         return insertLocation;
366     }

371      * @param element the element
372      * @return the org.w3c.dom. element

374     private org.w3c.dom.Element toDom(final org.jdom.Element element) {
375         return toDom(element.getDocument()).getDocumentElement();
376     }

378     /**
379      * Convert the received jdom doc to a Document element.
380      *
381      * @param doc the doc
382      * @return the org.w3c.dom. document
383      */
384     private org.w3c.dom.Document toDom(final Document doc) {
385         try {
386             final XMLOutputter xmlOutputter = new XMLOutputter();
387             final StringWriter elemStrWriter = new StringWriter();
388             xmlOutputter.output(doc, elemStrWriter);
389             final byte[] xmlBytes = elemStrWriter.toString().getBytes(Charset.defaultCharset());
390             final DocumentBuilderFactory dbf = DocumentBuilderFactory
391                     .newInstance();
392             dbf.setNamespaceAware(true);
393             return dbf.newDocumentBuilder().parse(
394                     new ByteArrayInputStream(xmlBytes));
395         } catch (final Exception e) {
396             logger.trace(e.getMessage(), e);
397             return null;
398         }
399     }
