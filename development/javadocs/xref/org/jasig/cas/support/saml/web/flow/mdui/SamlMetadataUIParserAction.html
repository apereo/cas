46  /**
47   * This is {@link SamlMetadataUIParserAction} that attempts to parse
48   * the mdui extension block for a SAML SP from the provided metadata locations.
49   * The result is put into the flow request context under the parameter
50   * {@link #MDUI_FLOW_PARAMETER_NAME}. The entity id parameter is
51   * specified by default at {@link #ENTITY_ID_PARAMETER_NAME}.
52   *
53   * <p>This action is best suited to be invoked when the CAS login page
54   * is about to render so that the page, once the MDUI info is obtained,
55   * has a chance to populate the UI with relevant info about the SP.</p>
56   *
57   * @author Misagh Moayyed
58   * @since 4.1.0
59   */
60  public class SamlMetadataUIParserAction extends AbstractAction {
61      /**
62       * The default entityId parameter name.
63       */
64      public static final String ENTITY_ID_PARAMETER_NAME = "entityId";
65  
66      /**
67       * The default entityId parameter name.
68       */
69      public static final String MDUI_FLOW_PARAMETER_NAME = "mduiContext";
70  
71      private final Logger logger = LoggerFactory.getLogger(this.getClass());
72  
73      @NotNull
74      private final String entityIdParameterName;
75  
76      @NotNull
77      private final MetadataResolverAdapter metadataAdapter;
78  
79      @Autowired
80      @NotNull
81      private ServicesManager servicesManager;
82  
83      /**
84       * Instantiates a new SAML mdui parser action.
85       * Defaults the parameter name to {@link #ENTITY_ID_PARAMETER_NAME}.
86       *
87       * @param metadataAdapter the metadata resources
88       */
89      public SamlMetadataUIParserAction(final MetadataResolverAdapter metadataAdapter) {
90          this(ENTITY_ID_PARAMETER_NAME, metadataAdapter);
91      }
92  
93      /**
94       * Instantiates a new SAML mdui parser action.
95       *
96       * @param entityIdParameterName the entity id parameter name
97       * @param metadataAdapter     the metadata adapter
98       */
99      public SamlMetadataUIParserAction(final String entityIdParameterName,
100                                       final MetadataResolverAdapter metadataAdapter) {
101         this.entityIdParameterName = entityIdParameterName;
102         this.metadataAdapter = metadataAdapter;
103     }
104 
105     @Override
106     protected Event doExecute(final RequestContext requestContext) throws Exception {
107         final HttpServletRequest request = WebUtils.getHttpServletRequest(requestContext);
108         final String entityId = request.getParameter(this.entityIdParameterName);
109         if (StringUtils.isBlank(entityId)) {
110             logger.debug("No entity id found for parameter [{}]", this.entityIdParameterName);
111             return success();
112         }
113 
114         final WebApplicationService service = new SimpleWebApplicationServiceImpl(entityId);
115         final RegisteredService registeredService = this.servicesManager.findServiceBy(service);
116         if (registeredService == null || !registeredService.getAccessStrategy().isServiceAccessAllowed()) {
117             logger.debug("Entity id [{}] is not recognized/allowed by the CAS service registry", entityId);
118             throw new UnauthorizedServiceException(UnauthorizedServiceException.CODE_UNAUTHZ_SERVICE,
119                     "Entity " + entityId + " not recognized");
120         }
121 
122         final EntityDescriptor entityDescriptor = this.metadataAdapter.getEntityDescriptorForEntityId(entityId);
123         if (entityDescriptor == null) {
124             logger.debug("Entity descriptor not found for [{}]", entityId);
125             return success();
126         }
127 
128         final SPSSODescriptor spssoDescriptor = getSPSSODescriptor(entityDescriptor);
129         if (spssoDescriptor == null) {
130             logger.debug("SP SSO descriptor not found for [{}]", entityId);
131             return success();
132         }
133 
134         final Extensions extensions = spssoDescriptor.getExtensions();
135         final List<XMLObject> spExtensions = extensions.getUnknownXMLObjects(UIInfo.DEFAULT_ELEMENT_NAME);
136         if (spExtensions.isEmpty()) {
137             logger.debug("No extensions are found for [{}]", UIInfo.DEFAULT_ELEMENT_NAME.getNamespaceURI());
138             return success();
139         }
140 
141         final SimpleMetadataUIInfo mdui = new SimpleMetadataUIInfo(registeredService);
142 
143         for (final XMLObject obj : spExtensions) {
144             if (obj instanceof UIInfo) {
145                 final UIInfo uiInfo = (UIInfo) obj;
146                 logger.debug("Found UI info for [{}] and added to flow context", entityId);
147                 mdui.setUIInfo(uiInfo);
148             }
149         }
150 
151         requestContext.getFlowScope().put(MDUI_FLOW_PARAMETER_NAME, mdui);
152         return success();
153     }
154 
155     /**
156      * Gets SP SSO descriptor.
157      *
158      * @param entityDescriptor the entity descriptor
159      * @return the sPSSO descriptor
160      */
161     private SPSSODescriptor getSPSSODescriptor(final EntityDescriptor entityDescriptor) {
162         logger.debug("Locating SP SSO descriptor for SAML2 protocol...");
163         SPSSODescriptor spssoDescriptor = entityDescriptor.getSPSSODescriptor(SAMLConstants.SAML20P_NS);
164         if (spssoDescriptor == null) {
165             logger.debug("Locating SP SSO descriptor for SAML11 protocol...");
166             spssoDescriptor = entityDescriptor.getSPSSODescriptor(SAMLConstants.SAML11P_NS);
167         }
168         if (spssoDescriptor == null) {
169             logger.debug("Locating SP SSO descriptor for SAML1 protocol...");
170             spssoDescriptor = entityDescriptor.getSPSSODescriptor(SAMLConstants.SAML10P_NS);
171         }
172         logger.debug("SP SSO descriptor resolved to be [{}]", spssoDescriptor);
173         return spssoDescriptor;
174     }
175 }
