39  /**
40   * Implementation of a view to return a SAML SOAP response and assertion, based on
41   * the SAML 1.1 specification.
42   * <p>
43   * If an AttributePrincipal is supplied, then the assertion will include the
44   * attributes from it (assuming a String key/Object value pair). The only
45   * Authentication attribute it will look at is the authMethod (if supplied).
46   * <p>
47   * Note that this class will currently not handle proxy authentication.
48   * <p>
49   *
50   * @author Scott Battaglia
51   * @author Marvin S. Addison
52   * @since 3.1
53   */
54  public final class Saml10SuccessResponseView extends AbstractSaml10ResponseView {
55      /** Namespace for custom attributes in the saml validation payload. */
56      private static final String VALIDATION_SAML_ATTRIBUTE_NAMESPACE = "http://www.ja-sig.org/products/cas/";
57  
58      private static final int DEFAULT_ISSUE_LENGTH = 30000;
59  
60      /** The issuer, generally the hostname. */
61      @NotNull
62      private String issuer;
63  
64      /**
65       * The amount of time in milliseconds this is valid for.
66       * Defaults to {@value}.
67       **/
68      @Min(1000)
69      private long issueLength = DEFAULT_ISSUE_LENGTH;
70  
71      @NotNull
72      private String rememberMeAttributeName = CasProtocolConstants.VALIDATION_REMEMBER_ME_ATTRIBUTE_NAME;
73  
74      @Override
75      protected void prepareResponse(final Response response, final Map<String, Object> model) {
76  
77          final DateTime issuedAt = response.getIssueInstant();
78          final Service service = getAssertionFrom(model).getService();
79  
80          final Authentication authentication = getPrimaryAuthenticationFrom(model);
81          final String authenticationMethod = (String) authentication.getAttributes().get(
82                  SamlAuthenticationMetaDataPopulator.ATTRIBUTE_AUTHENTICATION_METHOD);
83  
84          final AuthenticationStatement authnStatement = this.samlObjectBuilder.newAuthenticationStatement(
85                  authentication.getAuthenticationDate(), authenticationMethod, getPrincipal(model).getId());
86  
87          final Assertion assertion = this.samlObjectBuilder.newAssertion(authnStatement, this.issuer, issuedAt,
88                  this.samlObjectBuilder.generateSecureRandomId());
89          final Conditions conditions = this.samlObjectBuilder.newConditions(issuedAt, service.getId(), this.issueLength);
90          assertion.setConditions(conditions);
91  
92          final Subject subject = this.samlObjectBuilder.newSubject(getPrincipal(model).getId());
93          final Map<String, Object> attributesToSend = prepareSamlAttributes(model);
94  
95          if (!attributesToSend.isEmpty()) {
96              assertion.getAttributeStatements().add(this.samlObjectBuilder.newAttributeStatement(
97                      subject, attributesToSend, VALIDATION_SAML_ATTRIBUTE_NAMESPACE));
98          }
99  
100         response.setStatus(this.samlObjectBuilder.newStatus(StatusCode.SUCCESS, null));
101         response.getAssertions().add(assertion);
102     }
103 
104 
105     /**
106      * Prepare saml attributes. Combines both principal and authentication
107      * attributes. If the authentication is to be remembered, uses {@link #setRememberMeAttributeName(String)}
108      * for the remember-me attribute name.
109      *
110      * @param model the model
111      * @return the final map
112      * @since 4.1.0
113      */
114     private Map<String, Object> prepareSamlAttributes(final Map<String, Object> model) {
115         final Map<String, Object> authnAttributes =
116                 new HashMap<>(getAuthenticationAttributesAsMultiValuedAttributes(model));
117         if (isRememberMeAuthentication(model)) {
118             authnAttributes.remove(RememberMeCredential.AUTHENTICATION_ATTRIBUTE_REMEMBER_ME);
119             authnAttributes.put(this.rememberMeAttributeName, Boolean.TRUE.toString());
120         }
121         final Map<String, Object> attributesToReturn = new HashMap<>();
122         attributesToReturn.putAll(getPrincipalAttributesAsMultiValuedAttributes(model));
123         attributesToReturn.putAll(authnAttributes);
124         return attributesToReturn;
125     }
126 
127     public void setIssueLength(final long issueLength) {
128         this.issueLength = issueLength;
129     }
130 
131     public void setIssuer(final String issuer) {
132         this.issuer = issuer;
133     }
134 
135     public void setRememberMeAttributeName(final String rememberMeAttributeName) {
136         this.rememberMeAttributeName = rememberMeAttributeName;
137     }
138 }
