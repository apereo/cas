48  /**
49   * This is {@link AbstractSaml20ObjectBuilder}.
50   * to build saml2 objects.
51   * @author Misagh Moayyed mmoayyed@unicon.net
52   * @since 4.1
53   */
54  public abstract class AbstractSaml20ObjectBuilder extends AbstractSamlObjectBuilder {
55      private static final int HEX_HIGH_BITS_BITWISE_FLAG = 0x0f;
56  
57      /**
58       * Gets name id.
59       *
60       * @param nameIdFormat the name id format
61       * @param nameIdValue the name id value
62       * @return the name iD
63       */
64      protected NameID getNameID(final String nameIdFormat, final String nameIdValue) {
65          final NameID nameId = newSamlObject(NameID.class);
66          nameId.setFormat(nameIdFormat);
67          nameId.setValue(nameIdValue);
68          return nameId;
69      }
70  
71      /**
72       * Create a new SAML response object.
73       * @param id the id
74       * @param issueInstant the issue instant
75       * @param recipient the recipient
76       * @param service the service
77       * @return the response
78       */
79      public Response newResponse(final String id, final DateTime issueInstant,
80                                  final String recipient, final WebApplicationService service) {
81  
82          final Response samlResponse = newSamlObject(Response.class);
83          samlResponse.setID(id);
84          samlResponse.setIssueInstant(issueInstant);
85          samlResponse.setVersion(SAMLVersion.VERSION_20);
86          if (service instanceof SamlService) {
87              final SamlService samlService = (SamlService) service;
88  
89              final String requestId = samlService.getRequestID();
90              if (StringUtils.isNotBlank(requestId)) {
91                  samlResponse.setInResponseTo(requestId);
92              }
93          }
94          return samlResponse;
95      }
96  
97      /**
98       * Create a new SAML status object.
99       *
100      * @param codeValue the code value
101      * @param statusMessage the status message
102      * @return the status
103      */
104     public Status newStatus(final String codeValue, final String statusMessage) {
105         final Status status = newSamlObject(Status.class);
106         final StatusCode code = newSamlObject(StatusCode.class);
107         code.setValue(codeValue);
108         status.setStatusCode(code);
109         if (StringUtils.isNotBlank(statusMessage)) {
110             final StatusMessage message = newSamlObject(StatusMessage.class);
111             message.setMessage(statusMessage);
112             status.setStatusMessage(message);
113         }
114         return status;
115     }
116 
117     /**
118      * Create a new SAML1 response object.
119      *
120      * @param authnStatement the authn statement
121      * @param issuer the issuer
122      * @param issuedAt the issued at
123      * @param id the id
124      * @return the assertion
125      */
126     public Assertion newAssertion(final AuthnStatement authnStatement, final String issuer,
127                                   final DateTime issuedAt, final String id) {
128         final Assertion assertion = newSamlObject(Assertion.class);
129         assertion.setID(id);
130         assertion.setIssueInstant(issuedAt);
131         assertion.setIssuer(newIssuer(issuer));
132         assertion.getAuthnStatements().add(authnStatement);
133         return assertion;
134     }
135 
136     /**
137      * New issuer.
138      *
139      * @param issuerValue the issuer
140      * @return the issuer
141      */
142     public Issuer newIssuer(final String issuerValue) {
143         final Issuer issuer = newSamlObject(Issuer.class);
144         issuer.setValue(issuerValue);
145         return issuer;
146     }
147 
148     /**
149      * New authn statement.
150      *
151      * @param contextClassRef the context class ref such as {@link AuthnContext#PASSWORD_AUTHN_CTX}
152      * @param authnInstant the authn instant
153      * @return the authn statement
154      */
155     public AuthnStatement newAuthnStatement(final String contextClassRef, final DateTime authnInstant) {
156         final AuthnStatement stmt = newSamlObject(AuthnStatement.class);
157         final AuthnContext ctx = newSamlObject(AuthnContext.class);
158 
159         final AuthnContextClassRef classRef = newSamlObject(AuthnContextClassRef.class);
160         classRef.setAuthnContextClassRef(contextClassRef);
161 
162         ctx.setAuthnContextClassRef(classRef);
163         stmt.setAuthnContext(ctx);
164         stmt.setAuthnInstant(authnInstant);
165 
166         return stmt;
167     }
168 
169     /**
170      * New conditions element.
171      *
172      * @param notBefore the not before
173      * @param notOnOrAfter the not on or after
174      * @param audienceUri the service id
175      * @return the conditions
176      */
177     public Conditions newConditions(final DateTime notBefore, final DateTime notOnOrAfter, final String audienceUri) {
178         final Conditions conditions = newSamlObject(Conditions.class);
179         conditions.setNotBefore(notBefore);
180         conditions.setNotOnOrAfter(notOnOrAfter);
181 
182         final AudienceRestriction audienceRestriction = newSamlObject(AudienceRestriction.class);
183         final Audience audience = newSamlObject(Audience.class);
184         audience.setAudienceURI(audienceUri);
185         audienceRestriction.getAudiences().add(audience);
186         conditions.getAudienceRestrictions().add(audienceRestriction);
187         return conditions;
188     }
189 
190     /**
191      * New subject element.
192      *
193      * @param nameIdFormat the name id format
194      * @param nameIdValue the name id value
195      * @param recipient the recipient
196      * @param notOnOrAfter the not on or after
197      * @param inResponseTo the in response to
198      * @return the subject
199      */
200     public Subject newSubject(final String nameIdFormat, final String nameIdValue,
201                               final String recipient, final DateTime notOnOrAfter,
202                               final String inResponseTo) {
203 
204         final SubjectConfirmation confirmation = newSamlObject(SubjectConfirmation.class);
205         confirmation.setMethod(SubjectConfirmation.METHOD_BEARER);
206 
207         final SubjectConfirmationData data = newSamlObject(SubjectConfirmationData.class);
208         data.setRecipient(recipient);
209         data.setNotOnOrAfter(notOnOrAfter);
210         data.setInResponseTo(inResponseTo);
211 
212         confirmation.setSubjectConfirmationData(data);
213 
214         final Subject subject = newSamlObject(Subject.class);
215         subject.setNameID(getNameID(nameIdFormat, nameIdValue));
216         subject.getSubjectConfirmations().add(confirmation);
217         return subject;
218     }

220     @Override
221     public String generateSecureRandomId() {
222         final SecureRandom generator = new SecureRandom();
223         final char[] charMappings = {
224                 'a', 'b', 'c', 'd', 'e', 'f', 'g',
225                 'h', 'i', 'j', 'k', 'l', 'm', 'n', 'o',
226                 'p'};
227 
228         final int charsLength = 40;
229         final int generatorBytesLength = 20;
230         final int shiftLength = 4;
231 
232         // 160 bits
233         final byte[] bytes = new byte[generatorBytesLength];
234         generator.nextBytes(bytes);
235 
236         final char[] chars = new char[charsLength];
237         for (int i = 0; i < bytes.length; i++) {
238             final int left = bytes[i] >> shiftLength & HEX_HIGH_BITS_BITWISE_FLAG;
239             final int right = bytes[i] & HEX_HIGH_BITS_BITWISE_FLAG;
240             chars[i * 2] = charMappings[left];
241             chars[i * 2 + 1] = charMappings[right];
242         }
243         return String.valueOf(chars);
244     }

246     /**
247      * Decode authn request xml.
248      *
249      * @param encodedRequestXmlString the encoded request xml string
250      * @return the request
251      */
252     public String decodeSamlAuthnRequest(final String encodedRequestXmlString) {
253         if (StringUtils.isEmpty(encodedRequestXmlString)) {
254             return null;
255         }
256 
257         final byte[] decodedBytes = CompressionUtils.decodeBase64ToByteArray(encodedRequestXmlString);
258         if (decodedBytes == null) {
259             return null;
260         }
261 
262         final String inflated = CompressionUtils.inflate(decodedBytes);
263         if (!StringUtils.isEmpty(inflated)) {
264             return inflated;
265         }
266 
267         return CompressionUtils.decodeByteArrayToString(decodedBytes);
268     }
