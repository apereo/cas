19  package org.jasig.cas.support.pac4j.authentication.handler.support;

21  import java.security.GeneralSecurityException;
22  import javax.security.auth.login.FailedLoginException;
23  import javax.validation.constraints.NotNull;
24  
25  import org.apache.commons.lang.StringUtils;
26  import org.jasig.cas.authentication.BasicCredentialMetaData;
27  import org.jasig.cas.authentication.HandlerResult;
28  import org.jasig.cas.authentication.PreventedException;
29  import org.jasig.cas.authentication.handler.support.AbstractPreAndPostProcessingAuthenticationHandler;
30  import org.jasig.cas.authentication.Credential;
31  import org.jasig.cas.authentication.principal.SimplePrincipal;
32  import org.jasig.cas.support.pac4j.authentication.principal.ClientCredential;
33  import org.pac4j.core.client.Client;
34  import org.pac4j.core.client.Clients;
35  import org.pac4j.core.profile.UserProfile;
36  
37  /**
38   * This handler authenticates the client credentials : it uses them to get the user profile returned by the provider
39   * for an authenticated user.
40   *
41   * @author Jerome Leleu
42   * @since 3.5.0
43   */
44  @SuppressWarnings("unchecked")
45  public final class ClientAuthenticationHandler extends AbstractPreAndPostProcessingAuthenticationHandler {
46  
47      /**
48       * The clients for authentication.
49       */
50      @NotNull
51      private final Clients clients;
52  
53      /**
54       * Define the clients.
55       *
56       * @param theClients The clients for authentication
57       */
58      public ClientAuthenticationHandler(final Clients theClients) {
59          this.clients = theClients;
60      }
61  
62      @Override
63      public boolean supports(final Credential credential) {
64          return credential != null && ClientCredential.class.isAssignableFrom(credential.getClass());
65      }
66  
67      @Override
68      protected HandlerResult doAuthentication(final Credential credential) throws GeneralSecurityException, PreventedException {
69          final ClientCredential clientCredentials = (ClientCredential) credential;
70          logger.debug("clientCredentials : {}", clientCredentials);
71  
72          final String clientName = clientCredentials.getCredentials().getClientName();
73          logger.debug("clientName : {}", clientName);
74  
75          // get client
76          final Client<org.pac4j.core.credentials.Credentials, UserProfile> client = this.clients.findClient(clientName);
77          logger.debug("client : {}", client);
78  
79          // get user profile
80          final UserProfile userProfile = client.getUserProfile(clientCredentials.getCredentials());
81          logger.debug("userProfile : {}", userProfile);
82  
83          if (userProfile != null && StringUtils.isNotBlank(userProfile.getId())) {
84              clientCredentials.setUserProfile(userProfile);
85              return new HandlerResult(
86                      this,
87                      new BasicCredentialMetaData(credential),
88                      new SimplePrincipal(userProfile.getId(), userProfile.getAttributes()));
89          }
90  
91          throw new FailedLoginException("Provider did not produce profile for " + clientCredentials);
92      }
93  }
