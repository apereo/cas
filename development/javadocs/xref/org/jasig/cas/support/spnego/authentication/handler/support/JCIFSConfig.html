88      @Override
89      public void afterPropertiesSet() throws Exception {
90          if (System.getProperty(SYS_PROP_LOGIN_CONF) != null) {
91              logger.warn("found login config in system property, may overide : "
92                      + System.getProperty(SYS_PROP_LOGIN_CONF));
93          }
94  
95          URL url = getClass().getResource(
96                  this.loginConf == null ? DEFAULT_LOGIN_CONFIG : this.loginConf);
97          if (url != null) {
98              this.loginConf = url.toExternalForm();
99          }
100         if (this.loginConf != null) {
101             System.setProperty(SYS_PROP_LOGIN_CONF, this.loginConf);
102         } else {
103             url = getClass().getResource("/jcifs/http/login.conf");
104             if (url != null) {
105                 System.setProperty(SYS_PROP_LOGIN_CONF, url.toExternalForm());
106             }
107         }
108         logger.debug("configured login configuration path : "
109                 + System.getProperty(SYS_PROP_LOGIN_CONF));
110     }
