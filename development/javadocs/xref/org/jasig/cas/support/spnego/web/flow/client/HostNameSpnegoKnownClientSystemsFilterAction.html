20  package org.jasig.cas.support.spnego.web.flow.client;
21  
22  import javax.validation.constraints.NotNull;
23  import java.util.regex.Pattern;
24  
25  /**
26   * A simple implementation of {@link BaseSpnegoKnownClientSystemsFilterAction} to allow / skip SPNEGO / KRB /
27   * NTLM authentication based on a regex match against a reverse DNS lookup of the requesting
28   * system.
29   *
30   * @author Sean Baker
31   * @author Misagh Moayyed
32   * @since 4.1
33   */
34  public class HostNameSpnegoKnownClientSystemsFilterAction extends BaseSpnegoKnownClientSystemsFilterAction {
35      private final Pattern hostNamePatternString;
36  
37      /**
38       * Instantiates a new hostname spnego known client systems filter action.
39       *
40       * @param hostNamePatternString the host name pattern string.
41       *                              The pattern to match the retrieved hostname against.
42       */
43      public HostNameSpnegoKnownClientSystemsFilterAction(@NotNull final String hostNamePatternString) {
44          super();
45          this.hostNamePatternString = Pattern.compile(hostNamePatternString);
46      }
47  
48      /**
49       * {@inheritDoc}.
50       * <p>
51       * Checks whether the IP should even be paid attention to,
52       * then does a reverse DNS lookup, and if it matches the supplied pattern, performs SPNEGO
53       * else skips the process.
54       *
55       * @param remoteIp The remote ip address to validate
56       */
57      @Override
58      protected boolean shouldDoSpnego(final String remoteIp) {
59          final boolean ipCheck = ipPatternCanBeChecked(remoteIp);
60          if(ipCheck && !ipPatternMatches(remoteIp)) {
61              return false;
62          }
63          final String hostName = getRemoteHostName(remoteIp);
64          logger.debug("Retrieved host name for the remote ip is {}", hostName);
65          return this.hostNamePatternString.matcher(hostName).find();
66      }
67  }
