37  /**
38   * Implementation of an AuthenticationHandler for SPNEGO supports. This Handler
39   * support both NTLM and Kerberos. NTLM is disabled by default.
40   *
41   * @author Arnaud Lesueur
42   * @author Marc-Antoine Garrigue
43   * @author Scott Battaglia
44   * @author Marvin S. Addison
45   * @deprecated As of 4.1, the class name is abbreviated in a way that is not per camel-casing standards and will be renamed in the future.
46   * @since 3.1
47   */
48  @Deprecated
49  public final class JCIFSSpnegoAuthenticationHandler extends AbstractPreAndPostProcessingAuthenticationHandler {
50  
51      private Authentication authentication;
52  
53      /**
54       * Principal contains the DomainName ? (true by default).
55       */
56      private boolean principalWithDomainName = true;
57  
58      /**
59       * Allow SPNEGO/NTLM Token as valid credentials. (false by default)
60       */
61      private boolean isNTLMallowed;
62  
63      @Override
64      protected HandlerResult doAuthentication(final Credential credential) throws GeneralSecurityException, PreventedException {
65          final SpnegoCredential spnegoCredential = (SpnegoCredential) credential;
66          java.security.Principal principal;
67          byte[] nextToken;
68          if (!this.isNTLMallowed && spnegoCredential.isNtlm()) {
69              throw new FailedLoginException("NTLM not allowed");
70          }
71          try {
72              // proceed authentication using jcifs
73              synchronized (this) {
74                  this.authentication.reset();
75                  this.authentication.process(spnegoCredential.getInitToken());
76                  principal = this.authentication.getPrincipal();
77                  nextToken = this.authentication.getNextToken();
78              }
79          } catch (final jcifs.spnego.AuthenticationException e) {
80              throw new FailedLoginException(e.getMessage());
81          }
82  
83          // evaluate jcifs response
84          if (nextToken != null) {
85              logger.debug("Setting nextToken in credential");
86              spnegoCredential.setNextToken(nextToken);
87          } else {
88              logger.debug("nextToken is null");
89          }
90  
91          boolean success = false;
92          if (principal != null) {
93              if (spnegoCredential.isNtlm()) {
94                  logger.debug("NTLM Credential is valid for user [{}]", principal.getName());
95              } else {
96                  logger.debug("Kerberos Credential is valid for user [{}]", principal.getName());
97              }
98              spnegoCredential.setPrincipal(getPrincipal(principal.getName(), spnegoCredential.isNtlm()));
99              success = true;
100         }
101 
102         if (!success) {
103             throw new FailedLoginException("Principal is null, the processing of the SPNEGO Token failed");
104         }
105         return new DefaultHandlerResult(this, new BasicCredentialMetaData(credential), spnegoCredential.getPrincipal());
106     }
107 
108     @Override
109     public boolean supports(final Credential credential) {
110         return credential instanceof SpnegoCredential;
111     }
112 
113     public void setAuthentication(final Authentication authentication) {
114         this.authentication = authentication;
115     }
116 
117     public void setPrincipalWithDomainName(final boolean principalWithDomainName) {
118         this.principalWithDomainName = principalWithDomainName;
119     }
120 
121     public void setNTLMallowed(final boolean isNTLMallowed) {
122         this.isNTLMallowed = isNTLMallowed;
123     }
124 
125     /**
126      * @deprecated As of 4.1. Use {@link #getPrincipal(String, boolean)}
127      * Gets the simple principal from the given name.
128      *
129      * @param name the name
130      * @param isNtlm the is ntlm
131      * @return the simple principal
132      */
133     @Deprecated
134     protected SimplePrincipal getSimplePrincipal(final String name, final boolean isNtlm) {
135         logger.warn("getSimplePrincipal() is deprecated and will be removed. Consider getPrincipal() instead.");
136 
137         if (this.principalWithDomainName) {
138             return (SimplePrincipal) new DefaultPrincipalFactory().createPrincipal(name);
139         }
140         if (isNtlm) {
141             return Pattern.matches("\\S+\\\\\\S+", name)
142                     ? (SimplePrincipal) new DefaultPrincipalFactory().createPrincipal(name.split("\\\\")[1])
143                     : (SimplePrincipal) new DefaultPrincipalFactory().createPrincipal(name);
144         }
145         return (SimplePrincipal) new DefaultPrincipalFactory().createPrincipal(name.split("@")[0]);
146     }
147 
148     /**
149      * Gets the principal from the given name. The principal
150      * is created by the factory instance.
151      *
152      * @param name the name
153      * @param isNtlm the is ntlm
154      * @return the simple principal
155      */
156     protected Principal getPrincipal(final String name, final boolean isNtlm) {
157         if (this.principalWithDomainName) {
158             return this.principalFactory.createPrincipal(name);
159         }
160         if (isNtlm) {
161             return Pattern.matches("\\S+\\\\\\S+", name)
162                     ? this.principalFactory.createPrincipal(name.split("\\\\")[1])
163                     : this.principalFactory.createPrincipal(name);
164         }
165         return this.principalFactory.createPrincipal(name.split("@")[0]);
166     }
