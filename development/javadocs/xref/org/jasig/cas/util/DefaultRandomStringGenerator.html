19  package org.jasig.cas.util;
20  
21  import java.security.SecureRandom;
22  
23  /**
24   * Implementation of the RandomStringGenerator that allows you to define the
25   * length of the random part.
26   *
27   * @author Scott Battaglia
28  
29   * @since 3.0.0
30   */
31  public final class DefaultRandomStringGenerator implements RandomStringGenerator {
32  
33      /** The default maximum length. */
34      protected static final int DEFAULT_MAX_RANDOM_LENGTH = 35;
35  
36      /** The array of printable characters to be used in our random string. */
37      private static final char[] PRINTABLE_CHARACTERS = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ012345679"
38          .toCharArray();
39  
40      /** An instance of secure random to ensure randomness is secure. */
41      private final SecureRandom randomizer = new SecureRandom();
42  
43      /** The maximum length the random string can be. */
44      private final int maximumRandomLength;
45  
46      /**
47       * Instantiates a new default random string generator
48       * with length set to {@link #DEFAULT_MAX_RANDOM_LENGTH}.
49       */
50      public DefaultRandomStringGenerator() {
51          this.maximumRandomLength = DEFAULT_MAX_RANDOM_LENGTH;
52      }
53  
54      /**
55       * Instantiates a new default random string generator.
56       *
57       * @param maxRandomLength the max random length
58       */
59      public DefaultRandomStringGenerator(final int maxRandomLength) {
60          this.maximumRandomLength = maxRandomLength;
61      }
62  
63      public int getMinLength() {
64          return this.maximumRandomLength;
65      }
66  
67      public int getMaxLength() {
68          return this.maximumRandomLength;
69      }
70  
71      @Override
72      public String getNewString() {
73          final byte[] random = getNewStringAsBytes();
74  
75          return convertBytesToString(random);
76      }
77  
78      @Override
79      public byte[] getNewStringAsBytes() {
80          final byte[] random = new byte[this.maximumRandomLength];
81  
82          this.randomizer.nextBytes(random);
83  
84          return random;
85      }
86  
87      /**
88       * Convert bytes to string, taking into account {@link #PRINTABLE_CHARACTERS}.
89       *
90       * @param random the random
91       * @return the string
92       */
93      private String convertBytesToString(final byte[] random) {
94          final char[] output = new char[random.length];
95          for (int i = 0; i < random.length; i++) {
96              final int index = Math.abs(random[i] % PRINTABLE_CHARACTERS.length);
97              output[i] = PRINTABLE_CHARACTERS[index];
98          }
99  
100         return new String(output);
101     }
102 }
