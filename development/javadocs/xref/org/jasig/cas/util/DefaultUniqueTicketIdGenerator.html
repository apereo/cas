19  package org.jasig.cas.util;
20  
21  import org.apache.commons.lang3.StringUtils;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  /**
26   * Default implementation of {@link UniqueTicketIdGenerator}. Implementation
27   * utilizes a DefaultLongNumericGeneraor and a DefaultRandomStringGenerator to
28   * construct the ticket id.
29   * <p>
30   * Tickets are of the form [PREFIX]-[SEQUENCE NUMBER]-[RANDOM STRING]-[SUFFIX]
31   * </p>
32   *
33   * @author Scott Battaglia
34   * @since 3.0.0
35   */
36  public class DefaultUniqueTicketIdGenerator implements UniqueTicketIdGenerator {
37  
38      /** The logger instance. */
39      protected final Logger logger = LoggerFactory.getLogger(this.getClass());
40  
41      /** The numeric generator to generate the static part of the id. */
42      private final NumericGenerator numericGenerator;
43  
44      /** The RandomStringGenerator to generate the secure random part of the id. */
45      private final RandomStringGenerator randomStringGenerator;
46  
47      /**
48       * Optional suffix to ensure uniqueness across JVMs by specifying unique
49       * values.
50       */
51      private final String suffix;
52  
53      /**
54       * Creates an instance of DefaultUniqueTicketIdGenerator with default values
55       * including a {@link DefaultLongNumericGenerator} with a starting value of
56       * 1.
57       */
58      public DefaultUniqueTicketIdGenerator() {
59          this(DefaultRandomStringGenerator.DEFAULT_MAX_RANDOM_LENGTH);
60      }
61  
62      /**
63       * Creates an instance of DefaultUniqueTicketIdGenerator with a specified
64       * maximum length for the random portion.
65       *
66       * @param maxLength the maximum length of the random string used to generate
67       * the id.
68       */
69      public DefaultUniqueTicketIdGenerator(final int maxLength) {
70          this(maxLength, null);
71      }
72  
73      /**
74       * Creates an instance of DefaultUniqueTicketIdGenerator with a specified
75       * maximum length for the random portion.
76       *
77       * @param numericGenerator the numeric generator
78       * @param randomStringGenerator the random string generator
79       * @param suffix the value to append at the end of the unique id to ensure
80       * uniqueness across JVMs.
81       * @since 4.1.0
82       */
83      public DefaultUniqueTicketIdGenerator(final NumericGenerator numericGenerator,
84                                            final RandomStringGenerator randomStringGenerator,
85                                            final String suffix) {
86  
87          this.randomStringGenerator = randomStringGenerator;
88          this.numericGenerator = numericGenerator;
89          this.suffix = StringUtils.isNoneBlank(suffix) ? '-' + suffix : null;
90      }
91  
92      /**
93       * Creates an instance of DefaultUniqueTicketIdGenerator with a specified
94       * maximum length for the random portion.
95       *
96       * @param maxLength the maximum length of the random string used to generate
97       * the id.
98       * @param suffix the value to append at the end of the unique id to ensure
99       * uniqueness across JVMs.
100      */
101     public DefaultUniqueTicketIdGenerator(final int maxLength, final String suffix) {
102         this(new DefaultLongNumericGenerator(1), new DefaultRandomStringGenerator(maxLength), suffix);
103     }
104 
105 
106 
107     @Override
108     public final String getNewTicketId(final String prefix) {
109         final String number = this.numericGenerator.getNextNumberAsString();
110         final StringBuilder buffer = new StringBuilder(prefix.length() + 2
111             + (StringUtils.isNotBlank(this.suffix) ? this.suffix.length() : 0) + this.randomStringGenerator.getMaxLength()
112             + number.length());
113 
114         buffer.append(prefix);
115         buffer.append('-');
116         buffer.append(number);
117         buffer.append('-');
118         buffer.append(this.randomStringGenerator.getNewString());
119 
120         if (this.suffix != null) {
121             buffer.append(this.suffix);
122         }
123 
124         return buffer.toString();
125     }
126 }
