108     /**
109      * Sends a message to a particular endpoint.  Option of sending it without
110      * waiting to ensure a response was returned.
111      * <p>
112      * This is useful when it doesn't matter about the response as you'll perform no action based on the response.
113      *
114      * @param url the url to send the message to
115      * @param message the message itself
116      * @param async true if you don't want to wait for the response, false otherwise.
117      * @return boolean if the message was sent, or async was used.  false if the message failed.
118      */
119     public boolean sendMessageToEndPoint(final String url, final String message, final boolean async) {
120         final Future<Boolean> result = EXECUTOR_SERVICE.submit(new MessageSender(url, message,
121                 this.readTimeout, this.connectionTimeout, this.followRedirects));
122 
123         if (async) {
124             return true;
125         }
126 
127         try {
128             return result.get();
129         } catch (final Exception e) {
130             return false;
131         }
132     }

150     /**
151      * Make a synchronous HTTP(S) call to ensure that the url is reachable.
152      *
153      * @param url the url to call
154      * @return whether the url is valid
155      */
156     public boolean isValidEndPoint(final URL url) {
157         HttpURLConnection connection = null;
158         InputStream is = null;
159         try {
160             connection = (HttpURLConnection) url.openConnection();
161             connection.setConnectTimeout(this.connectionTimeout);
162             connection.setReadTimeout(this.readTimeout);
163             connection.setInstanceFollowRedirects(this.followRedirects);
164 
165             if (connection instanceof HttpsURLConnection) {
166                 final HttpsURLConnection httpsConnection = (HttpsURLConnection) connection;
167 
168                 if (this.sslSocketFactory != null) {
169                     httpsConnection.setSSLSocketFactory(this.sslSocketFactory);
170                 }
171 
172                 if (this.hostnameVerifier != null) {
173                     httpsConnection.setHostnameVerifier(this.hostnameVerifier);
174                 }
175             }
176 
177             connection.connect();
178 
179             final int responseCode = connection.getResponseCode();
180 
181             for (final int acceptableCode : this.acceptableCodes) {
182                 if (responseCode == acceptableCode) {
183                     LOGGER.debug("Response code from server matched {}.", responseCode);
184                     return true;
185                 }
186             }
187 
188             LOGGER.debug("Response Code did not match any of the acceptable response codes. Code returned was {}",
189                     responseCode);
190 
191             // if the response code is an error and we don't find that error acceptable above:
192             if (responseCode == 500) {
193                 is = connection.getInputStream();
194                 final String value = IOUtils.toString(is);
195                 LOGGER.error("There was an error contacting the endpoint: {}; The error was:\n{}", url.toExternalForm(),
196                         value);
197             }
198         } catch (final IOException e) {
199             LOGGER.error(e.getMessage(), e);
200         } finally {
201             IOUtils.closeQuietly(is);
202             if (connection != null) {
203                 connection.disconnect();
204             }
205         }
206         return false;
207     }

272     private static final class MessageSender implements Callable<Boolean> {
273 
274         private String url;
275 
276         private String message;
277 
278         private int readTimeout;
279 
280         private int connectionTimeout;
281 
282         private boolean followRedirects;
283 
284         public MessageSender(final String url, final String message, final int readTimeout,
285                 final int connectionTimeout, final boolean followRedirects) {
286             this.url = url;
287             this.message = message;
288             this.readTimeout = readTimeout;
289             this.connectionTimeout = connectionTimeout;
290             this.followRedirects = followRedirects;
291         }
292 
293         public Boolean call() throws Exception {
294             HttpURLConnection connection = null;
295             BufferedReader in = null;
296             try {
297                 LOGGER.debug("Attempting to access {}", url);
298                 final URL logoutUrl = new URL(url);
299                 final String output = "logoutRequest=" + URLEncoder.encode(message, "UTF-8");
300 
301                 connection = (HttpURLConnection) logoutUrl.openConnection();
302                 connection.setDoInput(true);
303                 connection.setDoOutput(true);
304                 connection.setRequestMethod("POST");
305                 connection.setReadTimeout(this.readTimeout);
306                 connection.setConnectTimeout(this.connectionTimeout);
307                 connection.setInstanceFollowRedirects(this.followRedirects);
308                 connection.setRequestProperty("Content-Length", Integer.toString(output.getBytes().length));
309                 connection.setRequestProperty("Content-Type", "application/x-www-form-urlencoded");
310                 final DataOutputStream printout = new DataOutputStream(connection.getOutputStream());
311                 printout.writeBytes(output);
312                 printout.flush();
313                 printout.close();
314 
315                 in = new BufferedReader(new InputStreamReader(connection.getInputStream()));
316 
317                 boolean readInput = true;
318                 while (readInput) {
319                     readInput =StringUtils.isNotBlank(in.readLine());
320                 }
321 
322                 LOGGER.debug("Finished sending message to {}", url);
323                 return true;
324             } catch (final SocketTimeoutException e) {
325                 LOGGER.warn("Socket Timeout Detected while attempting to send message to [{}]", url);
326                 return false;
327             } catch (final Exception e) {
328                 LOGGER.warn("Error Sending message to url endpoint [{}]. Error is [{}]", url, e.getMessage());
329                 return false;
330             } finally {
331                 IOUtils.closeQuietly(in);
332                 if (connection != null) {
333                     connection.disconnect();
334                 }
335             }
336         }
337
