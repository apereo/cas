335     @Override
336     public boolean sendMessageToEndPoint(@NotNull final HttpMessage message) {
337         Assert.notNull(this.httpClient);
338 
339         FutureRequestExecutionService service = null;
340         try {
341             final HttpPost request = new HttpPost(message.getUrl().toURI());
342             request.addHeader("Content-Type", message.getContentType());
343             
344             final StringEntity entity = new StringEntity(message.getMessage(), ContentType.create(message.getContentType()));
345             request.setEntity(entity);
346 
347             service = new FutureRequestExecutionService(this.httpClient, executorService);
348 
349             final HttpRequestFutureTask<String> task = service.execute(request,
350                     HttpClientContext.create(), new BasicResponseHandler());
351 
352             if (message.isAsynchronous()) {
353                 return true;
354             }
355             
356             return StringUtils.isNotBlank(task.get());
357         } catch (final Exception e) {
358             LOGGER.trace(e.getMessage(), e);
359             return false;
360         } finally {
361             IOUtils.closeQuietly(service);
362         }
363     }

376     @Override
377     public boolean isValidEndPoint(final URL url) {
378         Assert.notNull(this.httpClient);
379 
380         CloseableHttpResponse response = null;
381         HttpEntity entity = null;
382 
383         try {
384             final HttpGet request = new HttpGet(url.toURI());
385 
386             response = this.httpClient.execute(request);
387             final int responseCode = response.getStatusLine().getStatusCode();
388 
389             for (final int acceptableCode : this.acceptableCodes) {
390                 if (responseCode == acceptableCode) {
391                     LOGGER.debug("Response code from server matched {}.", responseCode);
392                     return true;
393                 }
394             }
395 
396             LOGGER.debug("Response code did not match any of the acceptable response codes. Code returned was {}",
397                     responseCode);
398 
399             if (responseCode == HttpStatus.SC_INTERNAL_SERVER_ERROR) {
400                 final String value = response.getStatusLine().getReasonPhrase();
401                 LOGGER.error("There was an error contacting the endpoint: {}; The error was:\n{}", url.toExternalForm(),
402                         value);
403             }
404 
405             entity = response.getEntity();
406         } catch (final Exception e) {
407             LOGGER.error(e.getMessage(), e);
408         } finally {
409             EntityUtils.consumeQuietly(entity);
410             IOUtils.closeQuietly(response);
411            
412         }
413         return false;
414     }

504     /**
505      * Prepare the http client with configured settings.
506      */
507     private void init() {
508         try {
509 
510             final ConnectionSocketFactory plainsf = PlainConnectionSocketFactory.getSocketFactory();
511             final LayeredConnectionSocketFactory sslsf = this.sslSocketFactory;
512 
513             final Registry<ConnectionSocketFactory> registry = RegistryBuilder.<ConnectionSocketFactory>create()
514                     .register("http", plainsf)
515                     .register("https", sslsf)
516                     .build();
517 
518             final PoolingHttpClientConnectionManager connMgmr = new PoolingHttpClientConnectionManager(registry);
519             connMgmr.setMaxTotal(this.maxPooledConnections);
520             connMgmr.setDefaultMaxPerRoute(this.maxConnectionsPerRoute);
521 
522             final HttpHost httpHost = new HttpHost(InetAddress.getLocalHost());
523             final HttpRoute httpRoute = new HttpRoute(httpHost);
524             connMgmr.setMaxPerRoute(httpRoute, MAX_CONNECTIONS_PER_ROUTE);
525     
526             final RequestConfig requestConfig = RequestConfig.custom()
527                     .setSocketTimeout(this.readTimeout)
528                     .setConnectTimeout(this.connectionTimeout)
529                     .setConnectionRequestTimeout(this.connectionTimeout)
530                     .setStaleConnectionCheckEnabled(true)
531                     .setCircularRedirectsAllowed(this.circularRedirectsAllowed)
532                     .setRedirectsEnabled(this.redirectsEnabled)
533                     .setAuthenticationEnabled(this.authenticationEnabled)
534                     .build();
535             
536             final HttpClientBuilder builder = HttpClients.custom()
537                     .setConnectionManager(connMgmr)
538                     .setDefaultRequestConfig(requestConfig)
539                     .setSSLSocketFactory(this.sslSocketFactory)
540                     .setHostnameVerifier(this.hostnameVerifier)
541                     .setRedirectStrategy(this.redirectionStrategy)
542                     .setDefaultCredentialsProvider(this.credentialsProvider)
543                     .setDefaultCookieStore(this.cookieStore)
544                     .setConnectionReuseStrategy(this.connectionReuseStrategy)
545                     .setConnectionBackoffStrategy(this.connectionBackoffStrategy)
546                     .setServiceUnavailableRetryStrategy(this.serviceUnavailableRetryStrategy)
547                     .setProxyAuthenticationStrategy(this.proxyAuthenticationStrategy)
548                     .setDefaultHeaders(this.defaultHeaders)
549                     .useSystemProperties();
550 
551 
552             this.httpClient = builder.build();
553         } catch (final Exception e) {
554             LOGGER.error(e.getMessage(), e);
555         }
556     }
