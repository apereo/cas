19  package org.jasig.cas;
20  
21  import com.google.common.collect.ImmutableList;
22  import org.apache.commons.lang3.builder.HashCodeBuilder;
23  import org.springframework.util.Assert;
24  
25  import java.io.Serializable;
26  import java.util.Arrays;
27  
28  /**
29   * Simple parameterized message descriptor with a code that refers to a message bundle key and a default
30   * message string to use if no message code can be resolved.
31   *
32   * @author Marvin S. Addison
33   * @since 4.0.0
34   */
35  public class DefaultMessageDescriptor implements MessageDescriptor {
36  
37      /** Serialization support. */
38      private static final long serialVersionUID = 1227390629186486032L;
39  
40      private final String code;
41  
42      private final String defaultMessage;
43  
44      private final Serializable[] params;
45  
46      /**
47       * Instantiates a new message.
48       *
49       * @param code the code
50       */
51      public DefaultMessageDescriptor(final String code) {
52          this(code, code);
53      }
54  
55      /**
56       * Instantiates a new message.
57       *
58       * @param code the code
59       * @param defaultMessage the default message
60       * @param params the params
61       */
62      public DefaultMessageDescriptor(final String code, final String defaultMessage, final Serializable... params) {
63          Assert.hasText(code, "Code cannot be null or empty");
64          Assert.hasText(defaultMessage, "Default message cannot be null or empty");
65          this.code = code;
66          this.defaultMessage = defaultMessage;
67          this.params = params;
68      }
69  
70      public String getCode() {
71          return this.code;
72      }
73  
74      public String getDefaultMessage() {
75          return this.defaultMessage;
76      }
77  
78      /**
79       * Get parameters for the message.
80       *
81       * @return the serializable [ ]
82       */
83      public Serializable[] getParams() {
84          if (this.params == null) {
85              return null;
86          }
87          return ImmutableList.copyOf(this.params).toArray(new Serializable[this.params.length]);
88      }
89  
90      @Override
91      public int hashCode() {
92          final HashCodeBuilder builder = new HashCodeBuilder(99, 31);
93          builder.append(this.code);
94          builder.append(this.defaultMessage);
95          builder.append(this.params);
96          return builder.toHashCode();
97      }
98  
99      @Override
100     public boolean equals(final Object other) {
101         if (other == null || !(other instanceof DefaultMessageDescriptor)) {
102             return false;
103         }
104         if (other == this) {
105             return true;
106         }
107         final DefaultMessageDescriptor m = (DefaultMessageDescriptor) other;
108         return this.code.equals(m.getCode())
109                 && this.defaultMessage.equals(m.getDefaultMessage())
110                 && Arrays.equals(this.params, m.getParams());
111     }
112 }
