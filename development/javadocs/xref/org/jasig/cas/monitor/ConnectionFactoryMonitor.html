19  package org.jasig.cas.monitor;
20  
21  import org.jasig.cas.util.LdapUtils;
22  import org.ldaptive.Connection;
23  import org.ldaptive.ConnectionFactory;
24  import org.ldaptive.LdapException;
25  import org.ldaptive.pool.Validator;
26  
27  /**
28   * Monitors an ldaptive {@link ConnectionFactory}.  While this class can be used with instances of
29   * {@link org.ldaptive.pool.PooledConnectionFactory}, the {@link PooledConnectionFactoryMonitor} class is preferable.
30   *
31   * @author Marvin S. Addison
32   * @since 4.0.0
33   */
34  public class ConnectionFactoryMonitor extends AbstractNamedMonitor<Status> {
35  
36      /** OK status. */
37      private static final Status OK = new Status(StatusCode.OK);
38  
39      /** Error status. */
40      private static final Status ERROR = new Status(StatusCode.ERROR);
41  
42      /** Source of connections to validate. */
43      private final ConnectionFactory connectionFactory;
44  
45      /** Connection validator. */
46      private final Validator<Connection> validator;
47  
48  
49      /**
50       * Creates a new instance that monitors the given connection factory.
51       *
52       * @param  factory  Connection factory to monitor.
53       * @param  validator  Validates connections from the factory.
54       */
55      public ConnectionFactoryMonitor(final ConnectionFactory factory, final Validator<Connection> validator) {
56          this.connectionFactory = factory;
57          this.validator = validator;
58      }
59  
60  
61      /**
62       * Gets a connection from the underlying connection factory and attempts to validate it.
63       *
64       * @return  Status with code {@link StatusCode#OK} on success otherwise {@link StatusCode#ERROR}.
65       */
66      @Override
67      public Status observe() {
68          Connection conn = null;
69          try {
70              conn = this.connectionFactory.getConnection();
71              if (!conn.isOpen()) {
72                  conn.open();
73              }
74              return this.validator.validate(conn) ? OK : ERROR;
75          } catch (final LdapException e) {
76              logger.warn("Validation failed with error.", e);
77          } finally {
78              LdapUtils.closeConnection(conn);
79          }
80          return ERROR;
81      }
82  }
