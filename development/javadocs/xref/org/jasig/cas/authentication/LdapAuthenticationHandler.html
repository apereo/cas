48  /**
49   * LDAP authentication handler that uses the ldaptive <code>Authenticator</code> component underneath.

170     @Override
171     public HandlerResult authenticate(final Credential credential) throws GeneralSecurityException,
172                                 PreventedException {
173         final AuthenticationResponse response;
174         final UsernamePasswordCredential upc = (UsernamePasswordCredential) credential;
175         try {
176             logger.debug("Attempting LDAP authentication for {}", credential);
177             final AuthenticationRequest request = new AuthenticationRequest(upc.getUsername(),
178                     new org.ldaptive.Credential(upc.getPassword()),
179                     this.authenticatedEntryAttributes);
180             response = this.authenticator.authenticate(request);
181         } catch (final LdapException e) {
182             throw new PreventedException("Unexpected LDAP error", e);
183         }
184         logger.debug("LDAP response: {}", response);
185 
186         final List<Message> messageList;
187         if (this.ldapPasswordPolicyConfiguration != null) {
188             logger.debug("Applying password policy to {}", response);
189             messageList = this.ldapPasswordPolicyConfiguration.getAccountStateHandler().handle(
190                     response, ldapPasswordPolicyConfiguration);
191         } else {
192             messageList = Collections.emptyList();
193         }
194         if (response.getResult()) {
195             return new HandlerResult(
196                     this,
197                     new BasicCredentialMetaData(credential),
198                     createPrincipal(upc.getUsername(), response.getLdapEntry()),
199                     messageList);
200         }
201 
202         if (AuthenticationResultCode.DN_RESOLUTION_FAILURE == response.getAuthenticationResultCode()) {
203             throw new AccountNotFoundException(upc.getUsername() + " not found.");
204         }
205         throw new FailedLoginException("Invalid credentials.");
206     }

218     /**
219      * Creates a CAS principal with attributes if the LDAP entry contains principal attributes.
220      *
221      * @param username Username that was successfully authenticated which is used for principal ID when
222      *                 {@link #setPrincipalIdAttribute(String)} is not specified.
223      * @param ldapEntry LDAP entry that may contain principal attributes.
224      *
225      * @return Principal if the LDAP entry contains at least a principal ID attribute value, null otherwise.
226      *
227      * @throws LoginException On security policy errors related to principal creation.
228      */
229     protected Principal createPrincipal(final String username, final LdapEntry ldapEntry) throws LoginException {
230         final String id;
231         if (this.principalIdAttribute != null) {
232             final LdapAttribute principalAttr = ldapEntry.getAttribute(this.principalIdAttribute);
233             if (principalAttr == null || principalAttr.size() == 0) {
234                 throw new LoginException(this.principalIdAttribute + " attribute not found for " + username);
235             }
236             if (principalAttr.size() > 1) {
237                 if (this.allowMultiplePrincipalAttributeValues) {
238                     logger.warn(
239                             "Found multiple values for principal ID attribute: {}. Using first value={}.",
240                             principalAttr,
241                             principalAttr.getStringValue());
242                 } else {
243                     throw new LoginException("Multiple principal values not allowed: " + principalAttr);
244                 }
245             }
246             id = principalAttr.getStringValue();
247         } else {
248             id = username;
249         }
250         final Map<String, Object> attributeMap = new LinkedHashMap<String, Object>(this.principalAttributeMap.size());
251         for (String ldapAttrName : this.principalAttributeMap.keySet()) {
252             final LdapAttribute attr = ldapEntry.getAttribute(ldapAttrName);
253             if (attr != null) {
254                 logger.debug("Found principal attribute: {}", attr);
255                 final String principalAttrName = this.principalAttributeMap.get(ldapAttrName);
256                 if (attr.size() > 1) {
257                     attributeMap.put(principalAttrName, attr.getStringValues());
258                 } else {
259                     attributeMap.put(principalAttrName, attr.getStringValue());
260                 }
261             }
262         }
263         return new SimplePrincipal(id, attributeMap);
264     }

266     @PostConstruct
267     public void initialize() {
268         final List<String> attributes = new ArrayList<String>();
269         if (this.principalIdAttribute != null) {
270             attributes.add(this.principalIdAttribute);
271         }
272         attributes.addAll(this.principalAttributeMap.keySet());
273         attributes.addAll(this.additionalAttributes);
274         this.authenticatedEntryAttributes = attributes.toArray(new String[attributes.size()]);
275     }
