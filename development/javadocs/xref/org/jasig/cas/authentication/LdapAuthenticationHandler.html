49  /**
50   * LDAP authentication handler that uses the ldaptive <code>Authenticator</code> component underneath.
51   * This handler provides simple attribute resolution machinery by reading attributes from the entry
52   * corresponding to the DN of the bound user (in the bound security context) upon successful authentication.
53   * Principal resolution is controlled by the following properties:
54   *
55   * <ul>
56   *     <li>{@link #setPrincipalIdAttribute(String)}</li>
57   *     <li>{@link #setPrincipalAttributeMap(java.util.Map)}</li>
58   * </ul>
59   *
60   * @author Marvin S. Addison
61   * @since 4.0.0
62   */
63  public class LdapAuthenticationHandler extends AbstractUsernamePasswordAuthenticationHandler {
64  
65      /** Mapping of LDAP attribute name to principal attribute name. */
66      @NotNull
67      protected Map<String, String> principalAttributeMap = Collections.emptyMap();
68  
69      /** List of additional attributes to be fetched but are not principal attributes. */
70      @NotNull
71      protected List<String> additionalAttributes = Collections.emptyList();
72  
73      /**
74       * Performs LDAP authentication given username/password.
75       **/
76      @NotNull
77      private final Authenticator authenticator;
78  
79      /** Component name. */
80      @NotNull
81      private String name = LdapAuthenticationHandler.class.getSimpleName();
82  
83      /** Name of attribute to be used for resolved principal. */
84      private String principalIdAttribute;
85  
86      /** Flag indicating whether multiple values are allowed fo principalIdAttribute. */
87      private boolean allowMultiplePrincipalAttributeValues;
88  
89      /** Set of LDAP attributes fetch from an entry as part of the authentication process. */
90      private String[] authenticatedEntryAttributes = ReturnAttributes.NONE.value();

120     /**
121      * Sets a flag that determines whether multiple values are allowed for the {@link #principalIdAttribute}.
122      * This flag only has an effect if {@link #principalIdAttribute} is configured. If multiple values are detected
123      * when the flag is false, the first value is used and a warning is logged. If multiple values are detected
124      * when the flag is true, an exception is raised.
125      *
126      * @param allowed True to allow multiple principal ID attribute values, false otherwise.

128     public void setAllowMultiplePrincipalAttributeValues(final boolean allowed) {
129         this.allowMultiplePrincipalAttributeValues = allowed;
130     }

132     /**
133      * Sets the mapping of additional principal attributes where the key is the LDAP attribute
134      * name and the value is the principal attribute name. The key set defines the set of
135      * attributes read from the LDAP entry at authentication time. Note that the principal ID attribute
136      * should not be listed among these attributes.
137      *
138      * @param attributeNameMap Map of LDAP attribute name to principal attribute name.
139      */
140     public void setPrincipalAttributeMap(final Map<String, String> attributeNameMap) {
141         this.principalAttributeMap = attributeNameMap;
142     }
143 
144     /**
145      * Sets the mapping of additional principal attributes where the key and value is the LDAP attribute
146      * name. Note that the principal ID attribute
147      * should not be listed among these attributes.
148      *
149      * @param attributeList List of LDAP attribute names
150      */
151     public void setPrincipalAttributeList(final List<String> attributeList) {
152         this.principalAttributeMap = Maps.uniqueIndex(attributeList, Functions.toStringFunction());
153     }

155     /**
156      * Sets the list of additional attributes to be fetched from the user entry during authentication.
157      * These attributes are <em>not</em> bound to the principal.
158      * <p>
159      * A common use case for these attributes is to support password policy machinery.
160      *
161      * @param additionalAttributes List of operational attributes to fetch when resolving an entry.
162      */
163     public void setAdditionalAttributes(final List<String> additionalAttributes) {
164         this.additionalAttributes = additionalAttributes;
165     }

167     @Override
168     protected HandlerResult authenticateUsernamePasswordInternal(final UsernamePasswordCredential upc)
169             throws GeneralSecurityException, PreventedException {
170         final AuthenticationResponse response;
171         try {
172             logger.debug("Attempting LDAP authentication for {}", upc);
173             final String password = getPasswordEncoder().encode(upc.getPassword());
174             final AuthenticationRequest request = new AuthenticationRequest(upc.getUsername(),
175                     new org.ldaptive.Credential(password),
176                     this.authenticatedEntryAttributes);
177             response = this.authenticator.authenticate(request);
178         } catch (final LdapException e) {
179             throw new PreventedException("Unexpected LDAP error", e);
180         }
181         logger.debug("LDAP response: {}", response);
182 
183         final List<Message> messageList;
184         
185         final LdapPasswordPolicyConfiguration ldapPasswordPolicyConfiguration =
186                 (LdapPasswordPolicyConfiguration) super.getPasswordPolicyConfiguration();
187         if (ldapPasswordPolicyConfiguration != null) {
188             logger.debug("Applying password policy to {}", response);
189             messageList = ldapPasswordPolicyConfiguration.getAccountStateHandler().handle(
190                     response, ldapPasswordPolicyConfiguration);
191         } else {
192             messageList = Collections.emptyList();
193         }
194         
195         if (response.getResult()) {
196             return createHandlerResult(upc, createPrincipal(upc.getUsername(), response.getLdapEntry()), messageList);
197         }
198 
199         if (AuthenticationResultCode.DN_RESOLUTION_FAILURE == response.getAuthenticationResultCode()) {
200             throw new AccountNotFoundException(upc.getUsername() + " not found.");
201         }
202         throw new FailedLoginException("Invalid credentials");
203     }

228     /**
229      * Creates a CAS principal with attributes if the LDAP entry contains principal attributes.
230      *
231      * @param username Username that was successfully authenticated which is used for principal ID when
232      *                 {@link #setPrincipalIdAttribute(String)} is not specified.
233      * @param ldapEntry LDAP entry that may contain principal attributes.
234      *
235      * @return Principal if the LDAP entry contains at least a principal ID attribute value, null otherwise.
236      *
237      * @throws LoginException On security policy errors related to principal creation.
238      */
239     protected Principal createPrincipal(final String username, final LdapEntry ldapEntry) throws LoginException {
240         final String id;
241         if (this.principalIdAttribute != null) {
242             final LdapAttribute principalAttr = ldapEntry.getAttribute(this.principalIdAttribute);
243             if (principalAttr == null || principalAttr.size() == 0) {
244                 throw new LoginException(this.principalIdAttribute + " attribute not found for " + username);
245             }
246             if (principalAttr.size() > 1) {
247                 if (this.allowMultiplePrincipalAttributeValues) {
248                     logger.warn(
249                             "Found multiple values for principal ID attribute: {}. Using first value={}.",
250                             principalAttr,
251                             principalAttr.getStringValue());
252                 } else {
253                     throw new LoginException("Multiple principal values not allowed: " + principalAttr);
254                 }
255             }
256             id = principalAttr.getStringValue();
257         } else {
258             id = username;
259         }
260         final Map<String, Object> attributeMap = new LinkedHashMap<>(this.principalAttributeMap.size());
261         for (final Map.Entry<String, String> ldapAttr : this.principalAttributeMap.entrySet()) {
262             final LdapAttribute attr = ldapEntry.getAttribute(ldapAttr.getKey());
263             if (attr != null) {
264                 logger.debug("Found principal attribute: {}", attr);
265                 final String principalAttrName = ldapAttr.getValue();
266                 if (attr.size() > 1) {
267                     attributeMap.put(principalAttrName, attr.getStringValues());
268                 } else {
269                     attributeMap.put(principalAttrName, attr.getStringValue());
270                 }
271             }
272         }
273         return this.principalFactory.createPrincipal(id, attributeMap);
274     }

276     /**
277      * Initialize the handler, setup the authentication entry attributes.
278      */
279     @PostConstruct
280     public void initialize() {
281         /**
282          * Use a set to ensure we ignore duplicates.
283          */
284         final Set<String> attributes = new HashSet<>();
285 
286         if (this.principalIdAttribute != null) {
287             attributes.add(this.principalIdAttribute);
288         }
289         if (!this.principalAttributeMap.isEmpty()) {
290             attributes.addAll(this.principalAttributeMap.keySet());
291         }
292         if (!this.additionalAttributes.isEmpty()) {
293             attributes.addAll(this.additionalAttributes);
294         }
295         if (!attributes.isEmpty()) {
296             this.authenticatedEntryAttributes = attributes.toArray(new String[attributes.size()]);
297         }
298     }
