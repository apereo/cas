19  package org.jasig.cas.authentication;
20  
21  import org.apache.commons.lang3.builder.EqualsBuilder;
22  import org.apache.commons.lang3.builder.HashCodeBuilder;
23  import org.jasig.cas.authentication.principal.Principal;
24  import org.joda.time.DateTime;
25  import org.springframework.util.Assert;
26  
27  import java.util.Collections;
28  import java.util.List;
29  import java.util.Map;
30  
31  /**
32   * Immutable authentication event whose attributes may not change after creation.
33   * This class is designed for serialization and is suitable for long-term storage.
34   *
35   * @author Dmitriy Kopylenko
36   * @author Scott Battaglia
37   * @author Marvin S. Addison
38   *
39   * @since 3.0.0
40   */
41  public final class ImmutableAuthentication implements Authentication {
42  
43      /** UID for serializing. */
44      private static final long serialVersionUID = 3206127526058061391L;
45  
46      /** Authentication date stamp. */
47      private final DateTime authenticationDate;
48  
49      /** List of metadata about credentials presented at authentication. */
50      private final List<CredentialMetaData> credentials;
51  
52      /** Authenticated principal. */
53      private final Principal principal;
54  
55      /** Authentication metadata attributes. */
56      private final Map<String, Object> attributes;
57  
58      /** Map of handler name to handler authentication success event. */
59      private final Map<String, HandlerResult> successes;
60  
61      /** Map of handler name to handler authentication failure cause. */
62      private final Map<String, Class<? extends Exception>> failures;
63  
64      /** No-arg constructor for serialization support. */
65      private ImmutableAuthentication() {
66          this.authenticationDate = null;
67          this.credentials = null;
68          this.principal = null;
69          this.attributes = null;
70          this.successes = null;
71          this.failures = null;
72      }
73  
74      /**
75       * Creates a new instance with the given data.
76       *
77       * @param date Non-null authentication date.
78       * @param credentials Non-null list of credential metadata containing at least one entry.
79       * @param principal Non-null authenticated principal.
80       * @param attributes Nullable map of authentication metadata.
81       * @param successes Non-null map of authentication successes containing at least one entry.
82       * @param failures Nullable map of authentication failures.
83       */
84      public ImmutableAuthentication(
85              final DateTime date,
86              final List<CredentialMetaData> credentials,
87              final Principal principal,
88              final Map<String, Object> attributes,
89              final Map<String, HandlerResult> successes,
90              final Map<String, Class<? extends Exception>> failures) {
91  
92          Assert.notNull(date, "Date cannot be null");
93          Assert.notNull(credentials, "Credential cannot be null");
94          Assert.notNull(principal, "Principal cannot be null");
95          Assert.notNull(successes, "Successes cannot be null");
96          Assert.notEmpty(credentials, "Credential cannot be empty");
97          Assert.notEmpty(successes, "Successes cannot be empty");
98  
99          this.authenticationDate = date;
100         this.credentials = credentials;
101         this.principal = principal;
102         this.attributes = attributes.isEmpty() ? null : attributes;
103         this.successes = successes;
104         this.failures = failures.isEmpty() ? null : failures;
105     }
106 
107     @Override
108     public Principal getPrincipal() {
109         return this.principal;
110     }
111 
112     @Override
113     public DateTime getAuthenticationDate() {
114         return authenticationDate;
115     }
116 
117     @Override
118     public Map<String, Object> getAttributes() {
119         return wrap(this.attributes);
120     }
121 
122     @Override
123     public List<CredentialMetaData> getCredentials() {
124         return Collections.unmodifiableList(this.credentials);
125     }
126 
127     @Override
128     public Map<String, HandlerResult> getSuccesses() {
129         return Collections.unmodifiableMap(this.successes);
130     }
131 
132     @Override
133     public Map<String, Class<? extends Exception>> getFailures() {
134         return wrap(this.failures);
135     }
136 
137     @Override
138     public int hashCode() {
139         final HashCodeBuilder builder = new HashCodeBuilder(97, 31);
140         builder.append(this.principal);
141         builder.append(this.authenticationDate);
142         builder.append(this.attributes);
143         builder.append(this.credentials);
144         builder.append(this.successes);
145         builder.append(this.failures);
146         return builder.toHashCode();
147     }
148 
149     @Override
150     public boolean equals(final Object obj) {
151         if (!(obj instanceof Authentication)) {
152             return false;
153         }
154         if (obj == this) {
155             return true;
156         }
157         final Authentication other = (Authentication) obj;
158         final EqualsBuilder builder = new EqualsBuilder();
159         builder.append(this.principal, other.getPrincipal());
160         builder.append(this.credentials, other.getCredentials());
161         builder.append(this.successes, other.getSuccesses());
162         builder.append(this.authenticationDate, other.getAuthenticationDate());
163         builder.append(wrap(this.attributes), other.getAttributes());
164         builder.append(wrap(this.failures), other.getFailures());
165         return builder.isEquals();
166     }
167 
168     /**
169      * Wraps a possibly null map in an immutable wrapper.
170      *
171      * @param <K> the key type
172      * @param <V> the value type
173      * @param source Nullable map to wrap.
174      * @return {@link Collections#unmodifiableMap(java.util.Map)} if given map is not null, otherwise
175      * {@link java.util.Collections#emptyMap()}.
176      */
177     private static <K, V> Map<K, V> wrap(final Map<K, V> source) {
178         if (source != null) {
179             return Collections.unmodifiableMap(source);
180         }
181         return Collections.emptyMap();
182     }
183 }
