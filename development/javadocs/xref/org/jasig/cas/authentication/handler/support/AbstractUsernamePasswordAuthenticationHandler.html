38  import javax.security.auth.login.AccountNotFoundException;

41  /**
42   * Abstract class to override supports so that we don't need to duplicate the
43   * check for UsernamePasswordCredential.
44   *
45   * @author Scott Battaglia
46   * @author Marvin S. Addison
47   *
48   * @since 3.0.0
49   */
50  public abstract class AbstractUsernamePasswordAuthenticationHandler extends
51      AbstractPreAndPostProcessingAuthenticationHandler {
52  
53      /**
54       * PasswordEncoder to be used by subclasses to encode passwords for
55       * comparing against a resource.
56       */
57      @NotNull
58      private PasswordEncoder passwordEncoder = new PlainTextPasswordEncoder();
59  
60      @NotNull
61      private PrincipalNameTransformer principalNameTransformer = new NoOpPrincipalNameTransformer();
62  
63      /** The password policy configuration to be used by extensions. */
64      private PasswordPolicyConfiguration passwordPolicyConfiguration;
65      
66      /**
67       * {@inheritDoc}
68       **/
69      @Override
70      protected final HandlerResult doAuthentication(final Credential credential)
71              throws GeneralSecurityException, PreventedException {
72          final UsernamePasswordCredential userPass = (UsernamePasswordCredential) credential;
73          if (userPass.getUsername() == null) {
74              throw new AccountNotFoundException("Username is null.");
75          }
76          
77          final String transformedUsername= this.principalNameTransformer.transform(userPass.getUsername());
78          if (transformedUsername == null) {
79              throw new AccountNotFoundException("Transformed username is null.");
80          }
81          userPass.setUsername(transformedUsername);
82          return authenticateUsernamePasswordInternal(userPass);
83      }
84  
85      /**
86       * Authenticates a username/password credential by an arbitrary strategy.
87       *
88       * @param transformedCredential the credential object bearing the transformed username and password.
89       *
90       * @return HandlerResult resolved from credential on authentication success or null if no principal could be resolved
91       * from the credential.
92       *
93       * @throws GeneralSecurityException On authentication failure.
94       * @throws PreventedException On the indeterminate case when authentication is prevented.
95       */
96      protected abstract HandlerResult authenticateUsernamePasswordInternal(final UsernamePasswordCredential transformedCredential)
97              throws GeneralSecurityException, PreventedException;
98  
99      /**
100      * Method to return the PasswordEncoder to be used to encode passwords.
101      *
102      * @return the PasswordEncoder associated with this class.
103      */
104     protected final PasswordEncoder getPasswordEncoder() {
105         return this.passwordEncoder;
106     }
107 
108     protected final PrincipalNameTransformer getPrincipalNameTransformer() {
109         return this.principalNameTransformer;
110     }
111     
112     protected final PasswordPolicyConfiguration getPasswordPolicyConfiguration() {
113         return this.passwordPolicyConfiguration;
114     }
115 
116     /**
117      * Helper method to construct a handler result
118      * on successful authentication events.
119      *
120      * @param credential the credential on which the authentication was successfully performed.
121      * Note that this credential instance may be different from what was originally provided
122      * as transformation of the username may have occurred, if one is in fact defined.
123      * @param principal the resolved principal
124      * @param warnings the warnings
125      * @return the constructed handler result
126      */
127     protected final HandlerResult createHandlerResult(final Credential credential, final Principal principal,
128             final List<MessageDescriptor> warnings) {
129         return new DefaultHandlerResult(this, new BasicCredentialMetaData(credential), principal, warnings);
130     }
131     /**
132      * Sets the PasswordEncoder to be used with this class.
133      *
134      * @param passwordEncoder the PasswordEncoder to use when encoding
135      * passwords.
136      */
137     public final void setPasswordEncoder(final PasswordEncoder passwordEncoder) {
138         this.passwordEncoder = passwordEncoder;
139     }
140 
141     public final void setPrincipalNameTransformer(final PrincipalNameTransformer principalNameTransformer) {
142         this.principalNameTransformer = principalNameTransformer;
143     }
144     
145     public final void setPasswordPolicyConfiguration(final PasswordPolicyConfiguration passwordPolicyConfiguration) {
146         this.passwordPolicyConfiguration = passwordPolicyConfiguration;
147     }
148 
149     /**
150      * {@inheritDoc}
151      * @return True if credential is a {@link UsernamePasswordCredential}, false otherwise.
152      */
153     @Override
154     public boolean supports(final Credential credential) {
155         return credential instanceof UsernamePasswordCredential;
156     }
157 }
