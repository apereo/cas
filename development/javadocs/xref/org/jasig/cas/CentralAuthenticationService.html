19  package org.jasig.cas;
20  
21  import org.apache.commons.collections4.Predicate;
22  import org.jasig.cas.authentication.AuthenticationException;
23  import org.jasig.cas.authentication.Credential;
24  import org.jasig.cas.authentication.principal.Service;
25  import org.jasig.cas.logout.LogoutRequest;
26  import org.jasig.cas.ticket.TicketException;
27  import org.jasig.cas.ticket.InvalidTicketException;
28  import org.jasig.cas.ticket.ServiceTicket;
29  import org.jasig.cas.ticket.Ticket;
30  import org.jasig.cas.ticket.TicketGrantingTicket;
31  import org.jasig.cas.validation.Assertion;
32  
33  import javax.validation.constraints.NotNull;
34  import java.util.Collection;
35  import java.util.List;
36  
37  /**
38   * CAS viewed as a set of services to generate and validate Tickets.
39   * <p>
40   * This is the interface between a Web HTML, Web Services, RMI, or any other
41   * request processing layer and the CAS Service viewed as a mechanism to
42   * generate, store, validate, and retrieve Tickets containing Authentication
43   * information. The features of the request processing layer (the HttpXXX
44   * Servlet objects) are not visible here or in any modules behind this layer. In
45   * theory, a standalone application could call these methods directly as a
46   * private authentication service.
47   * </p>
48   *
49   * @author William G. Thompson, Jr.
50   * @author Dmitry Kopylenko
51   * @author Scott Battaglia
52   * @author Marvin S. Addison
53   * @author Misagh Moayyed
54   * @since 3.0.0
55   */
56  public interface CentralAuthenticationService {
57  
58      /**
59       * Create a {@link org.jasig.cas.ticket.TicketGrantingTicket} by authenticating credentials.
60       * The details of the security policy around credential authentication and the definition
61       * of authentication success are dependent on the implementation, but it SHOULD be safe to assume
62       * that at least one credential MUST be authenticated for ticket creation to succeed.
63       *
64       * @param credentials One or more credentials that may be authenticated in order to create the ticket.
65       *
66       * @return Non-null ticket-granting ticket identifier.
67       *
68       * @throws AuthenticationException on errors authenticating the credentials
69       * @throws org.jasig.cas.ticket.TicketException if ticket cannot be created
70       */
71      TicketGrantingTicket createTicketGrantingTicket(@NotNull Credential... credentials)
72          throws AuthenticationException, TicketException;
73  
74  
75      /**
76       * Obtains the given ticket by its id and type
77       * and returns the CAS-representative object. Implementations
78       * need to check for the validity of the ticket by making sure
79       * it exists and has not expired yet, etc. This method is specifically
80       * designed to remove the need to access the ticket registry.
81       *
82       * @param ticketId the ticket granting ticket id
83       * @param clazz the ticket type that is reques to be found
84       * @param <T> the generic ticket type to return that extends {@link Ticket}
85       * @return the ticket object
86       * @throws org.jasig.cas.ticket.InvalidTicketException if ticket is not found or has expired.
87       * @since 4.1.0
88       */
89      <T extends Ticket> T getTicket(@NotNull String ticketId, @NotNull Class<? extends Ticket> clazz)
90              throws InvalidTicketException;
91  
92      /**
93       * Retrieve a collection of tickets from the underlying ticket registry.
94       * The retrieval operation must pass the predicate check that is solely
95       * used to filter the collection of tickets received. Implementations
96       * can use the predicate to request a collection of expired tickets,
97       * or tickets whose id matches a certain pattern, etc. The resulting
98       * collection will include ticktes that have been evaluated by the predicate.
99       *
100      * @param predicate the predicate
101      * @return the tickets
102      * @since 4.1.0
103      */
104     Collection<Ticket> getTickets(@NotNull Predicate predicate);
105 
106     /**
107      * Grants a {@link org.jasig.cas.ticket.ServiceTicket} that may be used to access the given service.
108      *
109      * @param ticketGrantingTicketId Proof of prior authentication.
110      * @param service The target service of the ServiceTicket.
111      *
112      * @return Non-null service ticket identifier.
113      *
114      * @throws org.jasig.cas.ticket.TicketException if the ticket could not be created.
115      */
116     ServiceTicket grantServiceTicket(@NotNull String ticketGrantingTicketId, @NotNull Service service) throws TicketException;
117 
118     /**
119      * Grant a {@link org.jasig.cas.ticket.ServiceTicket} that may be used to access the given service
120      * by authenticating the given credentials.
121      * The details of the security policy around credential authentication and the definition
122      * of authentication success are dependent on the implementation, but it SHOULD be safe to assume
123      * that at least one credential MUST be authenticated for ticket creation to succeed.
124      * <p>
125      * The principal that is resolved from the authenticated credentials MUST be the same as that to which
126      * the given ticket-granting ticket was issued.
127      * </p>
128      *
129      * @param ticketGrantingTicketId Proof of prior authentication.
130      * @param service The target service of the ServiceTicket.
131      * @param credentials One or more credentials to authenticate prior to granting the service ticket.
132      *
133      * @return Non-null service ticket identifier.
134      *
135      * @throws AuthenticationException on errors authenticating the credentials
136      * @throws org.jasig.cas.ticket.TicketException if the ticket could not be created.
137      */
138     ServiceTicket grantServiceTicket(
139             @NotNull final String ticketGrantingTicketId, @NotNull final Service service, final Credential... credentials)
140             throws AuthenticationException, TicketException;
141 
142     /**
143      * Validate a ServiceTicket for a particular Service.
144      *
145      * @param serviceTicketId Proof of prior authentication.
146      * @param service Service wishing to validate a prior authentication.
147      *
148      * @return Non-null ticket validation assertion.
149      *
150      * @throws org.jasig.cas.ticket.TicketException if there was an error validating the ticket.
151      */
152     Assertion validateServiceTicket(@NotNull final String serviceTicketId, @NotNull final Service service) throws TicketException;
153 
154     /**
155      * Destroy a TicketGrantingTicket and perform back channel logout. This has the effect of invalidating any
156      * Ticket that was derived from the TicketGrantingTicket being destroyed. May throw an
157      * {@link IllegalArgumentException} if the TicketGrantingTicket ID is null.
158      *
159      * @param ticketGrantingTicketId the id of the ticket we want to destroy
160      * @return the logout requests.
161      */
162     List<LogoutRequest> destroyTicketGrantingTicket(@NotNull final String ticketGrantingTicketId);
163 
164     /**
165      * Delegate a TicketGrantingTicket to a Service for proxying authentication
166      * to other Services.
167      *
168      * @param serviceTicketId The service ticket identifier that will delegate to a
169      * {@link org.jasig.cas.ticket.TicketGrantingTicket}.
170      * @param credentials One or more credentials to authenticate prior to delegating the ticket.
171      *
172      * @return Non-null ticket-granting ticket identifier that can grant {@link org.jasig.cas.ticket.ServiceTicket}
173      * that proxy authentication.
174      *
175      * @throws AuthenticationException on errors authenticating the credentials
176      * @throws org.jasig.cas.ticket.TicketException if there was an error creating the ticket
177      */
178     TicketGrantingTicket delegateTicketGrantingTicket(@NotNull final String serviceTicketId, @NotNull final Credential... credentials)
179             throws AuthenticationException, TicketException;
180 }
