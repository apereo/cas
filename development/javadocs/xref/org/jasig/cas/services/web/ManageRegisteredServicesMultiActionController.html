19  package org.jasig.cas.services.web;
20  
21  import java.util.ArrayList;
22  import java.util.HashMap;
23  import java.util.List;
24  import java.util.Map;
25  
26  import javax.validation.constraints.NotNull;
27  
28  import org.jasig.cas.services.RegisteredService;
29  import org.jasig.cas.services.ServicesManager;
30  import org.springframework.beans.factory.annotation.Autowired;
31  import org.springframework.beans.factory.annotation.Value;
32  import org.springframework.stereotype.Controller;
33  import org.springframework.web.bind.annotation.RequestMapping;
34  import org.springframework.web.bind.annotation.RequestMethod;
35  import org.springframework.web.bind.annotation.RequestParam;
36  import org.springframework.web.servlet.ModelAndView;
37  import org.springframework.web.servlet.view.RedirectView;
38  
39  /**
40   * MultiActionController to handle the deletion of RegisteredServices as well as
41   * displaying them on the Manage Services page.
42   *
43   * @author Scott Battaglia
44   * @since 3.1
45   */
46  @Controller
47  public final class ManageRegisteredServicesMultiActionController {
48  
49      /** Instance of ServicesManager. */
50      @NotNull
51      private final ServicesManager servicesManager;
52  
53      @NotNull
54      private final String defaultServiceUrl;
55  
56      /**
57       * Instantiates a new manage registered services multi action controller.
58       *
59       * @param servicesManager the services manager
60       * @param defaultServiceUrl the default service url
61       */
62      @Autowired
63      public ManageRegisteredServicesMultiActionController(final ServicesManager servicesManager,
64              @Value("${cas-management.securityContext.serviceProperties.service}") final String defaultServiceUrl) {
65          this.servicesManager = servicesManager;
66          this.defaultServiceUrl = defaultServiceUrl;
67      }
68  
69      /**
70       * Authorization failure handling. Simply returns the view name.
71       *
72       * @return the view name.
73       */
74      @RequestMapping(value="authorizationFailure.html", method={RequestMethod.GET})
75      public String authorizationFailureView() {
76          return "authorizationFailure";
77      }
78  
79      /**
80       * Logout handling. Simply returns the view name.
81       *
82       * @return the view name.
83       */
84      @RequestMapping(value="loggedOut.html", method={RequestMethod.GET})
85      public String logoutView() {
86          return "logout";
87      }
88  
89  
90      /**
91       * Method to delete the RegisteredService by its ID.
92       *
93       * @param idAsLong the id
94       * @return the Model and View to go to after the service is deleted.
95       */
96      @RequestMapping(value="deleteRegisteredService.html", method={RequestMethod.GET})
97      public ModelAndView deleteRegisteredService(
98              @RequestParam("id") final long idAsLong) {
99          
100         final ModelAndView modelAndView = new ModelAndView(new RedirectView(
101                 "manage.html", true), "status", "deleted");
102 
103         final RegisteredService r = this.servicesManager.delete(idAsLong);
104         modelAndView.addObject("serviceName", r != null ? r.getName() : "");
105 
106         return modelAndView;
107     }
108 
109     /**
110      * Method to show the RegisteredServices.
111      * @return the Model and View to go to after the services are loaded.
112      */
113     @RequestMapping(value="manage.html", method={RequestMethod.GET})
114     public ModelAndView manage() {
115         final Map<String, Object> model = new HashMap<>();
116 
117         final List<RegisteredService> services = new ArrayList<>(this.servicesManager.getAllServices());
118 
119         model.put("services", services);
120         model.put("pageTitle", "Manage");
121         model.put("defaultServiceUrl", this.defaultServiceUrl);
122 
123         return new ModelAndView("manage", model);
124     }
125 
126     /**
127      * Updates the {@link RegisteredService#getEvaluationOrder()}. Expects an <code>id</code> parameter to indicate
128      * the {@link RegisteredService#getId()} and the new <code>evaluationOrder</code> integer parameter from the request.
129      * as parameters.
130      *
131      * @param id the id
132      * @param evaluationOrder the evaluation order
133      * @return {@link ModelAndView} object that redirects to a <code>jsonView</code>. The model will contain a
134      * a parameter <code>error</code> whose value should describe the error occurred if the update is unsuccessful.
135      * There will also be a <code>successful</code> boolean parameter that indicates whether or not the update
136      * was successful.
137      */
138     @RequestMapping(value="updateRegisteredServiceEvaluationOrder.html", method={RequestMethod.GET})
139     public ModelAndView updateRegisteredServiceEvaluationOrder(@RequestParam("id") final long id,
140             @RequestParam("evaluationOrder") final int evaluationOrder) {
141         final RegisteredService svc = this.servicesManager.findServiceBy(id);
142         if (svc == null) {
143             throw new IllegalArgumentException("Service id " + id + " cannot be found.");
144         }
145 
146         svc.setEvaluationOrder(evaluationOrder);
147         this.servicesManager.save(svc);
148 
149         return new ModelAndView("jsonView");
150     }
151 }
