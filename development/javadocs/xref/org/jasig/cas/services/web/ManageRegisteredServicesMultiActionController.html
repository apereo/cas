46  /**
47   * MultiActionController to handle the deletion of RegisteredServices as well as
48   * displaying them on the Manage Services page.
49   *
50   * @author Scott Battaglia
51   * @since 3.1
52   */
53  @Controller
54  public final class ManageRegisteredServicesMultiActionController extends AbstractManagementController {
55      @NotNull
56      private final Service defaultService;
57  
58      /**
59       * Instantiates a new manage registered services multi action controller.
60       *
61       * @param servicesManager the services manager
62       * @param defaultServiceUrl the default service url
63       */
64      @Autowired
65      public ManageRegisteredServicesMultiActionController(final ServicesManager servicesManager,
66              @Value("${cas-management.securityContext.serviceProperties.service}") final String defaultServiceUrl) {
67          super(servicesManager);
68          this.defaultService = new SimpleWebApplicationServiceImpl(defaultServiceUrl);
69      }
70  
71      /**
72       * Ensure default service exists.
73       */
74      private void ensureDefaultServiceExists() {
75          final Collection<RegisteredService> c = this.servicesManager.getAllServices();
76          if (c == null) {
77              throw new IllegalStateException("Services cannot be empty");
78          }
79  
80          if (!this.servicesManager.matchesExistingService(defaultService)) {
81              final RegexRegisteredService svc = new RegexRegisteredService();
82              svc.setServiceId('^' + defaultService.getId());
83              svc.setName("Services Management Web Application");
84              svc.setDescription(svc.getName());
85              this.servicesManager.save(svc);
86          }
87      }
88      /**
89       * Authorization failure handling. Simply returns the view name.
90       *
91       * @return the view name.
92       */
93      @RequestMapping(value="authorizationFailure.html", method={RequestMethod.GET})
94      public String authorizationFailureView() {
95          return "authorizationFailure";
96      }
97  
98      /**
99       * Logout handling. Simply returns the view name.
100      *
101      * @param request the request
102      * @param session the session
103      * @return the view name.
104      */
105     @RequestMapping(value="logout.html", method={RequestMethod.GET})
106     public String logoutView(final HttpServletRequest request, final HttpSession session) {
107         logger.debug("Invalidating application session...");
108         session.invalidate();
109         return "logout";
110     }
111 
112     /**
113      * Method to delete the RegisteredService by its ID. Will make sure
114      * the default service that is the management app itself cannot be deleted
115      * or the user will be locked out.
116      *
117      * @param idAsLong the id
118      * @param response the response
119      */
120     @RequestMapping(value="deleteRegisteredService.html", method={RequestMethod.POST})
121     public void deleteRegisteredService(@RequestParam("id") final long idAsLong,
122                                         final HttpServletResponse response) {
123         final RegisteredService svc = this.servicesManager.findServiceBy(this.defaultService);
124         if (svc == null || svc.getId() == idAsLong) {
125             throw new IllegalArgumentException("The default service " + defaultService.getId() + " cannot be deleted. "
126                                        + "The definition is required for accessing the application.");
127         }
128 
129         final RegisteredService r = this.servicesManager.delete(idAsLong);
130         if (r == null) {
131             throw new IllegalArgumentException("Service id " + idAsLong + " cannot be found.");
132         }
133         final Map<String, Object> model = new HashMap<>();
134         model.put("serviceName", r.getName());
135         model.put("status", HttpServletResponse.SC_OK);
136         JsonViewUtils.render(model, response);
137     }
138 
139     /**
140      * Method to show the RegisteredServices.
141      * @param response the response
142      * @return the Model and View to go to after the services are loaded.
143      */
144     @RequestMapping(value="manage.html", method={RequestMethod.GET})
145     public ModelAndView manage(final HttpServletResponse response) {
146         ensureDefaultServiceExists();
147         final Map<String, Object> model = new HashMap<>();
148         model.put("defaultServiceUrl", this.defaultService.getId());
149         model.put("status", HttpServletResponse.SC_OK);
150         return new ModelAndView("manage", model);
151     }
152 
153     /**
154      * Gets services.
155      *
156      * @param response the response
157      */
158     @RequestMapping(value="getServices.html", method={RequestMethod.GET})
159     public void getServices(final HttpServletResponse response) {
160         ensureDefaultServiceExists();
161         final Map<String, Object> model = new HashMap<>();
162         final List<RegisteredServiceViewBean> serviceBeans = new ArrayList<>();
163         final List<RegisteredService> services = new ArrayList<>(this.servicesManager.getAllServices());
164         for (final RegisteredService svc : services) {
165             serviceBeans.add(RegisteredServiceViewBean.fromRegisteredService(svc));
166         }
167         model.put("services", serviceBeans);
168         model.put("status", HttpServletResponse.SC_OK);
169         JsonViewUtils.render(model, response);
170     }
171 
172     /**
173      * Updates the {@link RegisteredService#getEvaluationOrder()}.
174      *
175      * @param response the response
176      * @param id the service ids, whose order also determines the service evaluation order
177      */
178     @RequestMapping(value="updateRegisteredServiceEvaluationOrder.html", method={RequestMethod.POST})
179     public void updateRegisteredServiceEvaluationOrder(final HttpServletResponse response,
180                                                        @RequestParam("id") final long... id) {
181         if (id == null || id.length == 0) {
182             throw new IllegalArgumentException("No service id was received. Re-examine the request");
183         }
184         for (int i = 0; i < id.length; i++) {
185             final long svcId = id[i];
186             final RegisteredService svc = this.servicesManager.findServiceBy(svcId);
187             if (svc == null) {
188                 throw new IllegalArgumentException("Service id " + svcId + " cannot be found.");
189             }
190             svc.setEvaluationOrder(i);
191             this.servicesManager.save(svc);
192         }
193         final Map<String, Object> model = new HashMap<>();
194         model.put("status", HttpServletResponse.SC_OK);
195         JsonViewUtils.render(model, response);
196     }
