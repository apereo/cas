51  /**
52   * SimpleFormController to handle adding/editing of RegisteredServices.
53   *
54   * @author Scott Battaglia
55   * @since 3.1
56   */
57  @Controller
58  public final class RegisteredServiceSimpleFormController {
59  
60      private static final Logger LOGGER = LoggerFactory.getLogger(RegisteredServiceSimpleFormController.class);
61  
62      private static final String COMMAND_NAME = "registeredService";
63  
64      /** Instance of ServiceRegistryManager. */
65      @NotNull
66      private final ServicesManager servicesManager;
67  
68      /** Instance of AttributeRegistry. */
69      @NotNull
70      private final IPersonAttributeDao personAttributeDao;
71      
72      @NotNull
73      @Resource(name="registeredServiceValidator")
74      private Validator validator;
75  
76      /**
77       * Instantiates a new registered service simple form controller.
78       *
79       * @param servicesManager the services manager
80       * @param personAttributeDao the attribute repository
81       */
82      @Autowired
83      public RegisteredServiceSimpleFormController(final ServicesManager servicesManager, 
84              final IPersonAttributeDao personAttributeDao) {
85          this.personAttributeDao = personAttributeDao;
86          this.servicesManager = servicesManager;
87      }
88      
89      /**
90       * Instantiates a new registered service simple form controller.
91       *
92       * @param servicesManager the services manager
93       * @param personAttributeDao the person attribute dao
94       * @param validator the validator
95       */
96      RegisteredServiceSimpleFormController(final ServicesManager servicesManager, 
97              final IPersonAttributeDao personAttributeDao, final Validator validator) {
98          this(servicesManager, personAttributeDao);
99          this.validator = validator;
100     }
101     
102     /**
103      * Sets the require fields and the disallowed fields from the
104      * HttpServletRequest.
105      *
106      * @param request the request
107      * @param binder the binder
108      * @throws Exception the exception
109      */
110     @InitBinder
111     protected void initBinder(final HttpServletRequest request, final ServletRequestDataBinder binder) throws Exception {
112         binder.setRequiredFields("description", "serviceId",
113                 "name", "enabled", "ssoEnabled",
114                 "anonymousAccess", "evaluationOrder");
115         binder.setDisallowedFields("id");
116         binder.registerCustomEditor(String.class, new StringTrimmerEditor(true));
117     }
118 
119     /**
120      * Adds the service to the ServiceRegistry via the ServiceRegistryManager.
121      *
122      * @param service the service
123      * @param result the binding result
124      * @param model the page model
125      * @param request the http request
126      * @return the model and view
127      * @throws Exception the exception
128      */
129     @RequestMapping(method = RequestMethod.POST)
130     protected ModelAndView onSubmit(@ModelAttribute(COMMAND_NAME) final RegisteredService service,
131             final BindingResult result, final ModelMap model, final HttpServletRequest request) throws Exception {
132         
133         updateModelMap(model, request);
134         
135         this.validator.validate(service, result);
136         if (result.hasErrors()) {
137             model.addAttribute("validationErrors", result.getAllErrors());
138             return render(request, model);
139         }
140        
141         RegisteredService svcToUse = service;
142         if (service.getServiceId().startsWith("^") && service instanceof RegisteredServiceImpl) {
143             LOGGER.debug("Detected regular expression starting with ^");
144             final RegexRegisteredService regexService = new RegexRegisteredService();
145             regexService.copyFrom(service);
146             svcToUse = regexService;
147         } else if (!service.getServiceId().startsWith("^") && service instanceof RegexRegisteredService) {
148             LOGGER.debug("Detected ant expression {}", service.getServiceId());
149             final RegisteredServiceImpl regexService = new RegisteredServiceImpl();
150             regexService.copyFrom(service);
151             svcToUse = regexService;
152         } 
153 
154         this.servicesManager.save(svcToUse);
155         LOGGER.info("Saved changes to service {}", svcToUse.getId());
156 
157         final ModelAndView modelAndView = new ModelAndView(new RedirectView(
158                 "manage.html#" + svcToUse.getId(), true));
159         modelAndView.addObject("action", "add");
160         modelAndView.addObject("id", svcToUse.getId());
161 
162         
163         
164         return modelAndView;
165     }
166 
167     /**
168      * Handles displaying the add and edit view by first updating the view by calling
169      * {@link #updateModelMap(ModelMap, HttpServletRequest)}.
170      * @param request the request
171      * @param model the model
172      * @return the model and view
173      * @throws Exception the exception
174      */
175     @RequestMapping(method=RequestMethod.GET, value= {"add.html", "edit.html"})
176     protected ModelAndView render(final HttpServletRequest request, final ModelMap model)
177             throws Exception {
178         updateModelMap(model, request);
179         return new ModelAndView("add");
180     }
181     
182     /**
183      * Updates model map. The following objects will be available in the model:
184      * 
185      * <ul>
186      *  <li><code>availableAttributes</code></li>
187      *  <li><code>availableUsernameAttributes</code></li>
188      *  <li><code>pageTitle</code></li>
189      * </ul>
190      *
191      * @param model the model
192      * @param request the request
193      */
194     private void updateModelMap(final ModelMap model, final HttpServletRequest request) {
195         final List<String> possibleAttributeNames = new ArrayList<>();
196         possibleAttributeNames.addAll(this.personAttributeDao.getPossibleUserAttributeNames());
197         Collections.sort(possibleAttributeNames);
198         model.addAttribute("availableAttributes", possibleAttributeNames);
199 
200         final List<String> possibleUsernameAttributeNames = new ArrayList<>();
201         possibleUsernameAttributeNames.addAll(possibleAttributeNames);
202         possibleUsernameAttributeNames.add(0, "");
203         model.addAttribute("availableUsernameAttributes", possibleUsernameAttributeNames);
204         
205         final String path = request.getServletPath().replace("/", "").replace(".html", "").concat("ServiceView");
206         model.addAttribute("pageTitle", path);    
207     }
208     
209     /**
210      * Determines the registered service to be used. 
211      * If no <code>id</code> is specified, a service of type
212      * {@link RegisteredServiceImpl} will be created by default.
213      *
214      * @param id the id
215      * @return the service
216      */
217     @ModelAttribute(COMMAND_NAME)
218     public RegisteredService getCommand(@RequestParam(value="id", required=false) final String id) {
219 
220         if (!StringUtils.hasText(id)) {
221             final RegisteredService service = new RegisteredServiceImpl();
222             LOGGER.debug("Created new service of type {}", service.getClass().getName());
223             return service;
224         }
225 
226         final RegisteredService service = this.servicesManager.findServiceBy(Long.parseLong(id));
227 
228         if (service != null) {
229             LOGGER.debug("Loaded service {}", service.getServiceId());
230         } else {
231             LOGGER.debug("Invalid service id specified.");
232         }
233 
234         return service;
235     }
