104     /**
105      * Sets the require fields and the disallowed fields from the
106      * HttpServletRequest.
107      *
108      * @param request the request
109      * @param binder the binder

112     @InitBinder
113     protected void initBinder(final HttpServletRequest request, final ServletRequestDataBinder binder) throws Exception {

118         binder.registerCustomEditor(String.class, new StringTrimmerEditor(true));

121     /**
122      * Adds the service to the ServiceRegistry via the ServiceRegistryManager.
123      *
124      * @param service the service
125      * @param result the binding result
126      * @param model the page model
127      * @param request the http request
128      * @return the model and view
129      * @throws Exception the exception
130      */
131     @RequestMapping(method = RequestMethod.POST)
132     protected ModelAndView onSubmit(@ModelAttribute(COMMAND_NAME) final RegisteredService service,
133             final BindingResult result, final ModelMap model, final HttpServletRequest request) throws Exception {
134         
135         updateModelMap(model, request);
136         
137         this.validator.validate(service, result);
138         if (result.hasErrors()) {
139             model.addAttribute("validationErrors", result.getAllErrors());
140             return render(request, model);
141         }
142        
143         RegisteredService svcToUse = service;
144         if (service.getServiceId().startsWith("^") && service instanceof RegisteredServiceImpl) {
145             LOGGER.debug("Detected regular expression starting with ^");
146             final RegexRegisteredService regexService = new RegexRegisteredService();
147             regexService.copyFrom(service);
148             svcToUse = regexService;
149         } else if (!service.getServiceId().startsWith("^") && service instanceof RegexRegisteredService) {
150             LOGGER.debug("Detected ant expression {}", service.getServiceId());
151             final RegisteredServiceImpl regexService = new RegisteredServiceImpl();
152             regexService.copyFrom(service);
153             svcToUse = regexService;
154         } 
155 
156         this.servicesManager.save(svcToUse);
157         LOGGER.info("Saved changes to service {}", svcToUse.getId());
158 
159         final ModelAndView modelAndView = new ModelAndView(new RedirectView(
160                 "manage.html#" + svcToUse.getId(), true));
161         modelAndView.addObject("action", "add");
162         modelAndView.addObject("id", svcToUse.getId());
163 
164         
165         
166         return modelAndView;
167     }

169     /**
170      * Render the page noted by {@link #VIEW_NAME}.
171      * The view is first updated by {@link #updateModelMap(ModelMap, HttpServletRequest)}.
172      * @param request the request
173      * @param model the model
174      * @return the model and view
175      * @throws Exception the exception
176      */
177     @RequestMapping(method=RequestMethod.GET, value= {"add.html", "edit.html"})
178     protected ModelAndView render(final HttpServletRequest request, final ModelMap model)
179             throws Exception {
180         updateModelMap(model, request);
181         return new ModelAndView(VIEW_NAME);
182     }
183     
184     /**
185      * Updates model map. The following objects will be available in the model:
186      * 
187      * <ul>
188      *  <li><code>availableAttributes</code></li>
189      *  <li><code>availableUsernameAttributes</code></li>
190      *  <li><code>pageTitle</code></li>
191      * </ul>
192      *
193      * @param model the model
194      * @param request the request
195      */
196     private void updateModelMap(final ModelMap model, final HttpServletRequest request) {
197         final List<String> possibleAttributeNames = new ArrayList<String>();
198         possibleAttributeNames.addAll(this.personAttributeDao.getPossibleUserAttributeNames());
199         Collections.sort(possibleAttributeNames);
200         model.addAttribute("availableAttributes", possibleAttributeNames);
201 
202         final List<String> possibleUsernameAttributeNames = new ArrayList<String>();
203         possibleUsernameAttributeNames.addAll(possibleAttributeNames);
204         possibleUsernameAttributeNames.add(0, "");
205         model.addAttribute("availableUsernameAttributes", possibleUsernameAttributeNames);
206         
207         final String path = request.getServletPath().replace("/", "").replace(".html", "").concat("ServiceView");
208         model.addAttribute("pageTitle", path);    
209     }

211     /**
212      * Determines the registered service to be used. 
213      * If no <code>id</code> is specified, a service of type
214      * {@link RegisteredServiceImpl} will be created by default.
215      *
216      * @param id the id
217      * @return the service
218      */
219     @ModelAttribute(COMMAND_NAME)
220     public RegisteredService getCommand(@RequestParam(value="id", required=false) final String id) {
221 
222         if (!StringUtils.hasText(id)) {
223             final RegisteredService service = new RegisteredServiceImpl();
224             LOGGER.debug("Created new service of type {}", service.getClass().getName());
225             return service;
226         }
227 
228         final RegisteredService service = this.servicesManager.findServiceBy(Long.parseLong(id));
229 
230         if (service != null) {
231             LOGGER.debug("Loaded service {}", service.getServiceId());
232         } else {
233             LOGGER.debug("Invalid service id specified.");
234         }
235 
236         return service;
237     }
