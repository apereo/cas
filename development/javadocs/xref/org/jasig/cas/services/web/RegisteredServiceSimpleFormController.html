43  /**
44   * SimpleFormController to handle adding/editing of RegisteredServices.
45   *
46   * @author Scott Battaglia
47   * @since 3.1
48   */
49  public final class RegisteredServiceSimpleFormController extends SimpleFormController {
50  
51      /** Instance of ServiceRegistryManager. */
52      @NotNull
53      private final ServicesManager servicesManager;
54  
55      /** Instance of AttributeRegistry. */
56      @NotNull
57      private final IPersonAttributeDao personAttributeDao;
58  
59      public RegisteredServiceSimpleFormController(
60              final ServicesManager servicesManager,
61              final IPersonAttributeDao attributeRepository) {
62          this.servicesManager = servicesManager;
63          this.personAttributeDao = attributeRepository;
64      }
65  
66      /**
67       * {@inheritDoc}
68       * Sets the require fields and the disallowed fields from the
69       * HttpServletRequest.
70       *
71       * @see org.springframework.web.servlet.mvc.BaseCommandController#initBinder(javax.servlet.http.HttpServletRequest,
72       * org.springframework.web.bind.ServletRequestDataBinder)
73       */
74      @Override
75      protected void initBinder(final HttpServletRequest request,
76              final ServletRequestDataBinder binder) throws Exception {
77          binder.setRequiredFields(new String[] {"description", "serviceId",
78                  "name", "allowedToProxy", "enabled", "ssoEnabled",
79                  "anonymousAccess", "evaluationOrder"});
80          binder.setDisallowedFields(new String[] {"id"});
81          binder.registerCustomEditor(String.class, new StringTrimmerEditor(true));
82      }
83  
84      /**
85       * {@inheritDoc}
86       * Adds the service to the ServiceRegistry via the ServiceRegistryManager.
87       *
88       * @see org.springframework.web.servlet.mvc.SimpleFormController#onSubmit(javax.servlet.http.HttpServletRequest,
89       * javax.servlet.http.HttpServletResponse, java.lang.Object,
90       * org.springframework.validation.BindException)
91       */
92      @Override
93      protected ModelAndView onSubmit(final HttpServletRequest request,
94              final HttpServletResponse response, final Object command,
95              final BindException errors) throws Exception {
96          RegisteredService service = (RegisteredService) command;
97  
98          // only change object class if there isn't an explicit RegisteredService class set
99          if (this.getCommandClass() == null) {
100             // CAS-1071
101             // Treat _new_ patterns starting with ^ character as a regular expression
102             if (service.getId() == RegisteredService.INITIAL_IDENTIFIER_VALUE
103                     && service.getServiceId().startsWith("^")) {
104                 logger.debug("Detected regular expression starting with ^");
105                 final RegexRegisteredService regexService = new RegexRegisteredService();
106                 regexService.copyFrom(service);
107                 service = regexService;
108             }
109         }
110         this.servicesManager.save(service);
111         logger.info("Saved changes to service " + service.getId());
112 
113         final ModelAndView modelAndView = new ModelAndView(new RedirectView(
114                 "manage.html#" + service.getId(), true));
115         modelAndView.addObject("action", "add");
116         modelAndView.addObject("id", service.getId());
117 
118         return modelAndView;
119     }
120 
121     @Override
122     protected Object formBackingObject(final HttpServletRequest request)
123             throws Exception {
124         final String id = request.getParameter("id");
125 
126         if (!StringUtils.hasText(id)) {
127             // create a default RegisteredServiceImpl object if an explicit class isn't set
128             final Object service;
129             if (this.getCommandClass() != null) {
130                 service = this.createCommand();
131             } else {
132                 service = new RegisteredServiceImpl();
133             }
134             logger.debug("Created new service of type " + service.getClass().getName());
135             return service;
136         }
137 
138         final RegisteredService service = this.servicesManager.findServiceBy(Long.parseLong(id));
139 
140         if (service != null) {
141             logger.debug("Loaded service " + service.getServiceId());
142         } else {
143             logger.debug("Invalid service id specified.");
144         }
145 
146         return service;
147     }
148 
149     /**
150      * {@inheritDoc}
151      * @see org.springframework.web.servlet.mvc.SimpleFormController#referenceData(javax.servlet.http.HttpServletRequest)
152      */
153     @Override
154     protected Map<?, ?> referenceData(final HttpServletRequest request) throws Exception {
155 
156         final Map<String, Object> model = new HashMap<String, Object>();
157 
158         final List<String> possibleAttributeNames = new ArrayList<String>();
159         possibleAttributeNames.addAll(this.personAttributeDao.getPossibleUserAttributeNames());
160         Collections.sort(possibleAttributeNames);
161         model.put("availableAttributes", possibleAttributeNames);
162 
163         final List<String> possibleUsernameAttributeNames = new ArrayList<String>();
164         possibleUsernameAttributeNames.addAll(possibleAttributeNames);
165         possibleUsernameAttributeNames.add(0, "");
166         model.put("availableUsernameAttributes", possibleUsernameAttributeNames);
167 
168 
169         model.put("pageTitle", getFormView());
170         model.put("commandName", getCommandName());
171         return model;
172     }
