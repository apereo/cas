19  package org.jasig.cas.services;
20  
21  import org.jasig.cas.authentication.principal.Service;
22  
23  import java.io.Serializable;
24  import java.net.URL;
25  import java.util.Set;
26  
27  /**
28   * Interface for a service that can be registered by the Services Management
29   * interface.
30   *
31   * @author Scott Battaglia
32   * @since 3.1
33   */
34  public interface RegisteredService extends Cloneable, Serializable {
35  
36      /** Initial ID value of newly created (but not persisted) registered service. **/
37      long INITIAL_IDENTIFIER_VALUE = -Long.MAX_VALUE;
38  
39      /**
40       * Get the proxy policy rules for this service.
41       *
42       * @return the proxy policy
43       */
44      RegisteredServiceProxyPolicy getProxyPolicy();
45  
46      /**
47       * The unique identifier for this service.
48       *
49       * @return the unique identifier for this service.
50       */
51      String getServiceId();
52  
53      /**
54       * The numeric identifier for this service. Implementations
55       * are expected to initialize the id with the value of {@link #INITIAL_IDENTIFIER_VALUE}.
56       * @return the numeric identifier for this service.
57       */
58      long getId();
59  
60      /**
61       * Returns the name of the service.
62       *
63       * @return the name of the service.
64       */
65      String getName();
66  
67      /**
68       * Returns a short theme name. Services do not need to have unique theme
69       * names.
70       *
71       * @return the theme name associated with this service.
72       */
73      String getTheme();
74  
75      /**
76       * Returns the description of the service.
77       *
78       * @return the description of the service.
79       */
80      String getDescription();
81  
82      /**
83       * Gets the relative evaluation order of this service when determining
84       * matches.
85       * @return Evaluation order relative to other registered services.
86       * Services with lower values will be evaluated for a match before others.
87       */
88      int getEvaluationOrder();
89  
90      /**
91       * Sets the relative evaluation order of this service when determining
92       * matches.
93       * @param evaluationOrder the service evaluation order
94       */
95      void setEvaluationOrder(int evaluationOrder);
96  
97      /**
98       * Get the name of the attribute this service prefers to consume as username.
99       * @return an instance of {@link RegisteredServiceUsernameAttributeProvider}
100      */
101     RegisteredServiceUsernameAttributeProvider getUsernameAttributeProvider();
102 
103     /**
104      * Gets the set of handler names that must successfully authenticate credentials in order to access the service.
105      * An empty set indicates that there are no requirements on particular authentication handlers; any will suffice.
106      *
107      * @return Non-null set of required handler names.
108      */
109     Set<String> getRequiredHandlers();
110 
111     /**
112      * Gets the access strategy that decides whether this registered
113      * service is able to proceed with authentication requests.
114      *
115      * @return the access strategy
116      */
117     RegisteredServiceAccessStrategy getAccessStrategy();
118 
119     /**
120      * Returns whether the service matches the registered service.
121      * <p>Note, as of 3.1.2, matches are case insensitive.
122      *
123      * @param service the service to match.
124      * @return true if they match, false otherwise.
125      */
126     boolean matches(Service service);
127     
128     /**
129      * Clone this service.
130      *
131      * @return the registered service
132      * @throws CloneNotSupportedException the clone not supported exception
133      */
134     RegisteredService clone() throws CloneNotSupportedException;
135 
136     /**
137      * Returns the logout type of the service.
138      *
139      * @return the logout type of the service.
140      */
141     LogoutType getLogoutType();
142     
143     /**
144      * Gets the attribute filtering policy to determine
145      * how attributes are to be filtered and released for
146      * this service.
147      *
148      * @return the attribute release policy
149      */
150     AttributeReleasePolicy getAttributeReleasePolicy();
151 
152     /**
153      * Gets the logo image associated with this service.
154      * The image mostly is served on the user interface
155      * to identify this requesting service during authentication.
156      * @return URL of the image
157      * @since 4.1
158      */
159     URL getLogo();
160 
161     /**
162      * Identifies the logout url that that will be invoked
163      * upon sending single-logout callback notifications.
164      * This is an optional setting. When undefined, the service
165      * url as is defined by {@link #getServiceId()} will be used
166      * to handle logout invocations.
167      * @return the logout url for this service
168      * @since 4.1
169      */
170     URL getLogoutUrl();
171     
172     /**
173      * Gets the public key associated with this service
174      * that is used to authorize the request by
175      * encrypting certain elements and attributes in
176      * the CAS validation protocol response, such as
177      * the PGT.
178      * @return the public key instance used to authorize the request
179      * @since 4.1
180      */
181     RegisteredServicePublicKey getPublicKey();
182     
183 }
