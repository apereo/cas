40  /**
41   * Abstract class to handle retrieving the Assertion from the model.
42   *
43   * @author Scott Battaglia
44   * @since 3.1
45   */
46  public abstract class AbstractCasView extends AbstractView {
47  
48      /**
49       * Indicate whether this view will be generating the success response or not.
50       * By default, the view is treated as a failure.
51       */
52      protected boolean successResponse = false;
53  
54      /** Logger instance. **/
55      protected final Logger logger = LoggerFactory.getLogger(this.getClass());
56  
57      /**
58       * Gets the assertion from the model.
59       *
60       * @param model the model
61       * @return the assertion from
62       */
63      protected final Assertion getAssertionFrom(final Map<String, Object> model) {
64          return (Assertion) model.get("assertion");
65      }
66  
67      /**
68       * Gets the authentication from the model.
69       *
70       * @param model the model
71       * @return the assertion from
72       * @since 4.1
73       */
74      protected final Authentication getPrimaryAuthenticationFrom(final Map<String, Object> model) {
75          return getAssertionFrom(model).getPrimaryAuthentication();
76      }
77  
78      /**
79       * Gets the principal from the model.
80       *
81       * @param model the model
82       * @return the assertion from
83       * @since 4.1
84       */
85      protected final Principal getPrincipal(final Map<String, Object> model) {
86          return getPrimaryAuthenticationFrom(model).getPrincipal();
87      }
88  
89      /**
90       * Gets principal attributes.
91       * Single-valued attributes are converted to a collection
92       * so the review can easily loop through all.
93       * @param model the model
94       * @return the attributes
95       * @since 4.1
96       * @see #convertAttributeValuesToMultiValuedObjects(java.util.Map)
97       */
98      protected final Map<String, Object> getPrincipalAttributesAsMultiValuedAttributes(final Map<String, Object> model) {
99          return convertAttributeValuesToMultiValuedObjects(getPrincipal(model).getAttributes());
100     }
101 
102     /**
103      * Gets authentication attributes.
104      * Single-valued attributes are converted to a collection
105      * so the review can easily loop through all.
106      * @param model the model
107      * @return the attributes
108      * @since 4.1
109      * @see #convertAttributeValuesToMultiValuedObjects(java.util.Map)
110      */
111     protected final Map<String, Object> getAuthenticationAttributesAsMultiValuedAttributes(final Map<String, Object> model) {
112         return convertAttributeValuesToMultiValuedObjects(getPrimaryAuthenticationFrom(model).getAttributes());
113     }
114 
115     /**
116      * Is remember me authentication?
117      * looks at the authentication object to find {@link RememberMeCredential#AUTHENTICATION_ATTRIBUTE_REMEMBER_ME}
118      * and expects the assertion to also note a new login session.
119      * @param model the model
120      * @return true if remember-me, false if otherwise.
121      */
122     protected final boolean isRememberMeAuthentication(final Map<String, Object> model) {
123         final Map<String, Object> authnAttributes = getAuthenticationAttributesAsMultiValuedAttributes(model);
124         final Collection authnMethod = (Collection) authnAttributes.get(RememberMeCredential.AUTHENTICATION_ATTRIBUTE_REMEMBER_ME);
125         return authnMethod != null && authnMethod.contains(Boolean.TRUE) && isAssertionBackedByNewLogin(model);
126     }
127 
128 
129     /**
130      * Is assertion backed by new login?
131      *
132      * @param model the model
133      * @return true/false.
134      */
135     protected final boolean isAssertionBackedByNewLogin(final Map<String, Object> model) {
136         return getAssertionFrom(model).isFromNewLogin();
137     }
138 
139     /**
140      * Convert attribute values to multi valued objects.
141      *
142      * @param attributes the attributes
143      * @return the map of attributes to return
144      */
145     private Map<String, Object> convertAttributeValuesToMultiValuedObjects(final Map<String, Object> attributes) {
146         final Map<String, Object> attributesToReturn = new HashMap<>();
147         final Set<Map.Entry<String, Object>> entries = attributes.entrySet();
148         for (final Map.Entry<String, Object> entry : entries) {
149             final Object value = entry.getValue();
150             if (value instanceof Collection || value instanceof Map || value instanceof Object[]
151                     || value instanceof Iterator || value instanceof Enumeration) {
152                 attributesToReturn.put(entry.getKey(), value);
153             } else {
154                 attributesToReturn.put(entry.getKey(), Collections.singleton(value));
155             }
156         }
157         return attributesToReturn;
158     }

169     }

171     /**
172      * Gets chained authentications.
173      *
174      * @param model the model
175      * @return the chained authentications
176      */
177     protected final Collection<Authentication> getChainedAuthentications(final Map<String, Object> model) {
178         final List<Authentication> chainedAuthenticationsToReturn = new ArrayList<>();
179 
180         final Assertion assertion = getAssertionFrom(model);
181         final List<Authentication> chainedAuthentications = assertion.getChainedAuthentications();
182         for (int i = 0; i < chainedAuthentications.size() - 2; i++) {
183             chainedAuthenticationsToReturn.add(chainedAuthentications.get(i));
184         }
185         return chainedAuthenticationsToReturn;
186     }
