52  /**
53   * Process the /validate , /serviceValidate , and /proxyValidate URL requests.
54   * <p>
55   * Obtain the Service Ticket and Service information and present them to the CAS
56   * validation services. Receive back an Assertion containing the user Principal
57   * and (possibly) a chain of Proxy Principals. Store the Assertion in the Model
58   * and chain to a View to generate the appropriate response (CAS 1, CAS 2 XML,
59   * SAML, ...).
60   *
61   * @author Scott Battaglia
62   * @author Misagh Moayyed
63   * @since 3.0.0
64   */

100     /**
101      * Overrideable method to determine which credentials to use to grant a
102      * proxy granting ticket. Default is to use the pgtUrl.
103      *
104      * @param service the webapp service requesting proxy
105      * @param request the HttpServletRequest object.
106      * @return the credentials or null if there was an error or no credentials
107      * provided.
108      */
109     protected Credential getServiceCredentialsFromRequest(final WebApplicationService service, final HttpServletRequest request) {
110         final String pgtUrl = request.getParameter(CasProtocolConstants.PARAMETER_PROXY_CALLBACK_URL);
111         if (StringUtils.hasText(pgtUrl)) {
112             try {
113                 final RegisteredService registeredService = this.servicesManager.findServiceBy(service);
114                 verifyRegisteredServiceProperties(registeredService, service);
115                 return new HttpBasedServiceCredential(new URL(pgtUrl), registeredService);
116             } catch (final Exception e) {
117                 logger.error("Error constructing pgtUrl", e);
118             }
119         }
120 
121         return null;
122     }

124     /**
125      * Inits the binder with the required fields. <code>renew</code> is required.
126      *
127      * @param request the request
128      * @param binder the binder
129      */
130     protected void initBinder(final HttpServletRequest request, final ServletRequestDataBinder binder) {
131         binder.setRequiredFields("renew");
132     }

134     @Override
135     protected final ModelAndView handleRequestInternal(final HttpServletRequest request, final HttpServletResponse response)
136             throws Exception {
137         final WebApplicationService service = this.argumentExtractor.extractService(request);
138         final String serviceTicketId = service != null ? service.getArtifactId() : null;
139 
140         if (service == null || serviceTicketId == null) {
141             logger.debug("Could not identify service and/or service ticket for service: [{}]", service);
142             return generateErrorView(CasProtocolConstants.ERROR_CODE_INVALID_REQUEST,
143                     CasProtocolConstants.ERROR_CODE_INVALID_REQUEST, null);
144         }
145 
146         try {
147             final Credential serviceCredential = getServiceCredentialsFromRequest(service, request);
148             TicketGrantingTicket proxyGrantingTicketId = null;
149             
150             if (serviceCredential != null) {
151                 try {
152                     proxyGrantingTicketId = this.centralAuthenticationService.delegateTicketGrantingTicket(serviceTicketId,
153                                 serviceCredential);
154                     logger.debug("Generated PGT [{}] off of service ticket [{}] and credential [{}]",
155                             proxyGrantingTicketId.getId(), serviceTicketId, serviceCredential);
156                 } catch (final AuthenticationException e) {
157                     logger.info("Failed to authenticate service credential {}", serviceCredential);
158                 } catch (final TicketException e) {
159                     logger.error("Failed to create proxy granting ticket for {}", serviceCredential, e);
160                 }
161                 
162                 if (proxyGrantingTicketId == null) {
163                     return generateErrorView(CasProtocolConstants.ERROR_CODE_INVALID_PROXY_CALLBACK,
164                             CasProtocolConstants.ERROR_CODE_INVALID_PROXY_CALLBACK,
165                             new Object[] {serviceCredential.getId()});
166                 }
167             }
168 
169             final Assertion assertion = this.centralAuthenticationService.validateServiceTicket(serviceTicketId, service);
170 
171             final ValidationSpecification validationSpecification = this.getCommandClass();
172             final ServletRequestDataBinder binder = new ServletRequestDataBinder(validationSpecification, "validationSpecification");
173             initBinder(request, binder);
174             binder.bind(request);
175 
176             if (!validationSpecification.isSatisfiedBy(assertion)) {
177                 logger.debug("Service ticket [{}] does not satisfy validation specification.", serviceTicketId);
178                 return generateErrorView(CasProtocolConstants.ERROR_CODE_INVALID_TICKET,
179                         CasProtocolConstants.ERROR_CODE_INVALID_TICKET, null);
180             }
181 
182             String proxyIou = null;
183             if (serviceCredential != null && this.proxyHandler.canHandle(serviceCredential)) {
184                 proxyIou = this.proxyHandler.handle(serviceCredential, proxyGrantingTicketId);
185                 if (StringUtils.isEmpty(proxyIou)) {
186                     return generateErrorView(CasProtocolConstants.ERROR_CODE_INVALID_PROXY_CALLBACK,
187                             CasProtocolConstants.ERROR_CODE_INVALID_PROXY_CALLBACK,
188                             new Object[] {serviceCredential.getId()});
189                 }
190             }
191 
192             onSuccessfulValidation(serviceTicketId, assertion);
193             logger.debug("Successfully validated service ticket {} for service [{}]", serviceTicketId, service.getId());
194             return generateSuccessView(assertion, proxyIou, service, proxyGrantingTicketId);
195         } catch (final TicketValidationException e) {
196             final String code = e.getCode();
197             return generateErrorView(code, code,
198                     new Object[] {serviceTicketId, e.getOriginalService().getId(), service.getId()});
199         } catch (final TicketException te) {
200             return generateErrorView(te.getCode(), te.getCode(),
201                 new Object[] {serviceTicketId});
202         } catch (final UnauthorizedProxyingException e) {
203             return generateErrorView(e.getMessage(), e.getMessage(), new Object[] {service.getId()});
204         } catch (final UnauthorizedServiceException e) {
205             return generateErrorView(e.getMessage(), e.getMessage(), null);
206         }
207     }

220     /**
221      * Generate error view, set to {@link #setFailureView(String)}.
222      *
223      * @param code the code
224      * @param description the description
225      * @param args the args
226      * @return the model and view
227      */
228     private ModelAndView generateErrorView(final String code, final String description, final Object[] args) {
229         final ModelAndView modelAndView = new ModelAndView(this.failureView);
230         final String convertedDescription = getMessageSourceAccessor().getMessage(description, args, description);
231         modelAndView.addObject("code", code);
232         modelAndView.addObject("description", convertedDescription);
233 
234         return modelAndView;
235     }

237     /**
238      * Generate the success view. The result will contain the assertion and the proxy iou.
239      *
240      * @param assertion the assertion
241      * @param proxyIou the proxy iou
242      * @param service the validated service
243      * @param proxyGrantingTicket the proxy granting ticket
244      * @return the model and view, pointed to the view name set by
245      */
246     private ModelAndView generateSuccessView(final Assertion assertion, final String proxyIou,
247                                              final WebApplicationService service,
248                                              final TicketGrantingTicket proxyGrantingTicket) {
249 
250         final ModelAndView success = new ModelAndView(this.successView);
251         success.addObject(CasViewConstants.MODEL_ATTRIBUTE_NAME_ASSERTION, assertion);
252         success.addObject(CasViewConstants.MODEL_ATTRIBUTE_NAME_SERVICE, service);
253         success.addObject(CasViewConstants.MODEL_ATTRIBUTE_NAME_PROXY_GRANTING_TICKET_IOU, proxyIou);
254         if (proxyGrantingTicket != null) {
255             success.addObject(CasViewConstants.MODEL_ATTRIBUTE_NAME_PROXY_GRANTING_TICKET, proxyGrantingTicket.getId());
256         }
257         final Map<String, ?> augmentedModelObjects = augmentSuccessViewModelObjects(assertion);
258         if (augmentedModelObjects != null) {
259             success.addAllObjects(augmentedModelObjects);
260         }
261         return success;
262     }

271      * @return map of objects each keyed to a name

274         return Collections.emptyMap();

349     /**
350      * Ensure that the service is found and enabled in the service registry.
351      * @param registeredService the located entry in the registry
352      * @param service authenticating service
353      * @throws UnauthorizedServiceException
354      */
355     private void verifyRegisteredServiceProperties(final RegisteredService registeredService, final Service service) {
356         if (registeredService == null) {
357             final String msg = String.format("ServiceManagement: Unauthorized Service Access. "
358                     + "Service [%s] is not found in service registry.", service.getId());
359             logger.warn(msg);
360             throw new UnauthorizedServiceException(UnauthorizedServiceException.CODE_UNAUTHZ_SERVICE, msg);
361         }
362         if (!registeredService.getAccessStrategy().isServiceAccessAllowed()) {
363             final String msg = String.format("ServiceManagement: Unauthorized Service Access. "
364                     + "Service [%s] is not enabled in service registry.", service.getId());
365             
366             logger.warn(msg);
367             throw new UnauthorizedServiceException(UnauthorizedServiceException.CODE_UNAUTHZ_SERVICE, msg);
368         }
