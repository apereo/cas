49  /**
50   * Action to authenticate credential and retrieve a TicketGrantingTicket for
51   * those credential. If there is a request for renew, then it also generates
52   * the Service Ticket required.

55   * @since 3.0.4

57  public class AuthenticationViaFormAction {

94      /**
95       * Bind the request to credentials.
96       *
97       * @param context the context
98       * @param credential the credential
99       * @throws Exception the exception arising as the result of the bind op.
100      */
101     public final void doBind(final RequestContext context, final Credential credential) throws Exception {
102         final HttpServletRequest request = WebUtils.getHttpServletRequest(context);
103 
104         if (this.credentialsBinder != null && this.credentialsBinder.supports(credential.getClass())) {
105             this.credentialsBinder.bind(request, credential);
106         }
107     }

109     /**
110      * Handle the submission of credentials from the post.
111      *
112      * @param context the context
113      * @param credential the credential
114      * @param messageContext the message context
115      * @return the event
116      * @since 4.1
117      */
118     public final Event submit(final RequestContext context, final Credential credential,
119                               final MessageContext messageContext)  {
120         if (!checkLoginTicketIfExists(context)) {
121             return returnInvalidLoginTicketEvent(context, messageContext);
122         }
123 
124         if (isRequestAskingForServiceTicket(context)) {
125             return grantServiceTicket(context, credential);
126         }
127 
128         return createTicketGrantingTicket(context, credential, messageContext);
129     }
130 
131     /**
132      * Tries to to determine if the login ticket in the request flow scope
133      * matches the login ticket provided by the request. The comparison
134      * is case-sensitive.
135      *
136      * @param context the context
137      * @return true if valid
138      * @since 4.1
139      */
140     protected boolean checkLoginTicketIfExists(final RequestContext context) {
141         final String loginTicketFromFlowScope = WebUtils.getLoginTicketFromFlowScope(context);
142         final String loginTicketFromRequest = WebUtils.getLoginTicketFromRequest(context);
143 
144         logger.trace("Comparing login ticket in the flow scope [{}] with login ticket in the request [{}]",
145                 loginTicketFromFlowScope, loginTicketFromRequest);
146         return StringUtils.equals(loginTicketFromFlowScope, loginTicketFromRequest);
147     }

149     /**
150      * Return invalid login ticket event.
151      *
152      * @param context the context
153      * @param messageContext the message context
154      * @return the error event
155      * @since 4.1
156      */
157     protected Event returnInvalidLoginTicketEvent(final RequestContext context, final MessageContext messageContext) {
158         final String loginTicketFromRequest = WebUtils.getLoginTicketFromRequest(context);
159         logger.warn("Invalid login ticket [{}]", loginTicketFromRequest);
160         messageContext.addMessage(new MessageBuilder().code("error.invalid.loginticket").build());
161         return newEvent(ERROR);
162     }

164     /**
165      * Is request asking for service ticket?
166      *
167      * @param context the context
168      * @return true, if both service and tgt are found, and the request is not asking to renew.
169      * @since 4.1
170      */
171     protected boolean isRequestAskingForServiceTicket(final RequestContext context) {
172         final String ticketGrantingTicketId = WebUtils.getTicketGrantingTicketId(context);
173         final Service service = WebUtils.getService(context);
174         return (StringUtils.isNotBlank(context.getRequestParameters().get(CasProtocolConstants.PARAMETER_RENEW))
175                 && ticketGrantingTicketId != null
176                 && service != null);
177     }

179     /**
180      * Grant service ticket for the given credential based on the service and tgt
181      * that are found in the request context.
182      *
183      * @param context the context
184      * @param credential the credential
185      * @return the resulting event. Warning, authentication failure or error.
186      * @since 4.1
187      */
188     protected Event grantServiceTicket(final RequestContext context, final Credential credential) {
189         final String ticketGrantingTicketId = WebUtils.getTicketGrantingTicketId(context);
190         try {
191             final Service service = WebUtils.getService(context);
192             final String serviceTicketId = this.centralAuthenticationService.grantServiceTicket(
193                     ticketGrantingTicketId, service, credential);
194             WebUtils.putServiceTicketInRequestScope(context, serviceTicketId);
195             putWarnCookieIfRequestParameterPresent(context);
196             return newEvent(WARN);
197         } catch (final AuthenticationException e) {
198             return newEvent(AUTHENTICATION_FAILURE, e);
199         } catch (final TicketCreationException e) {
200             logger.warn(
201                     "Invalid attempt to access service using renew=true with different credential. "
202                             + "Ending SSO session.");
203             this.centralAuthenticationService.destroyTicketGrantingTicket(ticketGrantingTicketId);
204         } catch (final TicketException e) {
205             return newEvent(ERROR, e);
206         }
207         return newEvent(ERROR);
208 
209     }
210     /**
211      * Create ticket granting ticket for the given credentials.
212      * Adds all warnings into the message context.
213      *
214      * @param context the context
215      * @param credential the credential
216      * @param messageContext the message context
217      * @return the resulting event.
218      * @since 4.1
219      */
220     protected Event createTicketGrantingTicket(final RequestContext context, final Credential credential,
221                                                final MessageContext messageContext) {
222         try {
223             final String tgtId = this.centralAuthenticationService.createTicketGrantingTicket(credential);
224             WebUtils.putTicketGrantingTicketInFlowScope(context, tgtId);
225             putWarnCookieIfRequestParameterPresent(context);
226             if (addWarningMessagesToMessageContextIfNeeded(tgtId, messageContext)) {
227                 return newEvent(SUCCESS_WITH_WARNINGS);
228             }
229             return newEvent(SUCCESS);
230         } catch (final AuthenticationException e) {
231             return newEvent(AUTHENTICATION_FAILURE, e);
232         } catch (final Exception e) {
233             return newEvent(ERROR, e);
234         }
235     }
236 
237     /**
238      * Add warning messages to message context if needed.
239      *
240      * @param tgtId the tgt id
241      * @param messageContext the message context
242      * @return true if warnings were found and added, false otherwise.
243      * @since 4.1
244      */
245     protected boolean addWarningMessagesToMessageContextIfNeeded(final String tgtId, final MessageContext messageContext) {
246         boolean foundAndAddedWarnings = false;
247         final TicketGrantingTicket tgt = (TicketGrantingTicket) this.ticketRegistry.getTicket(tgtId);
248         for (final Map.Entry<String, HandlerResult> entry : tgt.getAuthentication().getSuccesses().entrySet()) {
249             for (final Message message : entry.getValue().getWarnings()) {
250                 addWarningToContext(messageContext, message);
251                 foundAndAddedWarnings = true;
252             }
253         }
254         return foundAndAddedWarnings;
255 
256     }
257     /**
258      * Put warn cookie if request parameter present.
259      *
260      * @param context the context
261      */
262     private void putWarnCookieIfRequestParameterPresent(final RequestContext context) {
263         final HttpServletResponse response = WebUtils.getHttpServletResponse(context);
264 
265         if (StringUtils.isNotBlank(context.getExternalContext().getRequestParameterMap().get("warn"))) {
266             this.warnCookieGenerator.addCookie(response, "true");
267         } else {
268             this.warnCookieGenerator.removeCookie(response);
269         }
270     }

297     /**
298      * Set a CredentialsBinder for additional binding of the HttpServletRequest
299      * to the Credential instance, beyond our default binding of the
300      * Credential as a Form Object in Spring WebMVC parlance. By the time we
301      * invoke this CredentialsBinder, we have already engaged in default binding
302      * such that for each HttpServletRequest parameter, if there was a JavaBean
303      * property of the Credential implementation of the same name, we have set
304      * that property to be the value of the corresponding request parameter.
305      * This CredentialsBinder plugin point exists to allow consideration of
306      * things other than HttpServletRequest parameters in populating the
307      * Credential (or more sophisticated consideration of the
308      * HttpServletRequest parameters).
309      *
310      * @param credentialsBinder the credential binder to set.
311      */
312     public final void setCredentialsBinder(final CredentialsBinder credentialsBinder) {
313         this.credentialsBinder = credentialsBinder;
314     }

324     /**
325      * Adds a warning message to the message context.
326      *
327      * @param context Message context.
328      * @param warning Warning message.
329      */
330     private void addWarningToContext(final MessageContext context, final Message warning) {
331         final MessageBuilder builder = new MessageBuilder()
332                 .warning()
333                 .code(warning.getCode())
334                 .defaultText(warning.getDefaultMessage())
335                 .args(warning.getParams());
336         context.addMessage(builder.build());
337     }
