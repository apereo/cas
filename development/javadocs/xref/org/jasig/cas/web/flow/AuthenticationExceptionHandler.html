19  package org.jasig.cas.web.flow;
20  
21  import java.util.ArrayList;
22  import java.util.Collections;
23  import java.util.List;
24  
25  import javax.validation.constraints.NotNull;
26  
27  import org.jasig.cas.authentication.AuthenticationException;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  import org.springframework.binding.message.MessageBuilder;
31  import org.springframework.binding.message.MessageContext;
32  
33  /**
34   * Performs two important error handling functions on an {@link AuthenticationException} raised from the authentication
35   * layer:
36   *
37   * <ol>
38   *     <li>Maps handler errors onto message bundle strings for display to user.</li>
39   *     <li>Determines the next webflow state by comparing handler erors against {@link #errors}
40   *     in list order. The first entry that matches determines the outcome state, which
41   *     is the simple class name of the exception.</li>
42   * </ol>
43   *
44   * @author Marvin S. Addison
45   * @since 4.0
46   */
47  public class AuthenticationExceptionHandler {
48  
49      /** State name when no matching exception is found. */
50      private static final String UNKNOWN = "UNKNOWN";
51  
52      /** Default message bundle prefix. */
53      private static final String DEFAULT_MESSAGE_BUNDLE_PREFIX = "authenticationFailure.";
54  
55      /** Default list of errors this class knows how to handle. */
56      private static final List<Class<? extends Exception>> DEFAULT_ERROR_LIST =
57              new ArrayList<Class<? extends Exception>>();
58  
59      private final Logger logger = LoggerFactory.getLogger(this.getClass());
60      
61      static {
62          DEFAULT_ERROR_LIST.add(javax.security.auth.login.AccountLockedException.class);
63          DEFAULT_ERROR_LIST.add(javax.security.auth.login.FailedLoginException.class);
64          DEFAULT_ERROR_LIST.add(javax.security.auth.login.CredentialExpiredException.class);
65          DEFAULT_ERROR_LIST.add(javax.security.auth.login.AccountNotFoundException.class);
66          DEFAULT_ERROR_LIST.add(org.jasig.cas.authentication.AccountDisabledException.class);
67          DEFAULT_ERROR_LIST.add(org.jasig.cas.authentication.InvalidLoginLocationException.class);
68          DEFAULT_ERROR_LIST.add(org.jasig.cas.authentication.InvalidLoginTimeException.class);
69      }
70  
71      /** Ordered list of error classes that this class knows how to handle. */
72      @NotNull
73      private List<Class<? extends Exception>> errors = DEFAULT_ERROR_LIST;
74  
75      /** String appended to exception class name to create a message bundle key for that particular error. */
76      private String messageBundlePrefix = DEFAULT_MESSAGE_BUNDLE_PREFIX;
77  
78      /**
79       * Sets the list of errors that this class knows how to handle.
80       *
81       * @param errors List of errors in order of descending precedence.
82       */
83      public void setErrors(final List<Class<? extends Exception>> errors) {
84          this.errors = errors;
85      }
86  
87      public final List<Class<? extends Exception>> getErrors() {
88          return Collections.unmodifiableList(this.errors);
89      }
90      
91      /**
92       * Sets the message bundle prefix appended to exception class names to create a message bundle key for that
93       * particular error.
94       *
95       * @param prefix Prefix appended to exception names.
96       */
97      public void setMessageBundlePrefix(final String prefix) {
98          this.messageBundlePrefix = prefix;
99      }
100 
101     /**
102      * Maps an authentication exception onto a state name equal to the simple class name of the
103      * {@link org.jasig.cas.authentication.AuthenticationException#getHandlerErrors()} with highest precedence.
104      * Also sets an ERROR severity message in the message context of the form
105      * <code>[messageBundlePrefix][exceptionClassSimpleName]</code> for each handler error that is
106      * configured. If not match is found, {@value #UNKNOWN} is returned.
107      *
108      * @param e Authentication error to handle.
109      *
110      * @return Name of next flow state to transition to or {@value #UNKNOWN}
111      */
112     public String handle(final AuthenticationException e, final MessageContext messageContext) {
113         if (e != null) {
114             for (final Class<? extends Exception> kind : this.errors) {
115                 for (final Class<? extends Exception> handlerError : e.getHandlerErrors().values()) {
116                     if (handlerError != null && handlerError.equals(kind)) {
117                         final String messageCode = this.messageBundlePrefix + handlerError.getSimpleName();
118                         messageContext.addMessage(new MessageBuilder().error().code(messageCode).build());
119                         return handlerError.getSimpleName();
120                     }
121                 }
122 
123             }
124         }
125         final String messageCode = this.messageBundlePrefix + UNKNOWN;
126         logger.trace("Unable to translate handler errors of the authentication exception {}. Returning {} by default...", e, messageCode);
127         messageContext.addMessage(new MessageBuilder().error().code(messageCode).build());
128         return UNKNOWN;
129     }
