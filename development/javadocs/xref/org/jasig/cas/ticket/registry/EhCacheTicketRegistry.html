19  package org.jasig.cas.ticket.registry;

23  import net.sf.ehcache.config.CacheConfiguration;
24  import org.apache.commons.lang3.BooleanUtils;
25  import org.apache.commons.lang3.StringUtils;
26  import org.jasig.cas.ticket.ServiceTicket;
27  import org.jasig.cas.ticket.Ticket;
28  import org.jasig.cas.ticket.TicketGrantingTicket;
29  import org.jasig.cas.ticket.registry.encrypt.AbstractCrypticTicketRegistry;
30  import org.springframework.beans.BeanInstantiationException;
31  import org.springframework.beans.factory.InitializingBean;
32  import org.springframework.core.style.ToStringCreator;
33  
34  import java.util.Collection;
35  import java.util.HashSet;

37  /**
38   * <p>
39   * <a href="http://ehcache.org/">Ehcache</a> based distributed ticket registry.
40   * </p>
41   *
42   * <p>
43   * Use distinct caches for ticket granting tickets (TGT) and service tickets (ST) for:
44   * <ul>
45   *   <li>Tuning : use cache level time to live with different values for TGT an ST.</li>
46   *   <li>Monitoring : follow separately the number of TGT and ST.</li>
47   * </ul>
48   *
49   * @author <a href="mailto:cleclerc@xebia.fr">Cyrille Le Clerc</a>
50   * @author Adam Rybicki
51   * @author Andrew Tillinghast
52   * @since 3.5
53   */
54  public final class EhCacheTicketRegistry extends AbstractCrypticTicketRegistry implements InitializingBean {
55  
56      private Cache serviceTicketsCache;
57  
58      private Cache ticketGrantingTicketsCache;
59  
60      /**
61       * @see #setSupportRegistryState(boolean)
62       **/
63      private boolean supportRegistryState = true;
64  
65      /**
66       * Instantiates a new EhCache ticket registry.
67       */
68      public EhCacheTicketRegistry() {
69          super();
70      }
71  
72      /**
73       * Instantiates a new EhCache ticket registry.
74       *
75       * @param serviceTicketsCache the service tickets cache
76       * @param ticketGrantingTicketsCache the ticket granting tickets cache
77       */
78      public EhCacheTicketRegistry(final Cache serviceTicketsCache, final Cache ticketGrantingTicketsCache) {
79          super();
80          setServiceTicketsCache(serviceTicketsCache);
81          setTicketGrantingTicketsCache(ticketGrantingTicketsCache);
82      }
83  
84      /**
85       * Instantiates a new EhCache ticket registry.
86       *
87       * @param serviceTicketsCache the service tickets cache
88       * @param ticketGrantingTicketsCache the ticket granting tickets cache
89       * @param supportRegistryState the support registry state
90       */
91      public EhCacheTicketRegistry(final Cache serviceTicketsCache, final Cache ticketGrantingTicketsCache,
92              final boolean supportRegistryState) {
93          this(serviceTicketsCache, ticketGrantingTicketsCache);
94          setSupportRegistryState(supportRegistryState);
95      }
96  
97      @Override
98      public void addTicket(final Ticket ticketToAdd) {
99          final Ticket ticket = encodeTicket(ticketToAdd);
100         final Element element = new Element(ticket.getId(), ticket);
101         if (ticket instanceof ServiceTicket) {
102             logger.debug("Adding service ticket {} to the cache {}", ticket.getId(), this.serviceTicketsCache.getName());
103             this.serviceTicketsCache.put(element);
104         } else if (ticket instanceof TicketGrantingTicket) {
105             logger.debug("Adding ticket granting ticket {} to the cache {}", ticket.getId(),
106                     this.ticketGrantingTicketsCache.getName());
107             this.ticketGrantingTicketsCache.put(element);
108         } else {
109             throw new IllegalArgumentException("Invalid ticket type " + ticket);
110         }
111     }
112 
113     @Override
114     public boolean deleteTicket(final String ticketIdToDelete) {
115         final String ticketId = encodeTicketId(ticketIdToDelete);
116         if (StringUtils.isBlank(ticketId)) {
117             return false;
118         }
119         return this.serviceTicketsCache.remove(ticketId) || this.ticketGrantingTicketsCache.remove(ticketId);
120     }
121 
122     @Override
123     public Ticket getTicket(final String ticketIdToGet) {
124         final String ticketId = encodeTicketId(ticketIdToGet);
125         if (ticketId == null) {
126             return null;
127         }
128 
129         Element element = this.serviceTicketsCache.get(ticketId);
130         if (element == null) {
131             element = this.ticketGrantingTicketsCache.get(ticketId);
132         }
133         if (element == null) {
134             logger.debug("No ticket by id [{}] is found in the registry", ticketId);
135             return null;
136         }
137 
138         final Ticket ticket = decodeTicket((Ticket) element.getObjectValue());
139         return getProxiedTicketInstance(ticket);
140     }
141 
142     @Override
143     public Collection<Ticket> getTickets() {
144         final Collection<Element> serviceTickets = this.serviceTicketsCache.getAll(
145                 this.serviceTicketsCache.getKeysWithExpiryCheck()).values();
146         final Collection<Element> tgtTicketsTickets = this.ticketGrantingTicketsCache.getAll(
147                 this.ticketGrantingTicketsCache.getKeysWithExpiryCheck()).values();
148 
149         final Collection<Ticket> allTickets = new HashSet<>(serviceTickets.size() + tgtTicketsTickets.size());
150 
151         for (final Element ticket : serviceTickets) {
152             allTickets.add((Ticket) ticket.getObjectValue());
153         }
154 
155         for (final Element ticket : tgtTicketsTickets) {
156             allTickets.add((Ticket) ticket.getObjectValue());
157         }
158 
159         return decodeTickets(allTickets);
160     }
161 
162     public void setServiceTicketsCache(final Cache serviceTicketsCache) {
163         this.serviceTicketsCache = serviceTicketsCache;
164     }
165 
166     public void setTicketGrantingTicketsCache(final Cache ticketGrantingTicketsCache) {
167         this.ticketGrantingTicketsCache = ticketGrantingTicketsCache;
168     }
169 
170     @Override
171     public String toString() {
172         return new ToStringCreator(this).append("ticketGrantingTicketsCache", this.ticketGrantingTicketsCache)
173                 .append("serviceTicketsCache", this.serviceTicketsCache).toString();
174     }
175 
176     @Override
177     protected void updateTicket(final Ticket ticket) {
178         addTicket(ticket);
179     }
180 
181     @Override
182     protected boolean needsCallback() {
183         return false;
184     }
185 
186     /**
187      * Flag to indicate whether this registry instance should participate in reporting its state with
188      * default value set to <code>true</code>.
189      * Based on the <a href="http://ehcache.org/apidocs/net/sf/ehcache/Ehcache.html#getKeysWithExpiryCheck()">EhCache documentation</a>,
190      * determining the number of service tickets and the total session count from the cache can be considered
191      * an expensive operation with the time taken as O(n), where n is the number of elements in the cache.
192      *
193      * <p>Therefore, the flag provides a level of flexibility such that depending on the cache and environment
194      * settings, reporting statistics
195      * can be set to false and disabled.</p>
196      *
197      * @param supportRegistryState true, if the registry is to support registry state
198      * @see #sessionCount()
199      * @see #serviceTicketCount()
200      * @see org.jasig.cas.monitor.SessionMonitor
201      */
202     public void setSupportRegistryState(final boolean supportRegistryState) {
203         this.supportRegistryState = supportRegistryState;
204     }
205 
206     @Override
207     public void afterPropertiesSet() throws Exception {
208         if (this.serviceTicketsCache == null || this.ticketGrantingTicketsCache == null) {
209             throw new BeanInstantiationException(this.getClass(),
210                     "Both serviceTicketsCache and ticketGrantingTicketsCache are required properties.");
211         }
212 
213         if (logger.isDebugEnabled()) {
214             CacheConfiguration config = this.serviceTicketsCache.getCacheConfiguration();
215             logger.debug("serviceTicketsCache.maxElementsInMemory={}", config.getMaxEntriesLocalHeap());
216             logger.debug("serviceTicketsCache.maxElementsOnDisk={}", config.getMaxElementsOnDisk());
217             logger.debug("serviceTicketsCache.isOverflowToDisk={}", config.isOverflowToDisk());
218             logger.debug("serviceTicketsCache.timeToLive={}", config.getTimeToLiveSeconds());
219             logger.debug("serviceTicketsCache.timeToIdle={}", config.getTimeToIdleSeconds());
220             logger.debug("serviceTicketsCache.cacheManager={}", this.serviceTicketsCache.getCacheManager().getName());
221 
222             config = this.ticketGrantingTicketsCache.getCacheConfiguration();
223             logger.debug("ticketGrantingTicketsCache.maxElementsInMemory={}", config.getMaxEntriesLocalHeap());
224             logger.debug("ticketGrantingTicketsCache.maxElementsOnDisk={}", config.getMaxElementsOnDisk());
225             logger.debug("ticketGrantingTicketsCache.isOverflowToDisk={}", config.isOverflowToDisk());
226             logger.debug("ticketGrantingTicketsCache.timeToLive={}", config.getTimeToLiveSeconds());
227             logger.debug("ticketGrantingTicketsCache.timeToIdle={}", config.getTimeToIdleSeconds());
228             logger.debug("ticketGrantingTicketsCache.cacheManager={}", this.ticketGrantingTicketsCache.getCacheManager()
229                     .getName());
230         }
231     }
232 
233     /**
234      * {@inheritDoc}
235      * @see Cache#getKeysWithExpiryCheck()
236      */
237     @Override
238     public int sessionCount() {
239         return BooleanUtils.toInteger(this.supportRegistryState, this.ticketGrantingTicketsCache
240                 .getKeysWithExpiryCheck().size(), super.sessionCount());
241     }
242 
243     /**
244      * {@inheritDoc}
245      * @see Cache#getKeysWithExpiryCheck()
246      */
247     @Override
248     public int serviceTicketCount() {
249         return BooleanUtils.toInteger(this.supportRegistryState, this.serviceTicketsCache.getKeysWithExpiryCheck()
250                 .size(), super.serviceTicketCount());
251     }
252 }
