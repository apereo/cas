63  /**
64   * {@link net.spy.memcached.MemcachedClient} transcoder implementation based on Kryo fast serialization framework
65   * suited for efficient serialization of tickets.
66   *
67   * @author Marvin S. Addison
68   * @since 3.0.0
69   */
70  @SuppressWarnings("rawtypes")
71  public class KryoTranscoder implements Transcoder<Object> {
72  
73      /** Kryo serializer. */
74      private final Kryo kryo = new Kryo();
75  
76      /** Logging instance. */
77      private final Logger logger = LoggerFactory.getLogger(getClass());
78  
79      /** Map of class to serializer that handles it. */
80      private Map<Class<?>, Serializer> serializerMap;
81  
82      /**
83       * Creates a Kryo-based transcoder.
84       */
85      public KryoTranscoder() {
86      }
87  
88      /**
89       * @deprecated
90       * Creates a Kryo-based transcoder.
91       *
92       * @param initialBufferSize Initial size for buffer holding encoded object data.
93       */
94      @Deprecated
95      public KryoTranscoder(final int initialBufferSize) {
96          logger.warn("It's no longer necessary to define the initialBufferSize. Use the empty constructor.");
97      }
98  
99      /**
100      * Sets a map of additional types that should be regisetered with Kryo,
101      * for example GoogleAccountsService and OpenIdService.
102      *
103      * @param map Map of class to the serializer instance that handles it.
104      */
105     public void setSerializerMap(final Map<Class<?>, Serializer> map) {
106         this.serializerMap = map;
107     }
108 
109     /**
110      * Initialize and register classes with kryo.
111      */
112     public void initialize() {
113         // Register types we know about and do not require external configuration
114         kryo.register(ArrayList.class);
115         kryo.register(BasicCredentialMetaData.class);
116         kryo.register(Class.class, new DefaultSerializers.ClassSerializer());
117         kryo.register(Date.class, new DefaultSerializers.DateSerializer());
118         kryo.register(HardTimeoutExpirationPolicy.class);
119         kryo.register(HashMap.class);
120         kryo.register(DefaultHandlerResult.class);
121         kryo.register(ImmutableAuthentication.class);
122         kryo.register(MultiTimeUseOrTimeoutExpirationPolicy.class);
123         kryo.register(NeverExpiresExpirationPolicy.class);
124         kryo.register(RememberMeDelegatingExpirationPolicy.class);
125         kryo.register(ServiceTicketImpl.class);
126         kryo.register(SimpleWebApplicationServiceImpl.class, new SimpleWebApplicationServiceSerializer());
127         kryo.register(ThrottledUseAndTimeoutExpirationPolicy.class);
128         kryo.register(TicketGrantingTicketExpirationPolicy.class);
129         kryo.register(TicketGrantingTicketImpl.class);
130         kryo.register(TimeoutExpirationPolicy.class);
131         kryo.register(URL.class, new URLSerializer());
132 
133         // we add these ones for tests only
134         kryo.register(RegisteredServiceImpl.class, new RegisteredServiceSerializer());
135         kryo.register(RegexRegisteredService.class, new RegisteredServiceSerializer());
136 
137         // new serializers to manage Joda dates and immutable collections
138         kryo.register(DateTime.class, new JodaDateTimeSerializer());
139         // from the kryo-serializers library (https://github.com/magro/kryo-serializers)
140         UnmodifiableCollectionsSerializer.registerSerializers(kryo);
141 
142         // Register other types
143         if (serializerMap != null) {
144             for (final Map.Entry<Class<?>, Serializer> clazz : serializerMap.entrySet()) {
145                 kryo.register(clazz.getKey(), clazz.getValue());
146             }
147         }
148 
149         // don't reinit the registered classes after every write or read
150         kryo.setAutoReset(false);
151         // don't replace objects by references
152         kryo.setReferences(false);
153         // Catchall for any classes not explicitly registered
154         kryo.setRegistrationRequired(false);
155     }
156 
157     /**
158      * Asynchronous decoding is not supported.
159      *
160      * @param d Data to decode.
161      * @return False.
162      */
163     public boolean asyncDecode(final CachedData d) {
164         return false;
165     }
166 
167     @Override
168     public CachedData encode(final Object obj) {
169         final ByteArrayOutputStream byteStream = new ByteArrayOutputStream();
170         try (final Output output = new Output(byteStream)) {
171             kryo.writeClassAndObject(output, obj);
172             output.flush();
173             final byte[] bytes = byteStream.toByteArray();
174             return new CachedData(0, bytes, bytes.length);
175         }
176     }
177 
178     @Override
179     public Object decode(final CachedData d) {
180         final byte[] bytes = d.getData();
181         try (final Input input = new Input(new ByteArrayInputStream(bytes))) {
182             final Object obj =  kryo.readClassAndObject(input);
183             return obj;
184         }
185     }
186 
187     /**
188      * Maximum size of encoded data supported by this transcoder.
189      *
190      * @return  <code>net.spy.memcached.CachedData#MAX_SIZE</code>.
191      */
192     public int getMaxSize() {
193         return CachedData.MAX_SIZE;
194     }
195 
196     /**
197      * Gets the kryo object that provides encoding and decoding services for this instance.
198      *
199      * @return Underlying Kryo instance.
200      */
201     public Kryo getKryo() {
202         return kryo;
203     }
204 }
