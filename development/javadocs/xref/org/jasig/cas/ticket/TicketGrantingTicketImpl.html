41  /**
42   * Concrete implementation of a TicketGrantingTicket. A TicketGrantingTicket is
43   * the global identifier of a principal into the system. It grants the Principal
44   * single-sign on access to any service that opts into single-sign on.
45   * Expiration of a TicketGrantingTicket is controlled by the ExpirationPolicy
46   * specified as object creation.
47   *
48   * @author Scott Battaglia
49   * @since 3.0.0
50   */
51  @Entity
52  @Table(name="TICKETGRANTINGTICKET")
53  public final class TicketGrantingTicketImpl extends AbstractTicket implements TicketGrantingTicket {
54  
55      /** Unique Id for serialization. */
56      private static final long serialVersionUID = -8608149809180911599L;
57  
58      /** Logger instance. */
59      private static final Logger LOGGER = LoggerFactory.getLogger(TicketGrantingTicketImpl.class);
60  
61      /** The authenticated object for which this ticket was generated for. */
62      @Lob
63      @Column(name="AUTHENTICATION", nullable=false, length = 1000000)
64      private Authentication authentication;
65  
66      /** Flag to enforce manual expiration. */
67      @Column(name="EXPIRED", nullable=false)
68      private Boolean expired = Boolean.FALSE;
69  
70      /** Service that produced a proxy-granting ticket. */
71      @Column(name="PROXIED_BY", nullable=true)
72      private Service proxiedBy;
73  
74      /** The services associated to this ticket. */
75      @Lob
76      @Column(name="SERVICES_GRANTED_ACCESS_TO", nullable=false, length = 1000000)
77      private final HashMap<String, Service> services = new HashMap<>();
78  
79      @Lob
80      @Column(name="SUPPLEMENTAL_AUTHENTICATIONS", nullable=false, length = 1000000)
81      private final ArrayList<Authentication> supplementalAuthentications = new ArrayList<>();
82  
83      /**
84       * Instantiates a new ticket granting ticket impl.
85       */
86      public TicketGrantingTicketImpl() {
87          // nothing to do
88      }
89  
90      /**
91       * Constructs a new TicketGrantingTicket.
92       * May throw an {@link IllegalArgumentException} if the Authentication object is null.
93       *
94       * @param id the id of the Ticket
95       * @param proxiedBy Service that produced this proxy ticket.
96       * @param parentTicketGrantingTicket the parent ticket
97       * @param authentication the Authentication request for this ticket
98       * @param policy the expiration policy for this ticket.
99       */
100     public TicketGrantingTicketImpl(final String id,
101         final Service proxiedBy,
102         final TicketGrantingTicket parentTicketGrantingTicket,
103         @NotNull final Authentication authentication, final ExpirationPolicy policy) {
104 
105         super(id, parentTicketGrantingTicket, policy);
106 
107         if (parentTicketGrantingTicket != null && proxiedBy == null) {
108             throw new IllegalArgumentException("Must specify proxiedBy when providing parent TGT");
109         }
110         Assert.notNull(authentication, "authentication cannot be null");
111         this.authentication = authentication;
112         this.proxiedBy = proxiedBy;
113     }
114 
115     /**
116      * Constructs a new TicketGrantingTicket without a parent
117      * TicketGrantingTicket.
118      *
119      * @param id the id of the Ticket
120      * @param authentication the Authentication request for this ticket
121      * @param policy the expiration policy for this ticket.
122      */
123     public TicketGrantingTicketImpl(final String id,
124         final Authentication authentication, final ExpirationPolicy policy) {
125         this(id, null, null, authentication, policy);
126     }
127 
128     /**
129      * {@inheritDoc}
130      */
131     @Override
132     public Authentication getAuthentication() {
133         return this.authentication;
134     }
135 
136     /**
137      * {@inheritDoc}
138      * <p>The state of the ticket is affected by this operation and the
139      * ticket will be considered used. The state update subsequently may
140      * impact the ticket expiration policy in that, depending on the policy
141      * configuration, the ticket may be considered expired.
142      */
143     @Override
144     public synchronized ServiceTicket grantServiceTicket(final String id,
145         final Service service, final ExpirationPolicy expirationPolicy,
146         final boolean credentialsProvided) {
147         final ServiceTicket serviceTicket = new ServiceTicketImpl(id, this,
148             service, this.getCountOfUses() == 0 || credentialsProvided,
149             expirationPolicy);
150 
151         updateState();
152 
153         final List<Authentication> authentications = getChainedAuthentications();
154         service.setPrincipal(authentications.get(authentications.size()-1).getPrincipal());
155 
156         this.services.put(id, service);
157 
158         return serviceTicket;
159     }
160 
161     /**
162      * Gets an immutable map of service ticket and services accessed by this ticket-granting ticket.
163      * Unlike {@link Collections#unmodifiableMap(java.util.Map)},
164      * which is a view of a separate map which can still change, an instance of {@link ImmutableMap}
165      * contains its own data and will never change.
166      *
167      * @return an immutable map of service ticket and services accessed by this ticket-granting ticket.
168     */
169     @Override
170     public synchronized Map<String, Service> getServices() {
171         return ImmutableMap.copyOf(this.services);
172     }
173 
174     /**
175      * Remove all services of the TGT (at logout).
176      */
177     @Override
178     public void removeAllServices() {
179         services.clear();
180     }
181 
182     /**
183      * Return if the TGT has no parent.
184      *
185      * @return if the TGT has no parent.
186      */
187     @Override
188     public boolean isRoot() {
189         return this.getGrantingTicket() == null;
190     }
191 
192     /**
193      * {@inheritDoc}
194      */
195     @Override
196     public void markTicketExpired() {
197         this.expired = Boolean.TRUE;
198     }
199 
200     /**
201      * {@inheritDoc}
202      */
203     @Override
204     public TicketGrantingTicket getRoot() {
205         TicketGrantingTicket current = this;
206         TicketGrantingTicket parent = current.getGrantingTicket();
207         while (parent != null) {
208             current = parent;
209             parent = current.getGrantingTicket();
210         }
211         return current;
212     }
213 
214     /**
215      * Return if the TGT is expired.
216      *
217      * @return if the TGT is expired.
218      */
219     @Override
220     public boolean isExpiredInternal() {
221         return this.expired;
222     }
223 
224     /**
225      * {@inheritDoc}
226      */
227     @Override
228     public List<Authentication> getSupplementalAuthentications() {
229         return this.supplementalAuthentications;
230     }
231 
232     /**
233      * {@inheritDoc}
234      */
235     @Override
236     public List<Authentication> getChainedAuthentications() {
237         final List<Authentication> list = new ArrayList<>();
238 
239         list.add(getAuthentication());
240 
241         if (getGrantingTicket() == null) {
242             return Collections.unmodifiableList(list);
243         }
244 
245         list.addAll(getGrantingTicket().getChainedAuthentications());
246         return Collections.unmodifiableList(list);
247     }
248 
249     @Override
250     public Service getProxiedBy() {
251         return this.proxiedBy;
252     }
253 
254     /**
255      * {@inheritDoc}
256      */
257     @Override
258     public boolean equals(final Object object) {
259         if (object == null) {
260             return false;
261         }
262         if (object == this) {
263             return true;
264         }
265         if (!(object instanceof TicketGrantingTicket)) {
266             return false;
267         }
268 
269         final Ticket ticket = (Ticket) object;
270 
271         return new EqualsBuilder()
272                 .append(ticket.getId(), this.getId())
273                 .isEquals();
274     }
