50  /**
51   * Handles the creation of Ticket Granting Tickets.

57  public class TicketResource extends ServerResource {

59      private static final Logger LOGGER = LoggerFactory.getLogger(TicketResource.class);
60  
61      @Autowired
62      private CentralAuthenticationService centralAuthenticationService;
63  
64      public TicketResource() {
65          setNegotiated(false);
66      }
67  
68      @Post
69      public final void acceptRepresentation(final Representation entity)  {
70          LOGGER.debug("Obtaining credentials...");
71          final Credential c = obtainCredentials();
72  
73          Formatter fmt = null;
74          try {
75              final String ticketGrantingTicketId = this.centralAuthenticationService.createTicketGrantingTicket(c);
76              getResponse().setStatus(determineStatus());
77              final Reference ticketReference = getRequest().getResourceRef().addSegment(ticketGrantingTicketId);
78              getResponse().setLocationRef(ticketReference);
79  
80              fmt = new Formatter();
81              fmt.format("<!DOCTYPE HTML PUBLIC \\\"-//IETF//DTD HTML 2.0//EN\\\"><html><head><title>");
82  
83              fmt.format("%s %s", getResponse().getStatus().getCode(), getResponse().getStatus().getDescription())
84                 .format("</title></head><body><h1>TGT Created</h1><form action=\"%s", ticketReference)
85                 .format("\" method=\"POST\">Service:<input type=\"text\" name=\"service\" value=\"\">")
86                 .format("<br><input type=\"submit\" value=\"Submit\"></form></body></html>");
87  
88              getResponse().setEntity(fmt.toString(), MediaType.TEXT_HTML);
89          } catch (final Exception e) {
90              LOGGER.error(e.getMessage(), e);
91              getResponse().setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
92          } finally {
93              IOUtils.closeQuietly(fmt);
94          }
95      }
96      /**
97       * Template method for determining which status to return on a successful ticket creation.
98       * This method exists for compatibility reasons with bad clients (i.e. Flash) that can't
99       * process 201 with a Location header.
100      *
101      * @return the status to return.
102      */
103     protected Status determineStatus() {
104         return Status.SUCCESS_CREATED;
105     }
106 
107     protected Credential obtainCredentials() {
108         final UsernamePasswordCredential c = new UsernamePasswordCredential();
109         final WebRequestDataBinder binder = new WebRequestDataBinder(c);
110         final RestletWebRequest webRequest = new RestletWebRequest(getRequest());
111 
112         logFormRequest(new Form(getRequest().getEntity()));
113         binder.bind(webRequest);
114 
115         return c;
116     }
117 
118     private void logFormRequest(final Form form) {
119         if (LOGGER.isDebugEnabled()) {
120             final Set<String> pairs = new HashSet<String>();
121             for (final String name : form.getNames()) {
122                 final StringBuilder builder = new StringBuilder();
123                 builder.append(name);
124                 builder.append(": ");
125                 if (!"password".equalsIgnoreCase(name)) {
126                     builder.append(form.getValues(name));
127                 } else {
128                     builder.append("*****");
129                 }
130                 pairs.add(builder.toString());
131             }
132             LOGGER.debug(StringUtils.join(pairs, ", "));
133         }
134     }
