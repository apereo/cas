80  /**
81   * Concrete implementation of a CentralAuthenticationService, and also the
82   * central, organizing component of CAS's internal implementation.
83   * <p>
84   * This class is threadsafe.
85   * </p>
86   * This class has the following properties that must be set:
87   * <ul>
88   * <li> <code>ticketRegistry</code> - The Ticket Registry to maintain the list
89   * of available tickets.</li>
90   * <li> <code>serviceTicketRegistry</code> - Provides an alternative to configure separate registries for
91   * TGTs and ST in order to store them in different locations (i.e. long term memory or short-term)</li>
92   * <li> <code>authenticationManager</code> - The service that will handle
93   * authentication.</li>
94   * <li> <code>ticketGrantingTicketUniqueTicketIdGenerator</code> - Plug in to
95   * generate unique secure ids for TicketGrantingTickets.</li>
96   * <li> <code>serviceTicketUniqueTicketIdGenerator</code> - Plug in to
97   * generate unique secure ids for ServiceTickets.</li>
98   * <li> <code>ticketGrantingTicketExpirationPolicy</code> - The expiration
99   * policy for TicketGrantingTickets.</li>
100  * <li> <code>serviceTicketExpirationPolicy</code> - The expiration policy for
101  * ServiceTickets.</li>
102  * </ul>
103  *
104  * @author William G. Thompson, Jr.
105  * @author Scott Battaglia
106  * @author Dmitry Kopylenko
107  * @since 3.0.0
108  */

209     /**
210      * {@inheritDoc}
211      * Destroy a TicketGrantingTicket and perform back channel logout. This has the effect of invalidating any
212      * Ticket that was derived from the TicketGrantingTicket being destroyed. May throw an
213      * {@link IllegalArgumentException} if the TicketGrantingTicket ID is null.
214      *
215      * @param ticketGrantingTicketId the id of the ticket we want to destroy
216      * @return the logout requests.
217      */
218     @Audit(
219             action="TICKET_GRANTING_TICKET_DESTROYED",
220             actionResolverName="DESTROY_TICKET_GRANTING_TICKET_RESOLVER",
221             resourceResolverName="DESTROY_TICKET_GRANTING_TICKET_RESOURCE_RESOLVER")
222     @Timed(name = "DESTROY_TICKET_GRANTING_TICKET_TIMER")
223     @Metered(name="DESTROY_TICKET_GRANTING_TICKET_METER")
224     @Counted(name="DESTROY_TICKET_GRANTING_TICKET_COUNTER", monotonic=true)
225     @Override
226     public List<LogoutRequest> destroyTicketGrantingTicket(@NotNull final String ticketGrantingTicketId) {
227         try {
228             logger.debug("Removing ticket [{}] from registry...", ticketGrantingTicketId);
229             final TicketGrantingTicket ticket = getTicket(ticketGrantingTicketId, TicketGrantingTicket.class);
230             logger.debug("Ticket found. Processing logout requests and then deleting the ticket...");
231             final List<LogoutRequest> logoutRequests = logoutManager.performLogout(ticket);
232             this.ticketRegistry.deleteTicket(ticketGrantingTicketId);
233             return logoutRequests;
234         } catch (final InvalidTicketException e) {
235             logger.debug("TicketGrantingTicket [{}] cannot be found in the ticket registry.", ticketGrantingTicketId);
236         }
237         return Collections.emptyList();
238     }

240     @Audit(
241         action="SERVICE_TICKET",
242         actionResolverName="GRANT_SERVICE_TICKET_RESOLVER",
243         resourceResolverName="GRANT_SERVICE_TICKET_RESOURCE_RESOLVER")
244     @Timed(name="GRANT_SERVICE_TICKET_TIMER")
245     @Metered(name="GRANT_SERVICE_TICKET_METER")
246     @Counted(name="GRANT_SERVICE_TICKET_COUNTER", monotonic=true)
247     @Override
248     public ServiceTicket grantServiceTicket(
249             final String ticketGrantingTicketId,
250             final Service service, final Credential... credentials)
251             throws AuthenticationException, TicketException {
252 
253         final TicketGrantingTicket ticketGrantingTicket = getTicket(ticketGrantingTicketId, TicketGrantingTicket.class);
254         final RegisteredService registeredService = this.servicesManager.findServiceBy(service);
255 
256         verifyRegisteredServiceProperties(registeredService, service);
257         final Set<Credential> sanitizedCredentials = sanitizeCredentials(credentials);
258 
259         Authentication currentAuthentication = null;
260         if (sanitizedCredentials.size() > 0) {
261             currentAuthentication = this.authenticationManager.authenticate(
262                     sanitizedCredentials.toArray(new Credential[] {}));
263             final Authentication original = ticketGrantingTicket.getAuthentication();
264             if (!currentAuthentication.getPrincipal().equals(original.getPrincipal())) {
265                 throw new MixedPrincipalException(
266                         currentAuthentication, currentAuthentication.getPrincipal(), original.getPrincipal());
267             }
268             ticketGrantingTicket.getSupplementalAuthentications().add(currentAuthentication);
269         }
270 
271         if (currentAuthentication == null && !registeredService.getAccessStrategy().isServiceAccessAllowedForSso()) {
272             logger.warn("ServiceManagement: Service [{}] is not allowed to use SSO.", service.getId());
273             throw new UnauthorizedSsoServiceException();
274         }
275 
276         final Service proxiedBy = ticketGrantingTicket.getProxiedBy();
277         if (proxiedBy != null) {
278             logger.debug("TGT is proxied by [{}]. Locating proxy service in registry...", proxiedBy.getId());
279             final RegisteredService proxyingService = servicesManager.findServiceBy(proxiedBy);
280 
281             if (proxyingService != null) {
282                 logger.debug("Located proxying service [{}] in the service registry", proxyingService);
283                 if (!proxyingService.getProxyPolicy().isAllowedToProxy()) {
284                     logger.warn("Found proxying service {}, but it is not authorized to fulfill the proxy attempt made by {}",
285                             proxyingService.getId(), service.getId());
286                     throw new UnauthorizedProxyingException("Proxying is not allowed for registered service "
287                             + registeredService.getId());
288                 }
289             } else {
290                 logger.warn("No proxying service found. Proxy attempt by service [{}] (registered service [{}]) is not allowed.",
291                         service.getId(), registeredService.getId());
292                 throw new UnauthorizedProxyingException("Proxying is not allowed for registered service "
293                         + registeredService.getId());
294             }
295         } else {
296             logger.trace("TGT is not proxied by another service");
297         }
298 
299         // Perform security policy check by getting the authentication that satisfies the configured policy
300         // This throws if no suitable policy is found
301         getAuthenticationSatisfiedByPolicy(ticketGrantingTicket, new ServiceContext(service, registeredService));
302 
303         final List<Authentication> authentications = ticketGrantingTicket.getChainedAuthentications();
304         final Principal principal = authentications.get(authentications.size() - 1).getPrincipal();
305 
306         final Map<String, Object> principalAttrs = registeredService.getAttributeReleasePolicy().getAttributes(principal);
307         if (!registeredService.getAccessStrategy().doPrincipalAttributesAllowServiceAccess(principalAttrs)) {
308             logger.warn("ServiceManagement: Cannot grant service ticket because Service [{}] is not authorized for use by [{}].",
309                     service.getId(), principal);
310             throw new UnauthorizedServiceForPrincipalException();
311         }
312 
313         final String uniqueTicketIdGenKey = service.getClass().getName();
314         logger.debug("Looking up service ticket id generator for [{}]", uniqueTicketIdGenKey);
315         UniqueTicketIdGenerator serviceTicketUniqueTicketIdGenerator =
316                 this.uniqueTicketIdGeneratorsForService.get(uniqueTicketIdGenKey);
317         if (serviceTicketUniqueTicketIdGenerator == null) {
318             serviceTicketUniqueTicketIdGenerator = this.defaultServiceTicketIdGenerator;
319             logger.debug("Service ticket id generator not found for [{}]. Using the default generator...",
320                     uniqueTicketIdGenKey);
321         }
322 
323         final String ticketPrefix = authentications.size() == 1 ? ServiceTicket.PREFIX : ServiceTicket.PROXY_TICKET_PREFIX;
324         final String ticketId = serviceTicketUniqueTicketIdGenerator.getNewTicketId(ticketPrefix);
325         final ServiceTicket serviceTicket = ticketGrantingTicket.grantServiceTicket(
326                 ticketId,
327                 service,
328                 this.serviceTicketExpirationPolicy,
329                 currentAuthentication != null);
330 
331         this.serviceTicketRegistry.addTicket(serviceTicket);
332 
333         logger.info("Granted ticket [{}] for service [{}] for user [{}]",
334                 serviceTicket.getId(), service.getId(), principal.getId());
335 
336         return serviceTicket;
337     }

339     /**
340      * Attempts to sanitize the array of credentials by removing
341      * all null elements from the collection.
342      * @param credentials credentials to sanitize
343      * @return a set of credentials with no null values
344      */
345     private static Set<Credential> sanitizeCredentials(final Credential[] credentials) {
346         if (credentials != null && credentials.length > 0) {
347             final Set<Credential> set = new HashSet<>(Arrays.asList(credentials));
348             final Iterator<Credential> it = set.iterator();
349             while (it.hasNext()) {
350                 if (it.next() == null) {
351                     it.remove();
352                 }
353             }
354             return set;
355         }
356         return Collections.emptySet();
357     }

376     @Audit(
377         action="PROXY_GRANTING_TICKET",
378         actionResolverName="GRANT_PROXY_GRANTING_TICKET_RESOLVER",
379         resourceResolverName="GRANT_PROXY_GRANTING_TICKET_RESOURCE_RESOLVER")
380     @Timed(name="GRANT_PROXY_GRANTING_TICKET_TIMER")
381     @Metered(name="GRANT_PROXY_GRANTING_TICKET_METER")
382     @Counted(name="GRANT_PROXY_GRANTING_TICKET_COUNTER", monotonic=true)
383     @Override
384     public TicketGrantingTicket delegateTicketGrantingTicket(final String serviceTicketId, final Credential... credentials)
385             throws AuthenticationException, TicketException {
386 
387         final ServiceTicket serviceTicket =  this.serviceTicketRegistry.getTicket(serviceTicketId, ServiceTicket.class);
388 
389         if (serviceTicket == null || serviceTicket.isExpired()) {
390             logger.debug("ServiceTicket [{}] has expired or cannot be found in the ticket registry", serviceTicketId);
391             throw new InvalidTicketException(serviceTicketId);
392         }
393 
394         final RegisteredService registeredService = this.servicesManager
395                 .findServiceBy(serviceTicket.getService());
396 
397         verifyRegisteredServiceProperties(registeredService, serviceTicket.getService());
398         
399         if (!registeredService.getProxyPolicy().isAllowedToProxy()) {
400             logger.warn("ServiceManagement: Service [{}] attempted to proxy, but is not allowed.", serviceTicket.getService().getId());
401             throw new UnauthorizedProxyingException();
402         }
403 
404         final Authentication authentication = this.authenticationManager.authenticate(credentials);
405 
406         final String pgtId = this.ticketGrantingTicketUniqueTicketIdGenerator.getNewTicketId(
407                 TicketGrantingTicket.PROXY_GRANTING_TICKET_PREFIX);
408         final TicketGrantingTicket proxyGrantingTicket = serviceTicket.grantTicketGrantingTicket(pgtId,
409                                     authentication, this.ticketGrantingTicketExpirationPolicy);
410 
411         logger.debug("Generated proxy granting ticket [{}] based off of [{}]", proxyGrantingTicket, serviceTicketId);
412         this.ticketRegistry.addTicket(proxyGrantingTicket);
413 
414         return proxyGrantingTicket;
415     }

417     @Audit(
418         action="SERVICE_TICKET_VALIDATE",
419         actionResolverName="VALIDATE_SERVICE_TICKET_RESOLVER",
420         resourceResolverName="VALIDATE_SERVICE_TICKET_RESOURCE_RESOLVER")
421     @Timed(name="VALIDATE_SERVICE_TICKET_TIMER")
422     @Metered(name="VALIDATE_SERVICE_TICKET_METER")
423     @Counted(name="VALIDATE_SERVICE_TICKET_COUNTER", monotonic=true)
424     @Override
425     public Assertion validateServiceTicket(final String serviceTicketId, final Service service) throws TicketException {
426         final RegisteredService registeredService = this.servicesManager.findServiceBy(service);
427         verifyRegisteredServiceProperties(registeredService, service);
428 
429         final ServiceTicket serviceTicket =  this.serviceTicketRegistry.getTicket(serviceTicketId, ServiceTicket.class);
430 
431         if (serviceTicket == null) {
432             logger.info("Service ticket [{}] does not exist.", serviceTicketId);
433             throw new InvalidTicketException(serviceTicketId);
434         }
435 
436         try {
437             synchronized (serviceTicket) {
438                 if (serviceTicket.isExpired()) {
439                     logger.info("ServiceTicket [{}] has expired.", serviceTicketId);
440                     throw new InvalidTicketException(serviceTicketId);
441                 }
442 
443                 if (!serviceTicket.isValidFor(service)) {
444                     logger.error("Service ticket [{}] with service [{}] does not match supplied service [{}]",
445                             serviceTicketId, serviceTicket.getService().getId(), service);
446                     throw new UnrecognizableServiceForServiceTicketValidationException(serviceTicket.getService());
447                 }
448             }
449 
450             final TicketGrantingTicket root = serviceTicket.getGrantingTicket().getRoot();
451             final Authentication authentication = getAuthenticationSatisfiedByPolicy(
452                     root, new ServiceContext(serviceTicket.getService(), registeredService));
453             final Principal principal = authentication.getPrincipal();
454 
455             final AttributeReleasePolicy attributePolicy = registeredService.getAttributeReleasePolicy();
456             logger.debug("Attribute policy [{}] is associated with service [{}]", attributePolicy, registeredService);
457             
458             @SuppressWarnings("unchecked")
459             final Map<String, Object> attributesToRelease = attributePolicy != null
460                     ? attributePolicy.getAttributes(principal) : Collections.EMPTY_MAP;
461             
462             final String principalId = registeredService.getUsernameAttributeProvider().resolveUsername(principal, service);
463             final Principal modifiedPrincipal = this.principalFactory.createPrincipal(principalId, attributesToRelease);
464             final AuthenticationBuilder builder = DefaultAuthenticationBuilder.newInstance(authentication);
465             builder.setPrincipal(modifiedPrincipal);
466 
467             return new ImmutableAssertion(
468                     builder.build(),
469                     serviceTicket.getGrantingTicket().getChainedAuthentications(),
470                     serviceTicket.getService(),
471                     serviceTicket.isFromNewLogin());
472         } finally {
473             if (serviceTicket.isExpired()) {
474                 this.serviceTicketRegistry.deleteTicket(serviceTicketId);
475             }
476         }
477     }

479     @Audit(
480         action="TICKET_GRANTING_TICKET",
481         actionResolverName="CREATE_TICKET_GRANTING_TICKET_RESOLVER",
482         resourceResolverName="CREATE_TICKET_GRANTING_TICKET_RESOURCE_RESOLVER")
483     @Timed(name = "CREATE_TICKET_GRANTING_TICKET_TIMER")
484     @Metered(name = "CREATE_TICKET_GRANTING_TICKET_METER")
485     @Counted(name="CREATE_TICKET_GRANTING_TICKET_COUNTER", monotonic=true)
486     @Override
487     public TicketGrantingTicket createTicketGrantingTicket(final Credential... credentials)
488             throws AuthenticationException, TicketException {
489 
490         final Set<Credential> sanitizedCredentials = sanitizeCredentials(credentials);
491         if (sanitizedCredentials.size() > 0) {
492             final Authentication authentication = this.authenticationManager.authenticate(credentials);
493 
494             final TicketGrantingTicket ticketGrantingTicket = new TicketGrantingTicketImpl(
495                     this.ticketGrantingTicketUniqueTicketIdGenerator
496                             .getNewTicketId(TicketGrantingTicket.PREFIX),
497                     authentication, this.ticketGrantingTicketExpirationPolicy);
498 
499             this.ticketRegistry.addTicket(ticketGrantingTicket);
500             return ticketGrantingTicket;
501         }
502         final String msg = "No credentials were specified in the request for creating a new ticket-granting ticket";
503         logger.warn(msg);
504         throw new TicketCreationException(new IllegalArgumentException(msg));
505     }

507     /**
508      * {@inheritDoc}
509      */
510     @Timed(name = "GET_TICKET_TIMER")
511     @Metered(name = "GET_TICKET_METER")
512     @Counted(name="GET_TICKET_COUNTER", monotonic=true)
513     @Override
514     public <T extends Ticket> T getTicket(final String ticketId, final Class<? extends Ticket> clazz)
515             throws InvalidTicketException {
516         Assert.notNull(ticketId, "ticketId cannot be null");
517         final Ticket ticket = this.ticketRegistry.getTicket(ticketId, clazz);
518 
519         if (ticket == null) {
520             logger.debug("Ticket [{}] by type [{}] cannot be found in the ticket registry.", ticketId, clazz.getSimpleName());
521             throw new InvalidTicketException(ticketId);
522         }
523 
524         if (ticket instanceof TicketGrantingTicket) {
525             synchronized (ticket) {
526                 if (ticket.isExpired()) {
527                     this.ticketRegistry.deleteTicket(ticketId);
528                     logger.debug("Ticket [{}] has expired and is now deleted from the ticket registry.", ticketId);
529                     throw new InvalidTicketException(ticketId);
530                 }
531             }
532         }
533         return (T) ticket;
534     }

537      */
538     @Timed(name = "GET_TICKETS_TIMER")
539     @Metered(name = "GET_TICKETS_METER")
540     @Counted(name="GET_TICKETS_COUNTER", monotonic=true)
541     @Override
542     public Collection<Ticket> getTickets(final Predicate predicate) {
543         final Collection<Ticket> c = new HashSet<>(this.ticketRegistry.getTickets());
544         final Iterator<Ticket> it = c.iterator();
545         while (it.hasNext()) {
546             if (!predicate.evaluate(it.next())) {
547                 it.remove();
548             }
549         }
550         return c;
551     }

593     /**
594      * Gets the authentication satisfied by policy.
595      *
596      * @param ticket the ticket
597      * @param context the context
598      * @return the authentication satisfied by policy
599      * @throws org.jasig.cas.ticket.TicketException the ticket exception
600      */
601     private Authentication getAuthenticationSatisfiedByPolicy(
602             final TicketGrantingTicket ticket, final ServiceContext context) throws TicketException {
603 
604         final ContextualAuthenticationPolicy<ServiceContext> policy =
605                 serviceContextAuthenticationPolicyFactory.createPolicy(context);
606         if (policy.isSatisfiedBy(ticket.getAuthentication())) {
607             return ticket.getAuthentication();
608         }
609         for (final Authentication auth : ticket.getSupplementalAuthentications()) {
610             if (policy.isSatisfiedBy(auth)) {
611                 return auth;
612             }
613         }
614         throw new UnsatisfiedAuthenticationPolicyException(policy);
615     }

617     /**
618      * Ensure that the service is found and enabled in the service registry.
619      * @param registeredService the located entry in the registry
620      * @param service authenticating service
621      * @throws UnauthorizedServiceException
622      */
623     private void verifyRegisteredServiceProperties(final RegisteredService registeredService, final Service service) {
624         if (registeredService == null) {
625             final String msg = String.format("ServiceManagement: Unauthorized Service Access. "
626                     + "Service [%s] is not found in service registry.", service.getId());
627             logger.warn(msg);
628             throw new UnauthorizedServiceException(UnauthorizedServiceException.CODE_UNAUTHZ_SERVICE, msg);
629         }
630         if (!registeredService.getAccessStrategy().isServiceAccessAllowed()) {
631             final String msg = String.format("ServiceManagement: Unauthorized Service Access. "
632                     + "Service [%s] is not enabled in service registry.", service.getId());
633 
634             logger.warn(msg);
635             throw new UnauthorizedServiceException(UnauthorizedServiceException.CODE_UNAUTHZ_SERVICE, msg);
636         }
637     }
