20  package org.jasig.cas.adaptors.x509.authentication.handler.support.ldap;
21  
22  import java.security.cert.CertificateException;
23  import java.security.cert.X509CRL;
24  import javax.validation.constraints.NotNull;
25  import org.jasig.cas.adaptors.x509.authentication.handler.support.ResourceCRLFetcher;
26  import org.jasig.cas.util.CompressionUtils;
27  import org.ldaptive.ConnectionConfig;
28  import org.ldaptive.ConnectionFactory;
29  import org.ldaptive.DefaultConnectionFactory;
30  import org.ldaptive.LdapAttribute;
31  import org.ldaptive.LdapEntry;
32  import org.ldaptive.LdapException;
33  import org.ldaptive.Response;
34  import org.ldaptive.ResultCode;
35  import org.ldaptive.SearchExecutor;
36  import org.ldaptive.SearchResult;
37  import org.springframework.core.io.ByteArrayResource;
38  
39  /**
40   * Fetches a CRL from an LDAP instance.
41   * @author Daniel Fisher
42   * @since 4.1
43   */
44  public class LdaptiveResourceCRLFetcher extends ResourceCRLFetcher {
45  
46      /** Search exec that looks for the attribute. */
47      protected final SearchExecutor searchExecutor;
48  
49      /** The connection config to prep for connections. **/
50      protected final ConnectionConfig connectionConfig;
51  
52      /**
53       * Instantiates a new Ldap resource cRL fetcher.
54  
55       * @param connectionConfig the connection configuration
56       * @param searchExecutor the search executor
57       */
58      public LdaptiveResourceCRLFetcher(
59              @NotNull final ConnectionConfig connectionConfig,
60              @NotNull final SearchExecutor searchExecutor) {
61          this.connectionConfig = connectionConfig;
62          this.searchExecutor = searchExecutor;
63      }
64  
65      @Override
66      protected X509CRL fetchInternal(final Object r) throws Exception {
67          if (r.toString().toLowerCase().startsWith("ldap")) {
68              return fetchCRLFromLdap(r);
69          }
70          return super.fetchInternal(r);
71      }
72  
73      /**
74       * Downloads a CRL from given LDAP url.
75       *
76       * @param r the resource that is the ldap url.
77       * @return the x 509 cRL
78       * @throws Exception if connection to ldap fails, or attribute to get the revocation list is unavailable
79       */
80      protected X509CRL fetchCRLFromLdap(final Object r) throws Exception {
81          try {
82              final String ldapURL = r.toString();
83              logger.debug("Fetching CRL from ldap {}", ldapURL);
84  
85              final Response<SearchResult> result = performLdapSearch(ldapURL);
86              if (result.getResultCode() == ResultCode.SUCCESS) {
87                  final LdapEntry entry = result.getResult().getEntry();
88                  final LdapAttribute attribute = entry.getAttribute();
89  
90                  logger.debug("Located entry [{}]. Retrieving first attribute [{}]",
91                          entry, attribute);
92                  return fetchX509CRLFromAttribute(attribute);
93              } else {
94                  logger.debug("Failed to execute the search [{}]", result);
95              }
96  
97              throw new CertificateException("Failed to establish a connection ldap and search.");
98  
99          } catch (final LdapException e) {
100             logger.error(e.getMessage(), e);
101             throw new CertificateException(e);
102         }
103     }
104 
105 
106     /**
107      * Gets x509 cRL from attribute. Retrieves the binary attribute value,
108      * decodes it to base64, and fetches it as a byte-array resource.
109      *
110      * @param aval the attribute, which may be null if it's not found
111      * @return the x 509 cRL from attribute
112      * @throws Exception if attribute not found or could could not be decoded.
113      */
114     protected X509CRL fetchX509CRLFromAttribute(final LdapAttribute aval) throws Exception {
115         if (aval != null) {
116             final byte[] val = aval.getBinaryValue();
117             if (val == null || val.length == 0) {
118                 throw new CertificateException("Empty attribute. Can not download CRL from ldap");
119             }
120             final byte[] decoded64 = CompressionUtils.decodeBase64ToByteArray(val);
121             if (decoded64 == null) {
122                 throw new CertificateException("Could not decode the attribute value to base64");
123             }
124             logger.debug("Retrieved CRL from ldap as byte array decoded in base64. Fetching...");
125             return super.fetch(new ByteArrayResource(decoded64));
126         }
127         throw new CertificateException("Attribute not found. Can not retrieve CRL");
128     }
129 
130     /**
131      * Executes an LDAP search against the supplied URL.
132      *
133      * @param ldapURL to search
134      * @return search result
135      * @throws LdapException if an error occurs performing the search
136      */
137     protected Response<SearchResult> performLdapSearch(final String ldapURL) throws LdapException {
138         final ConnectionFactory connectionFactory = prepareConnectionFactory(ldapURL);
139         return this.searchExecutor.search(connectionFactory);
140     }
141 
142     /**
143      * Prepare a new LDAP connection.
144      *
145      * @param ldapURL the ldap uRL
146      * @return connection factory
147      */
148     protected ConnectionFactory prepareConnectionFactory(final String ldapURL) {
149         final ConnectionConfig cc = ConnectionConfig.newConnectionConfig(this.connectionConfig);
150         cc.setLdapUrl(ldapURL);
151         return new DefaultConnectionFactory(cc);
152     }
153 }
