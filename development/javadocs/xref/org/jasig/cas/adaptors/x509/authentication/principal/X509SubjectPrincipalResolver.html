19  package org.jasig.cas.adaptors.x509.authentication.principal;

21  import java.security.cert.X509Certificate;
22  import java.util.HashMap;
23  import java.util.Map;
24  import java.util.regex.Matcher;
25  import java.util.regex.Pattern;
26  
27  import javax.validation.constraints.NotNull;
28  
29  import edu.vt.middleware.crypt.x509.DNUtils;
30  import edu.vt.middleware.crypt.x509.types.AttributeType;
31  
32  
33  /**
34   * Credential to principal resolver that extracts one or more attribute values
35   * from the certificate subject DN and combines them with intervening delimiters.
36   *
37   * @author Marvin S. Addison
38   * @since 3.4.4
39   *
40   */
41  public class X509SubjectPrincipalResolver extends AbstractX509PrincipalResolver {
42  
43      /** Pattern used to extract attribute names from descriptor. */
44      private static final Pattern ATTR_PATTERN = Pattern.compile("\\$(\\w+)");
45  
46      /** Descriptor representing an abstract format of the principal to be resolved.*/
47      @NotNull
48      private String descriptor;
49  
50      /**
51       * Sets the descriptor that describes for format of the principal ID to
52       * create from X.509 subject DN attributes.  The descriptor is made up of
53       * common X.509 attribute names prefixed by "$", which are replaced by
54       * attribute values extracted from DN attribute values.
55       * <p>
56       * EXAMPLE:
57       * </p>
58       * {@code
59       * <code>
60       * &lt;bean class="org.jasig.cas.adaptors.x509.authentication.principal.X509SubjectPrincipalResolver"
61       *   p:descriptor="$UID@$DC.$DC" /&gt;
62       * </code>
63       * }
64       *
65       * The above bean when applied to a certificate with the DN
66       * <p>
67       * <b>DC=edu, DC=vt/UID=jacky, CN=Jascarnella Ellagwonto</b></p>
68       * <p>
69       * produces the principal <strong>jacky@vt.edu</strong>.</p>
70       *
71       * @param s Descriptor string where attribute names are prefixed with "$"
72       * to identify replacement by real attribute values from the subject DN.
73       * Valid attributes include common X.509 DN attributes such as the following:
74       * <ul>
75       *  <li>C</li>
76       *  <li>CN</li>
77       *  <li>DC</li>
78       *  <li>EMAILADDRESS</li>
79       *  <li>L</li>
80       *  <li>O</li>
81       *  <li>OU</li>
82       *  <li>SERIALNUMBER</li>
83       *  <li>ST</li>
84       *  <li>UID</li>
85       *  <li>UNIQUEIDENTIFIER</li>
86       * </ul>
87       * For a complete list of supported attributes, see
88       * {@link edu.vt.middleware.crypt.x509.types.AttributeType}.
89       *
90       */
91      public void setDescriptor(final String s) {
92          this.descriptor = s;
93      }
94  
95      /**
96       * Replaces placeholders in the descriptor with values extracted from attribute
97       * values in relative distinguished name components of the DN.
98       *
99       * @param certificate X.509 certificate credential.
100      * @return Resolved principal ID.
101      * @see AbstractX509PrincipalResolver#resolvePrincipalInternal(java.security.cert.X509Certificate)
102      */
103     @Override
104     protected String resolvePrincipalInternal(final X509Certificate certificate) {
105         logger.debug("Resolving principal for {}", certificate);
106         final StringBuffer sb = new StringBuffer();
107         final Matcher m = ATTR_PATTERN.matcher(this.descriptor);
108         final Map<String, AttributeContext> attrMap = new HashMap<>();
109         String name;
110         String[] values;
111         AttributeContext context;
112         while (m.find()) {
113             name = m.group(1);
114             if (!attrMap.containsKey(name)) {
115                 values = DNUtils.getAttributeValues(
116                         certificate.getSubjectX500Principal(),
117                         AttributeType.fromName(name));
118                 attrMap.put(name, new AttributeContext(name, values));
119             }
120             context = attrMap.get(name);
121             m.appendReplacement(sb, context.nextValue());
122         }
123         m.appendTail(sb);
124         return sb.toString();
125     }
126 
127     private static final class AttributeContext {
128         private int currentIndex;
129         private String name;
130         private final String[] values;
131 
132         /**
133          * Instantiates a new attribute context.
134          *
135          * @param name the name
136          * @param values the values
137          */
138         public AttributeContext(final String name, final String[] values) {
139             this.values = values;
140         }
141 
142   
143         /**
144          * Retrieve the next value, by incrementing the current index.
145          *
146          * @return the string
147          * @throws IllegalStateException if no values are remaining.
148          */
149         public String nextValue() {
150             if (this.currentIndex == this.values.length) {
151                 throw new IllegalStateException("No values remaining for attribute " + this.name);
152             }
153             return this.values[this.currentIndex++];
154         }
155     }
156 }
