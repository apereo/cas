37  /**
38   * Credential to principal resolver that extracts Subject Alternative Name UPN extension
39   * from the provided certificate if available as a resolved principal id.
40   *
41   * @author Dmitriy Kopylenko
42   * @since 4.1
43   */
44  public class X509SubjectAlternativeNameUPNPrincipalResolver extends AbstractX509PrincipalResolver {
45  
46      /**
47       * ObjectID for upn altName for windows smart card logon.
48       */
49      public static final String UPN_OBJECTID = "1.3.6.1.4.1.311.20.2.3";
50  
51      /**
52       * Retrieves Subject Alternative Name UPN extension as a principal id String.
53       *
54       * @param certificate X.509 certificate credential.
55       *
56       * @return Resolved principal ID or null if no SAN UPN extension is available in provided certificate.
57       *
58       * @see AbstractX509PrincipalResolver#resolvePrincipalInternal(java.security.cert.X509Certificate)
59       * @see <a href="http://docs.oracle.com/javase/7/docs/api/java/security/cert/X509Certificate.html#getSubjectAlternativeNames()">
60       *     X509Certificate#getSubjectAlternativeNames</a>
61       */
62      @Override
63      protected String resolvePrincipalInternal(final X509Certificate certificate) {
64          logger.debug("Resolving principal from Subject Alternative Name UPN for {}", certificate);
65          try {
66              final Collection<List<?>> subjectAltNames = certificate.getSubjectAlternativeNames();
67              if (subjectAltNames != null) {
68                  for (final List<?> sanItem : subjectAltNames) {
69                      final ASN1Sequence seq = getAltnameSequence(sanItem);
70                      final String upnString = getUPNStringFromSequence(seq);
71                      if (upnString != null) {
72                          return upnString;
73                      }
74                  }
75              }
76          } catch (final CertificateParsingException e) {
77              logger.error("Error is encountered while trying to retrieve subject alternative names collection from certificate", e);
78              logger.debug("Returning null principal id...");
79              return null;
80          }
81          logger.debug("Returning null principal id...");
82          return null;
83      }
84  
85      /**
86       * Get UPN String.
87       *
88       * @param seq ASN1Sequence abstraction representing subject alternative name.
89       * First element is the object identifier, second is the object itself.
90       *
91       * @return UPN string or null
92       */
93      private String getUPNStringFromSequence(final ASN1Sequence seq) {
94          if (seq != null) {
95              // First in sequence is the object identifier, that we must check
96              final DERObjectIdentifier id = DERObjectIdentifier.getInstance(seq.getObjectAt(0));
97              if (id.getId().equals(UPN_OBJECTID)) {
98                  final ASN1TaggedObject obj = (ASN1TaggedObject) seq.getObjectAt(1);
99                  final DERUTF8String str = DERUTF8String.getInstance(obj.getObject());
100                 return str.getString();
101             }
102         }
103         return null;
104     }
105 
106     /**
107      * Get alt name seq.
108      *
109      * @param sanItem subject alternative name value encoded as a two elements List with elem(0) representing object id and elem(1)
110      * representing object (subject alternative name) itself.
111      *
112      * @return ASN1Sequence abstraction representing subject alternative name or null if the passed in
113      * List doesn't contain at least to elements
114      * as expected to be returned by implementation of {@code X509Certificate.html#getSubjectAlternativeNames}
115      *
116      * @see <a href="http://docs.oracle.com/javase/7/docs/api/java/security/cert/X509Certificate.html#getSubjectAlternativeNames()">
117      *     X509Certificate#getSubjectAlternativeNames</a>
118      */
119     private ASN1Sequence getAltnameSequence(final List sanItem) {
120         //Should not be the case, but still, a extra "safety" check
121         if (sanItem.size() < 2) {
122             logger.error("Subject Alternative Name List does not contain at least two required elements. Returning null principal id...");
123         }
124         final Integer itemType = (Integer) sanItem.get(0);
125         if (itemType == 0) {
126             final byte[] altName = (byte[]) sanItem.get(1);
127             return getAltnameSequence(altName);
128         }
129         return null;
130     }
131 
132     /**
133      * Get alt name seq.
134      *
135      * @param sanValue subject alternative name value encoded as byte[]
136      *
137      * @return ASN1Sequence abstraction representing subject alternative name
138      *
139      * @see <a href="http://docs.oracle.com/javase/7/docs/api/java/security/cert/X509Certificate.html#getSubjectAlternativeNames()">
140      *     X509Certificate#getSubjectAlternativeNames</a>
141      */
142     private ASN1Sequence getAltnameSequence(final byte[] sanValue) {
143         DERObject oct = null;
144         ASN1InputStream input = null;
145         ByteArrayInputStream bInput = null;
146         try {
147             bInput = new ByteArrayInputStream(sanValue);
148             input = new ASN1InputStream(bInput);
149             oct = input.readObject();
150         } catch (final IOException e) {
151             logger.error("Error on getting Alt Name as a DERSEquence: {}", e.getMessage(), e);
152         } finally {
153             IOUtils.closeQuietly(bInput);
154             IOUtils.closeQuietly(input);
155         }
156         //It is OK to pass null DERObject to this method (in case of handled IOException). null will be returned
157         return ASN1Sequence.getInstance(oct);
158     }
