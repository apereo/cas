19  package org.jasig.cas.adaptors.jdbc;
20  
21  import org.jasig.cas.authentication.HandlerResult;
22  import org.jasig.cas.authentication.PreventedException;
23  import org.jasig.cas.authentication.UsernamePasswordCredential;
24  import org.springframework.beans.factory.InitializingBean;
25  import org.springframework.dao.DataAccessException;
26  
27  import javax.security.auth.login.FailedLoginException;
28  import javax.validation.constraints.NotNull;
29  import java.security.GeneralSecurityException;
30  
31  /**
32   * Class that given a table, username field and password field will query a
33   * database table with the provided encryption technique to see if the user
34   * exists. This class defaults to a PasswordTranslator of
35   * PlainTextPasswordTranslator.
36   *
37   * @author Scott Battaglia
38   * @author Dmitriy Kopylenko
39   * @author Marvin S. Addison
40   *
41   * @since 3.0.0
42   */
43  
44  public class SearchModeSearchDatabaseAuthenticationHandler extends AbstractJdbcUsernamePasswordAuthenticationHandler
45          implements InitializingBean {
46  
47      private static final String SQL_PREFIX = "Select count('x') from ";
48  
49      @NotNull
50      private String fieldUser;
51  
52      @NotNull
53      private String fieldPassword;
54  
55      @NotNull
56      private String tableUsers;
57  
58      private String sql;
59  
60      /**
61       * {@inheritDoc}
62       */
63      @Override
64      protected final HandlerResult authenticateUsernamePasswordInternal(final UsernamePasswordCredential credential)
65              throws GeneralSecurityException, PreventedException {
66  
67          final String username = credential.getUsername();
68          final String encyptedPassword = getPasswordEncoder().encode(credential.getPassword());
69          final int count;
70          try {
71              count = getJdbcTemplate().queryForObject(this.sql, Integer.class, username, encyptedPassword);
72          } catch (final DataAccessException e) {
73              throw new PreventedException("SQL exception while executing query for " + username, e);
74          }
75          if (count == 0) {
76              throw new FailedLoginException(username + " not found with SQL query.");
77          }
78          return createHandlerResult(credential, this.principalFactory.createPrincipal(username), null);
79      }
80  
81      @Override
82      public void afterPropertiesSet() throws Exception {
83          this.sql = SQL_PREFIX + this.tableUsers + " WHERE " + this.fieldUser + " = ? AND " + this.fieldPassword
84                  + " = ?";
85      }
86  
87      /**
88       * @param fieldPassword The fieldPassword to set.
89       */
90      public final void setFieldPassword(final String fieldPassword) {
91          this.fieldPassword = fieldPassword;
92      }
93  
94      /**
95       * @param fieldUser The fieldUser to set.
96       */
97      public final void setFieldUser(final String fieldUser) {
98          this.fieldUser = fieldUser;
99      }
100 
101     /**
102      * @param tableUsers The tableUsers to set.
103      */
104     public final void setTableUsers(final String tableUsers) {
105         this.tableUsers = tableUsers;
106     }
107 
108 }
