41  /**
42   * A JDBC querying handler that will pull back the password and
43   * the private salt value for a user and validate the encoded
44   * password using the public salt value. Assumes everything
45   * is inside the same database table. Supports settings for
46   * number of iterations as well as private salt.
47   *
48   * <p>
49   * This handler uses the hashing method defined by Apache Shiro's
50   * {@link org.apache.shiro.crypto.hash.DefaultHashService}. Refer to the Javadocs
51   * to learn more about the behavior. If the hashing behavior and/or configuration
52   * of private and public salts does nto meet your needs, a extension can be developed
53   * to specify alternative methods of encoding and digestion of the encoded password.
54   * </p>
55   * @author Misagh Moayyed
56   * @author Charles Hasegawa (mailto:chasegawa@unicon.net)
57   */

104     /**
105      * Instantiates a new Query and encode database authentication handler.
106      *
107      * @param datasource The database datasource
108      * @param sql the sql query to execute which must include a parameter placeholder
109      *            for the userid. (i.e. <code>SELECT * FROM table WHERE username = ?</code>
110      * @param algorithmName the algorithm name (i.e. <code>MessageDigestAlgorithms.SHA_512</code>)
111      * @see {@link org.apache.commons.codec.digest.MessageDigestAlgorithms}
112      */
113     public QueryAndEncodeDatabaseAuthenticationHandler(final DataSource datasource,
114                                                        final String sql,
115                                                        final String algorithmName) {
116         super();
117         setDataSource(datasource);
118         this.sql = sql;
119         this.algorithmName = algorithmName;
120     }

122     @Override
123     protected final HandlerResult authenticateUsernamePasswordInternal(final UsernamePasswordCredential transformedCredential)
124             throws GeneralSecurityException, PreventedException {
125         final String username = getPrincipalNameTransformer().transform(transformedCredential.getUsername());
126         final String encodedPsw = this.getPasswordEncoder().encode(transformedCredential.getPassword());
127 
128         try {
129             final Map<String, Object> values = getJdbcTemplate().queryForMap(this.sql, username);
130             final String digestedPassword = digestEncodedPassword(encodedPsw, values);
131 
132             if (!values.get(this.passwordFieldName).equals(digestedPassword)) {
133                 throw new FailedLoginException("Password does not match value on record.");
134             }
135             return createHandlerResult(transformedCredential, new SimplePrincipal(username), null);
136 
137         } catch (final IncorrectResultSizeDataAccessException e) {
138             if (e.getActualSize() == 0) {
139                 throw new AccountNotFoundException(username + " not found with SQL query");
140             } else {
141                 throw new FailedLoginException("Multiple records found for " + username);
142             }
143         } catch (final DataAccessException e) {
144             throw new PreventedException("SQL exception while executing query for " + username, e);
145         }
146 
147     }

149     /**
150      * Digest encoded password.
151      *
152      * @param encodedPassword the encoded password
153      * @param values the values retrieved from database
154      * @return the digested password
155      */
156     protected String digestEncodedPassword(final String encodedPassword, final Map<String, Object> values) {
157         final ConfigurableHashService hashService = new DefaultHashService();
158 
159         if (StringUtils.isNotBlank(this.staticSalt)) {
160             hashService.setPrivateSalt(ByteSource.Util.bytes(this.staticSalt));
161         }
162         hashService.setHashAlgorithmName(this.algorithmName);
163 
164         Long numOfIterations = this.numberOfIterations;
165         if (values.containsKey(this.numberOfIterationsFieldName)) {
166             final String longAsStr = values.get(this.numberOfIterationsFieldName).toString();
167             numOfIterations = Long.valueOf(longAsStr);
168         }
169 
170         hashService.setHashIterations(numOfIterations.intValue());
171         if (!values.containsKey(this.saltFieldName)) {
172             throw new RuntimeException("Specified field name for salt does not exist in the results");
173         }
174 
175         final String dynaSalt = values.get(this.saltFieldName).toString();
176         final HashRequest request = new HashRequest.Builder()
177                                     .setSalt(dynaSalt)
178                                     .setSource(encodedPassword)
179                                     .build();
180         return hashService.computeHash(request).toHex();
181     }

183     /**
184      * Sets static/private salt to be combined with the dynamic salt retrieved
185      * from the database. Optional.
186      *
187      * <p>If using this implementation as part of a password hashing strategy,
188      * it might be desirable to configure a private salt.
189      * A hash and the salt used to compute it are often stored together.
190      * If an attacker is ever able to access the hash (e.g. during password cracking)
191      * and it has the full salt value, the attacker has all of the input necessary
192      * to try to brute-force crack the hash (source + complete salt).</p>
193      *
194      * <p>However, if part of the salt is not available to the attacker (because it is not stored with the hash),
195      * it is much harder to crack the hash value since the attacker does not have the complete inputs necessary.
196      * The privateSalt property exists to satisfy this private-and-not-shared part of the salt.</p>
197      * <p>If you configure this attribute, you can obtain this additional very important safety feature.</p>
198      * @param staticSalt the static salt
199      */
200     public final void setStaticSalt(final String staticSalt) {
201         this.staticSalt = staticSalt;
202     }
