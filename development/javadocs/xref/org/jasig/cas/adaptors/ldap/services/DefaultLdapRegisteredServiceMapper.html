43  /**
44   * Default implementation of {@link LdapRegisteredServiceMapper} that is able
45   * to map ldap entries to {@link RegisteredService} instances based on
46   * certain attributes names. This implementation also respects the object class
47   * attribute of LDAP entries via {@link LdapUtils#OBJECTCLASS_ATTRIBUTE}.
48   * @author Misagh Moayyed
49   * @since 4.1.0
50   */
51  public final class DefaultLdapRegisteredServiceMapper implements LdapRegisteredServiceMapper {
52  
53      private static final Logger LOGGER = LoggerFactory.getLogger(DefaultLdapRegisteredServiceMapper.class);
54  
55      @NotNull
56      private JsonSerializer<RegisteredService> jsonSerializer = new RegisteredServiceJsonSerializer();
57  
58      @NotNull
59      private String objectClass = "casRegisteredService";
60  
61      @NotNull
62      private String idAttribute = "uid";
63  
64      @NotNull
65      private String serviceDefinitionAttribute = "description";
66  
67      @Override
68      public LdapEntry mapFromRegisteredService(final String dn, final RegisteredService svc) {
69          try {
70              if (svc.getId() == RegisteredService.INITIAL_IDENTIFIER_VALUE) {
71                  ((AbstractRegisteredService) svc).setId(System.nanoTime());
72              }
73              final String newDn = getDnForRegisteredService(dn, svc);
74              LOGGER.debug("Creating entry {}", newDn);
75  
76              final Collection<LdapAttribute> attrs = new ArrayList<>();
77              attrs.add(new LdapAttribute(this.idAttribute, String.valueOf(svc.getId())));
78  
79              final StringWriter writer = new StringWriter();
80              this.jsonSerializer.toJson(writer, svc);
81              attrs.add(new LdapAttribute(this.serviceDefinitionAttribute, writer.toString()));
82              attrs.add(new LdapAttribute(LdapUtils.OBJECTCLASS_ATTRIBUTE, "top", this.objectClass));
83  
84              return new LdapEntry(newDn, attrs);
85          } catch (final Exception e) {
86              throw new RuntimeException(e);
87          }
88      }
89  
90      @Override
91      public RegisteredService mapToRegisteredService(final LdapEntry entry) {
92          try {
93              final String value = LdapUtils.getString(entry, this.serviceDefinitionAttribute);
94              if (StringUtils.hasText(value)) {
95                  final RegisteredService service = this.jsonSerializer.fromJson(value);
96                  return service;
97              }
98  
99              return null;
100         } catch (final Exception e) {
101             throw new RuntimeException(e);
102         }
103     }
104 
105     public String getObjectClass() {
106         return this.objectClass;
107     }
108 
109     public void setObjectClass(final String objectClass) {
110         this.objectClass = objectClass;
111     }
112 
113     public String getIdAttribute() {
114         return this.idAttribute;
115     }
116 
117     public void setIdAttribute(final String idAttribute) {
118         this.idAttribute = idAttribute;
119     }
120 
121     public String getServiceDefinitionAttribute() {
122         return serviceDefinitionAttribute;
123     }
124 
125     public void setServiceDefinitionAttribute(final String serviceDefinitionAttribute) {
126         this.serviceDefinitionAttribute = serviceDefinitionAttribute;
127     }
128 
129     public void setJsonSerializer(final JsonSerializer<RegisteredService> jsonSerializer) {
130         this.jsonSerializer = jsonSerializer;
131     }
132 
133     @Override
134     public String getDnForRegisteredService(final String parentDn, final RegisteredService svc) {
135         return String.format("%s=%s,%s", this.idAttribute, svc.getId(), parentDn);
136     }
137 
138     /**
139      * Checks if is valid regex pattern.
140      *
141      * @param pattern the pattern
142      * @return true, if  valid regex pattern
143      */
144     private boolean isValidRegexPattern(final String pattern) {
145         try {
146             Pattern.compile(pattern);
147         } catch (final PatternSyntaxException e) {
148             LOGGER.debug("Failed to identify [{}] as a regular expression", pattern);
149             return false;
150         }
151         return true;
152     }
153 
154     /**
155      * Gets the attribute values if more than one, otherwise an empty list.
156      *
157      * @param entry the entry
158      * @param attrName the attr name
159      * @return the collection of attribute values
160      */
161     private Collection<String> getMultiValuedAttributeValues(@NotNull final LdapEntry entry, @NotNull final String attrName) {
162         final LdapAttribute attrs = entry.getAttribute(attrName);
163         if (attrs != null) {
164             return attrs.getStringValues();
165         }
166         return Collections.emptyList();
167     }
168 
169     /**
170      * Gets the registered service by id that would either match an ant or regex pattern.
171      *
172      * @param id the id
173      * @return the registered service
174      */
175     private AbstractRegisteredService getRegisteredService(@NotNull final String id) {
176         if (isValidRegexPattern(id)) {
177             return new RegexRegisteredService();
178         }
179 
180         if (new AntPathMatcher().isPattern(id)) {
181             return new RegisteredServiceImpl();
182         }
183         return null;
184     }
