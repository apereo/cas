108     @Override
109     public RadiusResponse authenticate(final String username, final String password) throws PreventedException {
110 
111         final AttributeList attributeList = new AttributeList();
112         
113         attributeList.add(new Attr_UserName(username));
114         attributeList.add(new Attr_UserPassword(password));
115 
116         if (StringUtils.isNotBlank(this.nasIpAddress)) {
117             attributeList.add(new Attr_NASIPAddress(this.nasIpAddress));
118         }
119         if (StringUtils.isNotBlank(this.nasIpv6Address)) {
120             attributeList.add(new Attr_NASIPv6Address(this.nasIpv6Address));
121         }
122 
123         if (this.nasPort != -1) {
124             attributeList.add(new Attr_NASPort(this.nasPort));
125         }
126         if (this.nasPortId != -1) {
127             attributeList.add(new Attr_NASPortId(this.nasPortId));
128         }
129         if (this.nasIdentifier != -1) {
130             attributeList.add(new Attr_NASIdentifier(this.nasIdentifier));
131         }
132         if (this.nasRealPort != -1) {
133             attributeList.add(new Attr_NASRealPort(this.nasRealPort));
134         }
135         if (this.nasPortType != -1) {
136             attributeList.add(new Attr_NASPortType(this.nasPortType));
137         }
138         
139         RadiusClient client = null;
140         try {
141             client = this.radiusClientFactory.newInstance();
142             final AccessRequest request = new AccessRequest(client, attributeList);
143             final RadiusPacket response = client.authenticate(
144                     request,
145                     RadiusClient.getAuthProtocol(this.protocol.getName()),
146                     this.retries);
147 
148             LOGGER.debug("RADIUS response from {}: {}",
149                     client.getRemoteInetAddress().getCanonicalHostName(),
150                     response.getClass().getName());
151 
152             if (response instanceof AccessAccept) {
153                 final AccessAccept acceptedResponse = (AccessAccept) response;
154                
155                 return new RadiusResponse(acceptedResponse.getCode(),
156                         acceptedResponse.getIdentifier(),
157                         acceptedResponse.getAttributes().getAttributeList());
158             }
159         } catch (final Exception e) {
160             throw new PreventedException(e);            
161         } finally {
162             if (client != null) {
163                 client.close();
164             }
165         }
166         return null;
167     }
