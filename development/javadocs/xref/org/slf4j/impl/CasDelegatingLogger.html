34  /**
35   * The CAS logger wrapper, which uses a substitute logger to route the logs.
36   * This component only exists to intercept logging calls before they are
37   * sent to the logging engine (log4j, etc) and serves to manipulate
38   * logging messages if needed, such as removing sensitive ticket id from
39   * the log message.
40   * @author Misagh Moayyed
41   * @since 4.1
42   */

47      private static final Pattern TICKET_ID_PATTERN = Pattern.compile("(" + TicketGrantingTicket.PREFIX + "|"
48              + TicketGrantingTicket.PROXY_GRANTING_TICKET_PREFIX
49              + ")(-)*(\\w)*(-)*(\\w)*");
50  
51      /**
52       * Specifies the ending tail length of the ticket id that would still be visible in the output
53       * for troubleshooting purposes.
54       */
55      private static final int VISIBLE_ID_TAIL_LENGTH = 10;
56  
57      private final Logger delegate;

68      /**
69       * Manipulate the log message. For now, removes ticket ids from the log.
70       * @param msg log message
71       * @return message to log
72       */
73      private String manipulateLogMessage(final String msg) {
74          return removeTicketId(msg);
75      }
76  
77      /**
78       * Manipulate the log arguments. For now, removes ticket ids from the log.
79       * @param args log args
80       * @return sanitized arguments
81       */
82      private Object[] manipulateLogArguments(final Object... args) {
83          for (int i = 0; i < args.length; i++) {
84              if (args[i] != null) {
85                  args[i] = removeTicketId(args[i].toString());
86              }
87          }
88          return args;
89      }
90  
91      /**
92       * Remove ticket id from the log message.
93       *
94       * @param msg the message
95       * @return the modified message with tgt id removed
96       */
97      private String removeTicketId(final String msg) {
98          final StringBuilder builder = new StringBuilder(msg.length());
99  
100         String modifiedMessage = msg;
101 
102         final Matcher matcher = TICKET_ID_PATTERN.matcher(msg);
103         while (matcher.find()) {
104             final String match = matcher.group();
105             final String newId = matcher.group(1) + "-"
106                     + StringUtils.repeat("*", match.length() - VISIBLE_ID_TAIL_LENGTH)
107                     + StringUtils.right(match, VISIBLE_ID_TAIL_LENGTH);
108 
109             modifiedMessage = modifiedMessage.replaceAll(match, newId);
110         }
111         return modifiedMessage;
112     }

114     /**
115      * Gets exception to log.
116      *
117      * @param msg the msg
118      * @param t the t
119      * @return the exception to log
120      */
121     private String getExceptionToLog(final String msg, final Throwable t) {
122         final StringWriter sW = new StringWriter();
123         final PrintWriter w = new PrintWriter(sW);
124         w.println(manipulateLogMessage(msg));
125         t.printStackTrace(w);
126 
127         final String log = sW.getBuffer().toString();
128         return manipulateLogMessage(log);
129     }
