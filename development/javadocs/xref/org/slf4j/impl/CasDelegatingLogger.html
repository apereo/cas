33  /**
34   * The CAS logger wrapper, which uses a substitute logger to route the logs.
35   * This component only exists to intercept logging calls before they are
36   * sent to the logging engine (log4j, etc) and serves to manipulate
37   * logging messages if needed, such as removing sensitive ticket id from
38   * the log message.
39   * @author Misagh Moayyed
40   * @since 4.1.0
41   */
42  public final class CasDelegatingLogger extends MarkerIgnoringBase implements Serializable {
43  
44      private static final long serialVersionUID = 6182834493563598289L;
45  
46      private static final Pattern TICKET_ID_PATTERN = Pattern.compile("(" + TicketGrantingTicket.PREFIX + "|"
47              + TicketGrantingTicket.PROXY_GRANTING_TICKET_IOU_PREFIX + "|" + TicketGrantingTicket.PROXY_GRANTING_TICKET_PREFIX
48              + ")(-)*(\\w)*(-)*(\\w)*");
49  
50      /**
51       * Specifies the ending tail length of the ticket id that would still be visible in the output
52       * for troubleshooting purposes.
53       */
54      private static final int VISIBLE_ID_TAIL_LENGTH = 10;
55  
56      private final Logger delegate;
57

75      /**
76       * Manipulate the log message. For now, removes ticket ids from the log.
77       * @param msg log message
78       * @return message to log
79       */
80      private String manipulateLogMessage(final String msg) {
81          return removeTicketId(msg);
82      }
83  
84      /**
85       * Manipulate the log arguments. For now, removes ticket ids from the log.
86       * @param args log args
87       * @return sanitized arguments
88       */
89      private Object[] manipulateLogArguments(final Object... args) {
90          for (int i = 0; i < args.length; i++) {
91              if (args[i] != null) {
92                  args[i] = removeTicketId(args[i].toString());
93              }
94          }
95          return args;
96      }
97  
98      /**
99       * Remove ticket id from the log message.
100      *
101      * @param msg the message
102      * @return the modified message with tgt id removed
103      */
104     private String removeTicketId(final String msg) {
105         String modifiedMessage = msg;
106 
107         if (StringUtils.isNotBlank(msg)) {
108             final Matcher matcher = TICKET_ID_PATTERN.matcher(msg);
109             while (matcher.find()) {
110                 final String match = matcher.group();
111                 final String newId = matcher.group(1) + '-'
112                         + StringUtils.repeat("*", match.length() - VISIBLE_ID_TAIL_LENGTH)
113                         + StringUtils.right(match, VISIBLE_ID_TAIL_LENGTH);
114 
115                 modifiedMessage = modifiedMessage.replaceAll(match, newId);
116             }
117         }
118         return modifiedMessage;
119     }
120 
121     /**
122      * Gets exception to log.
123      *
124      * @param msg the error msg
125      * @param t the exception to log. May be null if
126      *          the underlying call did not specify an inner exception.
127      * @return the exception message to log
128      */
129     private String getExceptionToLog(final String msg, final Throwable t) {
130         final StringWriter sW = new StringWriter();
131         final PrintWriter w = new PrintWriter(sW);
132         w.println(manipulateLogMessage(msg));
133         if (t != null) {
134             t.printStackTrace(w);
135         }
136 
137         final String log = sW.getBuffer().toString();
138         return manipulateLogMessage(log);
139     }
