7   public final class DuoWeb {
8       private static final String DUO_PREFIX = "TX";
9       private static final String APP_PREFIX = "APP";
10      private static final String AUTH_PREFIX = "AUTH";
11  
12      private static final int DUO_EXPIRE = 300;
13      private static final int APP_EXPIRE = 3600;
14  
15      private static final int IKEY_LEN = 20;
16      private static final int SKEY_LEN = 40;
17      private static final int AKEY_LEN = 40;
18  
19      public static final String ERR_USER = "ERR|The username passed to sign_request() is invalid.";
20      public static final String ERR_IKEY = "ERR|The Duo integration key passed to sign_request() is invalid.";
21      public static final String ERR_SKEY = "ERR|The Duo secret key passed to sign_request() is invalid.";
22      public static final String ERR_AKEY = "ERR|The application secret key passed to sign_request() must be at least " + AKEY_LEN + " characters.";
23      public static final String ERR_UNKNOWN = "ERR|An unknown error has occurred.";
24  
25      public static String signRequest(final String ikey, final String skey, final String akey, final String username) {
26          return signRequest(ikey, skey, akey, username, System.currentTimeMillis() / 1000);
27      }

29      public static String signRequest(final String ikey, final String skey, final String akey, final String username, final long time) {
30          final String duo_sig;
31          final String app_sig;
32  
33          if (username.equals("")) {
34              return ERR_USER;
35          }
36          if (username.indexOf('|') != -1) {
37              return ERR_USER;
38          }
39          if (ikey.equals("") || ikey.length() != IKEY_LEN) {
40              return ERR_IKEY;
41          }
42          if (skey.equals("") || skey.length() != SKEY_LEN) {
43              return ERR_SKEY;
44          }
45          if (akey.equals("") || akey.length() < AKEY_LEN) {
46              return ERR_AKEY;
47          }
48  
49          try {
50              duo_sig = signVals(skey, username, ikey, DUO_PREFIX, DUO_EXPIRE, time);
51              app_sig = signVals(akey, username, ikey, APP_PREFIX, APP_EXPIRE, time);
52          } catch (Exception e) {
53              return ERR_UNKNOWN;
54          }
55  
56          return duo_sig + ":" + app_sig;
57      }

59      public static String verifyResponse(final String ikey, final String skey, final String akey, final String sig_response)
60          throws DuoWebException, NoSuchAlgorithmException, InvalidKeyException, IOException {
61          return verifyResponse(ikey, skey, akey, sig_response, System.currentTimeMillis() / 1000);
62      }

64      public static String verifyResponse(final String ikey, final String skey, final String akey, final String sig_response, final long time)
65          throws DuoWebException, NoSuchAlgorithmException, InvalidKeyException, IOException {
66          String auth_user = null;
67          String app_user = null;
68  
69          final String[] sigs = sig_response.split(":");
70          final String auth_sig = sigs[0];
71          final String app_sig = sigs[1];
72  
73          auth_user = parseVals(skey, auth_sig, AUTH_PREFIX, ikey, time);
74          app_user = parseVals(akey, app_sig, APP_PREFIX, ikey, time);
75  
76          if (!auth_user.equals(app_user)) {
77              throw new DuoWebException("Authentication failed.");
78          }
79  
80          return auth_user;
81      }

83      private static String signVals(final String key, final String username, final String ikey, final String prefix, final int expire, final long time)
84          throws InvalidKeyException, NoSuchAlgorithmException {
85          final long expire_ts = time + expire;
86          final String exp = Long.toString(expire_ts);
87  
88          final String val = username + "|" + ikey + "|" + exp;
89          final String cookie = prefix + "|" + Base64.encodeBytes(val.getBytes());
90          final String sig = Util.hmacSign(key, cookie);
91  
92          return cookie + "|" + sig;
93      }
94  
95      private static String parseVals(final String key, final String val, final String prefix, final String ikey, final long time)
96          throws InvalidKeyException, NoSuchAlgorithmException, IOException, DuoWebException {
97  
98          final String[] parts = val.split("\\|");
99          if (parts.length != 3) {
100             throw new DuoWebException("Invalid response");
101         }
102 
103         final String u_prefix = parts[0];
104         final String u_b64 = parts[1];
105         final String u_sig = parts[2];
106 
107         final String sig = Util.hmacSign(key, u_prefix + "|" + u_b64);
108         if (!Util.hmacSign(key, sig).equals(Util.hmacSign(key, u_sig))) {
109             throw new DuoWebException("Invalid response");
110         }
111 
112         if (!u_prefix.equals(prefix)) {
113             throw new DuoWebException("Invalid response");
114         }
115 
116         final byte[] decoded = Base64.decode(u_b64);
117         final String cookie = new String(decoded);
118 
119         final String[] cookie_parts = cookie.split("\\|");
120         if (cookie_parts.length != 3) {
121             throw new DuoWebException("Invalid response");
122         }
123         final String username = cookie_parts[0];
124         final String u_ikey = cookie_parts[1];
125         final String expire = cookie_parts[2];
126 
127         if (!u_ikey.equals(ikey)) {
128             throw new DuoWebException("Invalid response");
129         }
130 
131         final long expire_ts = Long.parseLong(expire);
132         if (time >= expire_ts) {
133             throw new DuoWebException("Transaction has expired. Please check that the system time is correct.");
134         }
135 
136         return username;
137     }
