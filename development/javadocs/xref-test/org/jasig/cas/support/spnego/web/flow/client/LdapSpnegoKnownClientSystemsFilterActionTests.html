42  /**
43   * Test cases for {@link LdapSpnegoKnownClientSystemsFilterAction}.
44   * @author Misagh Moayyed
45   * @since 4.1
46   */
47  @RunWith(SpringJUnit4ClassRunner.class)
48  @ContextConfiguration({"/ldap-context.xml"})
49  public class LdapSpnegoKnownClientSystemsFilterActionTests extends AbstractLdapTests {
50  
51      @Autowired
52      @Qualifier("provisioningConnectionFactory")
53      private ConnectionFactory connectionFactory;
54  
55      @Autowired
56      private SearchRequest searchRequest;
57  
58      @BeforeClass
59      public static void bootstrap() throws Exception {
60          initDirectoryServer();
61      }
62  
63      @Test
64      public void ensureLdapAttributeShouldDoSpnego() {
65          final LdapSpnegoKnownClientSystemsFilterAction action =
66                  new LdapSpnegoKnownClientSystemsFilterAction(this.connectionFactory,
67                  this.searchRequest, "mail");
68          final MockRequestContext ctx = new MockRequestContext();
69          final MockHttpServletRequest req = new MockHttpServletRequest();
70          req.setRemoteAddr("127.0.0.1");
71          final ServletExternalContext extCtx = new ServletExternalContext(
72                  new MockServletContext(), req,
73                  new MockHttpServletResponse());
74          ctx.setExternalContext(extCtx);
75  
76          final Event ev = action.doExecute(ctx);
77          assertEquals(ev.getId(), new EventFactorySupport().yes(this).getId());
78      }
79  }
