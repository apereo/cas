42  /**
43   * Tests the {@link ClientAuthenticationHandler}.
44   *
45   * @author Jerome Leleu
46   * @since 4.1
47   *
48   */
49  public final class ClientAuthenticationHandlerTests {
50  
51      private static final String CALLBACK_URL = "http://localhost:8080/callback";
52      private static final String ID = "123456789";
53      
54      private MockFacebookClient fbClient;
55  
56      private ClientAuthenticationHandler handler;
57      
58      private ClientCredential clientCredential;
59  
60      @Before
61      public void setUp() {
62          this.fbClient = new MockFacebookClient();
63          final Clients clients = new Clients(CALLBACK_URL, fbClient);
64          this.handler = new ClientAuthenticationHandler(clients);
65          final Credentials credentials = new OAuthCredentials(null, MockFacebookClient.CLIENT_NAME);
66          this.clientCredential = new ClientCredential(credentials);
67          ExternalContextHolder.setExternalContext(mock(ServletExternalContext.class));
68      }
69      
70      @Test
71      public void testOk() throws GeneralSecurityException, PreventedException {
72          final FacebookProfile facebookProfile = new FacebookProfile();
73          facebookProfile.setId(ID);
74          this.fbClient.setFacebookProfile(facebookProfile);
75          final HandlerResult result = this.handler.authenticate(this.clientCredential);
76          final Principal principal = result.getPrincipal();
77          assertEquals(FacebookProfile.class.getSimpleName() + "#" + ID, principal.getId());
78      }
79  
80      @Test
81      public void testOkWithSimpleIdentifier() throws GeneralSecurityException, PreventedException {
82          this.handler.setTypedIdUsed(false);
83          final FacebookProfile facebookProfile = new FacebookProfile();
84          facebookProfile.setId(ID);
85          this.fbClient.setFacebookProfile(facebookProfile);
86          final HandlerResult result = this.handler.authenticate(this.clientCredential);
87          final Principal principal = result.getPrincipal();
88          assertEquals(ID, principal.getId());
89      }
90  
91      @Test(expected = FailedLoginException.class)
92      public void testNoProfile() throws GeneralSecurityException, PreventedException {
93          this.handler.authenticate(this.clientCredential);
94      }
95  }
