47  public class SamlAuthenticationMetaDataPopulatorTests {
48  
49      private SamlAuthenticationMetaDataPopulator populator;
50  
51      @Before
52      public void setUp() throws Exception {
53          this.populator = new SamlAuthenticationMetaDataPopulator();
54      }
55  
56      @Test
57      public void verifyAuthenticationTypeFound() {
58          final UsernamePasswordCredential credentials = new UsernamePasswordCredential();
59          final AuthenticationBuilder builder = newAuthenticationBuilder(TestUtils.getPrincipal());
60          this.populator.populateAttributes(builder, credentials);
61          final Authentication auth = builder.build();
62  
63          assertEquals(
64                  auth.getAttributes().get(SamlAuthenticationMetaDataPopulator.ATTRIBUTE_AUTHENTICATION_METHOD),
65                  SamlAuthenticationMetaDataPopulator.AUTHN_METHOD_PASSWORD);
66      }
67  
68      @Test
69      public void verifyAuthenticationTypeNotFound() {
70          final CustomCredential credentials = new CustomCredential();
71          final AuthenticationBuilder builder = newAuthenticationBuilder(TestUtils.getPrincipal());
72          this.populator.populateAttributes(builder, credentials);
73          final Authentication auth = builder.build();
74  
75          assertNull(auth.getAttributes().get(SamlAuthenticationMetaDataPopulator.ATTRIBUTE_AUTHENTICATION_METHOD));
76      }
77  
78      @Test
79      public void verifyAuthenticationTypeFoundCustom() {
80          final CustomCredential credentials = new CustomCredential();
81  
82          final Map<String, String> added = new HashMap<>();
83          added.put(CustomCredential.class.getName(), "FF");
84  
85          this.populator.setUserDefinedMappings(added);
86  
87          final AuthenticationBuilder builder = newAuthenticationBuilder(TestUtils.getPrincipal());
88          this.populator.populateAttributes(builder, credentials);
89          final Authentication auth = builder.build();
90  
91          assertEquals(
92                  "FF",
93                  auth.getAttributes().get(SamlAuthenticationMetaDataPopulator.ATTRIBUTE_AUTHENTICATION_METHOD));
94      }
95  
96      private static class CustomCredential implements Credential {
97  
98          public String getId() {
99              return "nobody";
100         }
101     }
102 
103     private static AuthenticationBuilder newAuthenticationBuilder(final Principal principal) {
104         final CredentialMetaData meta = new BasicCredentialMetaData(new UsernamePasswordCredential());
105         final AuthenticationHandler handler = new SimpleTestUsernamePasswordAuthenticationHandler();
106         return new DefaultAuthenticationBuilder(principal)
107                 .addCredential(meta)
108                 .addSuccess("test", new DefaultHandlerResult(handler, meta));
109     }
110 }
