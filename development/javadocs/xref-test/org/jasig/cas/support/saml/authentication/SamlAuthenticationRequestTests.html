20  package org.jasig.cas.support.saml.authentication;
21  
22  import org.jasig.cas.support.saml.AbstractOpenSamlTests;
23  import org.jasig.cas.support.saml.util.AbstractSaml20ObjectBuilder;
24  import org.jasig.cas.support.saml.util.GoogleSaml20ObjectBuilder;
25  import org.jasig.cas.util.CompressionUtils;
26  import org.junit.Test;
27  
28  import java.io.ByteArrayOutputStream;
29  import java.io.IOException;
30  import java.util.zip.DeflaterOutputStream;
31  
32  import static org.junit.Assert.*;
33  
34  /**
35   * Utility class to ensure authentication requests are properly encoded and decoded.
36   * @author Misagh Moayyed
37   * @since 4.1
38   */
39  public class SamlAuthenticationRequestTests extends AbstractOpenSamlTests {
40      private static final String SAML_REQUEST = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
41              + "<samlp:AuthnRequest xmlns:samlp=\"urn:oasis:names:tc:SAML:2.0:protocol\" "
42              + "ID=\"5545454455\" Version=\"2.0\" IssueInstant=\"Value\" "
43              + "ProtocolBinding=\"urn:oasis:names.tc:SAML:2.0:bindings:HTTP-Redirect\" "
44              + "ProviderName=\"https://localhost:8443/myRutgers\" "
45              + "AssertionConsumerServiceURL=\"https://localhost:8443/myRutgers\"/>";
46  
47      @Test
48      public void ensureDeflation() throws Exception {
49          final String deflator = CompressionUtils.deflate(SAML_REQUEST);
50          final String deflatorStream = deflateViaStream(SAML_REQUEST);
51          assertEquals(deflatorStream, deflator);
52      }
53  
54      @Test
55      public void ensureInflation() throws Exception {
56          final String deflator = CompressionUtils.deflate(SAML_REQUEST);
57          final AbstractSaml20ObjectBuilder builder = new GoogleSaml20ObjectBuilder();
58          final String msg = builder.decodeSamlAuthnRequest(deflator);
59          assertEquals(msg, SAML_REQUEST);
60      }
61  
62      private String deflateViaStream(final String samlRequest) throws IOException {
63          final byte[] xmlBytes = samlRequest.getBytes("UTF-8");
64          final ByteArrayOutputStream byteOutputStream = new ByteArrayOutputStream();
65          final DeflaterOutputStream deflaterOutputStream = new DeflaterOutputStream(
66                  byteOutputStream);
67          deflaterOutputStream.write(xmlBytes, 0, xmlBytes.length);
68          deflaterOutputStream.close();
69          return CompressionUtils.encodeBase64(byteOutputStream.toByteArray());
70      }
71  }
