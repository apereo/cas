20  package org.jasig.cas.support.oauth.services;
21  
22  import org.apache.commons.io.FileUtils;
23  import org.jasig.cas.services.AbstractRegisteredService;
24  import org.jasig.cas.services.JsonServiceRegistryDao;
25  import org.jasig.cas.services.RegisteredService;
26  import org.jasig.cas.services.ServiceRegistryDao;
27  import org.junit.BeforeClass;
28  import org.junit.Test;
29  import org.springframework.core.io.ClassPathResource;
30  
31  import static org.junit.Assert.*;
32  
33  /**
34   * @author Misagh Moayyed
35   * @since 4.1
36   */
37  public class OAuthRegisteredServiceTests {
38  
39      private static final ClassPathResource RESOURCE = new ClassPathResource("services");
40  
41      private final ServiceRegistryDao dao;
42  
43      public OAuthRegisteredServiceTests() throws Exception {
44          this.dao = new JsonServiceRegistryDao(RESOURCE.getFile());
45      }
46  
47      @BeforeClass
48      public static void prepTests() throws Exception {
49          FileUtils.cleanDirectory(RESOURCE.getFile());
50      }
51  
52      @Test
53      public void checkCloning() {
54          final AbstractRegisteredService r = new OAuthRegisteredService();
55          r.setName("checkCloning");
56          r.setServiceId("testId");
57          r.setTheme("theme");
58          r.setDescription("description");
59  
60          final OAuthRegisteredService r2 = (OAuthRegisteredService) r.clone();
61          assertEquals(r, r2);
62      }
63  
64      @Test
65      public void checkSaveMethod() {
66          final OAuthRegisteredService r = new OAuthRegisteredService();
67          r.setName("checkSaveMethod");
68          r.setServiceId("testId");
69          r.setTheme("theme");
70          r.setDescription("description");
71          r.setClientId("clientid");
72          r.setServiceId("secret");
73          r.setBypassApprovalPrompt(true);
74          final RegisteredService r2 = this.dao.save(r);
75          assertTrue(r2 instanceof OAuthRegisteredService);
76          this.dao.load();
77          final RegisteredService r3 = this.dao.findServiceById(r2.getId());
78          assertTrue(r3 instanceof OAuthRegisteredService);
79          assertEquals(r, r2);
80          assertEquals(r2, r3);
81      }
82  }
