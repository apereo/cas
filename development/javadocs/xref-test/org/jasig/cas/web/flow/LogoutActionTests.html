47  /**
48   * @author Scott Battaglia
49   * @since 3.0.0

51  public class LogoutActionTests extends AbstractCentralAuthenticationServiceTest {
52  
53      private static final String COOKIE_TGC_ID = "CASTGC";
54  
55      private LogoutAction logoutAction;
56  
57      private CookieRetrievingCookieGenerator warnCookieGenerator;
58  
59      private CookieRetrievingCookieGenerator ticketGrantingTicketCookieGenerator;
60  
61      private InMemoryServiceRegistryDaoImpl serviceRegistryDao;
62  
63      private DefaultServicesManagerImpl serviceManager;
64  
65      private MockHttpServletRequest request;
66  
67      private MockHttpServletResponse response;
68  
69      private RequestContext requestContext;
70  
71      @Before
72      public void onSetUp() throws Exception {
73          this.request = new MockHttpServletRequest();
74          this.response = new MockHttpServletResponse();
75          this.requestContext = mock(RequestContext.class);
76          final ServletExternalContext servletExternalContext = mock(ServletExternalContext.class);
77          when(this.requestContext.getExternalContext()).thenReturn(servletExternalContext);
78          when(servletExternalContext.getNativeRequest()).thenReturn(request);
79          when(servletExternalContext.getNativeResponse()).thenReturn(response);
80          final LocalAttributeMap flowScope = new LocalAttributeMap();
81          when(this.requestContext.getFlowScope()).thenReturn(flowScope);
82  
83          this.warnCookieGenerator = new CookieRetrievingCookieGenerator();
84          this.serviceRegistryDao = new InMemoryServiceRegistryDaoImpl();
85          this.serviceManager = new DefaultServicesManagerImpl(serviceRegistryDao);
86          this.serviceManager.reload();
87  
88          this.warnCookieGenerator.setCookieName("test");
89  
90          this.ticketGrantingTicketCookieGenerator = new CookieRetrievingCookieGenerator();
91          this.ticketGrantingTicketCookieGenerator.setCookieName(COOKIE_TGC_ID);
92  
93          this.logoutAction = new LogoutAction();
94          this.logoutAction.setServicesManager(this.serviceManager);
95      }
96  
97      @Test
98      public void verifyLogoutNoCookie() throws Exception {
99          final Event event = this.logoutAction.doExecute(this.requestContext);
100         assertEquals(LogoutAction.FINISH_EVENT, event.getId());
101     }
102 
103     @Test
104     public void verifyLogoutForServiceWithFollowRedirectsAndMatchingService() throws Exception {
105         this.request.addParameter("service", "TestService");
106         final RegisteredServiceImpl impl = new RegisteredServiceImpl();
107         impl.setServiceId("TestService");
108         impl.setName("TestService");
109         this.serviceManager.save(impl);
110         this.logoutAction.setFollowServiceRedirects(true);
111         final Event event = this.logoutAction.doExecute(this.requestContext);
112         assertEquals(LogoutAction.FINISH_EVENT, event.getId());
113         assertEquals("TestService", this.requestContext.getFlowScope().get("logoutRedirectUrl"));
114     }
115 
116     @Test
117     public void logoutForServiceWithNoFollowRedirects() throws Exception {
118         this.request.addParameter("service", "TestService");
119         this.logoutAction.setFollowServiceRedirects(false);
120         final Event event = this.logoutAction.doExecute(this.requestContext);
121         assertEquals(LogoutAction.FINISH_EVENT, event.getId());
122         assertNull(this.requestContext.getFlowScope().get("logoutRedirectUrl"));
123     }
124 
125     @Test
126     public void logoutForServiceWithFollowRedirectsNoAllowedService() throws Exception {
127         this.request.addParameter("service", "TestService");
128         final RegisteredServiceImpl impl = new RegisteredServiceImpl();
129         impl.setServiceId("http://FooBar");
130         impl.setName("FooBar");
131         this.serviceManager.save(impl);
132         this.logoutAction.setFollowServiceRedirects(true);
133         final Event event = this.logoutAction.doExecute(this.requestContext);
134         assertEquals(LogoutAction.FINISH_EVENT, event.getId());
135         assertNull(this.requestContext.getFlowScope().get("logoutRedirectUrl"));
136     }
137 
138     @Test
139     public void verifyLogoutCookie() throws Exception {
140         final Cookie cookie = new Cookie(COOKIE_TGC_ID, "test");
141         this.request.setCookies(cookie);
142         final Event event = this.logoutAction.doExecute(this.requestContext);
143         assertEquals(LogoutAction.FINISH_EVENT, event.getId());
144     }
145 
146     @Test
147     public void verifyLogoutRequestBack() throws Exception {
148         final Cookie cookie = new Cookie(COOKIE_TGC_ID, "test");
149         this.request.setCookies(cookie);
150         final LogoutRequest logoutRequest = new LogoutRequest("", null, null);
151         logoutRequest.setStatus(LogoutRequestStatus.SUCCESS);
152         WebUtils.putLogoutRequests(this.requestContext, Arrays.asList(logoutRequest));
153         final Event event = this.logoutAction.doExecute(this.requestContext);
154         assertEquals(LogoutAction.FINISH_EVENT, event.getId());
155     }
156 
157     @SuppressWarnings("unchecked")
158     @Test
159     public void verifyLogoutRequestFront() throws Exception {
160         final Cookie cookie = new Cookie(COOKIE_TGC_ID, "test");
161         this.request.setCookies(cookie);
162         final LogoutRequest logoutRequest = new LogoutRequest("", null, null);
163         WebUtils.putLogoutRequests(this.requestContext, Arrays.asList(logoutRequest));
164         final Event event = this.logoutAction.doExecute(this.requestContext);
165         assertEquals(LogoutAction.FRONT_EVENT, event.getId());
166         final List<LogoutRequest> logoutRequests = WebUtils.getLogoutRequests(this.requestContext);
167         assertEquals(1, logoutRequests.size());
168         assertEquals(logoutRequest, logoutRequests.get(0));
169     }
