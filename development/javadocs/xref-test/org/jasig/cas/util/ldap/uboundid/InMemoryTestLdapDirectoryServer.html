43  /**
44   * @author Misagh Moayyed
45   * @since 4.1.0
46   */
47  public final class InMemoryTestLdapDirectoryServer implements Closeable {
48      private static final Logger LOGGER = LoggerFactory.getLogger(InMemoryTestLdapDirectoryServer.class);
49  
50      private final InMemoryDirectoryServer directoryServer;
51  
52      private Collection<LdapEntry> ldapEntries;
53  
54      /**
55       * Instantiates a new Ldap directory server.
56       */
57      public InMemoryTestLdapDirectoryServer(final File properties, final File ldifFile, final File... schemaFile) {
58  
59          try {
60  
61              final Properties p = new Properties();
62              p.load(new FileInputStream(properties));
63  
64              final InMemoryDirectoryServerConfig config =
65                      new InMemoryDirectoryServerConfig(p.getProperty("ldap.rootDn"));
66              config.addAdditionalBindCredentials(p.getProperty("ldap.managerDn"), p.getProperty("ldap.managerPassword"));
67  
68              final String serverKeyStorePath = new ClassPathResource("/ldapServerTrustStore").getFile().getCanonicalPath();
69              final SSLUtil serverSSLUtil = new SSLUtil(
70                      new KeyStoreKeyManager(serverKeyStorePath, "changeit".toCharArray()), new TrustStoreTrustManager(serverKeyStorePath));
71              final SSLUtil clientSSLUtil = new SSLUtil(new TrustStoreTrustManager(serverKeyStorePath));
72              config.setListenerConfigs(
73                      InMemoryListenerConfig.createLDAPConfig("LDAP", // Listener name
74                              null, // Listen address. (null = listen on all interfaces)
75                              1389, // Listen port (0 = automatically choose an available port)
76                              serverSSLUtil.createSSLSocketFactory()), // StartTLS factory
77                      InMemoryListenerConfig.createLDAPSConfig("LDAPS", // Listener name
78                              null, // Listen address. (null = listen on all interfaces)
79                              1636, // Listen port (0 = automatically choose an available port)
80                              serverSSLUtil.createSSLServerSocketFactory(), // Server factory
81                              clientSSLUtil.createSSLSocketFactory())); // Client factory
82  
83              config.setEnforceSingleStructuralObjectClass(false);
84              config.setEnforceAttributeSyntaxCompliance(true);
85  
86              final Schema s = Schema.mergeSchemas(Schema.getSchema(schemaFile));
87              config.setSchema(s);
88  
89              this.directoryServer = new InMemoryDirectoryServer(config);
90  
91              LOGGER.debug("Populating directory with {}", ldifFile);
92              this.directoryServer.importFromLDIF(true, ldifFile.getCanonicalPath());
93              this.directoryServer.startListening();
94  
95              final LDAPConnection c = getConnection();
96              LOGGER.debug("Connected to {}:{}", c.getConnectedAddress(), c.getConnectedPort());
97  
98              populateEntries();
99  
100         } catch (final Exception e) {
101             throw new RuntimeException(e);
102         }
103     }
104 
105     private void populateEntries() throws Exception {
106         this.ldapEntries = LdapTestUtils.readLdif(new ClassPathResource("ldif/users-groups.ldif"), getBaseDn());
107         final LDAPConnection c = getConnection();
108         LdapTestUtils.createLdapEntries(c, ldapEntries);
109         c.close();
110     }
111 
112     public String getBaseDn() {
113         return this.directoryServer.getBaseDNs().get(0).toNormalizedString();
114     }
115 
116     public  Collection<LdapEntry> getLdapEntries() {
117         return this.ldapEntries;
118     }
119 
120     public LDAPConnection getConnection() throws LDAPException {
121         return this.directoryServer.getConnection();
122     }
123 
124     @Override
125     public void close() throws IOException {
126          this.directoryServer.shutDown(true);
127     }
128 }
