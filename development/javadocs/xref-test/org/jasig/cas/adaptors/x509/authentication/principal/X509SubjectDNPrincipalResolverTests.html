19  package org.jasig.cas.adaptors.x509.authentication.principal;
20  
21  import static org.junit.Assert.*;
22  
23  import java.security.cert.X509Certificate;
24  
25  import org.jasig.cas.authentication.UsernamePasswordCredential;
26  import org.junit.Test;
27  
28  /**
29   * @author Scott Battaglia
30   * @author Jan Van der Velpen
31   * @since 3.0.0.6
32   *
33   */
34  public class X509SubjectDNPrincipalResolverTests extends AbstractX509CertificateTests {
35  
36      private final X509SubjectDNPrincipalResolver
37          resolver = new X509SubjectDNPrincipalResolver();
38  
39      @Test
40      public void verifyResolvePrincipalInternal() {
41          final X509CertificateCredential c = new X509CertificateCredential(new X509Certificate[] {VALID_CERTIFICATE});
42          c.setCertificate(VALID_CERTIFICATE);
43          assertEquals(VALID_CERTIFICATE.getSubjectDN().getName(), this.resolver.resolve(c).getId());
44      }
45  
46      @Test
47      public void verifySupport() {
48          final X509CertificateCredential c = new X509CertificateCredential(new X509Certificate[] {VALID_CERTIFICATE});
49          assertTrue(this.resolver.supports(c));
50      }
51  
52      @Test
53      public void verifySupportFalse() {
54          assertFalse(this.resolver.supports(new UsernamePasswordCredential()));
55      }
56  
57  }
