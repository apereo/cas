19  package org.jasig.cas.ticket.support;
20  
21  import org.jasig.cas.TestUtils;
22  import org.jasig.cas.authentication.Authentication;
23  import org.jasig.cas.authentication.RememberMeCredential;
24  import org.jasig.cas.authentication.principal.DefaultPrincipalFactory;
25  import org.jasig.cas.authentication.principal.PrincipalFactory;
26  import org.jasig.cas.ticket.TicketGrantingTicketImpl;
27  import org.junit.Before;
28  import org.junit.Test;
29  
30  import javax.validation.constraints.NotNull;
31  import java.util.Collections;
32  
33  import static org.junit.Assert.*;
34  
35  /**
36   * Tests for RememberMeDelegatingExpirationPolicy.
37   *
38   * @author Scott Battaglia
39   * @since 3.2.1
40   *
41   */
42  public final class RememberMeDelegatingExpirationPolicyTests {
43  
44      /** Factory to create the principal type. **/
45      @NotNull
46      protected PrincipalFactory principalFactory = new DefaultPrincipalFactory();
47  
48  
49      private RememberMeDelegatingExpirationPolicy p;
50  
51      @Before
52      public void setUp() throws Exception {
53          this.p = new RememberMeDelegatingExpirationPolicy();
54          this.p.setRememberMeExpirationPolicy(new MultiTimeUseOrTimeoutExpirationPolicy(1, 20000));
55          this.p.setSessionExpirationPolicy(new MultiTimeUseOrTimeoutExpirationPolicy(5, 20000));
56      }
57  
58      @Test
59      public void verifyTicketExpirationWithRememberMe() {
60          final Authentication authentication = TestUtils.getAuthentication(
61                  this.principalFactory.createPrincipal("test"),
62                  Collections.<String, Object>singletonMap(
63                          RememberMeCredential.AUTHENTICATION_ATTRIBUTE_REMEMBER_ME, true));
64          final TicketGrantingTicketImpl t = new TicketGrantingTicketImpl("111", authentication, this.p);
65          assertFalse(t.isExpired());
66          t.grantServiceTicket("55", TestUtils.getService(), this.p, false);
67          assertTrue(t.isExpired());
68  
69      }
70  
71      @Test
72      public void verifyTicketExpirationWithoutRememberMe() {
73          final Authentication authentication = TestUtils.getAuthentication();
74          final TicketGrantingTicketImpl t = new TicketGrantingTicketImpl("111", authentication, this.p);
75          assertFalse(t.isExpired());
76          t.grantServiceTicket("55", TestUtils.getService(), this.p, false);
77          assertFalse(t.isExpired());
78  
79      }
80  
81  }
