85      public KryoTranscoderTests() {
86          transcoder = new KryoTranscoder();
87          final Map<Class<?>, Serializer> serializerMap = new HashMap<Class<?>, Serializer>();
88          serializerMap.put(
89                  MockServiceTicket.class,
90                  new FieldSerializer(transcoder.getKryo(), MockServiceTicket.class));
91          serializerMap.put(
92                  MockTicketGrantingTicket.class,
93                  new FieldSerializer(transcoder.getKryo(), MockTicketGrantingTicket.class));
94          transcoder.setSerializerMap(serializerMap);
95          transcoder.initialize();
96  
97          this.principalAttributes = new HashMap<String, Object>();
98          this.principalAttributes.put(NICKNAME_KEY, NICKNAME_VALUE);
99      }

101     @Test
102     public void verifyEncodeDecode() throws Exception {
103         final ServiceTicket expectedST =
104                 new MockServiceTicket(ST_ID);
105         assertEquals(expectedST, transcoder.decode(transcoder.encode(expectedST)));
106 
107         final Credential userPassCredential = new UsernamePasswordCredential(USERNAME, PASSWORD);
108         final TicketGrantingTicket expectedTGT = new MockTicketGrantingTicket(TGT_ID, userPassCredential, this.principalAttributes);
109         expectedTGT.grantServiceTicket(ST_ID, null, null, false);
110         assertEquals(expectedTGT, transcoder.decode(transcoder.encode(expectedTGT)));

112         internalProxyTest("http://localhost");

114     }

257     static class MockTicketGrantingTicket implements TicketGrantingTicket {
258 
259         private static final long serialVersionUID = 4829406617873497061L;
260 
261         private final String id;
262 
263         private int usageCount;
264 
265         private Service proxiedBy;
266 
267         private final Date creationDate = new Date();
268 
269         private final Authentication authentication;
270 
271         /** Factory to create the principal type. **/
272         @NotNull
273         private final PrincipalFactory principalFactory = new DefaultPrincipalFactory();
274 
275         /** Constructor for serialization support. */
276         MockTicketGrantingTicket() {
277             this.id = null;
278             this.authentication = null;
279         }
280 
281         public MockTicketGrantingTicket(final String id, final Credential credential, final Map<String, Object> principalAttributes) {
282             this.id = id;
283             final CredentialMetaData credentialMetaData = new BasicCredentialMetaData(credential);
284             final DefaultAuthenticationBuilder builder = new DefaultAuthenticationBuilder();
285             builder.setPrincipal(this.principalFactory.createPrincipal(USERNAME, principalAttributes));
286             builder.setAuthenticationDate(new Date());
287             builder.addCredential(credentialMetaData);
288             builder.addAttribute(RememberMeCredential.AUTHENTICATION_ATTRIBUTE_REMEMBER_ME, Boolean.TRUE);
289             final AuthenticationHandler handler = new MockAuthenticationHandler();
290             try {
291                 builder.addSuccess(handler.getName(), handler.authenticate(credential));
292             } catch (final Exception e) {
293                 throw new RuntimeException(e);
294             }
295             builder.addFailure(handler.getName(), FailedLoginException.class);
296             this.authentication = builder.build();
297         }
298 
299         @Override
300         public Authentication getAuthentication() {
301             return this.authentication;
302         }
303 
304         @Override
305         public List<Authentication> getSupplementalAuthentications() {
306             return Collections.emptyList();
307         }
308 
309         @Override
310         public ServiceTicket grantServiceTicket(
311                 final String id,
312                 final Service service,
313                 final ExpirationPolicy expirationPolicy,
314                 final boolean credentialsProvided) {
315             this.usageCount++;
316             return new MockServiceTicket(id);
317         }
318 
319         @Override
320         public Service getProxiedBy() {
321             return proxiedBy;
322         }
323 
324         @Override
325         public Map<String, Service> getServices() {
326             return Collections.emptyMap();
327         }

338         }

342             return this;
343         }

346         public List<Authentication> getChainedAuthentications() {
347             return Collections.emptyList();
348         }
349 
350         @Override

353         }
354 
355         @Override
356         public boolean isExpired() {
357             return false;
358         }

360         @Override

363         }

367             return this.creationDate.getTime();
368         }
369

371         public int getCountOfUses() {
372             return this.usageCount;
373         }

375         @Override
376         public boolean equals(final Object other) {
377             return other instanceof MockTicketGrantingTicket
378                     && ((MockTicketGrantingTicket) other).getId().equals(this.id)
379                     && ((MockTicketGrantingTicket) other).getCountOfUses() == this.usageCount
380                     && ((MockTicketGrantingTicket) other).getCreationTime() == this.creationDate.getTime()
381                     && ((MockTicketGrantingTicket) other).getAuthentication().equals(this.authentication);
382         }
383 
384         @Override
385         public int hashCode() {
386             final HashCodeBuilder bldr = new HashCodeBuilder(17, 33);
387             return bldr.append(this.id)
388                         .append(this.usageCount)
389                         .append(this.creationDate.getTime())
390                         .append(this.authentication).toHashCode();
391         }

394     public static class MockAuthenticationHandler implements AuthenticationHandler {
395 
396         @Override
397         public DefaultHandlerResult authenticate(final Credential credential) throws GeneralSecurityException, PreventedException {
398             if (credential instanceof HttpBasedServiceCredential) {
399                 return new DefaultHandlerResult(this, (HttpBasedServiceCredential) credential);
400             } else {
401                 return new DefaultHandlerResult(this, new BasicCredentialMetaData(credential));
402             }
403         }
404 
405         @Override
406         public boolean supports(final Credential credential) {
407             return true;
408         }
409 
410         @Override
411         public String getName() {
412             return "MockAuthenticationHandler";
413         }
