249     @Test
250     public void verifySaveAttributeReleasePolicyAllowedAttrRulesWithCaching() {
251         final RegisteredServiceImpl r = new RegisteredServiceImpl();
252         r.setName("testSaveAttributeReleasePolicyAllowedAttrRulesWithCaching");
253         r.setServiceId("testId");
254 
255         final ReturnAllowedAttributeReleasePolicy policy = new ReturnAllowedAttributeReleasePolicy();
256         policy.setAllowedAttributes(Arrays.asList("1", "2", "3"));
257 
258         final Map<String, List<Object>> attributes = new HashMap<>();
259         attributes.put("values", Arrays.asList(new Object[]{"v1", "v2", "v3"}));
260 
261         final CachingPrincipalAttributesRepository repository =
262                 new CachingPrincipalAttributesRepository(
263                         new StubPersonAttributeDao(attributes),
264                         TimeUnit.MILLISECONDS, 100);
265         repository.setMergingStrategy(new ReplacingAttributeAdder());
266 
267         policy.setPrincipalAttributesRepository(repository);
268         r.setAttributeReleasePolicy(policy);
269 
270         final RegisteredService r2 = this.dao.save(r);
271         final RegisteredService r3 = this.dao.findServiceById(r2.getId());
272 
273         assertEquals(r, r2);
274         assertEquals(r2, r3);
275         assertNotNull(r3.getAttributeReleasePolicy());
276         assertEquals(r2.getAttributeReleasePolicy(), r3.getAttributeReleasePolicy());
277     }

279     @Test
280     public void verifyServiceRemovals() {
281         final List<RegisteredService> list = new ArrayList<>(5);
282         for (int i = 1; i < 5; i++) {
283             final RegexRegisteredService r = new RegexRegisteredService();
284             r.setServiceId("^https://.+");
285             r.setName("testServiceType");
286             r.setTheme("testtheme");
287             r.setEvaluationOrder(1000);
288             r.setId(i * 100);
289             list.add(this.dao.save(r));
290         }
291 
292         for (final RegisteredService r2 : list) {
293             this.dao.delete(r2);
294             assertNull(this.dao.findServiceById(r2.getId()));
295         }

299     @Test
300     public void checkForAuthorizationStrategy() {
301         final RegexRegisteredService r = new RegexRegisteredService();
302         r.setServiceId("^https://.+");
303         r.setName("checkForAuthorizationStrategy");
304         r.setId(42);
305 
306         final DefaultRegisteredServiceAccessStrategy authz =
307                 new DefaultRegisteredServiceAccessStrategy(false, false);
308 
309         final Map<String, Set<String>> attrs = new HashMap<>();
310         attrs.put("cn", Sets.newHashSet("v1, v2, v3"));
311         attrs.put("memberOf", Sets.newHashSet(Arrays.asList("v4, v5, v6")));
312         authz.setRequiredAttributes(attrs);
313         r.setAccessStrategy(authz);
314 
315         this.dao.save(r);
316         final List<RegisteredService> list = this.dao.load();
317         assertEquals(list.size(), 1);
318     }
