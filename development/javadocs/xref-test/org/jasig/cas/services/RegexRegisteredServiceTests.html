51          final String globalCatchallHttpImap = "(https*|imaps*)://.*";
52          return Arrays.asList(new Object[][]{
53                  // CAS-1071 domain-specific HTTP catch-all #1
54                  {
55                          newService(domainCatchallHttp),
56                          "https://service.vt.edu/webapp?a=1",
57                          true,
58                  },
59                  // CAS-1071 domain-specific HTTP catch-all #2
60                  {
61                          newService(domainCatchallHttp),

63                          true,
64                  },
65                  // CAS-1071 domain-specific HTTP catch-all #3
66                  {
67                          newService(domainCatchallHttp),
68                          "https://thepiratebay.se?service.vt.edu/webapp?a=1",
69                          false,
70                  },
71                  // Domain-specific catch-all for HTTP(S)/IMAP(S) #1

75                          true,
76                  },
77                  // Domain-specific catch-all for HTTP(S)/IMAP(S) #2
78                  {

83                  // Global catch-all for HTTP(S)/IMAP(S) #1

90                  {
91                          newService(globalCatchallHttpImap),

93                          true,
94                  },
95                  // Null case
96                  {
97                          newService(globalCatchallHttpImap),
98                          null,
99                          false,
100                 },
101         });
102     }

105     public RegexRegisteredServiceTests(
106             final RegexRegisteredService service,
107             final String serviceToMatch,
108             final boolean expectedResult) {
109         this.service = service;
110         this.serviceToMatch = serviceToMatch;
111         this.expected = expectedResult;
112     }
113 
114 
115     @Test
116     public void testMatches() throws Exception {
117         final Service testService;
118         if (serviceToMatch == null) {
119             testService = null;
120         } else {
121             testService = new MockService(serviceToMatch);
122         }
123         assertEquals(expected, service.matches(testService));
124     }
