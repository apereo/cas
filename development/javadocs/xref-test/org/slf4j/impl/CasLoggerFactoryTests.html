340     private void validateLogData() {
341         try {
342             assertTrue("Log file " + LOG_FILE.getCanonicalPath() + " does not exist", LOG_FILE.exists());
343 
344             final String data = FileUtils.readFileToString(LOG_FILE);
345             assertTrue("Logged buffer data is blank in " + LOG_FILE.getCanonicalPath(), StringUtils.isNotBlank(data));
346             assertFalse("Logged buffer data should not contain " + ID1, data.contains(ID1));
347             assertFalse("Logged buffer data should not contain " + ID2, data.contains(ID2));
348         } catch (final IOException e) {
349             fail(e.getMessage());
350         }
351     }
