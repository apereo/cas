19  package org.jasig.cas.adaptors.x509.authentication.handler.support;
20  
21  import java.security.GeneralSecurityException;
22  import java.util.Date;
23  
24  /**
25   * Exception describing an expired CRL condition.
26   *
27   * @author Marvin S. Addison
28   * @since 3.4.6
29   *
30   */
31  public class ExpiredCRLException extends GeneralSecurityException {
32      /** Serialization version marker. */
33      private static final long serialVersionUID = 5157864033250359972L;
34  
35      /** Identifier/name of CRL. */
36      private final String id;
37  
38      /** CRL expiration date. */
39      private final Date expirationDate;
40  
41      /** Leniency of expiration. */
42      private final int leniency;
43  
44      /**
45       * Creates a new instance describing a CRL that expired on the given date.
46       *
47       * @param identifier Identifier or name that describes CRL.
48       * @param expirationDate CRL expiration date.
49       */
50      public ExpiredCRLException(final String identifier, final Date expirationDate) {
51          this(identifier, expirationDate, 0);
52      }
53  
54      /**
55       * Creates a new instance describing a CRL that expired on a date that is
56       * more than leniency seconds beyond its expiration date.
57       *
58       * @param identifier Identifier or name that describes CRL.
59       * @param expirationDate CRL expiration date.
60       * @param leniency Number of seconds beyond the expiration date at which
61       * the CRL is considered expired.  MUST be non-negative integer.
62       */
63      public ExpiredCRLException(final String identifier, final Date expirationDate, final int leniency) {
64          this.id = identifier;
65          this.expirationDate = expirationDate;
66          if (leniency < 0) {
67              throw new IllegalArgumentException("Leniency cannot be negative.");
68          }
69          this.leniency = leniency;
70      }
71  
72      /**
73       * @return Returns the id.
74       */
75      public String getId() {
76          return this.id;
77      }
78  
79      /**
80       * @return Returns the expirationDate.
81       */
82      public Date getExpirationDate() {
83          return this.expirationDate;
84      }
85  
86      /**
87       * @return Returns the leniency.
88       */
89      public int getLeniency() {
90          return this.leniency;
91      }
92  
93      /** {@inheritDoc} */
94      @Override
95      public String getMessage() {
96          if (this.leniency > 0) {
97              return String.format("CRL %s expired on %s and is beyond the leniency period of %s seconds.",
98                      this.id, this.expirationDate, this.leniency);
99          }
100         return String.format("CRL %s expired on %s", this.id, this.expirationDate);
101     }
102 }
