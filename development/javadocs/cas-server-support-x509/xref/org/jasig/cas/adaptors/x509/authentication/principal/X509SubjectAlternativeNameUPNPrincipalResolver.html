19  package org.jasig.cas.adaptors.x509.authentication.principal;
20  
21  import org.bouncycastle.asn1.ASN1InputStream;
22  import org.bouncycastle.asn1.ASN1Sequence;
23  import org.bouncycastle.asn1.ASN1TaggedObject;
24  import org.bouncycastle.asn1.DERObject;
25  import org.bouncycastle.asn1.DERObjectIdentifier;
26  import org.bouncycastle.asn1.DERUTF8String;
27  import java.io.ByteArrayInputStream;
28  import java.io.IOException;
29  import java.security.cert.CertificateParsingException;
30  import java.security.cert.X509Certificate;
31  import java.util.Collection;
32  import java.util.List;
33  
34  /**
35   * Credential to principal resolver that extracts Subject Alternative Name UPN extension
36   * from the provided certificate if available as a resolved principal id.
37   *
38   * @author Dmitriy Kopylenko
39   * @since 4.1.0
40   */
41  public class X509SubjectAlternativeNameUPNPrincipalResolver extends AbstractX509PrincipalResolver {
42  
43      /**
44       * ObjectID for upn altName for windows smart card logon.
45       */
46      public static final String UPN_OBJECTID = "1.3.6.1.4.1.311.20.2.3";
47  
48      /**
49       * Retrieves Subject Alternative Name UPN extension as a principal id String.
50       *
51       * @param certificate X.509 certificate credential.
52       *
53       * @return Resolved principal ID or null if no SAN UPN extension is available in provided certificate.
54       *
55       * @see AbstractX509PrincipalResolver#resolvePrincipalInternal(java.security.cert.X509Certificate)
56       * @see java.security.cert.X509Certificate#getSubjectAlternativeNames()
57       */
58      @Override
59      protected String resolvePrincipalInternal(final X509Certificate certificate) {
60          logger.debug("Resolving principal from Subject Alternative Name UPN for {}", certificate);
61          try {
62              final Collection<List<?>> subjectAltNames = certificate.getSubjectAlternativeNames();
63              if (subjectAltNames != null) {
64                  for (final List<?> sanItem : subjectAltNames) {
65                      final ASN1Sequence seq = getAltnameSequence(sanItem);
66                      final String upnString = getUPNStringFromSequence(seq);
67                      if (upnString != null) {
68                          return upnString;
69                      }
70                  }
71              }
72          } catch (final CertificateParsingException e) {
73              logger.error("Error is encountered while trying to retrieve subject alternative names collection from certificate", e);
74              logger.debug("Returning null principal id...");
75              return null;
76          }
77          logger.debug("Returning null principal id...");
78          return null;
79      }
80  
81      /**
82       * Get UPN String.
83       *
84       * @param seq ASN1Sequence abstraction representing subject alternative name.
85       * First element is the object identifier, second is the object itself.
86       *
87       * @return UPN string or null
88       */
89      private String getUPNStringFromSequence(final ASN1Sequence seq) {
90          if (seq != null) {
91              // First in sequence is the object identifier, that we must check
92              final DERObjectIdentifier id = DERObjectIdentifier.getInstance(seq.getObjectAt(0));
93              if (id != null && UPN_OBJECTID.equals(id.getId())) {
94                  final ASN1TaggedObject obj = (ASN1TaggedObject) seq.getObjectAt(1);
95                  final DERUTF8String str = DERUTF8String.getInstance(obj.getObject());
96                  return str.getString();
97              }
98          }
99          return null;
100     }
101 
102     /**
103      * Get alt name seq.
104      *
105      * @param sanItem subject alternative name value encoded as a two elements List with elem(0) representing object id and elem(1)
106      * representing object (subject alternative name) itself.
107      *
108      * @return ASN1Sequence abstraction representing subject alternative name or null if the passed in
109      * List doesn't contain at least to elements
110      * as expected to be returned by implementation of {@code X509Certificate.html#getSubjectAlternativeNames}
111      *
112      * @see <a href="http://docs.oracle.com/javase/7/docs/api/java/security/cert/X509Certificate.html#getSubjectAlternativeNames()">
113      *     X509Certificate#getSubjectAlternativeNames</a>
114      */
115     private ASN1Sequence getAltnameSequence(final List sanItem) {
116         //Should not be the case, but still, a extra "safety" check
117         if (sanItem.size() < 2) {
118             logger.error("Subject Alternative Name List does not contain at least two required elements. Returning null principal id...");
119         }
120         final Integer itemType = (Integer) sanItem.get(0);
121         if (itemType == 0) {
122             final byte[] altName = (byte[]) sanItem.get(1);
123             return getAltnameSequence(altName);
124         }
125         return null;
126     }
127 
128     /**
129      * Get alt name seq.
130      *
131      * @param sanValue subject alternative name value encoded as byte[]
132      *
133      * @return ASN1Sequence abstraction representing subject alternative name
134      *
135      * @see <a href="http://docs.oracle.com/javase/7/docs/api/java/security/cert/X509Certificate.html#getSubjectAlternativeNames()">
136      *     X509Certificate#getSubjectAlternativeNames</a>
137      */
138     private ASN1Sequence getAltnameSequence(final byte[] sanValue) {
139         DERObject oct = null;
140         try (final ByteArrayInputStream bInput = new ByteArrayInputStream(sanValue)) {
141             try (final ASN1InputStream input = new ASN1InputStream(bInput)) {
142                 oct = input.readObject();
143             } catch (final IOException e) {
144                 logger.error("Error on getting Alt Name as a DERSEquence: {}", e.getMessage(), e);
145             }
146             return ASN1Sequence.getInstance(oct);
147         } catch (final IOException e) {
148             logger.error("An error has occurred while reading the subject alternative name value", e);
149         }
150         return null;
151     }
152 }
