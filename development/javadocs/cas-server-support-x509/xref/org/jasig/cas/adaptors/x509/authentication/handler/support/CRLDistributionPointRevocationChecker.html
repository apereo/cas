46  /**
47   * Performs CRL-based revocation checking by consulting resources defined in
48   * the CRLDistributionPoints extension field on the certificate.  Although RFC
49   * 2459 allows the distribution point name to have arbitrary meaning, this class
50   * expects the name to define an absolute URL, which is the most common
51   * implementation.  This implementation caches CRL resources fetched from remote
52   * URLs to improve performance by avoiding CRL fetching on every revocation
53   * check.
54   *
55   * @author Marvin S. Addison
56   * @since 3.4.6
57   *
58   */
59  public class CRLDistributionPointRevocationChecker extends AbstractCRLRevocationChecker {
60  
61      /** CRL cache. */
62      private final Cache crlCache;
63  
64      /** The CRL fetcher instance. **/
65      private final CRLFetcher fetcher;
66  
67      private boolean throwOnFetchFailure;
68  
69      /**
70       * Creates a new instance that uses the given cache instance for CRL caching.
71       *
72       * @param crlCache Cache for CRL data.
73       */
74      public CRLDistributionPointRevocationChecker(@NotNull final Cache crlCache) {
75          this(crlCache, new ResourceCRLFetcher());
76      }
77  
78      /**
79       * Creates a new instance that uses the given cache instance for CRL caching.
80       *
81       * @param crlCache Cache for CRL data.
82       * @param throwOnFetchFailure the throw on fetch failure
83       */
84      public CRLDistributionPointRevocationChecker(@NotNull final Cache crlCache,
85                                                   final boolean throwOnFetchFailure) {
86          this(crlCache, new ResourceCRLFetcher());
87          setThrowOnFetchFailure(throwOnFetchFailure);
88      }
89  
90      /**
91       * Instantiates a new CRL distribution point revocation checker.
92       *
93       * @param crlCache the crl cache
94       * @param fetcher the fetcher
95       */
96      public CRLDistributionPointRevocationChecker(@NotNull final Cache crlCache,
97                                                   @NotNull final CRLFetcher fetcher) {
98          this.crlCache = crlCache;
99          this.fetcher = fetcher;
100     }
101 
102 
103     /**
104      * Throws exceptions if fetching crl fails. Defaults to false.
105      *
106      * @param throwOnFetchFailure the throw on fetch failure
107      */
108     public void setThrowOnFetchFailure(final boolean throwOnFetchFailure) {
109         this.throwOnFetchFailure = throwOnFetchFailure;
110     }
111 
112     /**
113      * {@inheritDoc}
114      * @see AbstractCRLRevocationChecker#getCRL(X509Certificate)
115      */
116     @Override
117     protected X509CRL getCRL(final X509Certificate cert) {
118         final URI[] urls = getDistributionPoints(cert);
119         logger.debug("Distribution points for {}: {}.", CertUtils.toString(cert), Arrays.asList(urls));
120 
121         try {
122             for (final URI url : urls) {
123                 final Element item = this.crlCache.get(url);
124                 if (item != null) {
125                     logger.debug("Found CRL in cache for {}", CertUtils.toString(cert));
126                     final byte[] encodedCrl = (byte[]) item.getObjectValue();
127                     return this.fetcher.fetch(new ByteArrayResource(encodedCrl));
128                 }
129             }
130         } catch (final Exception e) {
131             throw new RuntimeException(e);
132         }
133 
134         // Try all distribution points and stop at first fetch that succeeds
135         X509CRL crl = null;
136         for (int i = 0; i < urls.length && crl == null; i++) {
137             final URI url = urls[i];
138             logger.info("Attempting to fetch CRL at {}", url);
139             try {
140                 crl = this.fetcher.fetch(url);
141                 logger.info("Success. Caching fetched CRL at {}.", url);
142                 addCRL(url, crl);
143             } catch (final Exception e) {
144                 logger.error("Error fetching CRL at {}", url, e);
145                 if (this.throwOnFetchFailure) {
146                     throw new RuntimeException(e);
147                 }
148             }
149         }
150 
151         return crl;
152     }
153 
154     @Override
155     protected boolean addCRL(final Object id, final X509CRL crl) {
156         try {
157             if (crl == null) {
158                 return this.crlCache.remove(id);
159             }
160 
161             this.crlCache.put(new Element(id, crl.getEncoded()));
162             return this.crlCache.get(id) != null;
163 
164         } catch (final Exception e) {
165             logger.warn("Failed to add the crl entry [{}] to the cache", crl);
166             throw new RuntimeException(e);
167         }
168     }
169 
170 
171     /**
172      * Gets the distribution points.
173      *
174      * @param cert the cert
175      * @return the url distribution points
176      */
177     private URI[] getDistributionPoints(final X509Certificate cert) {
178         final DistributionPointList points;
179         try {
180             points = new ExtensionReader(cert).readCRLDistributionPoints();
181         } catch (final Exception e) {
182             logger.error("Error reading CRLDistributionPoints extension field on {}", CertUtils.toString(cert), e);
183             return new URI[0];
184         }
185 
186         final List<URI> urls = new ArrayList<>();
187         for (final DistributionPoint point : points.getItems()) {
188             final Object location = point.getDistributionPoint();
189             if (location instanceof String) {
190                 addURL(urls, (String) location);
191             } else if (location instanceof GeneralNameList) {
192                 for (final GeneralName gn : ((GeneralNameList) location).getItems()) {
193                     addURL(urls, gn.getName());
194                 }
195             } else {
196                 logger.warn("{} not supported. String or GeneralNameList expected.", location);
197             }
198         }
199 
200         return urls.toArray(new URI[urls.size()]);
201     }
202 
203     /**
204      * Adds the url to the list.
205      * Build URI by components to facilitate proper encoding of querystring.
206      * e.g. http://example.com:8085/ca?action=crl&issuer=CN=CAS Test User CA
207      *
208      * <p>If <code>uriString</code> is encoded, it will be decoded with <code>UTF-8</code>
209      * first before it's added to the list.</p>
210      * @param list the list
211      * @param uriString the uri string
212      */
213     private void addURL(final List<URI> list, final String uriString) {
214         try {
215             URI uri = null;
216             try {
217                 final URL url = new URL(URLDecoder.decode(uriString, "UTF-8"));
218                 uri = new URI(url.getProtocol(), url.getAuthority(), url.getPath(), url.getQuery(), null);
219             } catch (final MalformedURLException e) {
220                 uri = new URI(uriString);
221             }
222             list.add(uri);
223         } catch (final Exception e) {
224             logger.warn("{} is not a valid distribution point URI.", uriString);
225         }
226     }
