40  /**
41   * Authentication Handler that accepts X509 Certificates, determines their
42   * validity and ensures that they were issued by a trusted issuer. (targeted at
43   * X509v3) Optionally checks KeyUsage extension in the user certificate
44   * (container should do that too). Note that this handler trusts the servlet
45   * container to do some initial checks like path validation. Deployers can
46   * supply an optional pattern to match subject dns against to further restrict
47   * certificates in case they are not using their own issuer. It's also possible
48   * to specify a maximum pathLength for the SUPPLIED certificates. (note that
49   * this does not include a pathLength check for the root certificate)
50   * [PathLength is 0 for the CA certificate that issues the end-user certificate]
51   *
52   * @author Scott Battaglia
53   * @author Jan Van der Velpen
54   * @since 3.0.4
55   */
56  public class X509CredentialsAuthenticationHandler extends AbstractPreAndPostProcessingAuthenticationHandler {

113     @Override
114     public boolean supports(final Credential credential) {
115         return credential != null
116                 && X509CertificateCredential.class.isAssignableFrom(credential
117                         .getClass());

120     /** {@inheritDoc} */
121     @Override
122     protected final HandlerResult doAuthentication(final Credential credential) throws GeneralSecurityException, PreventedException {
123 
124         final X509CertificateCredential x509Credential = (X509CertificateCredential) credential;
125         final X509Certificate[] certificates = x509Credential.getCertificates();
126 
127         X509Certificate clientCert = null;
128         boolean hasTrustedIssuer = false;
129         for (int i = certificates.length - 1; i >= 0; i--) {
130             final X509Certificate certificate = certificates[i];
131             logger.debug("Evaluating {}", CertUtils.toString(certificate));
132 
133             validate(certificate);
134 
135             if (!hasTrustedIssuer) {
136                 hasTrustedIssuer = isCertificateFromTrustedIssuer(certificate);
137             }
138 
139             // getBasicConstraints returns pathLenContraint which is generally
140             // >=0 when this is a CA cert and -1 when it's not
141             int pathLength = certificate.getBasicConstraints();
142             if (pathLength < 0) {
143                 logger.debug("Found valid client certificate");
144                 clientCert = certificate;
145             } else {
146                 logger.debug("Found valid CA certificate");
147             }
148         }
149         if (hasTrustedIssuer && clientCert != null) {
150             x509Credential.setCertificate(clientCert);
151             return new HandlerResult(this, x509Credential, new SimplePrincipal(x509Credential.getId()));
152         }
153         throw new FailedLoginException();
154     }

204     private void validate(final X509Certificate cert) throws GeneralSecurityException {
205         cert.checkValidity();
206         this.revocationChecker.check(cert);
207 
208         int pathLength = cert.getBasicConstraints();
209         if (pathLength < 0) {
210             if (!isCertificateAllowed(cert)) {
211                 throw new FailedLoginException(
212                         "Certificate subject does not match pattern " + this.regExSubjectDnPattern.pattern());
213             }
214             if (this.checkKeyUsage && !isValidKeyUsage(cert)) {
215                 throw new FailedLoginException(
216                         "Certificate keyUsage constraint forbids SSL client authentication.");
217             }
218         } else {
219             // Check pathLength for CA cert
220             if (pathLength == Integer.MAX_VALUE && this.maxPathLengthAllowUnspecified != true) {
221                 throw new FailedLoginException("Unlimited certificate path length not allowed by configuration.");
222             } else if (pathLength > this.maxPathLength && pathLength < Integer.MAX_VALUE) {
223                 throw new FailedLoginException(String.format(
224                         "Certificate path length %s exceeds maximum value %s.", pathLength, this.maxPathLength));
225             }
226         }
227     }

229     private boolean isValidKeyUsage(final X509Certificate certificate) {
230         logger.debug("Checking certificate keyUsage extension");
231 
232         /*
233          * KeyUsage ::= BIT STRING { digitalSignature (0), nonRepudiation (1),
234          * keyEncipherment (2), dataEncipherment (3), keyAgreement (4),
235          * keyCertSign (5), cRLSign (6), encipherOnly (7), decipherOnly (8) }
236          */
237         final boolean[] keyUsage = certificate.getKeyUsage();
238         if (keyUsage == null) {
239             logger.warn("Configuration specifies checkKeyUsage but keyUsage extension not found in certificate.");
240             return !this.requireKeyUsage;
241         }
242 
243         final boolean valid;
244         if (isCritical(certificate, KEY_USAGE_OID) || this.requireKeyUsage) {
245             logger.debug("KeyUsage extension is marked critical or required by configuration.");
246             valid = keyUsage[0];
247         } else {
248             logger.debug(
249                "KeyUsage digitalSignature=%s, Returning true since keyUsage validation not required by configuration.");
250             valid = true;
251         }
252         return valid;
253     }
254 
255     private boolean isCritical(final X509Certificate certificate, final String extensionOid) {
256         final Set<String> criticalOids = certificate.getCriticalExtensionOIDs();
257 
258         if (criticalOids == null || criticalOids.isEmpty()) {
259             return false;
260         }
261 
262         return criticalOids.contains(extensionOid);
263     }
264 
265     private boolean isCertificateAllowed(final X509Certificate cert) {
266         return doesNameMatchPattern(cert.getSubjectDN(), this.regExSubjectDnPattern);
267     }
268 
269     private boolean isCertificateFromTrustedIssuer(final X509Certificate cert) {
270         return doesNameMatchPattern(cert.getIssuerDN(), this.regExTrustedIssuerDnPattern);
271     }
272 
273     private boolean doesNameMatchPattern(final Principal principal,
274             final Pattern pattern) {
275         final String name = principal.getName();
276         final boolean result = pattern.matcher(name).matches();
277         logger.debug(String.format("%s matches %s == %s", pattern.pattern(), name, result));
278         return result;
279     }
