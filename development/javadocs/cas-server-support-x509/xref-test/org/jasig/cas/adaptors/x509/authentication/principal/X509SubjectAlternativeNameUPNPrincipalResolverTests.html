19  package org.jasig.cas.adaptors.x509.authentication.principal;
20  
21  import org.junit.Assert;
22  import org.junit.Test;
23  import org.junit.runner.RunWith;
24  import org.junit.runners.Parameterized;
25  import org.junit.runners.Parameterized.Parameters;
26  
27  import java.io.FileInputStream;
28  import java.security.cert.CertificateFactory;
29  import java.security.cert.X509Certificate;
30  import java.util.ArrayList;
31  import java.util.Collection;
32  
33  /**
34   * Unit test for {@link org.jasig.cas.adaptors.x509.authentication.principal.X509SubjectAlternativeNameUPNPrincipalResolver}.
35   *
36   * @author Dmitriy Kopylenko
37   * @since 3.0.0
38   */
39  @RunWith(Parameterized.class)
40  public class X509SubjectAlternativeNameUPNPrincipalResolverTests {
41  
42      private X509Certificate certificate;
43      private final X509SubjectAlternativeNameUPNPrincipalResolver resolver;
44      private final String expected;
45  
46      /**
47       * Creates a new test instance with the given parameters.
48       *
49       * @param certPath path to the cert
50       * @param expectedResult the result expected from the test
51       */
52      public X509SubjectAlternativeNameUPNPrincipalResolverTests(
53              final String certPath,
54              final String expectedResult) {
55  
56          this.resolver = new X509SubjectAlternativeNameUPNPrincipalResolver();
57          try {
58              this.certificate = (X509Certificate) CertificateFactory.getInstance("X509").generateCertificate(
59                      new FileInputStream(getClass().getResource(certPath).getPath()));
60          } catch (final Exception e) {
61              Assert.fail(String.format("Error parsing certificate %s: %s", certPath, e.getMessage()));
62          }
63          this.expected = expectedResult;
64      }
65  
66      /**
67       * Gets the unit test parameters.
68       *
69       * @return Test parameter data.
70       */
71      @Parameters
72      public static Collection<Object[]> getTestParameters() {
73          final Collection<Object[]> params = new ArrayList<>();
74  
75          params.add(new Object[] {
76                  "/x509-san-upn-resolver.crt",
77                  "test-user@some-company-domain"
78          });
79          return params;
80      }
81  
82      @Test
83      public void verifyResolvePrincipalInternal() {
84          Assert.assertEquals(this.expected, this.resolver.resolvePrincipalInternal(this.certificate));
85      }
86  
87  }
