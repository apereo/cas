61  /**
62   * Controller that exposes the CAS internal state and beans
63   * as JSON. The report is available at <code>/status/config</code>.
64   * @author Misagh Moayyed
65   * @since 4.1
66   */
67  @Controller("internalConfigController")
68  @RequestMapping("/status/config")
69  public final class InternalConfigStateController {
70  
71      private static final Logger LOGGER = LoggerFactory.getLogger(InternalConfigStateController.class);
72  
73      private static final String VIEW_CONFIG = "monitoring/viewConfig";
74  
75      private static final String[] INCLUDE_PACKAGES = new String[] {"org.jasig"};
76  
77      @Autowired(required = true)
78      private ApplicationContext applicationContext;
79  
80      @Autowired(required = true)
81      @Qualifier("casProperties")
82      private Properties casProperties;
83  
84      /**
85       * Handle request.
86       *
87       * @param request the request
88       * @param response the response
89       * @return the model and view
90       * @throws Exception the exception
91       */
92      @RequestMapping(method = RequestMethod.GET)
93      protected ModelAndView handleRequestInternal(
94              final HttpServletRequest request, final HttpServletResponse response)
95              throws Exception {
96  
97          final Map<String, Object> list = getBeans(this.applicationContext);
98          LOGGER.debug("Found [{}] beans to report", list.size());
99  
100         final JsonSerializer<Object> serializer = new BeanObjectJsonSerializer();
101         final StringBuilder builder = new StringBuilder();
102         builder.append('[');
103 
104         final Set<Map.Entry<String, Object>> entries = list.entrySet();
105         final Iterator<Map.Entry<String, Object>> it = entries.iterator();
106 
107         while (it.hasNext()) {
108             final Map.Entry<String, Object> entry = it.next();
109             final Object obj = entry.getValue();
110 
111             final StringWriter writer = new StringWriter();
112             writer.append('{');
113             writer.append('\"' + entry.getKey() + "\":");
114             serializer.toJson(writer, obj);
115             writer.append('}');
116             builder.append(writer);
117 
118             if (it.hasNext()) {
119                 builder.append(',');
120             }
121         }
122         builder.append(']');
123         final ModelAndView mv = new ModelAndView(VIEW_CONFIG);
124         final String jsonData = StringEscapeUtils.escapeJson(builder.toString());
125 
126         mv.addObject("jsonData", jsonData);
127         mv.addObject("properties", casProperties.entrySet());
128         return mv;
129     }
130 
131     private static final class BeanObjectJsonSerializer extends AbstractJacksonBackedJsonSerializer<Object> {
132         private static final long serialVersionUID = 691461175315322624L;
133 
134         /**
135          * Instantiates a new Bean object json serializer.
136          */
137         public BeanObjectJsonSerializer() {
138             super(new MinimalPrettyPrinter());
139         }
140 
141         @Override
142         protected ObjectMapper initializeObjectMapper() {
143             final ObjectMapper mapper = super.initializeObjectMapper();
144 
145             final FilterProvider filters = new SimpleFilterProvider()
146                     .addFilter("beanObjectFilter", new CasSimpleBeanObjectFilter());
147             mapper.setFilters(filters);
148 
149             mapper.setVisibility(PropertyAccessor.ALL, JsonAutoDetect.Visibility.ANY);
150             mapper.configure(SerializationFeature.WRITE_EMPTY_JSON_ARRAYS, false);
151             mapper.configure(SerializationFeature.WRITE_NULL_MAP_VALUES, false);
152             mapper.setSerializationInclusion(JsonInclude.Include.NON_EMPTY);
153             mapper.setSerializationInclusion(JsonInclude.Include.NON_NULL);
154             mapper.addMixIn(Object.class, CasSimpleBeanObjectFilter.class);
155             mapper.disable(SerializationFeature.INDENT_OUTPUT);
156             mapper.disable(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS);
157 
158             return mapper;
159         }
160 
161         @Override
162         protected Class<Object> getTypeToSerialize() {
163             return Object.class;
164         }
165 
166         @JsonFilter("beanObjectFilter")
167         private static class CasSimpleBeanObjectFilter extends SimpleBeanPropertyFilter {
168             private static final Logger LOGGER = getLogger(CasSimpleBeanObjectFilter.class);
169 
170             private static final String[] EXCLUDE_FIELDS = new String[] {"logger"};
171 
172             @Override
173             public void serializeAsField(final Object pojo, final JsonGenerator jgen,
174                                          final SerializerProvider provider,
175                                          final PropertyWriter writer) throws Exception {
176                 try {
177                     if (!canSerializeField(pojo, writer)) {
178                         return;
179                     }
180                     super.serializeAsField(pojo, jgen, provider, writer);
181                 } catch (final Exception e) {
182                     LOGGER.debug(e.getMessage());
183                 }
184             }
185 
186             /**
187              * Can serialize field?
188              *
189              * @param pojo the pojo
190              * @param writer the writer
191              * @return the boolean
192              */
193             private boolean canSerializeField(final Object pojo, final PropertyWriter writer) {
194                 boolean foundPackage = false;
195                 final String packageName = pojo.getClass().getPackage().getName();
196                 for (int i = 0; !foundPackage && i < INCLUDE_PACKAGES.length; i++) {
197                     foundPackage = (packageName.startsWith(INCLUDE_PACKAGES[i]));
198                 }
199                 if (!foundPackage) {
200                     LOGGER.trace("Package [{}] is ignored", packageName);
201                     return false;
202                 }
203 
204 
205                 boolean foundField = true;
206                 final String fieldName = writer.getFullName().getSimpleName();
207                 for (int i = 0; foundField && i < EXCLUDE_FIELDS.length; i++) {
208                     foundField = !fieldName.equalsIgnoreCase(EXCLUDE_FIELDS[i]);
209                 }
210 
211                 if (!foundField) {
212                     LOGGER.trace("Field [{}] is excluded", fieldName);
213                     return false;
214                 }
215                 return true;
216             }
217         }
218     }
219 
220     /**
221      * Gets beans in the application context.
222      *
223      * @param ctx the ctx
224      * @return the beans
225      */
226     private static Map<String, Object> getBeans(final ApplicationContext ctx) {
227         final String[] all = BeanFactoryUtils.beanNamesIncludingAncestors(ctx);
228 
229         final Map<String, Object> singletons = new HashMap<>(all.length);
230         for (final String name : all) {
231             try {
232                 final Object object = ctx.getBean(name);
233                 if (object != null) {
234 
235                     boolean foundPackage = false;
236                     final String packageName = object.getClass().getPackage().getName();
237                     for (int i = 0; !foundPackage && i < INCLUDE_PACKAGES.length; i++) {
238                         foundPackage = (packageName.startsWith(INCLUDE_PACKAGES[i]));
239                     }
240                     if (foundPackage) {
241                         singletons.put(name, object);
242                     }
243 
244                 }
245             } catch (final BeanIsAbstractException e){
246                 LOGGER.debug("Skipping abstract bean definition. {}", e.getMessage());
247             } catch (final Throwable e){
248                 LOGGER.trace(e.getMessage(), e);
249             }
250         }
251 
252         return singletons;
253     }
