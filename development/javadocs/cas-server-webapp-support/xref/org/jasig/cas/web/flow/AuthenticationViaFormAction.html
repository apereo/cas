48  /**
49   * Action to authenticate credential and retrieve a TicketGrantingTicket for
50   * those credential. If there is a request for renew, then it also generates
51   * the Service Ticket required.
52   *
53   * @author Scott Battaglia
54   * @since 3.0.0.4
55   */
56  public class AuthenticationViaFormAction {
57  
58      /** Authentication success result. */
59      public static final String SUCCESS = "success";
60  
61      /** Authentication succeeded with warnings from authn subsystem that should be displayed to user. */
62      public static final String SUCCESS_WITH_WARNINGS = "successWithWarnings";
63  
64      /** Authentication success with "warn" enabled. */
65      public static final String WARN = "warn";
66  
67      /** Authentication failure result. */
68      public static final String AUTHENTICATION_FAILURE = "authenticationFailure";
69  
70      /** Error result. */
71      public static final String ERROR = "error";
72  
73      /** Logger instance. **/
74      protected final Logger logger = LoggerFactory.getLogger(getClass());
75  
76      /** Core we delegate to for handling all ticket related tasks. */
77      @NotNull
78      private CentralAuthenticationService centralAuthenticationService;
79  
80      @NotNull
81      private CookieGenerator warnCookieGenerator;
82  
83      /**
84       * Handle the submission of credentials from the post.
85       *
86       * @param context the context
87       * @param credential the credential
88       * @param messageContext the message context
89       * @return the event
90       * @since 4.1.0
91       */
92      public final Event submit(final RequestContext context, final Credential credential,
93                                final MessageContext messageContext)  {
94          if (!checkLoginTicketIfExists(context)) {
95              return returnInvalidLoginTicketEvent(context, messageContext);
96          }
97  
98          if (isRequestAskingForServiceTicket(context)) {
99              return grantServiceTicket(context, credential);
100         }
101 
102         return createTicketGrantingTicket(context, credential, messageContext);
103     }
104 
105     /**
106      * Tries to to determine if the login ticket in the request flow scope
107      * matches the login ticket provided by the request. The comparison
108      * is case-sensitive.
109      *
110      * @param context the context
111      * @return true if valid
112      * @since 4.1.0
113      */
114     protected boolean checkLoginTicketIfExists(final RequestContext context) {
115         final String loginTicketFromFlowScope = WebUtils.getLoginTicketFromFlowScope(context);
116         final String loginTicketFromRequest = WebUtils.getLoginTicketFromRequest(context);
117 
118         logger.trace("Comparing login ticket in the flow scope [{}] with login ticket in the request [{}]",
119                 loginTicketFromFlowScope, loginTicketFromRequest);
120         return StringUtils.equals(loginTicketFromFlowScope, loginTicketFromRequest);
121     }
122 
123     /**
124      * Return invalid login ticket event.
125      *
126      * @param context the context
127      * @param messageContext the message context
128      * @return the error event
129      * @since 4.1.0
130      */
131     protected Event returnInvalidLoginTicketEvent(final RequestContext context, final MessageContext messageContext) {
132         final String loginTicketFromRequest = WebUtils.getLoginTicketFromRequest(context);
133         logger.warn("Invalid login ticket [{}]", loginTicketFromRequest);
134         messageContext.addMessage(new MessageBuilder().error().code("error.invalid.loginticket").build());
135         return newEvent(ERROR);
136     }
137 
138     /**
139      * Is request asking for service ticket?
140      *
141      * @param context the context
142      * @return true, if both service and tgt are found, and the request is not asking to renew.
143      * @since 4.1.0
144      */
145     protected boolean isRequestAskingForServiceTicket(final RequestContext context) {
146         final String ticketGrantingTicketId = WebUtils.getTicketGrantingTicketId(context);
147         final Service service = WebUtils.getService(context);
148         return (StringUtils.isNotBlank(context.getRequestParameters().get(CasProtocolConstants.PARAMETER_RENEW))
149                 && ticketGrantingTicketId != null
150                 && service != null);
151     }
152 
153     /**
154      * Grant service ticket for the given credential based on the service and tgt
155      * that are found in the request context.
156      *
157      * @param context the context
158      * @param credential the credential
159      * @return the resulting event. Warning, authentication failure or error.
160      * @since 4.1.0
161      */
162     protected Event grantServiceTicket(final RequestContext context, final Credential credential) {
163         final String ticketGrantingTicketId = WebUtils.getTicketGrantingTicketId(context);
164         try {
165             final Service service = WebUtils.getService(context);
166             final ServiceTicket serviceTicketId = this.centralAuthenticationService.grantServiceTicket(
167                     ticketGrantingTicketId, service, credential);
168             WebUtils.putServiceTicketInRequestScope(context, serviceTicketId);
169             putWarnCookieIfRequestParameterPresent(context);
170             return newEvent(WARN);
171         } catch (final AuthenticationException e) {
172             return newEvent(AUTHENTICATION_FAILURE, e);
173         } catch (final TicketCreationException e) {
174             logger.warn(
175                     "Invalid attempt to access service using renew=true with different credential. "
176                             + "Ending SSO session.");
177             this.centralAuthenticationService.destroyTicketGrantingTicket(ticketGrantingTicketId);
178         } catch (final TicketException e) {
179             return newEvent(ERROR, e);
180         }
181         return newEvent(ERROR);
182 
183     }
184     /**
185      * Create ticket granting ticket for the given credentials.
186      * Adds all warnings into the message context.
187      *
188      * @param context the context
189      * @param credential the credential
190      * @param messageContext the message context
191      * @return the resulting event.
192      * @since 4.1.0
193      */
194     protected Event createTicketGrantingTicket(final RequestContext context, final Credential credential,
195                                                final MessageContext messageContext) {
196         try {
197             final TicketGrantingTicket tgt = this.centralAuthenticationService.createTicketGrantingTicket(credential);
198             WebUtils.putTicketGrantingTicketInScopes(context, tgt);
199             putWarnCookieIfRequestParameterPresent(context);
200             if (addWarningMessagesToMessageContextIfNeeded(tgt, messageContext)) {
201                 return newEvent(SUCCESS_WITH_WARNINGS);
202             }
203             return newEvent(SUCCESS);
204         } catch (final AuthenticationException e) {
205             return newEvent(AUTHENTICATION_FAILURE, e);
206         } catch (final Exception e) {
207             return newEvent(ERROR, e);
208         }
209     }
210 
211     /**
212      * Add warning messages to message context if needed.
213      *
214      * @param tgtId the tgt id
215      * @param messageContext the message context
216      * @return true if warnings were found and added, false otherwise.
217      * @since 4.1.0
218      */
219     protected boolean addWarningMessagesToMessageContextIfNeeded(final TicketGrantingTicket tgtId, final MessageContext messageContext) {
220         boolean foundAndAddedWarnings = false;
221         for (final Map.Entry<String, HandlerResult> entry : tgtId.getAuthentication().getSuccesses().entrySet()) {
222             for (final Message message : entry.getValue().getWarnings()) {
223                 addWarningToContext(messageContext, message);
224                 foundAndAddedWarnings = true;
225             }
226         }
227         return foundAndAddedWarnings;
228 
229     }
230     /**
231      * Put warn cookie if request parameter present.
232      *
233      * @param context the context
234      */
235     private void putWarnCookieIfRequestParameterPresent(final RequestContext context) {
236         final HttpServletResponse response = WebUtils.getHttpServletResponse(context);
237 
238         if (StringUtils.isNotBlank(context.getExternalContext().getRequestParameterMap().get("warn"))) {
239             this.warnCookieGenerator.addCookie(response, "true");
240         } else {
241             this.warnCookieGenerator.removeCookie(response);
242         }
243     }

245     /**
246      * New event based on the given id.
247      *
248      * @param id the id
249      * @return the event
250      */
251     private Event newEvent(final String id) {
252         return new Event(this, id);
253     }
254 
255     /**
256      * New event based on the id, which contains an error attribute referring to the exception occurred.
257      *
258      * @param id the id
259      * @param error the error
260      * @return the event
261      */
262     private Event newEvent(final String id, final Exception error) {
263         return new Event(this, id, new LocalAttributeMap("error", error));
264     }

267         this.centralAuthenticationService = centralAuthenticationService;
268     }

271         this.warnCookieGenerator = warnCookieGenerator;
272     }

286     /**
287      * Adds a warning message to the message context.
288      *
289      * @param context Message context.
290      * @param warning Warning message.
291      */
292     private void addWarningToContext(final MessageContext context, final Message warning) {
293         final MessageBuilder builder = new MessageBuilder()
294                 .warning()
295                 .code(warning.getCode())
296                 .defaultText(warning.getDefaultMessage())
297                 .args(warning.getParams());
298         context.addMessage(builder.build());
299     }
