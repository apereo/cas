19  package org.jasig.cas.ticket.registry;
20  
21  import java.util.ArrayList;
22  import java.util.Collection;
23  import java.util.Date;
24  import java.util.List;
25  
26  import javax.persistence.EntityManager;
27  import javax.persistence.LockModeType;
28  import javax.persistence.PersistenceContext;
29  import javax.validation.constraints.NotNull;
30  
31  import org.jasig.cas.ticket.ServiceTicket;
32  import org.jasig.cas.ticket.ServiceTicketImpl;
33  import org.jasig.cas.ticket.Ticket;
34  import org.jasig.cas.ticket.TicketGrantingTicketImpl;
35  import org.springframework.transaction.annotation.Transactional;
36  
37  /**
38   * JPA implementation of a CAS {@link TicketRegistry}. This implementation of
39   * ticket registry is suitable for HA environments.
40   *
41   * @author Scott Battaglia
42   * @author Marvin S. Addison
43   *
44   * @since 3.2.1
45   *
46   */
47  public final class JpaTicketRegistry extends AbstractDistributedTicketRegistry {
48  
49      @NotNull
50      @PersistenceContext
51      private EntityManager entityManager;
52  
53      @NotNull
54      private String ticketGrantingTicketPrefix = "TGT";
55  
56      @Transactional(readOnly = false)
57      @Override
58      protected void updateTicket(final Ticket ticket) {
59          entityManager.merge(ticket);
60          logger.debug("Updated ticket [{}].", ticket);
61      }
62  
63      @Transactional(readOnly = false)
64      @Override
65      public void addTicket(final Ticket ticket) {
66          entityManager.persist(ticket);
67          logger.debug("Added ticket [{}] to registry.", ticket);
68      }
69  
70      @Transactional(readOnly = false)
71      @Override
72      public boolean deleteTicket(final String ticketId) {
73          final Ticket ticket = getRawTicket(ticketId);
74  
75          if (ticket == null) {
76              return false;
77          }
78  
79          if (ticket instanceof ServiceTicket) {
80              removeTicket(ticket);
81              logger.debug("Deleted ticket [{}] from the registry.", ticket);
82              return true;
83          }
84  
85          deleteTicketAndChildren(ticket);
86          logger.debug("Deleted ticket [{}] and its children from the registry.", ticket);
87          return true;
88      }
89  
90      /**
91       * Delete the TGt and all of its service tickets.
92       *
93       * @param ticket the ticket
94       */
95      private void deleteTicketAndChildren(final Ticket ticket) {
96          final List<TicketGrantingTicketImpl> ticketGrantingTicketImpls = entityManager
97              .createQuery("select t from TicketGrantingTicketImpl t where t.ticketGrantingTicket.id = :id",
98                      TicketGrantingTicketImpl.class)
99              .setLockMode(LockModeType.PESSIMISTIC_WRITE)
100             .setParameter("id", ticket.getId())
101             .getResultList();
102         final List<ServiceTicketImpl> serviceTicketImpls = entityManager
103                 .createQuery("select s from ServiceTicketImpl s where s.ticketGrantingTicket.id = :id",
104                         ServiceTicketImpl.class)
105                 .setParameter("id", ticket.getId())
106                 .getResultList();
107 
108         for (final ServiceTicketImpl s : serviceTicketImpls) {
109             removeTicket(s);
110         }
111 
112         for (final TicketGrantingTicketImpl t : ticketGrantingTicketImpls) {
113             deleteTicketAndChildren(t);
114         }
115 
116         removeTicket(ticket);
117     }
118 
119     /**
120      * Removes the ticket.
121      *
122      * @param ticket the ticket
123      */
124     private void removeTicket(final Ticket ticket) {
125         try {
126             if (logger.isDebugEnabled()) {
127                 final Date creationDate = new Date(ticket.getCreationTime());
128                 logger.debug("Removing Ticket [{}] created: {}", ticket, creationDate.toString());
129              }
130             entityManager.remove(ticket);
131         } catch (final Exception e) {
132             logger.error("Error removing {} from registry.", ticket, e);
133         }
134     }
135 
136     @Transactional(readOnly=true)
137     @Override
138     public Ticket getTicket(final String ticketId) {
139         return getProxiedTicketInstance(getRawTicket(ticketId));
140     }
141 
142     /**
143      * Gets the ticket from the database, as is.
144      *
145      * @param ticketId the ticket id
146      * @return the raw ticket
147      */
148     private Ticket getRawTicket(final String ticketId) {
149         try {
150             if (ticketId.startsWith(this.ticketGrantingTicketPrefix)) {
151                 return entityManager.find(TicketGrantingTicketImpl.class, ticketId, LockModeType.PESSIMISTIC_WRITE);
152             }
153 
154             return entityManager.find(ServiceTicketImpl.class, ticketId);
155         } catch (final Exception e) {
156             logger.error("Error getting ticket {} from registry.", ticketId, e);
157         }
158         return null;
159     }
160 
161     @Transactional(readOnly=true)
162     @Override
163     public Collection<Ticket> getTickets() {
164         final List<TicketGrantingTicketImpl> tgts = entityManager
165             .createQuery("select t from TicketGrantingTicketImpl t", TicketGrantingTicketImpl.class)
166             .getResultList();
167         final List<ServiceTicketImpl> sts = entityManager
168             .createQuery("select s from ServiceTicketImpl s", ServiceTicketImpl.class)
169             .getResultList();
170 
171         final List<Ticket> tickets = new ArrayList<>();
172         tickets.addAll(tgts);
173         tickets.addAll(sts);
174 
175         return tickets;
176     }
177 
178     public void setTicketGrantingTicketPrefix(final String ticketGrantingTicketPrefix) {
179         this.ticketGrantingTicketPrefix = ticketGrantingTicketPrefix;
180     }
181 
182     @Override
183     protected boolean needsCallback() {
184         return false;
185     }
186 
187     @Transactional(readOnly=true)
188     @Override
189     public int sessionCount() {
190         return countToInt(entityManager.createQuery(
191                 "select count(t) from TicketGrantingTicketImpl t").getSingleResult());
192     }
193 
194     @Transactional(readOnly=true)
195     @Override
196     public int serviceTicketCount() {
197         return countToInt(entityManager.createQuery("select count(t) from ServiceTicketImpl t").getSingleResult());
198     }
199 
200     /**
201      * Count the result into a numeric value.
202      *
203      * @param result the result
204      * @return the int
205      */
206     private int countToInt(final Object result) {
207         final int intval;
208         if (result instanceof Long) {
209             intval = ((Long) result).intValue();
210         } else if (result instanceof Integer) {
211             intval = (Integer) result;
212         } else {
213             // Must be a Number of some kind
214             intval = ((Number) result).intValue();
215         }
216         return intval;
217     }
218 }
