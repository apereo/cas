45  public final class OpenIdService extends AbstractWebApplicationService {
46      
47      /** The Constant LOGGER. */
48      protected static final Logger LOGGER = LoggerFactory.getLogger(OpenIdService.class);
49  
50      private static final long serialVersionUID = 5776500133123291301L;
51  
52      private static final String CONST_PARAM_SERVICE = "openid.return_to";
53  
54      private final String identity;
55  
56      private final String artifactId;
57  
58      private final ParameterList requestParameters;
59  
60      /**
61       * Instantiates a new OpenID service.
62       *
63       * @param id the id
64       * @param originalUrl the original url
65       * @param artifactId the artifact id
66       * @param openIdIdentity the OpenID identity
67       * @param signature the signature
68       * @param parameterList the parameter list
69       */
70      protected OpenIdService(final String id, final String originalUrl,
71              final String artifactId, final String openIdIdentity,
72              final String signature, final ParameterList parameterList) {
73          super(id, originalUrl, artifactId);
74          this.identity = openIdIdentity;
75          this.artifactId = artifactId;
76          this.requestParameters = parameterList;
77      }
78  
79      /**
80       * Generates an Openid response.
81       * If no ticketId is found, response is negative.
82       * If we have a ticket id, then we check if we have an association.
83       * If so, we ask OpenId server manager to generate the answer according with the existing association.
84       * If not, we send back an answer with the ticket id as association handle.
85       * This will force the consumer to ask a verification, which will validate the service ticket.
86       * @param ticketId the service ticket to provide to the service.
87       * @return the generated authentication answer
88       */
89      @Override
90      public Response getResponse(final String ticketId) {
91          final Map<String, String> parameters = new HashMap<>();
92          if (ticketId != null) {
93  
94              final ServerManager manager = (ServerManager) ApplicationContextProvider.getApplicationContext().getBean("serverManager");
95              final CentralAuthenticationService cas = ApplicationContextProvider.getApplicationContext()
96                                                  .getBean("centralAuthenticationService", CentralAuthenticationService.class);
97              boolean associated = false;
98              boolean associationValid = true;
99              try {
100                 final AuthRequest authReq = AuthRequest.createAuthRequest(requestParameters, manager.getRealmVerifier());
101                 final Map parameterMap = authReq.getParameterMap();
102                 if (parameterMap != null && parameterMap.size() > 0) {
103                     final String assocHandle = (String) parameterMap.get("openid.assoc_handle");
104                     if (assocHandle != null) {
105                         final Association association = manager.getSharedAssociations().load(assocHandle);
106                         if (association != null) {
107                             associated = true;
108                             if (association.hasExpired()) {
109                                 associationValid = false;
110                             }
111                         }
112 
113                     }
114                 }
115             } catch (final MessageException me) {
116                 LOGGER.error("Message exception : {}", me.getMessage(), me);
117             }
118 
119             boolean successFullAuthentication = true;
120             try {
121                 if (associated) {
122                     if (associationValid) {
123                         cas.validateServiceTicket(ticketId, this);
124                         LOGGER.info("Validated openid ticket");
125                     } else {
126                         successFullAuthentication = false;
127                     }
128                 }
129             } catch (final TicketException te) {
130                 LOGGER.error("Could not validate ticket : {}", te.getMessage(), te);
131                 successFullAuthentication = false;
132             }
133 
134             // We sign directly (final 'true') because we don't add extensions
135             // response message can be either a DirectError or an AuthSuccess here.
136             // Anyway, handling is the same : send the response message
137             final Message response = manager.authResponse(requestParameters,
138                     this.identity,
139                     this.identity,
140                     successFullAuthentication,
141                     true);
142             parameters.putAll(response.getParameterMap());
143             if (!associated) {
144                 parameters.put("openid.assoc_handle", ticketId);
145             }
146         } else {
147             parameters.put("openid.mode", "cancel");
148         }
149         return Response.getRedirectResponse(getOriginalUrl(), parameters);
150     }
151 
152     /**
153      * Return that the service is already logged out.
154      *
155      * @return that the service is already logged out.
156      */
157     @Override
158     public boolean isLoggedOutAlready() {
159         return true;
160     }
161 
162     /**
163      * Creates the service from the request.
164      *
165      * @param request the request
166      * @return the OpenID service
167      */
168     public static OpenIdService createServiceFrom(
169             final HttpServletRequest request) {
170         final String service = request.getParameter(CONST_PARAM_SERVICE);
171         final String openIdIdentity = request.getParameter("openid.identity");
172         final String signature = request.getParameter("openid.sig");
173 
174         if (openIdIdentity == null || !StringUtils.hasText(service)) {
175             return null;
176         }
177 
178         final String id = cleanupUrl(service);
179         final String artifactId = request.getParameter("openid.assoc_handle");
180         final ParameterList paramList = new ParameterList(request.getParameterMap());
181 
182         return new OpenIdService(id, service, artifactId, openIdIdentity,
183                 signature, paramList);
184     }
185 
186     @Override
187     public int hashCode() {
188         return new HashCodeBuilder()
189                 .append(this.identity)
190                 .toHashCode();
191     }
192 
193     @Override
194     public boolean equals(final Object obj) {
195         if (this == obj) {
196             return true;
197         }
198         if (!super.equals(obj)) {
199             return false;
200         }
201         if (getClass() != obj.getClass()) {
202             return false;
203         }
204         final OpenIdService other = (OpenIdService) obj;
205         if (this.identity == null) {
206             if (other.identity != null) {
207                 return false;
208             }
209         } else if (!this.identity.equals(other.identity)) {
210             return false;
211         }
212         return true;
213     }
214 
215     public String getIdentity() {
216         return this.identity;
217     }
