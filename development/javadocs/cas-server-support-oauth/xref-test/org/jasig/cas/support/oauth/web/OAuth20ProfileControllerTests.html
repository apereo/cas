63      @Test
64      public void verifyNoAccessToken() throws Exception {
65          final MockHttpServletRequest mockRequest = new MockHttpServletRequest("GET", CONTEXT
66                  + OAuthConstants.PROFILE_URL);
67          final MockHttpServletResponse mockResponse = new MockHttpServletResponse();
68          final OAuth20WrapperController oauth20WrapperController = new OAuth20WrapperController();
69          oauth20WrapperController.afterPropertiesSet();
70          oauth20WrapperController.handleRequest(mockRequest, mockResponse);
71          assertEquals(200, mockResponse.getStatus());
72          assertEquals(CONTENT_TYPE, mockResponse.getContentType());
73          assertEquals("{\"error\":\"" + OAuthConstants.MISSING_ACCESS_TOKEN + "\"}", mockResponse.getContentAsString());
74      }

112     @Test
113     public void verifyOK() throws Exception {
114         final MockHttpServletRequest mockRequest = new MockHttpServletRequest("GET", CONTEXT
115                 + OAuthConstants.PROFILE_URL);
116         mockRequest.setParameter(OAuthConstants.ACCESS_TOKEN, TGT_ID);
117         final MockHttpServletResponse mockResponse = new MockHttpServletResponse();
118         final OAuth20WrapperController oauth20WrapperController = new OAuth20WrapperController();
119         final TicketRegistry ticketRegistry = mock(TicketRegistry.class);
120         final TicketGrantingTicket ticketGrantingTicket = mock(TicketGrantingTicket.class);
121         when(ticketGrantingTicket.isExpired()).thenReturn(false);
122         when(ticketRegistry.getTicket(TGT_ID)).thenReturn(ticketGrantingTicket);
123         final Authentication authentication = mock(Authentication.class);
124         final Principal principal = mock(Principal.class);
125         when(principal.getId()).thenReturn(ID);
126         final Map<String, Object> map = new HashMap<>();
127         map.put(NAME, VALUE);
128         final List<String> list = Arrays.asList(VALUE, VALUE);
129         map.put(NAME2, list);
130         when(principal.getAttributes()).thenReturn(map);
131         when(authentication.getPrincipal()).thenReturn(principal);
132         when(ticketGrantingTicket.getAuthentication()).thenReturn(authentication);
133         oauth20WrapperController.setTicketRegistry(ticketRegistry);
134         oauth20WrapperController.afterPropertiesSet();
135         oauth20WrapperController.handleRequest(mockRequest, mockResponse);
136         assertEquals(200, mockResponse.getStatus());
137         assertEquals(CONTENT_TYPE, mockResponse.getContentType());
138 
139         final ObjectMapper mapper = new ObjectMapper();
140 
141         final String expected = "{\"id\":\"" + ID + "\",\"attributes\":[{\"" + NAME + "\":\"" + VALUE + "\"},{\"" + NAME2
142                 + "\":[\"" + VALUE + "\",\"" + VALUE + "\"]}]}";
143         final JsonNode expectedObj = mapper.readTree(expected);
144         final JsonNode receivedObj = mapper.readTree(mockResponse.getContentAsString());
145         assertEquals(expectedObj.get("id").asText(), receivedObj.get("id").asText());
146 
147         final JsonNode expectedAttributes = expectedObj.get("attributes");
148         final JsonNode receivedAttributes = receivedObj.get("attributes");
149 
150         assertEquals(expectedAttributes.findValue(NAME).asText(), receivedAttributes.findValue(NAME).asText());
151         assertEquals(expectedAttributes.findValues(NAME2), receivedAttributes.findValues(NAME2));
152     }
