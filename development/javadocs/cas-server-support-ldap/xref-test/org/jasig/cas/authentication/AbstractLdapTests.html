21  import java.util.Collection;

23  import org.jasig.cas.util.LdapTestUtils;
24  import org.jasig.cas.util.LdapUtils;
25  import org.junit.After;
26  import org.junit.Assume;
27  import org.junit.Before;
28  import org.ldaptive.Connection;

30  import org.ldaptive.LdapEntry;
31  import org.ldaptive.LdapException;
32  import org.slf4j.Logger;
33  import org.slf4j.LoggerFactory;
34  import org.springframework.context.ApplicationContext;
35  import org.springframework.context.support.ClassPathXmlApplicationContext;
36  import org.springframework.core.io.Resource;

38  /**
39   * Base class for LDAP tests that provision and deprovision directory data as part of test setup/teardown.
40   * <p>
41   * NOTE: The <code>enableLdapTests</code> system property must be set to execute tests that derive from this class.
42   *
43   * @author Marvin S. Addison
44   */
45  public class AbstractLdapTests {
46  
47      private Resource usersLdif;
48  
49      private String usernameAttribute;
50  
51      private ConnectionFactory provisioningConnectionFactory;
52  
53      protected final Logger logger = LoggerFactory.getLogger(getClass());
54  
55      protected String baseDn;
56  
57      protected String[] contextPaths;
58  
59      protected LdapTestUtils.DirectoryType directoryType;
60  
61      protected ApplicationContext context;
62  
63      protected Collection<LdapEntry> testEntries;
64  
65      protected boolean enableLdapTests;
66  
67      @Before
68      public void setUp() throws Exception {
69          // Environment check
70          this.enableLdapTests = System.getProperty("enableLdapTests") != null;
71          Assume.assumeTrue("enableLdapTests system property not set", this.enableLdapTests);
72  
73          this.context = new ClassPathXmlApplicationContext(this.contextPaths);
74          this.baseDn = this.context.getBean("baseDn", String.class);
75          this.usersLdif = this.context.getBean("usersLdif", Resource.class);
76          this.usernameAttribute = this.context.getBean("usernameAttribute", String.class);
77          this.provisioningConnectionFactory = this.context.getBean(
78                  "provisioningConnectionFactory", ConnectionFactory.class);
79          this.testEntries = LdapTestUtils.readLdif(this.usersLdif, this.baseDn);
80          final Connection connection = getConnection();
81          try {
82              connection.open();
83              LdapTestUtils.createLdapEntries(connection, this.directoryType, this.testEntries);
84          } finally {
85              LdapUtils.closeConnection(connection);
86          }
87      }
88  
89      @After
90      public void tearDown() throws Exception {
91          if (!this.enableLdapTests) {
92              return;
93          }
94          final Connection connection = getConnection();
95          try {
96              connection.open();
97              LdapTestUtils.removeLdapEntries(connection, this.testEntries);
98          } finally {
99              LdapUtils.closeConnection(connection);
100         }
101     }
102 
103     /**
104      * Gets a connection for provisioning/deprovisioning test data.
105      *
106      * @return LDAP connection.
107      * @throws LdapException On errors.
108      */
109     protected Connection getConnection() throws LdapException {
110         return provisioningConnectionFactory.getConnection();
111     }
112 
113     protected String getUsername(final LdapEntry entry) {
114         return entry.getAttribute(usernameAttribute).getStringValue();
115     }
116 }
