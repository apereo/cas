122     /**
123      * Initializes the object after properties are set.
124      */
125     @PostConstruct
126     public void initialize() {
127         for (final SearchScope scope : SearchScope.values()) {
128             if (scope.ordinal() == this.searchControls.getSearchScope()) {
129                 this.searchScope = scope;
130             }
131         }
132         this.attributes = getResultAttributeMapping().keySet().toArray(new String[getResultAttributeMapping().size()]);
133     }

135     @Override
136     protected List<IPersonAttributes> getPeopleForQuery(final SearchFilter filter, final String userName) {
137         Connection connection = null;
138         try {
139             try {
140                 connection = this.connectionFactory.getConnection();
141             } catch (final LdapException e) {
142                 throw new RuntimeException("Failed getting LDAP connection", e);
143             }
144             final Response<SearchResult> response;
145             try {
146                 response = new SearchOperation(connection).execute(createRequest(filter));
147             } catch (final LdapException e) {
148                 throw new RuntimeException("Failed executing LDAP query " + filter, e);
149             }
150             final SearchResult result = response.getResult();
151             final List<IPersonAttributes> peopleAttributes = new ArrayList<IPersonAttributes>(result.size());
152             for (final LdapEntry entry : result.getEntries()) {
153                 final IPersonAttributes person;
154                 final String userNameAttribute = this.getConfiguredUserNameAttribute();
155                 final Map<String, List<Object>> attributes = convertLdapEntryToMap(entry);
156                 if (attributes.containsKey(userNameAttribute)) {
157                     person = new CaseInsensitiveAttributeNamedPersonImpl(userNameAttribute, attributes);
158                 } else {
159                     person = new CaseInsensitiveNamedPersonImpl(userName , attributes);
160                 }
161                 peopleAttributes.add(person);
162             }
163 
164             return peopleAttributes;
165         } finally {
166             LdapUtils.closeConnection(connection);
167         }
168     }
169 
170     @Override
171     protected SearchFilter appendAttributeToQuery(
172             final SearchFilter filter, final String attribute, final List<Object> values) {
173         final SearchFilter query;
174         if (filter == null && values.size() > 0) {
175             query = new SearchFilter(this.searchFilter);
176             query.setParameter(0, values.get(0).toString());
177             logger.debug("Constructed LDAP search query [{}]", query.format());
178         } else {
179             throw new UnsupportedOperationException("Multiple attributes not supported.");
180         }
181         return query;
182     }

184     /**
185      * Creates a search request from a search filter.
186      *
187      * @param filter LDAP search filter.
188      *
189      * @return ldaptive search request.
190      */
191     private SearchRequest createRequest(final SearchFilter filter) {
192         final SearchRequest request = new SearchRequest();
193         request.setBaseDn(this.baseDN);
194         request.setSearchFilter(filter);
195         request.setReturnAttributes(this.attributes);
196         request.setSearchScope(this.searchScope);
197         request.setSizeLimit(this.searchControls.getCountLimit());
198         request.setTimeLimit(this.searchControls.getTimeLimit());
199         return request;
200     }

202     /**
203      * Converts an ldaptive <code>LdapEntry</code> containing result entry attributes into an attribute map as needed
204      * by Person Directory components.
205      *
206      * @param entry Ldap entry.
207      *
208      * @return Attribute map.
209      */
210     private Map<String, List<Object>> convertLdapEntryToMap(final LdapEntry entry) {
211         final Map<String, List<Object>> attributeMap = new LinkedHashMap<String, List<Object>>(entry.size());
212         for (final LdapAttribute attr : entry.getAttributes()) {
213             attributeMap.put(attr.getName(), new ArrayList<Object>(attr.getStringValues()));
214         }
215         logger.debug("Converted ldap DN entry [{}] to attribute map {}", entry.getDn(), attributeMap.toString());
216         return attributeMap;
217     }
