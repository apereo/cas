54   * Implementation of the ServiceRegistryDao interface which stores the services in a LDAP Directory.

62      private final Logger logger = LoggerFactory.getLogger(this.getClass());
63  
64      @NotNull
65      private ConnectionFactory connectionFactory;

67      @NotNull
68      private LdapRegisteredServiceMapper ldapServiceMapper = new DefaultLdapRegisteredServiceMapper();
69

72

74      private String loadFilter;
75

77      private SearchRequest searchRequest;

80      /**
81       * Inits the dao with the search filter and load filters.
82       */
83      @PostConstruct
84      public void init() {
85          this.searchFilter = '(' + this.ldapServiceMapper.getIdAttribute() +  "={0})";
86          this.loadFilter = "(objectClass=" + this.ldapServiceMapper.getObjectClass() + ')';
87      }

89      @Override
90      public RegisteredService save(final RegisteredService rs) {
91          if (rs.getId() != RegisteredService.INITIAL_IDENTIFIER_VALUE) {
92              return update(rs);
93          }
94  
95          Connection connection = null;
96          try {
97              connection = getConnection();
98              final AddOperation operation = new AddOperation(connection);
99  
100             final LdapEntry entry = this.ldapServiceMapper.mapFromRegisteredService(this.searchRequest.getBaseDn(), rs);
101             operation.execute(new AddRequest(entry.getDn(), entry.getAttributes()));
102         } catch (final LdapException e) {
103             logger.error(e.getMessage(), e);
104         } finally {
105             LdapUtils.closeConnection(connection);
106         }
107         return rs;
108     }
109 
110     /**
111      * Update the ldap entry with the given registered service.
112      *
113      * @param rs the rs
114      * @return the registered service
115      */
116     private RegisteredService update(final RegisteredService rs) {
117         Connection searchConnection = null;
118         try {
119             searchConnection = getConnection();
120             final Response<SearchResult> response = searchForServiceById(searchConnection, rs.getId());
121             if (hasResults(response)) {
122                 final String currentDn = response.getResult().getEntry().getDn();
123 
124                 Connection modifyConnection = null;
125                 try {
126                     modifyConnection = getConnection();
127                     final ModifyOperation operation = new ModifyOperation(searchConnection);
128 
129                     final List<AttributeModification> mods = new ArrayList<>();
130 
131                     final LdapEntry entry = this.ldapServiceMapper.mapFromRegisteredService(this.searchRequest.getBaseDn(), rs);
132                     for (final LdapAttribute attr : entry.getAttributes()) {
133                         if (!attr.getName().equals(this.ldapServiceMapper.getIdAttribute())) {
134                             mods.add(new AttributeModification(AttributeModificationType.REPLACE, attr));
135                         }
136                     }
137                     final ModifyRequest request = new ModifyRequest(currentDn, mods.toArray(new AttributeModification[]{}));
138                     operation.execute(request);
139                 }catch (final LdapException e) {
140                     logger.error(e.getMessage(), e);
141                 } finally {
142                     LdapUtils.closeConnection(modifyConnection);
143                 }
144             }
145         } catch (final LdapException e) {
146             logger.error(e.getMessage(), e);
147         } finally {
148             LdapUtils.closeConnection(searchConnection);
149         }
150         return rs;
151     }

153     @Override
154     public boolean delete(final RegisteredService registeredService) {
155         Connection connection = null;
156         try {
157             connection = getConnection();
158 
159             final Response<SearchResult> response = searchForServiceById(connection, registeredService.getId());
160             if (hasResults(response)) {
161                 final LdapEntry entry = response.getResult().getEntry();
162                 final DeleteOperation delete = new DeleteOperation(connection);
163                 final DeleteRequest request = new DeleteRequest(entry.getDn());
164                 final Response<Void> res = delete.execute(request);
165                 return res.getResultCode() == ResultCode.SUCCESS;
166             }
167         } catch (final LdapException e) {
168             logger.error(e.getMessage(), e);
169         } finally {
170             LdapUtils.closeConnection(connection);
171         }
172 
173         return false;
174     }
175 
176     @Override
177     public List<RegisteredService> load() {
178         Connection connection = null;
179         final List<RegisteredService> list = new LinkedList<>();
180         try {
181             connection = getConnection();
182             final Response<SearchResult> response =
183                     executeSearchOperation(connection, new SearchFilter(this.loadFilter));
184             if (hasResults(response)) {
185                 for (final LdapEntry entry : response.getResult().getEntries()) {
186                     final RegisteredService svc = this.ldapServiceMapper.mapToRegisteredService(entry);
187                     list.add(svc);
188                 }
189             }
190         } catch (final LdapException e) {
191             logger.error(e.getMessage(), e);
192         } finally {
193             LdapUtils.closeConnection(connection);
194         }
195         return list;
196     }
197 
198     @Override
199     public RegisteredService findServiceById(final long id) {
200         Connection connection = null;
201         try {
202             connection = getConnection();
203 
204             final Response<SearchResult> response = searchForServiceById(connection, id);
205             if (hasResults(response)) {
206                 return this.ldapServiceMapper.mapToRegisteredService(response.getResult().getEntry());
207             }
208         } catch (final LdapException e) {
209             logger.error(e.getMessage(), e);
210         } finally {
211             LdapUtils.closeConnection(connection);
212         }
213 
214         return null;
215     }

217     /**
218      * Search for service by id.
219      *
220      * @param connection the connection
221      * @param id the id
222      * @return the response
223      * @throws LdapException the ldap exception
224      */
225     private Response<SearchResult> searchForServiceById(final Connection connection, final long id)
226             throws LdapException {
227 
228         final SearchFilter filter = new SearchFilter(this.searchFilter);
229         filter.setParameter(0, id);
230         return executeSearchOperation(connection, filter);
231     }
232 
233     /**
234      * Execute search operation.
235      *
236      * @param connection the connection
237      * @param filter the filter
238      * @return the response
239      * @throws LdapException the ldap exception
240      */
241     private Response<SearchResult> executeSearchOperation(final Connection connection, final SearchFilter filter)
242             throws LdapException {
243 
244         final SearchOperation searchOperation = new SearchOperation(connection);
245         final SearchRequest request = newRequest(filter);
246         logger.debug("Using search request {}", request.toString());
247         return searchOperation.execute(request);
248     }

262     /**
263      * Checks to see if response has a result.
264      *
265      * @param response the response
266      * @return true, if successful
267      */
268     private boolean hasResults(final Response<SearchResult> response) {
269         final SearchResult result = response.getResult();
270         if (result != null && result.getEntry() != null) {
271             return true;
272         }
273 
274         logger.trace("Requested ldap operation did not return a result or an ldap entry. Code: {}, Message: {}",
275                 response.getResultCode(), response.getMessage());
276         return false;
277     }

279     /**
280      * Builds a new request.
281      *
282      * @param filter the filter
283      * @return the search request
284      */
285     private SearchRequest newRequest(final SearchFilter filter) {
286 
287         final SearchRequest sr = new SearchRequest(this.searchRequest.getBaseDn(), filter);
288         sr.setBinaryAttributes(ReturnAttributes.ALL_USER.value());
289         sr.setDerefAliases(this.searchRequest.getDerefAliases());
290         sr.setSearchEntryHandlers(this.searchRequest.getSearchEntryHandlers());
291         sr.setSearchReferenceHandlers(this.searchRequest.getSearchReferenceHandlers());
292         sr.setFollowReferrals(this.searchRequest.getFollowReferrals());
293         sr.setReturnAttributes(ReturnAttributes.ALL_USER.value());
294         sr.setSearchScope(this.searchRequest.getSearchScope());
295         sr.setSizeLimit(this.searchRequest.getSizeLimit());
296         sr.setSortBehavior(this.searchRequest.getSortBehavior());
297         sr.setTimeLimit(this.searchRequest.getTimeLimit());
298         sr.setTypesOnly(this.searchRequest.getTypesOnly());
299         sr.setControls(this.searchRequest.getControls());
300         return sr;
301     }

303     /**
304      * Gets connection from the factory.
305      * Opens the connection if needed.
306      *
307      * @return the connection
308      * @throws LdapException the ldap exception
309      */
310     private Connection getConnection() throws LdapException {
311         final Connection c = this.connectionFactory.getConnection();
312         if (!c.isOpen()) {
313             c.open();
314         }
315         return c;
316     }
