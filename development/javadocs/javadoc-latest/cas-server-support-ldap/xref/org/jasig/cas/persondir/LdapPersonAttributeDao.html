49  /**
50   * Person directory <code>IPersonAttribute</code> implementation that queries an LDAP directory
51   * with ldaptive components to populate person attributes.
52   *
53   * @author Marvin S. Addison
54   * @since 4.0.0
55   */
56  public class LdapPersonAttributeDao extends AbstractQueryPersonAttributeDao<SearchFilter> {
57  
58      /** Logger instance. **/
59      protected final Logger logger = LoggerFactory.getLogger(this.getClass());
60  
61      /** Search base DN. */
62      @NotNull
63      private String baseDN;
64  
65      /** Search controls. */
66      @NotNull
67      private SearchControls searchControls;
68  
69      /** LDAP connection factory. */
70      @NotNull
71      private ConnectionFactory connectionFactory;
72  
73      /** LDAP search scope. */
74      private SearchScope searchScope;
75  
76      /** LDAP search filter. */
77      @NotNull
78      private String searchFilter;
79  
80      /** LDAP attributes to fetch from search results. */
81      private String[] attributes;
82  
83      /**
84       * Sets the base DN of the LDAP search for attributes.
85       *
86       * @param dn LDAP base DN of search.
87       */
88      public void setBaseDN(final String dn) {
89          this.baseDN = dn;
90      }
91  
92      /**
93       * Sets the LDAP search filter used to query for person attributes.
94       *
95       * @param filter Search filter of the form "(usernameAttribute={0})" where {0} and similar ordinal placeholders
96       *               are replaced with query parameters.
97       */
98      public void setSearchFilter(final String filter) {
99          this.searchFilter = filter;
100     }
101 
102     /**
103      * Sets a number of parameters that control LDAP search semantics including search scope,
104      * maximum number of results retrieved, and search timeout.
105      *
106      * @param searchControls LDAP search controls.
107      */
108     public void setSearchControls(final SearchControls searchControls) {
109         this.searchControls = searchControls;
110     }
111 
112     /**
113      * Sets the connection factory that produces LDAP connections on which searches occur. It is strongly recommended
114      * that this be a <code>PooledConnecitonFactory</code> object.
115      *
116      * @param connectionFactory LDAP connection factory.
117      */
118     public void setConnectionFactory(final ConnectionFactory connectionFactory) {
119         this.connectionFactory = connectionFactory;
120     }
121 
122     /**
123      * Initializes the object after properties are set.
124      */
125     @PostConstruct
126     public void initialize() {
127         for (final SearchScope scope : SearchScope.values()) {
128             if (scope.ordinal() == this.searchControls.getSearchScope()) {
129                 this.searchScope = scope;
130             }
131         }
132         this.attributes = getResultAttributeMapping().keySet().toArray(new String[getResultAttributeMapping().size()]);
133     }
134 
135     @Override
136     protected List<IPersonAttributes> getPeopleForQuery(final SearchFilter filter, final String userName) {
137         Connection connection = null;
138         try {
139             try {
140                 connection = this.connectionFactory.getConnection();
141                 connection.open();
142             } catch (final LdapException e) {
143                 throw new RuntimeException("Failed getting LDAP connection", e);
144             }
145             final Response<SearchResult> response;
146             try {
147                 response = new SearchOperation(connection).execute(createRequest(filter));
148             } catch (final LdapException e) {
149                 throw new RuntimeException("Failed executing LDAP query " + filter, e);
150             }
151             final SearchResult result = response.getResult();
152             final List<IPersonAttributes> peopleAttributes = new ArrayList<>(result.size());
153             for (final LdapEntry entry : result.getEntries()) {
154                 final IPersonAttributes person;
155                 final String userNameAttribute = this.getConfiguredUserNameAttribute();
156                 final Map<String, List<Object>> attributes = convertLdapEntryToMap(entry);
157                 if (attributes.containsKey(userNameAttribute)) {
158                     person = new CaseInsensitiveAttributeNamedPersonImpl(userNameAttribute, attributes);
159                 } else {
160                     person = new CaseInsensitiveNamedPersonImpl(userName , attributes);
161                 }
162                 peopleAttributes.add(person);
163             }
164 
165             return peopleAttributes;
166         } finally {
167             LdapUtils.closeConnection(connection);
168         }
169     }
170 
171     @Override
172     protected SearchFilter appendAttributeToQuery(
173             final SearchFilter filter, final String attribute, final List<Object> values) {
174         final SearchFilter query;
175         if (filter == null && values.size() > 0) {
176             query = new SearchFilter(this.searchFilter);
177             query.setParameter(0, values.get(0).toString());
178             logger.debug("Constructed LDAP search query [{}]", query.format());
179         } else {
180             throw new UnsupportedOperationException("Multiple attributes not supported.");
181         }
182         return query;
183     }
184 
185     /**
186      * Creates a search request from a search filter.
187      *
188      * @param filter LDAP search filter.
189      *
190      * @return ldaptive search request.
191      */
192     private SearchRequest createRequest(final SearchFilter filter) {
193         final SearchRequest request = new SearchRequest();
194         request.setBaseDn(this.baseDN);
195         request.setSearchFilter(filter);
196         request.setReturnAttributes(this.attributes);
197         request.setSearchScope(this.searchScope);
198         request.setSizeLimit(this.searchControls.getCountLimit());
199         request.setTimeLimit(this.searchControls.getTimeLimit());
200         return request;
201     }
202 
203     /**
204      * Converts an ldaptive <code>LdapEntry</code> containing result entry attributes into an attribute map as needed
205      * by Person Directory components.
206      *
207      * @param entry Ldap entry.
208      *
209      * @return Attribute map.
210      */
211     private Map<String, List<Object>> convertLdapEntryToMap(final LdapEntry entry) {
212         final Map<String, List<Object>> attributeMap = new LinkedHashMap<>(entry.size());
213         for (final LdapAttribute attr : entry.getAttributes()) {
214             attributeMap.put(attr.getName(), new ArrayList<Object>(attr.getStringValues()));
215         }
216         logger.debug("Converted ldap DN entry [{}] to attribute map {}", entry.getDn(), attributeMap.toString());
217         return attributeMap;
218     }
