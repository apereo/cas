19  package org.jasig.cas.support.spnego.authentication.principal;
20  
21  import org.jasig.cas.authentication.UsernamePasswordCredential;
22  import org.jasig.cas.authentication.principal.DefaultPrincipalFactory;
23  import org.junit.Before;
24  import org.junit.Test;
25  
26  import static org.junit.Assert.*;
27  
28  /**
29   * @author Marc-Antoine Garrigue
30   * @author Arnaud Lesueur
31   * @since 3.1
32   *
33   */
34  public class SpnegoCredentialsToPrincipalResolverTests {
35      private SpnegoPrincipalResolver resolver;
36  
37      private SpnegoCredential spnegoCredentials;
38  
39      @Before
40      public void setUp() throws Exception {
41          this.resolver = new SpnegoPrincipalResolver();
42          this.spnegoCredentials = new SpnegoCredential(new byte[] {0, 1, 2});
43      }
44  
45      @Test
46      public void verifyValidCredentials() {
47          this.spnegoCredentials.setPrincipal(new DefaultPrincipalFactory().createPrincipal("test"));
48          assertEquals("test", this.resolver.resolve(this.spnegoCredentials)
49                  .getId());
50      }
51  
52      @Test
53      public void verifySupports() {
54          assertFalse(this.resolver.supports(null));
55          assertTrue(this.resolver.supports(this.spnegoCredentials));
56          assertFalse(this.resolver.supports(new UsernamePasswordCredential()));
57      }
58  }
