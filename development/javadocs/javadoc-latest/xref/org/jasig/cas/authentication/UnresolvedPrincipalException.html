19  package org.jasig.cas.authentication;
20  
21  /**
22   * Describes an error condition where a principal could not be resolved.
23   *
24   * @author Marvin S. Addison
25   * @since 4.0.0
26   */
27  public class UnresolvedPrincipalException extends PrincipalException {
28  
29      /** Serialization version marker. */
30      private static final long serialVersionUID = 380456166081802820L;
31  
32      /** Error message when there was no error that prevent principal resolution. */
33      private static final String UNRESOLVED_PRINCIPAL = "No resolver produced a principal.";
34  
35      /**
36       * Creates a new instance from an authentication event that was successful prior to principal resolution.
37       *
38       * @param authentication Authentication event.
39       */
40      public UnresolvedPrincipalException(final Authentication authentication) {
41          super(UNRESOLVED_PRINCIPAL, authentication.getFailures(), authentication.getSuccesses());
42      }
43  
44      /**
45       * Creates a new instance from an authentication event that was successful prior to principal resolution.
46       * This form should be used when a resolver exception prevented principal resolution.
47       *
48       * @param authentication Authentication event.
49       * @param cause Exception that prevented principal resolution.
50       */
51      public UnresolvedPrincipalException(final Authentication authentication, final Exception cause) {
52          super(cause.getMessage(), authentication.getFailures(), authentication.getSuccesses());
53      }
54  }
