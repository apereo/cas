19  package org.jasig.cas.monitor;
20  
21  /**
22   * Abstract base class for monitors that observe cache storage systems.
23   *
24   * @author Marvin S. Addison
25   * @since 3.5.1
26   */
27  public abstract class AbstractCacheMonitor extends AbstractNamedMonitor<CacheStatus> {
28  
29      /** Default free capacity threshold is 10%. */
30      public static final int DEFAULT_WARN_FREE_THRESHOLD = 10;
31  
32      /** Default eviction threshold is 0. */
33      public static final long DEFAULT_EVICTION_THRESHOLD = 0;
34  
35      /** Percent free capacity threshold below which a warning is issued.*/
36      private int warnFreeThreshold = DEFAULT_WARN_FREE_THRESHOLD;
37  
38      /** Threshold for number of acceptable evictions above which an error is issued. */
39      private long evictionThreshold = DEFAULT_EVICTION_THRESHOLD;
40  
41  
42      /**
43       * Sets the percent free capacity threshold below which a warning is issued.
44       *
45       * @param percent Warning threshold percent.
46       */
47      public void setWarnFreeThreshold(final int percent) {
48          this.warnFreeThreshold = percent;
49      }
50  
51  
52      /**
53       * Sets the eviction threshold count above which an error is issued.
54       *
55       * @param count Threshold for number of cache evictions.
56       */
57      public void setEvictionThreshold(final long count) {
58          this.evictionThreshold = count;
59      }
60  
61      @Override
62      public CacheStatus observe() {
63          CacheStatus status;
64          try {
65              final CacheStatistics[] statistics = getStatistics();
66              if (statistics == null || statistics.length == 0) {
67                  return new CacheStatus(StatusCode.ERROR, "Cache statistics not available.");
68              }
69              StatusCode overall = StatusCode.OK;
70              StatusCode code;
71              for (final CacheStatistics stats : statistics) {
72                  code = status(stats);
73                  // Record highest status which is equivalent to worst case
74                  if (code.value() > overall.value()) {
75                      overall = code;
76                  }
77              }
78              status = new CacheStatus(overall, null, statistics);
79          } catch (final Exception e) {
80              status = new CacheStatus(e);
81          }
82          return status;
83      }
84  
85  
86      /**
87       * Gets the statistics from this monitor.
88       *
89       * @return the statistics
90       */
91      protected abstract CacheStatistics[] getStatistics();
92  
93  
94      /**
95       * Computes the status code for a given set of cache statistics.
96       *
97       * @param statistics Cache statistics.
98       *
99       * @return {@link StatusCode#WARN} if eviction count is above threshold or if
100      * percent free space is below threshold, otherwise {@link StatusCode#OK}.
101      */
102     protected StatusCode status(final CacheStatistics statistics) {
103         final StatusCode code;
104         if (statistics.getEvictions() > this.evictionThreshold) {
105             code = StatusCode.WARN;
106         } else if (statistics.getPercentFree() < this.warnFreeThreshold) {
107             code = StatusCode.WARN;
108         } else {
109             code = StatusCode.OK;
110         }
111         return code;
112     }
113 }
