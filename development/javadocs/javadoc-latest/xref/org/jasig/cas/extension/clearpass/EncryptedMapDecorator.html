146     /**
147      * Decorates a map using the provided algorithms.
148      * <p>Takes a salt and secretKey so that it can work with a distributed cache.
149      *
150      * @param decoratedMap the map to decorate.  CANNOT be NULL.
151      * @param hashAlgorithm the algorithm to use for hashing.  CANNOT BE NULL.
152      * @param salt the salt, as a String. Gets converted to bytes.   CANNOT be NULL.
153      * @param secretKeyAlgorithm the encryption algorithm. CANNOT BE NULL.
154      * @param secretKey the secret to use.  CANNOT be NULL.
155      * @throws RuntimeException if the algorithm cannot be found or the iv size cant be determined.
156      */
157     public EncryptedMapDecorator(final Map<String, String> decoratedMap, final String hashAlgorithm, final byte[] salt,
158             final String secretKeyAlgorithm, final Key secretKey) {
159         try {
160             this.decoratedMap = decoratedMap;
161             this.key = secretKey;
162             this.salt = ByteSource.wrap(salt);
163             this.secretKeyAlgorithm = secretKeyAlgorithm;
164             this.messageDigest = MessageDigest.getInstance(hashAlgorithm);
165             this.ivSize = getIvSize();
166         } catch (final Exception e) {
167             throw new RuntimeException(e);
168         }
169     }

260     /**
261      * Construct hashed key.
262      *
263      * @param key the key
264      * @return the string
265      */
266     protected String constructHashedKey(final String key) {
267         if (key == null) {
268             return null;
269         }
270 
271         final MessageDigest messageDigest = getMessageDigest();
272         messageDigest.update(consumeByteSourceOrNull(this.salt));
273         messageDigest.update(key.toLowerCase().getBytes(Charset.defaultCharset()));
274         final String hash = getFormattedText(messageDigest.digest());
275 
276         logger.debug("Generated hash of value [{}] for key [{}].", hash, key);
277         return hash;
278     }

280     /**
281      * Decrypt the value.
282      *
283      * @param value the value
284      * @param hashedKey the hashed key
285      * @return the string
286      */
287     protected String decrypt(final String value, final String hashedKey) {
288         if (value == null) {
289             return null;
290         }
291 
292         try {
293             final Cipher cipher = getCipherObject();
294             final byte[] ivCiphertext = CompressionUtils.decodeBase64ToByteArray(value);
295             final int ivSize = byte2int(Arrays.copyOfRange(ivCiphertext, 0, INTEGER_LEN));
296             final byte[] ivValue = Arrays.copyOfRange(ivCiphertext, INTEGER_LEN, (INTEGER_LEN + ivSize));
297             final byte[] ciphertext = Arrays.copyOfRange(ivCiphertext, INTEGER_LEN + ivSize, ivCiphertext.length);
298             final IvParameterSpec ivSpec = new IvParameterSpec(ivValue);
299             
300             cipher.init(Cipher.DECRYPT_MODE, this.key, ivSpec);
301 
302             final byte[] plaintext = cipher.doFinal(ciphertext);
303 
304             return new String(plaintext, Charset.defaultCharset());
305         } catch (final Exception e) {
306             throw new RuntimeException(e);
307         }
308     }

310     /**
311      * Read the contents of the source into a byte array.
312      * @param source  the byte array source
313      * @return the byte[] read from the source or null
314      */
315     private byte[] consumeByteSourceOrNull(final ByteSource source) {
316         try {
317             if (source == null || source.isEmpty()) {
318                 return null;
319             }
320             return source.read();
321         } catch (final IOException e) {
322             logger.warn("Could not consume the byte array source", e);
323             return null;
324         }
325     }

362     /**
363      * Encrypt.
364      *
365      * @param value the value
366      * @param hashedKey the hashed key
367      * @return the string
368      */
369     protected String encrypt(final String value, final String hashedKey) {
370         if (value == null) {
371             return null;
372         }
373 
374         try {
375             final Cipher cipher = getCipherObject();
376             final byte[] ivValue = generateIV(this.ivSize);
377             final IvParameterSpec ivSpec = new IvParameterSpec(ivValue);
378 
379             cipher.init(Cipher.ENCRYPT_MODE, this.key, ivSpec);
380             
381             final byte[] ciphertext = cipher.doFinal(value.getBytes(Charset.defaultCharset()));
382             final byte[] ivCiphertext = new byte[INTEGER_LEN + this.ivSize + ciphertext.length];
383 
384             System.arraycopy(int2byte(this.ivSize), 0, ivCiphertext, 0, INTEGER_LEN);
385             System.arraycopy(ivValue, 0, ivCiphertext, INTEGER_LEN, this.ivSize);
386             System.arraycopy(ciphertext, 0, ivCiphertext, INTEGER_LEN + this.ivSize, ciphertext.length);
387 
388             return CompressionUtils.encodeBase64(ivCiphertext);
389         } catch(final Exception e) {
390             throw new RuntimeException(e);
391         }
392     }

438     /**
439      * Tries to clone the {@link MessageDigest} that was created during construction. If the clone fails
440      * that is remembered and from that point on new {@link MessageDigest} instances will be created on
441      * every call.
442      * <p>
443      * Adopted from the Spring EhCache Annotations project.
444      *
445      * @return Generates a {@link MessageDigest} to use
446      */
447     protected MessageDigest getMessageDigest() {
448         if (this.cloneNotSupported) {
449             final String algorithm = this.messageDigest.getAlgorithm();
450             try {
451                 return MessageDigest.getInstance(algorithm);
452             } catch (final NoSuchAlgorithmException e) {
453                 throw new IllegalStateException("MessageDigest algorithm '" + algorithm + "' was supported when "
454                         + this.getClass().getSimpleName()
455                         + " was created but is not now. This should not be possible.", e);
456             }
457         }
458 
459         try {
460             return (MessageDigest) this.messageDigest.clone();
461         } catch (final CloneNotSupportedException e) {
462             this.cloneNotSupported = true;
463             final String msg = String.format("Could not clone MessageDigest using algorithm '%s'. "
464                         + "MessageDigest.getInstance will be used from now on which will be much more expensive.",
465                         this.messageDigest.getAlgorithm());
466             logger.warn(msg, e);
467             return this.getMessageDigest();
468         }
469     }

471     /**
472      * Takes the raw bytes from the digest and formats them.
473      *
474      * @param bytes the raw bytes from the digest.
475      * @return the formatted bytes.
476      */
477     private static String getFormattedText(final byte[] bytes) {
478         final StringBuilder buf = new StringBuilder(bytes.length * 2);
479 
480         for (byte b : bytes) {
481             buf.append(HEX_DIGITS[b >> HEX_RIGHT_SHIFT_COEFFICIENT & HEX_HIGH_BITS_BITWISE_FLAG]);
482             buf.append(HEX_DIGITS[b & HEX_HIGH_BITS_BITWISE_FLAG]);
483         }
484         return buf.toString();
485     }

500     /**
501      * Gets the secret key.
502      *
503      * @param secretKeyAlgorithm the secret key algorithm
504      * @param secretKey the secret key
505      * @param salt the salt
506      * @return the secret key
507      * @throws Exception the exception
508      */
509     private static Key getSecretKey(final String secretKeyAlgorithm, final String secretKey,
510             final String salt) throws Exception {
511 
512         final SecretKeyFactory factory = SecretKeyFactory.getInstance(SECRET_KEY_FACTORY_ALGORITHM);
513         final KeySpec spec = new PBEKeySpec(secretKey.toCharArray(), char2byte(salt), 65536, 128);
514         final SecretKey tmp = factory.generateSecret(spec);
515         return new SecretKeySpec(tmp.getEncoded(), secretKeyAlgorithm);
516     }
