41  /**
42   * This logout manager handles the Single Log Out process.
43   *
44   * @author Jerome Leleu
45   * @since 4.0.0
46   */
47  public final class LogoutManagerImpl implements LogoutManager {
48  
49      /** The logger. */
50      private static final Logger LOGGER = LoggerFactory.getLogger(LogoutManagerImpl.class);
51  
52      /** The parameter name that contains the logout request. */
53      private static final String LOGOUT_PARAMETER_NAME = "logoutRequest";
54  
55      /** ASCII character set. */
56      private static final Charset ASCII = Charset.forName("ASCII");
57  
58      /** The services manager. */
59      @NotNull
60      private final ServicesManager servicesManager;
61  
62      /** An HTTP client. */
63      @NotNull
64      private final HttpClient httpClient;
65  
66      @NotNull
67      private final LogoutMessageCreator logoutMessageBuilder;
68      
69      /** Whether single sign out is disabled or not. */
70      private boolean singleLogoutCallbacksDisabled;
71      
72      /** 
73       * Whether messages to endpoints would be sent in an asynchronous fashion.
74       * True by default.
75       **/
76      private boolean asynchronous = true;
77      
78      /**
79       * Build the logout manager.
80       * @param servicesManager the services manager.
81       * @param httpClient an HTTP client.
82       * @param logoutMessageBuilder the builder to construct logout messages.
83       */
84      public LogoutManagerImpl(final ServicesManager servicesManager, final HttpClient httpClient,
85                               final LogoutMessageCreator logoutMessageBuilder) {
86          this.servicesManager = servicesManager;
87          this.httpClient = httpClient;
88          this.logoutMessageBuilder = logoutMessageBuilder;
89      }
90  
91      /**
92       * Set if messages are sent in an asynchronous fashion.
93       *
94       * @param asyncCallbacks if message is synchronously sent
95       * @since 4.1.0
96       */
97      public void setAsynchronous(final boolean asyncCallbacks) {
98          this.asynchronous = asyncCallbacks;
99      }
100     
101     /**
102      * Set if messages are sent in an asynchronous fashion.
103      *
104      * @param asyncCallbacks if message is synchronously sent
105      * @deprecated As of 4.1. Use {@link #setAsynchronous(boolean)} instead
106      */
107     @Deprecated
108     public void setIssueAsynchronousCallbacks(final boolean asyncCallbacks) {
109         this.asynchronous = asyncCallbacks;
110         LOGGER.warn("setIssueAsynchronousCallbacks() is deprecated. Use setAsynchronous() instead.");
111     }
112     
113     /**
114      * Perform a back channel logout for a given ticket granting ticket and returns all the logout requests.
115      *
116      * @param ticket a given ticket granting ticket.
117      * @return all logout requests.
118      */
119     @Override
120     public List<LogoutRequest> performLogout(final TicketGrantingTicket ticket) {
121         final Map<String, Service> services;
122         // synchronize the retrieval of the services and their cleaning for the TGT
123         // to avoid concurrent logout mess ups
124         synchronized (ticket) {
125             services = ticket.getServices();
126             ticket.removeAllServices();
127         }
128         ticket.markTicketExpired();
129 
130         final List<LogoutRequest> logoutRequests = new ArrayList<>();
131         // if SLO is not disabled
132         if (!this.singleLogoutCallbacksDisabled) {
133             // through all services
134             for (final Map.Entry<String, Service> entry : services.entrySet()) {
135                 // it's a SingleLogoutService, else ignore
136                 final Service service = entry.getValue();
137                 if (service instanceof SingleLogoutService) {
138                     final LogoutRequest logoutRequest = handleLogoutForSloService((SingleLogoutService) service, entry.getKey());
139                     if (logoutRequest != null) {
140                         LOGGER.debug("Captured logout request [{}]", logoutRequest);
141                         logoutRequests.add(logoutRequest);
142                     }
143                 }
144             }
145         }
146 
147         return logoutRequests;
148     }
149 
150     /**
151      * Service supports back channel single logout?
152      * Service must be found in the registry. enabled and logout type must not be {@link LogoutType#NONE}.
153      * @param registeredService the registered service
154      * @return true, if support is available.
155      */
156     private boolean serviceSupportsSingleLogout(final RegisteredService registeredService) {
157         return registeredService != null
158                 && registeredService.getAccessStrategy().isServiceAccessAllowed()
159                 && registeredService.getLogoutType() != LogoutType.NONE;
160     }
161 
162     /**
163      * Handle logout for slo service.
164      *
165      * @param singleLogoutService the service
166      * @param ticketId the ticket id
167      * @return the logout request
168      */
169     private LogoutRequest handleLogoutForSloService(final SingleLogoutService singleLogoutService, final String ticketId) {
170         if (!singleLogoutService.isLoggedOutAlready()) {
171 
172             final RegisteredService registeredService = servicesManager.findServiceBy(singleLogoutService);
173             if (serviceSupportsSingleLogout(registeredService)) {
174 
175                 final URL logoutUrl = determineLogoutUrl(registeredService, singleLogoutService);
176                 final LogoutRequest logoutRequest = new LogoutRequest(ticketId, singleLogoutService, logoutUrl);
177                 final LogoutType type = registeredService.getLogoutType() == null
178                         ? LogoutType.BACK_CHANNEL : registeredService.getLogoutType();
179 
180                 switch (type) {
181                     case BACK_CHANNEL:
182                         if (performBackChannelLogout(logoutRequest)) {
183                             logoutRequest.setStatus(LogoutRequestStatus.SUCCESS);
184                         } else {
185                             logoutRequest.setStatus(LogoutRequestStatus.FAILURE);
186                             LOGGER.warn("Logout message not sent to [{}]; Continuing processing...", singleLogoutService.getId());
187                         }
188                         break;
189                     default:
190                         logoutRequest.setStatus(LogoutRequestStatus.NOT_ATTEMPTED);
191                         break;
192                 }
193                 return logoutRequest;
194             }
195 
196         }
197         return null;
198     }
199 
200     /**
201      * Determine logout url.
202      *
203      * @param registeredService the registered service
204      * @param singleLogoutService the single logout service
205      * @return the uRL
206      */
207     private URL determineLogoutUrl(final RegisteredService registeredService, final SingleLogoutService singleLogoutService) {
208         try {
209             URL logoutUrl = new URL(singleLogoutService.getOriginalUrl());
210             final URL serviceLogoutUrl = registeredService.getLogoutUrl();
211 
212             if (serviceLogoutUrl != null) {
213                 LOGGER.debug("Logout request will be sent to [{}] for service [{}]",
214                         serviceLogoutUrl, singleLogoutService);
215                 logoutUrl = serviceLogoutUrl;
216             }
217             return logoutUrl;
218         } catch (final Exception e) {
219             throw new IllegalArgumentException(e);
220         }
221     }
222 
223     /**
224      * Log out of a service through back channel.
225      *
226      * @param request the logout request.
227      * @return if the logout has been performed.
228      */
229     private boolean performBackChannelLogout(final LogoutRequest request) {
230         try {
231             final String logoutRequest = this.logoutMessageBuilder.create(request);
232             final SingleLogoutService logoutService = request.getService();
233             logoutService.setLoggedOutAlready(true);
234     
235             LOGGER.debug("Sending logout request for: [{}]", logoutService.getId());
236             final LogoutHttpMessage msg = new LogoutHttpMessage(request.getLogoutUrl(), logoutRequest);
237             LOGGER.debug("Prepared logout message to send is [{}]", msg);
238             return this.httpClient.sendMessageToEndPoint(msg);
239         } catch (final Exception e) {
240             LOGGER.error(e.getMessage(), e);
241         }
242         return false;
243     }
244 
245     /**
246      * Create a logout message for front channel logout.
247      *
248      * @param logoutRequest the logout request.
249      * @return a front SAML logout message.
250      */
251     public String createFrontChannelLogoutMessage(final LogoutRequest logoutRequest) {
252         final String logoutMessage = this.logoutMessageBuilder.create(logoutRequest);
253         LOGGER.trace("Attempting to deflate the logout message [{}]", logoutMessage);
254         return CompressionUtils.deflate(logoutMessage);
255     }

262     public void setSingleLogoutCallbacksDisabled(final boolean singleLogoutCallbacksDisabled) {
263         this.singleLogoutCallbacksDisabled = singleLogoutCallbacksDisabled;
264     }

266     /**
267      * A logout http message that is accompanied by a special content type
268      * and formatting.
269      * @since 4.1.0
270      */
271     private final class LogoutHttpMessage extends HttpMessage {
272         
273         /**
274          * Constructs a logout message, whose method of submission
275          * is controlled by the {@link LogoutManagerImpl#asynchronous}.
276          * 
277          * @param url The url to send the message to
278          * @param message Message to send to the url
279          */
280         public LogoutHttpMessage(final URL url, final String message) {
281             super(url, message, LogoutManagerImpl.this.asynchronous);
282             setContentType(MediaType.APPLICATION_FORM_URLENCODED_VALUE);
283         }
284 
285         /**
286          * {@inheritDoc}.
287          * Prepends the string "<code>logoutRequest=</code>" to the message body.
288          */
289         @Override
290         protected String formatOutputMessageInternal(final String message) {
291             return LOGOUT_PARAMETER_NAME + "=" + super.formatOutputMessageInternal(message);
292         }        
293     }
