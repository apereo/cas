42  /**
43   * Base class for mutable, persistable registered services.
44   *
45   * @author Marvin S. Addison
46   * @author Scott Battaglia
47   * @author Misagh Moayyed
48   * @since 3.0.0
49   */
50  @Entity
51  @Inheritance
52  @DiscriminatorColumn(name = "expression_type", length = 15, discriminatorType = DiscriminatorType.STRING,
53                       columnDefinition = "VARCHAR(15) DEFAULT 'ant'")
54  @Table(name = "RegisteredServiceImpl")
55  public abstract class AbstractRegisteredService implements RegisteredService, Comparable<RegisteredService> {
56  
57      private static final long serialVersionUID = 7645279151115635245L;
58  
59      /**
60       * The unique identifier for this service.
61       */
62      @Column(length = 255, updatable = true, insertable = true, nullable = false)
63      protected String serviceId;
64  
65      @Column(length = 255, updatable = true, insertable = true, nullable = false)
66      private String name;
67  
68      @Column(length = 255, updatable = true, insertable = true, nullable = true)
69      private String theme;
70  
71      @Id
72      @GeneratedValue(strategy = GenerationType.AUTO)
73      private long id = RegisteredService.INITIAL_IDENTIFIER_VALUE;
74  
75      @Column(length = 255, updatable = true, insertable = true, nullable = false)
76      private String description;
77  
78      /**
79       * Proxy policy for the service.
80       * By default, the policy is {@link RefuseRegisteredServiceProxyPolicy}.
81       */
82      @Lob
83      @Column(name = "proxy_policy", nullable = false)
84      private RegisteredServiceProxyPolicy proxyPolicy = new RefuseRegisteredServiceProxyPolicy();
85  
86      @Column(name = "evaluation_order", nullable = false)
87      private int evaluationOrder;
88  
89      /**
90       * Resolve the username for this service.
91       * By default the resolver is {@link DefaultRegisteredServiceUsernameProvider}.
92       */
93      @Lob
94      @Column(name = "username_attr", nullable = true)
95      private RegisteredServiceUsernameAttributeProvider usernameAttributeProvider =
96          new DefaultRegisteredServiceUsernameProvider();
97  
98      /**
99       * The logout type of the service. 
100      * The default logout type is the back channel one.
101      */
102     @Column(name = "logout_type", nullable = true)
103     private LogoutType logoutType = LogoutType.BACK_CHANNEL;
104 
105     @Lob
106     @Column(name = "required_handlers")
107     private HashSet<String> requiredHandlers = new HashSet<>();
108 
109     /** The attribute filtering policy. */
110     @Lob
111     @Column(name = "attribute_release")
112     private AttributeReleasePolicy attributeReleasePolicy = new ReturnAllowedAttributeReleasePolicy();
113 
114     @Column(name = "logo")
115     private URL logo;
116 
117     @Lob
118     @Column(name = "access_strategy")
119     private RegisteredServiceAccessStrategy accessStrategy =
120             new DefaultRegisteredServiceAccessStrategy();
121 
122     @Lob
123     @Column(name = "public_key")
124     private RegisteredServicePublicKey publicKey;
125

158         return this.accessStrategy;

161     @Override
162     public boolean equals(final Object o) {
163         if (o == null) {
164             return false;
165         }
166 
167         if (this == o) {
168             return true;
169         }
170 
171         if (!(o instanceof AbstractRegisteredService)) {
172             return false;
173         }
174 
175         final AbstractRegisteredService that = (AbstractRegisteredService) o;
176 
177         return new EqualsBuilder()
178                 .append(this.proxyPolicy, that.proxyPolicy)
179                 .append(this.evaluationOrder, that.evaluationOrder)
180                 .append(this.description, that.description)
181                 .append(this.name, that.name)
182                 .append(this.serviceId, that.serviceId)
183                 .append(this.theme, that.theme)
184                 .append(this.usernameAttributeProvider, that.usernameAttributeProvider)
185                 .append(this.logoutType, that.logoutType)
186                 .append(this.attributeReleasePolicy, that.attributeReleasePolicy)
187                 .append(this.accessStrategy, that.accessStrategy)
188                 .append(this.logo, that.logo)
189                 .append(this.publicKey, that.publicKey)
190                 .isEquals();
191     }
192 
193     @Override
194     public int hashCode() {
195         return new HashCodeBuilder(7, 31)
196                 .append(this.description)
197                 .append(this.serviceId)
198                 .append(this.name)
199                 .append(this.theme)
200                 .append(this.evaluationOrder)
201                 .append(this.usernameAttributeProvider)
202                 .append(this.accessStrategy)
203                 .append(this.logoutType)
204                 .append(this.attributeReleasePolicy)
205                 .append(this.accessStrategy)
206                 .append(this.logo)
207                 .append(this.publicKey)
208                 .toHashCode();
209     }

288     /**
289      * Copies the properties of the source service into this instance.
290      *
291      * @param source Source service from which to copy properties.
292      */
293     public void copyFrom(final RegisteredService source) {
294         this.setId(source.getId());
295         this.setProxyPolicy(source.getProxyPolicy());
296         this.setDescription(source.getDescription());
297         this.setName(source.getName());
298         this.setServiceId(source.getServiceId());
299         this.setTheme(source.getTheme());
300         this.setEvaluationOrder(source.getEvaluationOrder());
301         this.setUsernameAttributeProvider(source.getUsernameAttributeProvider());
302         this.setLogoutType(source.getLogoutType());
303         this.setAttributeReleasePolicy(source.getAttributeReleasePolicy());
304         this.setAccessStrategy(source.getAccessStrategy());
305         this.setLogo(source.getLogo());
306         this.setPublicKey(source.getPublicKey());
307     }
308 
309     /**
310      * {@inheritDoc}
311      * Compares this instance with the <code>other</code> registered service based on
312      * evaluation order, name. The name comparison is case insensitive.
313      *
314      * @see #getEvaluationOrder()
315      */
316     @Override
317     public int compareTo(final RegisteredService other) {
318         return new CompareToBuilder()
319                   .append(this.getEvaluationOrder(), other.getEvaluationOrder())
320                   .append(this.getName().toLowerCase(), other.getName().toLowerCase())
321                   .append(this.getServiceId(), other.getServiceId())
322                   .toComparison();
323     }

325     @Override
326     public String toString() {
327         final ToStringBuilder toStringBuilder = new ToStringBuilder(null, ToStringStyle.SHORT_PREFIX_STYLE);
328         toStringBuilder.append("id", this.id);
329         toStringBuilder.append("name", this.name);
330         toStringBuilder.append("description", this.description);
331         toStringBuilder.append("serviceId", this.serviceId);
332         toStringBuilder.append("usernameAttributeProvider", this.usernameAttributeProvider);
333         toStringBuilder.append("theme", this.theme);
334         toStringBuilder.append("evaluationOrder", this.evaluationOrder);
335         toStringBuilder.append("logoutType", this.logoutType);
336         toStringBuilder.append("attributeReleasePolicy", this.attributeReleasePolicy);
337         toStringBuilder.append("accessStrategy", this.accessStrategy);
338         toStringBuilder.append("publicKey", this.publicKey);
339         toStringBuilder.append("proxyPolicy", this.proxyPolicy);
340         toStringBuilder.append("logo", this.logo);
341 
342         return toStringBuilder.toString();
343     }

352     @Override
353     public Set<String> getRequiredHandlers() {
354         if (this.requiredHandlers == null) {
355             this.requiredHandlers = new HashSet<>();
356         }
357         return this.requiredHandlers;
358     }
359 
360     /**
361      * Sets the required handlers for this service.
362      *
363      * @param handlers the new required handlers
364      */
365     public void setRequiredHandlers(final Set<String> handlers) {
366         getRequiredHandlers().clear();
367         if (handlers == null) {
368             return;
369         }
370         for (final String handler : handlers) {
371             getRequiredHandlers().add(handler);
372         }
373     }
