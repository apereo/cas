19  package org.jasig.cas.web;
20  
21  import org.apache.commons.lang3.StringEscapeUtils;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  import org.springframework.web.servlet.HandlerExceptionResolver;
25  import org.springframework.web.servlet.ModelAndView;
26  import org.springframework.web.servlet.view.RedirectView;
27  import org.springframework.webflow.execution.repository.BadlyFormattedFlowExecutionKeyException;
28  import org.springframework.webflow.execution.repository.FlowExecutionRepositoryException;
29  
30  import javax.servlet.http.HttpServletRequest;
31  import javax.servlet.http.HttpServletResponse;
32  import javax.validation.constraints.NotNull;
33  import java.util.HashMap;
34  import java.util.Map;
35  
36  /**
37   * The FlowExecutionExceptionResolver catches the FlowExecutionRepositoryException
38   * thrown by Spring Webflow when the given flow id no longer exists. This can
39   * occur if a particular flow has reached an end state (the id is no longer
40   * valid)
41   * <p>
42   * It will redirect back to the requested URI which should start a new workflow.
43   * </p>
44   *
45   * @author Scott Battaglia
46   * @author Misagh Moayyed
47   * @since 3.0.0
48   */
49  public final class FlowExecutionExceptionResolver implements HandlerExceptionResolver {
50  
51      /** Instance of a logger. */
52      private final Logger logger = LoggerFactory.getLogger(this.getClass());
53  
54      @NotNull
55      private String modelKey = "exception.message";
56  
57      @Override
58      public ModelAndView resolveException(final HttpServletRequest request,
59          final HttpServletResponse response, final Object handler,
60          final Exception exception) {
61  
62          /*
63           * Since FlowExecutionRepositoryException is a common ancestor to these exceptions and other
64           * error cases we would likely want to hide from the user, it seems reasonable to check for
65           * FlowExecutionRepositoryException.
66           *
67           * BadlyFormattedFlowExecutionKeyException is specifically ignored by this handler
68           * because redirecting to the requested URI with this exception may cause an infinite
69           * redirect loop (i.e. when invalid "execution" parameter exists as part of the query string
70           */
71          if (!(exception instanceof FlowExecutionRepositoryException)
72                || exception instanceof BadlyFormattedFlowExecutionKeyException) {
73              logger.debug("Ignoring the received exception due to a type mismatch", exception);
74              return null;
75          }
76  
77          final String urlToRedirectTo = request.getRequestURI()
78                  + (request.getQueryString() != null ? "?"
79                  + request.getQueryString() : "");
80  
81          logger.debug("Error getting flow information for URL [{}]", urlToRedirectTo, exception);
82          final Map<String, Object> model = new HashMap<>();
83          model.put(this.modelKey, StringEscapeUtils.escapeHtml4(exception.getMessage()));
84  
85          return new ModelAndView(new RedirectView(urlToRedirectTo), model);
86      }
87  
88      public void setModelKey(final String modelKey) {
89          this.modelKey = modelKey;
90      }
91  }
