39  /**
40   * Class to automatically set the paths for the CookieGenerators.
41   * <p>
42   * Note: This is technically not threadsafe, but because its overriding with a
43   * constant value it doesn't matter.
44   * <p>
45   * Note: As of CAS 3.1, this is a required class that retrieves and exposes the
46   * values in the two cookies for subclasses to use.
47   *
48   * @author Scott Battaglia
49   * @since 3.1
50   */
51  public final class InitialFlowSetupAction extends AbstractAction {
52      private final Logger logger = LoggerFactory.getLogger(this.getClass());
53  
54      /** The services manager with access to the registry. **/
55      @NotNull
56      private ServicesManager servicesManager;
57  
58      /** CookieGenerator for the Warnings. */
59      @NotNull
60      private CookieRetrievingCookieGenerator warnCookieGenerator;
61  
62      /** CookieGenerator for the TicketGrantingTickets. */
63      @NotNull
64      private CookieRetrievingCookieGenerator ticketGrantingTicketCookieGenerator;
65  
66      /** Extractors for finding the service. */
67      @NotNull
68      @Size(min=1)
69      private List<ArgumentExtractor> argumentExtractors;
70  
71      /** Boolean to note whether we've set the values on the generators or not. */
72      private boolean pathPopulated;
73  
74      @Override
75      protected Event doExecute(final RequestContext context) throws Exception {
76          final HttpServletRequest request = WebUtils.getHttpServletRequest(context);
77          if (!this.pathPopulated) {
78              final String contextPath = context.getExternalContext().getContextPath();
79              final String cookiePath = StringUtils.hasText(contextPath) ? contextPath + "/" : "/";
80              logger.info("Setting path for cookies to: {} ", cookiePath);
81              this.warnCookieGenerator.setCookiePath(cookiePath);
82              this.ticketGrantingTicketCookieGenerator.setCookiePath(cookiePath);
83              this.pathPopulated = true;
84          }
85  
86          WebUtils.putTicketGrantingTicketInScopes(context,
87                  this.ticketGrantingTicketCookieGenerator.retrieveCookieValue(request));
88  
89          WebUtils.putWarningCookie(context,
90                  Boolean.valueOf(this.warnCookieGenerator.retrieveCookieValue(request)));
91  
92          final Service service = WebUtils.getService(this.argumentExtractors, context);
93  
94          if (service != null) {
95              logger.debug("Placing service in context scope: [{}]", service.getId());
96  
97              final RegisteredService registeredService = this.servicesManager.findServiceBy(service);
98              if (registeredService != null && registeredService.getAccessStrategy().isServiceAccessAllowed()) {
99                  logger.debug("Placing registered service [{}] with id [{}] in context scope",
100                         registeredService.getServiceId(),
101                         registeredService.getId());
102                 WebUtils.putRegisteredService(context, registeredService);
103             }
104         }
105         WebUtils.putService(context, service);
106         return result("success");
107     }
108 
109     public void setTicketGrantingTicketCookieGenerator(
110         final CookieRetrievingCookieGenerator ticketGrantingTicketCookieGenerator) {
111         this.ticketGrantingTicketCookieGenerator = ticketGrantingTicketCookieGenerator;
112     }
113 
114     public void setWarnCookieGenerator(final CookieRetrievingCookieGenerator warnCookieGenerator) {
115         this.warnCookieGenerator = warnCookieGenerator;
116     }
117 
118     public void setArgumentExtractors(final List<ArgumentExtractor> argumentExtractors) {
119         this.argumentExtractors = argumentExtractors;
120     }
121 
122     /**
123      * Set the service manager to allow access to the registry
124      * to retrieve the registered service details associated
125      * with an incoming service.
126      * Since 4.1
127      * @param servicesManager the services manager
128      */
129     public void setServicesManager(final ServicesManager servicesManager) {
130         this.servicesManager = servicesManager;
131     }
132 }
