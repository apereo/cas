19  package org.jasig.cas.util;
20  
21  import java.util.concurrent.atomic.AtomicLong;
22  
23  /**
24   * The default numeric generator for generating long values. Implementation
25   * allows for wrapping (to restart count) if the maximum is reached.
26   *
27   * @author Scott Battaglia
28  
29   * @since 3.0.0
30   */
31  public final class DefaultLongNumericGenerator implements LongNumericGenerator {
32  
33      /** The maximum length the string can be. */
34      private static final int MAX_STRING_LENGTH = Long.toString(Long.MAX_VALUE)
35          .length();
36  
37      /** The minimum length the String can be. */
38      private static final int MIN_STRING_LENGTH = 1;
39  
40      private final AtomicLong count;
41  
42      /**
43       * Instantiates a new default long numeric generator.
44       */
45      public DefaultLongNumericGenerator() {
46          this(0);
47          // nothing to do
48      }
49  
50      /**
51       * Instantiates a new default long numeric generator.
52       *
53       * @param initialValue the initial value
54       */
55      public DefaultLongNumericGenerator(final long initialValue) {
56          this.count = new AtomicLong(initialValue);
57      }
58  
59      public long getNextLong() {
60          return this.getNextValue();
61      }
62  
63      public String getNextNumberAsString() {
64          return Long.toString(this.getNextValue());
65      }
66  
67      @Override
68      public int maxLength() {
69          return DefaultLongNumericGenerator.MAX_STRING_LENGTH;
70      }
71      
72      @Override
73      public int minLength() {
74          return DefaultLongNumericGenerator.MIN_STRING_LENGTH;
75      }
76  
77      
78      /**
79       * Gets the next value.
80       *
81       * @return the next value. If the count has reached {@link Long#MAX_VALUE}, 
82       * then {@link Long#MAX_VALUE} is returned. Otherwise, the next increment.
83       */
84      protected long getNextValue() {
85          if (this.count.compareAndSet(Long.MAX_VALUE, 0)) {
86              return Long.MAX_VALUE;
87          }
88          return this.count.getAndIncrement();
89      }
90  }
