19  package org.jasig.cas.adaptors.radius;
20  
21  /**
22   * RADIUS protocol enumeration.
23   *
24   * @author Marvin S. Addison
25   * @since 4.0.0
26   */
27  public enum RadiusProtocol {
28      
29      /** The chap. */
30      CHAP("chap"),
31      
32      /** The EA p_ m d5. */
33      EAP_MD5("eap-md5"),
34      
35      /** The EA p_ mscha pv2. */
36      EAP_MSCHAPv2("eap-mschapv2"),
37      
38      /** The eap tls. */
39      EAP_TLS("eap-tls"),
40      
41      /** The eap ttls pap. */
42      EAP_TTLS_PAP("eap-ttls:innerProtocol=pap"),
43      
44      /** The EA p_ ttl s_ ea p_ m d5. */
45      EAP_TTLS_EAP_MD5("eap-ttls:innerProtocol=eap-md5"),
46      
47      /** The EA p_ ttl s_ ea p_ mscha pv2. */
48      EAP_TTLS_EAP_MSCHAPv2("eap-ttls:innerProtocol=eap-mschapv2"),
49      
50      /** The MSCHA pv1. */
51      MSCHAPv1("mschapv1"),
52      
53      /** The MSCHA pv2. */
54      MSCHAPv2("mschapv2"),
55      
56      /** The pap. */
57      PAP("pap"),
58      
59      /** The peap. */
60      PEAP("peap");
61  
62      /** The name. */
63      private final String name;
64  
65      /**
66       * Instantiates a new radius protocol.
67       *
68       * @param name the name
69       */
70      RadiusProtocol(final String name) {
71          this.name = name;
72      }
73  
74      /**
75       * Gets the radius protocol name required by {@link net.jradius.client.RadiusClient#getAuthProtocol(String)}.
76       *
77       * @return RADIUS protocol name known to {@link net.jradius.client.RadiusClient}.
78       */
79      public String getName() {
80          return this.name;
81      }
82  }
