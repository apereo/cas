19  package org.jasig.cas.adaptors.generic.remote;
20  
21  import org.jasig.cas.authentication.AbstractAuthenticationHandler;
22  import org.jasig.cas.authentication.Credential;
23  import org.jasig.cas.authentication.HandlerResult;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import javax.security.auth.login.FailedLoginException;
28  import javax.validation.constraints.NotNull;
29  import java.net.InetAddress;
30  import java.net.UnknownHostException;
31  import java.security.GeneralSecurityException;
32  
33  /**
34   * Checks if the remote address is in the range of allowed addresses.
35   *
36   * @author David Harrison
37   * @author Scott Battaglia
38   * @since 3.2.1
39   *
40   */
41  public final class RemoteAddressAuthenticationHandler extends AbstractAuthenticationHandler {
42  
43      private static final int HEX_RIGHT_SHIFT_COEFFICIENT = 0xff;
44      private final Logger logger = LoggerFactory.getLogger(getClass());
45  
46      /** The network netmask. */
47      @NotNull
48      private InetAddress inetNetmask;
49  
50      /** The network base address. */
51      @NotNull
52      private InetAddress inetNetwork;
53  
54      @Override
55      public HandlerResult authenticate(final Credential credential) throws GeneralSecurityException {
56          final RemoteAddressCredential c = (RemoteAddressCredential) credential;
57          try {
58              final InetAddress inetAddress = InetAddress.getByName(c.getRemoteAddress().trim());
59              if (containsAddress(this.inetNetwork, this.inetNetmask, inetAddress)) {
60                  return new HandlerResult(this, c, this.principalFactory.createPrincipal(c.getId()));
61              }
62          } catch (final UnknownHostException e) {
63              logger.debug("Unknown host {}", c.getRemoteAddress());
64          }
65          throw new FailedLoginException(c.getRemoteAddress() + " not in allowed range.");
66      }
67  
68      @Override
69      public boolean supports(final Credential credential) {
70          return credential instanceof RemoteAddressCredential;
71      }
72  
73      /**
74       * The following code is from the Apache Software Foundations's Lenya project
75       * See InetAddressUtil.java
76       * Distributed under the Apache 2.0 software license
77       */
78  
79      /**
80       * Checks if a subnet contains a specific IP address.
81       *
82       * @param network The network address.
83       * @param netmask The subnet mask.
84       * @param ip The IP address to check.
85       * @return A boolean value.
86       */
87      private boolean containsAddress(final InetAddress network, final InetAddress netmask, final InetAddress ip) {
88          logger.debug("Checking IP address: {} in {} by {}", ip, network, netmask);
89  
90          final byte[] networkBytes = network.getAddress();
91          final byte[] netmaskBytes = netmask.getAddress();
92          final byte[] ipBytes = ip.getAddress();
93  
94          /* check IPv4/v6-compatibility or parameters: */
95          if(networkBytes.length != netmaskBytes.length
96                  || netmaskBytes.length != ipBytes.length) {
97              logger.debug("Network address {}, subnet mask {} and/or host address {}"
98                      + " have different sizes! (return false ...)", network, netmask, ip);
99              return false;
100         }
101 
102         /* Check if the masked network and ip addresses match: */
103         for(int i=0; i < netmaskBytes.length; i++) {
104             final int mask = netmaskBytes[i] & HEX_RIGHT_SHIFT_COEFFICIENT;
105             if((networkBytes[i] & mask) != (ipBytes[i] & mask)) {
106                 logger.debug("{} is not in {}/{}", ip, network, netmask);
107                 return false;
108             }
109         }
110         logger.debug("{} is in {}/{}", ip, network, netmask);
111         return true;
112     }
113 
114     /**
115      * @param ipAddressRange the IP address range that should be allowed trusted logins     *
116      */
117     public void setIpNetworkRange(final String ipAddressRange) {
118 
119         if(ipAddressRange != null) {
120 
121             final String[] splitAddress = ipAddressRange.split("/");
122 
123             if (splitAddress.length == 2) {
124                 // A valid ip address/netmask was supplied parse values
125                 final String network = splitAddress[0].trim();
126                 final String netmask = splitAddress[1].trim();
127 
128                 try {
129                     this.inetNetwork = InetAddress.getByName(network);
130                     logger.debug("InetAddress network: {}", this.inetNetwork.toString());
131                 } catch (final UnknownHostException e) {
132                     logger.error("The network address was not valid: {}", e.getMessage());
133                 }
134 
135                 try {
136                     this.inetNetmask = InetAddress.getByName(netmask);
137                     logger.debug("InetAddress netmask: {}", this.inetNetmask.toString());
138                 } catch (final UnknownHostException e) {
139                     logger.error("The network netmask was not valid: {}", e.getMessage());
140                 }
141             }
142         }
143     }
144 }
