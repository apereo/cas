41  /**
42   * Provides a simple {@link UserDetailsService} implementation that obtains user details from an LDAP search.
43   * Two searches are performed by this component for every user details lookup:
44   *
45   * <ol>
46   *     <li>Search for an entry to resolve the username. In most cases the search should return exactly one result,
47   *     but the {@link #setAllowMultipleResults(boolean)} property may be toggled to change that behavior.</li>
48   *     <li>Search for groups of which the user is a member. This search commonly occurs on a separate directory
49   *     branch than that of the user search.</li>
50   * </ol>
51   *
52   * @author Marvin S. Addison
53   * @author Misagh Moayyed
54   * @since 4.0.0
55   */
56  public class LdapUserDetailsService implements UserDetailsService {
57  
58      /** Default role prefix. */
59      public static final String DEFAULT_ROLE_PREFIX = "ROLE_";
60  
61      /** Placeholder for unknown password given to user details. */
62      public static final String UNKNOWN_PASSWORD = "<UNKNOWN>";
63  
64      /** Logger instance. */
65      private final Logger logger = LoggerFactory.getLogger(getClass());
66  
67      /** Source of LDAP connections. */
68      @NotNull
69      private final ConnectionFactory connectionFactory;
70  
71      /** Executes the search query for user data. */
72      @NotNull
73      private final SearchExecutor userSearchExecutor;
74  
75      /** Executes the search query for roles. */
76      @NotNull
77      private final SearchExecutor roleSearchExecutor;
78  
79      /** Specify the name of LDAP attribute to use as principal identifier. */
80      @NotNull
81      private final String userAttributeName;
82  
83      /** Specify the name of LDAP attribute to be used as the basis for role granted authorities. */
84      @NotNull
85      private final String roleAttributeName;
86  
87      /** Prefix appended to the uppercased
88       * {@link #roleAttributeName} per the normal Spring Security convention.
89       **/
90      @NotNull
91      private String rolePrefix = DEFAULT_ROLE_PREFIX;
92  
93      /** Flag that indicates whether multiple search results are allowed for a given credential. */
94      private boolean allowMultipleResults;
95  
96      /**
97       * Creates a new instance with the given required parameters.
98       *
99       * @param  factory  Source of LDAP connections for searches.
100      * @param  userSearchExecutor  Executes the LDAP search for user data.
101      * @param roleSearchExecutor  Executes the LDAP search for role data.
102      * @param userAttributeName  Name of LDAP attribute that contains username for user details.
103      * @param roleAttributeName  Name of LDAP attribute that contains role membership data for the user.
104      */
105     public LdapUserDetailsService(
106             final ConnectionFactory factory,
107             final SearchExecutor userSearchExecutor,
108             final SearchExecutor roleSearchExecutor,
109             final String userAttributeName,
110             final String roleAttributeName) {
111 
112         this.connectionFactory = factory;
113         this.userSearchExecutor = userSearchExecutor;
114         this.roleSearchExecutor = roleSearchExecutor;
115         this.userAttributeName = userAttributeName;
116         this.roleAttributeName = roleAttributeName;
117     }
118 
119 
120     /**
121      * Sets the prefix appended to the uppercase {@link #roleAttributeName} per the normal Spring Security convention.
122      * The default value {@value #DEFAULT_ROLE_PREFIX} is sufficient in most cases.
123      *
124      * @param  rolePrefix  Role prefix.
125      */
126     public void setRolePrefix(final String rolePrefix) {
127         this.rolePrefix = rolePrefix;
128     }
129 
130 
131     /**
132      * Sets whether to allow multiple search results for user details given a username.
133      * This is false by default, which is sufficient and secure for more deployments.
134      * Setting this to true may have security consequences.
135      *
136      * @param  allowMultiple  True to allow multiple search results in which case the first result
137      *                        returned is used to construct user details, or false to indicate that
138      *                        a runtime exception should be raised on multiple search results for user details.
139      */
140     public void setAllowMultipleResults(final boolean allowMultiple) {
141         this.allowMultipleResults = allowMultiple;
142     }
143 
144     @Override
145     public UserDetails loadUserByUsername(final String username) {
146         final SearchResult userResult;
147         try {
148             logger.debug("Attempting to get details for user {}.", username);
149             final Response<SearchResult> response = this.userSearchExecutor.search(
150                     this.connectionFactory,
151                     createSearchFilter(this.userSearchExecutor, username));
152             logger.debug("LDAP user search response: {}", response);
153             userResult = response.getResult();
154         } catch (final LdapException e) {
155             throw new RuntimeException("LDAP error fetching details for user.", e);
156         }
157         if (userResult.size() == 0) {
158             throw new UsernameNotFoundException(username + " not found.");
159         }
160         if (userResult.size() > 1 && !this.allowMultipleResults) {
161             throw new IllegalStateException(
162                     "Found multiple results for user which is not allowed (allowMultipleResults=false).");
163         }
164         final String userDn = userResult.getEntry().getDn();
165         final LdapAttribute userAttribute = userResult.getEntry().getAttribute(this.userAttributeName);
166         if (userAttribute == null) {
167             throw new IllegalStateException(this.userAttributeName + " attribute not found in results.");
168         }
169         final String id = userAttribute.getStringValue();
170 
171         final SearchResult roleResult;
172         try {
173             logger.debug("Attempting to get roles for user {}.", userDn);
174             final Response<SearchResult> response = this.roleSearchExecutor.search(
175                     this.connectionFactory,
176                     createSearchFilter(this.roleSearchExecutor, userDn));
177             logger.debug("LDAP role search response: {}", response);
178             roleResult = response.getResult();
179         } catch (final LdapException e) {
180             throw new RuntimeException("LDAP error fetching roles for user.", e);
181         }
182         LdapAttribute roleAttribute;
183         final Collection<SimpleGrantedAuthority> roles = new ArrayList<>(roleResult.size());
184         for (final LdapEntry entry : roleResult.getEntries()) {
185             roleAttribute = entry.getAttribute(this.roleAttributeName);
186             if (roleAttribute == null) {
187                 logger.warn("Role attribute not found on entry {}", entry);
188                 continue;
189             }
190 
191             for (final String role : roleAttribute.getStringValues()) {
192                 roles.add(new SimpleGrantedAuthority(this.rolePrefix + role.toUpperCase()));
193             }
194 
195         }
196 
197         return new User(id, UNKNOWN_PASSWORD, roles);
198     }
199 
200     /**
201      * Constructs a new search filter using {@link SearchExecutor#searchFilter} as a template and
202      * the username as a parameter.
203      *
204      * @param executor the executor
205      * @param username the username
206      * @return  Search filter with parameters applied.
207      */
208     private SearchFilter createSearchFilter(final SearchExecutor executor, final String username) {
209         final SearchFilter filter = new SearchFilter();
210         filter.setFilter(executor.getSearchFilter().getFilter());
211         filter.setParameter(0, username);
212 
213         logger.debug("Constructed LDAP search filter [{}]", filter.format());
214         return filter;
215     }
216 }
