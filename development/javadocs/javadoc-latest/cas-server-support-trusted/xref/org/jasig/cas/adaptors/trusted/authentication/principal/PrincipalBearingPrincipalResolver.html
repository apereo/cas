19  package org.jasig.cas.adaptors.trusted.authentication.principal;
20  
21  import org.jasig.cas.authentication.principal.PersonDirectoryPrincipalResolver;
22  import org.jasig.cas.authentication.Credential;
23  
24  /**
25   * Extracts the Principal out of PrincipalBearingCredential. It is very simple
26   * to resolve PrincipalBearingCredential to a Principal since the credentials
27   * already bear the ready-to-go Principal.
28   *
29   * @author Andrew Petro
30   * @since 3.0.0.5
31   */
32  public final class PrincipalBearingPrincipalResolver extends PersonDirectoryPrincipalResolver {
33  
34      @Override
35      protected String extractPrincipalId(final Credential credential) {
36          return ((PrincipalBearingCredential) credential).getPrincipal().getId();
37      }
38  
39      @Override
40      public boolean supports(final Credential credential) {
41          return credential instanceof PrincipalBearingCredential;
42      }
43  }
