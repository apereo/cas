19  package org.jasig.cas.adaptors.generic;
20  
21  import org.apache.commons.lang3.StringUtils;
22  import org.jasig.cas.authentication.HandlerResult;
23  import org.jasig.cas.authentication.PreventedException;
24  import org.jasig.cas.authentication.UsernamePasswordCredential;
25  import org.jasig.cas.authentication.handler.support.AbstractUsernamePasswordAuthenticationHandler;
26  import org.springframework.core.io.Resource;
27  import javax.security.auth.login.AccountNotFoundException;
28  import javax.security.auth.login.FailedLoginException;
29  import javax.validation.constraints.NotNull;
30  import java.io.BufferedReader;
31  import java.io.IOException;
32  import java.io.InputStreamReader;
33  import java.nio.charset.Charset;
34  import java.security.GeneralSecurityException;
35  
36  /**
37   * Class designed to read data from a file in the format of USERNAME SEPARATOR
38   * PASSWORD that will go line by line and look for the username. If it finds the
39   * username it will compare the supplied password (first put through a
40   * PasswordTranslator) that is compared to the password provided in the file. If
41   * there is a match, the user is authenticated. Note that the default password
42   * translator is a plaintext password translator and the default separator is
43   * "::" (without quotes).
44   *
45   * @author Scott Battaglia
46   * @author Marvin S. Addison
47   * @since 3.0.0
48   */
49  public class FileAuthenticationHandler extends AbstractUsernamePasswordAuthenticationHandler {
50  
51      /** The default separator in the file. */
52      private static final String DEFAULT_SEPARATOR = "::";
53  
54      /** The separator to use. */
55      @NotNull
56      private String separator = DEFAULT_SEPARATOR;
57  
58      /** The filename to read the list of usernames from. */
59      @NotNull
60      private Resource fileName;
61  
62      /**
63       * {@inheritDoc}
64       */
65      @Override
66      protected final HandlerResult authenticateUsernamePasswordInternal(final UsernamePasswordCredential credential)
67              throws GeneralSecurityException, PreventedException {
68          try {
69              final String username = credential.getUsername();
70              final String passwordOnRecord = getPasswordOnRecord(username);
71              if (StringUtils.isBlank(passwordOnRecord)) {
72                  throw new AccountNotFoundException(username + " not found in backing file.");
73              }
74              final String password = credential.getPassword();
75              if (StringUtils.isNotBlank(password) && this.getPasswordEncoder().encode(password).equals(passwordOnRecord)) {
76                  return createHandlerResult(credential, this.principalFactory.createPrincipal(username), null);
77              }
78          } catch (final IOException e) {
79              throw new PreventedException("IO error reading backing file", e);
80          }
81          throw new FailedLoginException();
82      }
83  
84      /**
85       * @param fileName The fileName to set.
86       */
87      public final void setFileName(final Resource fileName) {
88          this.fileName = fileName;
89      }
90  
91      /**
92       * @param separator The separator to set.
93       */
94      public final void setSeparator(final String separator) {
95          this.separator = separator;
96      }
97  
98      /**
99       * Gets the password on record.
100      *
101      * @param username the username
102      * @return the password on record
103      * @throws IOException Signals that an I/O exception has occurred.
104      */
105     private String getPasswordOnRecord(final String username) throws IOException {
106         try (BufferedReader bufferedReader =
107                      new BufferedReader(
108                      new InputStreamReader(this.fileName.getInputStream(), Charset.defaultCharset()))) {
109             String line = bufferedReader.readLine();
110             while (line != null) {
111                 final String[] lineFields = line.split(this.separator);
112                 final String userOnRecord = lineFields[0];
113                 final String passOnRecord = lineFields[1];
114                 if (username.equals(userOnRecord)) {
115                     return passOnRecord;
116                 }
117                 line = bufferedReader.readLine();
118             }
119         }
120         return null;
121     }
122 }
