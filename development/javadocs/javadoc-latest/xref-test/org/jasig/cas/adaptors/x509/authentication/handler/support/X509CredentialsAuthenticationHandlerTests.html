241     /**
242      * Tests the {@link X509CredentialsAuthenticationHandler#authenticate(org.jasig.cas.authentication.Credential)} method.
243      */
244     @Test
245     public void verifyAuthenticate() {
246         try {
247             if (this.handler.supports(this.credential)) {
248                 final HandlerResult result = this.handler.authenticate(this.credential);
249                 if (this.expectedResult instanceof HandlerResult) {
250                     assertEquals(this.expectedResult, result);
251                 } else {
252                     fail("Authentication succeeded when it should have failed with " + this.expectedResult);
253                 }
254             }
255         } catch (final Exception e) {
256             if (this.expectedResult instanceof Exception) {
257                 assertEquals(this.expectedResult.getClass(), e.getClass());
258             } else {
259                 fail("Authentication failed when it should have succeeded.");
260             }
261         }
262     }

265      * Tests the {@link X509CredentialsAuthenticationHandler#supports(org.jasig.cas.authentication.Credential)} method.
266      */
267     @Test
268     public void verifySupports() {
269         assertEquals(this.expectedSupports, this.handler.supports(this.credential));
270     }

272     protected static X509Certificate[] createCertificates(final String ... files) {
273         final X509Certificate[] certs = new X509Certificate[files.length];
274 
275         int i = 0;
276         for (String file : files) {
277             try {
278                 certs[i++] = (X509Certificate) CryptReader.readCertificate(
279                         new ClassPathResource(file).getInputStream());
280             } catch (final Exception e) {
281                 throw new RuntimeException("Error creating certificate at " + file, e);
282             }
283         }
284         return certs;
285     }
