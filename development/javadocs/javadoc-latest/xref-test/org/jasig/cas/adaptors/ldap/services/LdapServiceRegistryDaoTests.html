46  /**
47   * Unit test for {@link LdapServiceRegistryDao} class.
48   * @author Misagh Moayyed
49   * @author Marvin S. Addison
50   * @since 4.0.0
51   */
52  @RunWith(SpringJUnit4ClassRunner.class)
53  @ContextConfiguration({"/ldap-context.xml", "/ldap-regservice-test.xml"})
54  public class LdapServiceRegistryDaoTests extends AbstractLdapTests {
55  
56      @Autowired
57      private ServiceRegistryDao dao;
58  
59      @Before
60      public void setUp() throws Exception {
61          for (final RegisteredService service : this.dao.load()) {
62              this.dao.delete(service);
63          }
64      }
65  
66      @Test
67      public void verifyEmptyRegistry() {
68          assertEquals(0, this.dao.load().size());
69      }
70  
71      @Test
72      public void verifyNonExistingService() {
73          assertNull(this.dao.findServiceById(9999991));
74      }
75  
76      @Test
77      public void verifySavingServices() {
78          this.dao.save(getRegisteredService());
79          this.dao.save(getRegexRegisteredService());
80          final List<RegisteredService> services = this.dao.load();
81          assertEquals(2, services.size());
82      }
83  
84      @Test
85      public void verifyUpdatingServices() {
86          this.dao.save(getRegisteredService());
87          final List<RegisteredService> services = this.dao.load();
88  
89          final AbstractRegisteredService rs = (AbstractRegisteredService) this.dao.findServiceById(services.get(0).getId());
90          assertNotNull(rs);
91          rs.setEvaluationOrder(9999);
92          rs.setUsernameAttributeProvider(new DefaultRegisteredServiceUsernameProvider());
93          rs.setName("Another Test Service");
94          rs.setDescription("The new description");
95          rs.setServiceId("https://hello.world");
96          rs.setProxyPolicy(new RegexMatchingRegisteredServiceProxyPolicy("https"));
97          rs.setAttributeReleasePolicy(new ReturnAllowedAttributeReleasePolicy());
98          assertNotNull(this.dao.save(rs));
99  
100         final RegisteredService rs3 = this.dao.findServiceById(rs.getId());
101         assertEquals(rs3.getName(), rs.getName());
102         assertEquals(rs3.getDescription(), rs.getDescription());
103         assertEquals(rs3.getEvaluationOrder(), rs.getEvaluationOrder());
104         assertEquals(rs3.getUsernameAttributeProvider(), rs.getUsernameAttributeProvider());
105         assertEquals(rs3.getProxyPolicy(), rs.getProxyPolicy());
106         assertEquals(rs3.getUsernameAttributeProvider(), rs.getUsernameAttributeProvider());
107         assertEquals(rs3.getServiceId(), rs.getServiceId());
108     }
109 
110     @Test
111     public void verifySavingServiceChangesDn() {
112         this.dao.save(getRegisteredService());
113         final List<RegisteredService> services = this.dao.load();
114 
115         final AbstractRegisteredService rs = (AbstractRegisteredService) this.dao.findServiceById(services.get(0).getId());
116         final long originalId = rs.getId();
117         assertNotNull(rs);
118         rs.setId(666);
119         assertNotNull(this.dao.save(rs));
120         assertNotEquals(rs.getId(), originalId);
121     }
122 
123     @Test
124     public void verifyDeletingSingleService() throws Exception {
125         final RegisteredService rs = getRegexRegisteredService();
126         final RegisteredService rs2 = getRegisteredService();
127         this.dao.save(rs2);
128         this.dao.save(rs);
129         List<RegisteredService> services = this.dao.load();
130         this.dao.delete(rs2);
131 
132         services = this.dao.load();
133         assertEquals(1, services.size());
134         assertEquals(services.get(0).getId(), rs.getId());
135         assertEquals(services.get(0).getName(), rs.getName());
136     }
137 
138     @Test
139     public void verifyDeletingServices() throws Exception {
140         this.dao.save(getRegisteredService());
141         this.dao.save(getRegexRegisteredService());
142         final List<RegisteredService> services = this.dao.load();
143         for (final RegisteredService registeredService : services) {
144             this.dao.delete(registeredService);
145         }
146         assertEquals(0, this.dao.load().size());
147     }
148 
149     private RegisteredService getRegisteredService() {
150         final AbstractRegisteredService rs = new RegisteredServiceImpl();
151         rs.setName("Service Name1");
152         rs.setProxyPolicy(new RefuseRegisteredServiceProxyPolicy());
153         rs.setUsernameAttributeProvider(new AnonymousRegisteredServiceUsernameAttributeProvider());
154         rs.setDescription("Service description");
155         rs.setServiceId("https://?.edu/**");
156         rs.setTheme("the theme name");
157         rs.setEvaluationOrder(123);
158         rs.setAttributeReleasePolicy(new ReturnAllAttributeReleasePolicy());
159         rs.setRequiredHandlers(new HashSet<String>(Arrays.asList("handler8", "handle92")));
160         return rs;
161     }
162 
163     private RegisteredService getRegexRegisteredService() {
164         final AbstractRegisteredService rs  = new RegexRegisteredService();
165         rs.setName("Service Name Regex");
166         rs.setProxyPolicy(new RefuseRegisteredServiceProxyPolicy());
167         rs.setUsernameAttributeProvider(new AnonymousRegisteredServiceUsernameAttributeProvider());
168         rs.setDescription("Service description");
169         rs.setServiceId("^http?://.+");
170         rs.setTheme("the theme name");
171         rs.setEvaluationOrder(123);
172         rs.setDescription("Here is another description");
173         rs.setRequiredHandlers(new HashSet<String>(Arrays.asList("handler1", "handler2")));
174         return rs;
175     }
