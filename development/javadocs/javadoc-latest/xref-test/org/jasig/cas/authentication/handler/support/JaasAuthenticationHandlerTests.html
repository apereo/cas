19  package org.jasig.cas.authentication.handler.support;
20  
21  import javax.security.auth.login.LoginException;
22  
23  import org.jasig.cas.TestUtils;
24  import org.junit.Before;
25  import org.junit.Test;
26  
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  import static org.junit.Assert.*;
31  
32  /**
33   * @author Marvin S. Addison
34   * @since 3.0.0
35   */
36  public class JaasAuthenticationHandlerTests {
37  
38      private final Logger logger = LoggerFactory.getLogger(this.getClass());
39  
40      private JaasAuthenticationHandler handler;
41  
42      @Before
43      public void setUp() throws Exception {
44          String pathPrefix = System.getProperty("user.dir");
45          pathPrefix = !pathPrefix.contains("cas-server-core") ? pathPrefix
46              + "/cas-server-core" : pathPrefix;
47          logger.info("PATH PREFIX: {}", pathPrefix);
48  
49          final String pathToConfig = pathPrefix
50              + "/src/test/resources/org/jasig/cas/authentication/handler/support/jaas.conf";
51          System.setProperty("java.security.auth.login.config", "="+pathToConfig);
52          this.handler = new JaasAuthenticationHandler();
53      }
54  
55      @Test(expected = LoginException.class)
56      public void verifyWithAlternativeRealm() throws Exception {
57  
58          this.handler.setRealm("TEST");
59          this.handler.authenticate(TestUtils.getCredentialsWithDifferentUsernameAndPassword("test", "test1"));
60      }
61  
62      @Test
63      public void verifyWithAlternativeRealmAndValidCredentials() throws Exception {
64          this.handler.setRealm("TEST");
65          assertNotNull(this.handler.authenticate(
66                  TestUtils.getCredentialsWithDifferentUsernameAndPassword("test", "test")));
67      }
68  
69      @Test
70      public void verifyWithValidCredenials() throws Exception {
71          assertNotNull(this.handler.authenticate(TestUtils.getCredentialsWithSameUsernameAndPassword()));
72      }
73  
74      @Test(expected = LoginException.class)
75      public void verifyWithInvalidCredentials() throws Exception {
76          this.handler.authenticate(TestUtils.getCredentialsWithDifferentUsernameAndPassword("test", "test1"));
77      }
78  
79  }
