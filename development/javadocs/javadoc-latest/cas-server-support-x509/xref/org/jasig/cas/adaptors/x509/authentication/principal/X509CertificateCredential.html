19  package org.jasig.cas.adaptors.x509.authentication.principal;
20  
21  import com.google.common.collect.ImmutableList;
22  import org.jasig.cas.adaptors.x509.util.CertUtils;
23  import org.jasig.cas.authentication.AbstractCredential;
24  
25  import javax.validation.constraints.NotNull;
26  import java.security.cert.X509Certificate;
27  
28  /**
29   * An X.509 certificate credential.
30   *
31   * @author Scott Battaglia
32   * @author Marvin S. Addison
33   * @since 3.0.0.4
34   *
35   */
36  public final class X509CertificateCredential extends AbstractCredential {
37  
38      /** Unique Id for serialization. */
39      private static final long serialVersionUID = 631753409512746474L;
40  
41      /** The collection of certificates sent with the request. */
42      private final X509Certificate[] certificates;
43  
44      /** The certificate that we actually use. */
45      private X509Certificate certificate;
46  
47      /**
48       * Instantiates a new x509 certificate credential.
49       *
50       * @param certificates the certificates
51       */
52      public X509CertificateCredential(@NotNull final X509Certificate[] certificates) {
53          this.certificates = ImmutableList.copyOf(certificates).toArray(new X509Certificate[certificates.length]);
54      }
55  
56      public X509Certificate[] getCertificates() {
57          return ImmutableList.copyOf(this.certificates).toArray(new X509Certificate[this.certificates.length]);
58      }
59  
60      public void setCertificate(final X509Certificate certificate) {
61          this.certificate = certificate;
62      }
63  
64      public X509Certificate getCertificate() {
65          return this.certificate;
66      }
67  
68      @Override
69      public String getId() {
70          X509Certificate cert = null;
71          if (this.certificate != null) {
72              cert = this.certificate;
73          } else if (this.certificates.length > 0) {
74              cert = this.certificates[0];
75          }
76  
77          if (cert != null) {
78              return CertUtils.toString(cert);
79          }
80          return UNKNOWN_ID;
81      }
82  }
