19  package org.jasig.cas.adaptors.x509.authentication.handler.support;
20  
21  import org.joda.time.DateTime;
22  
23  import java.security.GeneralSecurityException;
24  import java.util.Date;
25  
26  /**
27   * Exception describing an expired CRL condition.
28   *
29   * @author Marvin S. Addison
30   * @since 3.4.6
31   *
32   */
33  public class ExpiredCRLException extends GeneralSecurityException {
34      /** Serialization version marker. */
35      private static final long serialVersionUID = 5157864033250359972L;
36  
37      /** Identifier/name of CRL. */
38      private final String id;
39  
40      /** CRL expiration date. */
41      private final DateTime expirationDate;
42  
43      /** Leniency of expiration. */
44      private final int leniency;
45  
46      /**
47       * Creates a new instance describing a CRL that expired on the given date.
48       *
49       * @param identifier Identifier or name that describes CRL.
50       * @param expirationDate CRL expiration date.
51       */
52      public ExpiredCRLException(final String identifier, final Date expirationDate) {
53          this(identifier, expirationDate, 0);
54      }
55  
56      /**
57       * Creates a new instance describing a CRL that expired on a date that is
58       * more than leniency seconds beyond its expiration date.
59       *
60       * @param identifier Identifier or name that describes CRL.
61       * @param expirationDate CRL expiration date.
62       * @param leniency Number of seconds beyond the expiration date at which
63       * the CRL is considered expired.  MUST be non-negative integer.
64       */
65      public ExpiredCRLException(final String identifier, final Date expirationDate, final int leniency) {
66          this.id = identifier;
67          this.expirationDate = new DateTime(expirationDate);
68          if (leniency < 0) {
69              throw new IllegalArgumentException("Leniency cannot be negative.");
70          }
71          this.leniency = leniency;
72      }
73  
74      /**
75       * @return Returns the id.
76       */
77      public String getId() {
78          return this.id;
79      }
80  
81      /**
82       * @return Returns the expirationDate.
83       */
84      public DateTime getExpirationDate() {
85          return this.expirationDate == null ? null : new DateTime(this.expirationDate);
86      }
87  
88      /**
89       * @return Returns the leniency.
90       */
91      public int getLeniency() {
92          return this.leniency;
93      }
94  
95      /**
96       * {@inheritDoc}
97       */
98      @Override
99      public String getMessage() {
100         if (this.leniency > 0) {
101             return String.format("CRL %s expired on %s and is beyond the leniency period of %s seconds.",
102                     this.id, this.expirationDate, this.leniency);
103         }
104         return String.format("CRL %s expired on %s", this.id, this.expirationDate);
105     }
106 }
