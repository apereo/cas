250     @Test
251     public void verifySaveAttributeReleasePolicyAllowedAttrRulesWithCaching() {
252         final RegisteredServiceImpl r = new RegisteredServiceImpl();
253         r.setName("testSaveAttributeReleasePolicyAllowedAttrRulesWithCaching");
254         r.setServiceId("testId");
255 
256         final ReturnAllowedAttributeReleasePolicy policy = new ReturnAllowedAttributeReleasePolicy();
257         policy.setAllowedAttributes(Arrays.asList("1", "2", "3"));
258 
259         final Map<String, List<Object>> attributes = new HashMap<String, List<Object>>();
260         attributes.put("values", Arrays.asList(new Object[]{"v1", "v2", "v3"}));
261 
262         final CachingPrincipalAttributesRepository repository =
263                 new CachingPrincipalAttributesRepository(
264                         new StubPersonAttributeDao(attributes),
265                         TimeUnit.MILLISECONDS, 100);
266         repository.setMergingStrategy(new ReplacingAttributeAdder());
267 
268         policy.setPrincipalAttributesRepository(repository);
269         r.setAttributeReleasePolicy(policy);
270 
271         final RegisteredService r2 = this.dao.save(r);
272         final RegisteredService r3 = this.dao.findServiceById(r2.getId());
273 
274         assertEquals(r, r2);
275         assertEquals(r2, r3);
276         assertNotNull(r3.getAttributeReleasePolicy());
277         assertEquals(r2.getAttributeReleasePolicy(), r3.getAttributeReleasePolicy());
278     }

280     @Test
281     public void verifyServiceRemovals() {
282         final List<RegisteredService> list = new ArrayList<>(5);
283         for (int i = 1; i < 5; i++) {
284             final RegexRegisteredService r = new RegexRegisteredService();
285             r.setServiceId("^https://.+");
286             r.setName("testServiceType");
287             r.setEnabled(true);
288             r.setTheme("testtheme");
289             r.setEvaluationOrder(1000);
290             r.setId(i * 100);
291             list.add(this.dao.save(r));
292         }
293 
294         for (final RegisteredService r2 : list) {
295             this.dao.delete(r2);
296             assertNull(this.dao.findServiceById(r2.getId()));
297         }
298 
299     }
