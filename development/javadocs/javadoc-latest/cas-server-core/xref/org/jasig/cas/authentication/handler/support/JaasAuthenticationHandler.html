38  /**
39   * JAAS Authentication Handler for CAAS. This is a simple bridge from CAS'
40   * authentication to JAAS.
41   *
42   * <p>
43   * Using the JAAS Authentication Handler requires you to configure the
44   * appropriate JAAS modules. You can specify the location of a jass.conf file
45   * using the following VM parameter:
46   * <pre>
47   * -Djava.security.auth.login.config=$PATH_TO_JAAS_CONF/jaas.conf
48   * </pre>
49   *
50   * <p>
51   * This example jaas.conf would try Kerberos based authentication, then try LDAP
52   * authentication:
53   * <pre>
54   * CAS {
55   *   com.sun.security.auth.module.Krb5LoginModule sufficient
56   *     client=TRUE
57   *     debug=FALSE
58   *     useTicketCache=FALSE;
59   *   edu.uconn.netid.jaas.LDAPLoginModule sufficient
60   *     java.naming.provider.url="ldap://ldapserver.my.edu:389/dc=my,dc=edu"
61   *     java.naming.security.principal="uid=jaasauth,dc=my,dc=edu"
62   *     java.naming.security.credentials="password"
63   *     Attribute="uid"
64   *     startTLS="true";
65   * };
66   * </pre>
67   *
68   * @author <a href="mailto:dotmatt@uconn.edu">Matthew J. Smith</a>
69   * @author Marvin S. Addison
70   * @author Misagh Moayyed
71   *
72   * @see javax.security.auth.callback.CallbackHandler
73   * @see javax.security.auth.callback.PasswordCallback
74   * @see javax.security.auth.callback.NameCallback
75   * @since 3.0.0.5
76   */
77  public class JaasAuthenticationHandler extends AbstractUsernamePasswordAuthenticationHandler {
78  
79      /** If no realm is specified, we default to CAS. */
80      private static final String DEFAULT_REALM = "CAS";
81  
82      /**
83       * System property key to specify kerb5 realm.
84       */
85      private static final String SYS_PROP_KRB5_REALM = "java.security.krb5.realm";
86      
87      /**
88       * System property key to specify kerb5 kdc.
89       */
90      private static final String SYS_PROP_KERB5_KDC = "java.security.krb5.kdc";
91      
92      /** The realm that contains the login module information. */
93      @NotNull
94      private String realm = DEFAULT_REALM;
95  
96      /** System property value to overwrite the realm in krb5 config. */
97      private String kerberosRealmSystemProperty;
98      
99      /** System property value to overwrite the kdc in krb5 config. */
100     private String kerberosKdcSystemProperty;
101     
102     /**
103      * Instantiates a new Jaas authentication handler,
104      * and attempts to load/verify the configuration.
105      */
106     public JaasAuthenticationHandler() {
107         Assert.notNull(Configuration.getConfiguration(),
108                 "Static Configuration cannot be null. Did you remember to specify \"java.security.auth.login.config\"?");
109     }
110 
111     /**
112      * {@inheritDoc}
113      */
114     @Override
115     protected final HandlerResult authenticateUsernamePasswordInternal(final UsernamePasswordCredential credential)
116             throws GeneralSecurityException, PreventedException {
117 
118         if (this.kerberosKdcSystemProperty != null) {
119             logger.debug("Setting kerberos system property {} to {}", SYS_PROP_KERB5_KDC, this.kerberosKdcSystemProperty);
120             System.setProperty(SYS_PROP_KERB5_KDC, this.kerberosKdcSystemProperty);
121         }
122         if (this.kerberosRealmSystemProperty != null) {
123             logger.debug("Setting kerberos system property {} to {}", SYS_PROP_KRB5_REALM, this.kerberosRealmSystemProperty);
124             System.setProperty(SYS_PROP_KRB5_REALM, this.kerberosRealmSystemProperty);
125         }
126         
127         final String username = credential.getUsername();
128         final String password = getPasswordEncoder().encode(credential.getPassword());
129         final LoginContext lc = new LoginContext(
130                 this.realm,
131                 new UsernamePasswordCallbackHandler(username, password));
132         try {
133             logger.debug("Attempting authentication for: {}", username);
134             lc.login();
135         } finally {
136             lc.logout();
137         }
138 
139         Principal principal = null;
140         final Set<java.security.Principal> principals = lc.getSubject().getPrincipals();
141         if (principals != null && principals.size() > 0) {
142             principal = this.principalFactory.createPrincipal(principals.iterator().next().getName());
143         }
144         return createHandlerResult(credential, principal, null);
145     }
146 
147     public void setRealm(final String realm) {
148         this.realm = realm;
149     }

151     /**
152      * Typically, the default realm and the KDC for that realm are indicated in the Kerberos <code>krb5.conf</code> configuration file.
153      * However, if you like, you can instead specify the realm value by setting this following system property value.
154      * <p>If you set the realm property, you SHOULD also configure the {@link #setKerberosKdcSystemProperty(String)}.
155      * <p>Also note that if you set these properties, then no cross-realm authentication is possible unless
156      * a <code>krb5.conf</code> file is also provided from which the additional information required for cross-realm authentication
157      * may be obtained.
158      * <p>If you set values for these properties, then they override the default realm and KDC values specified
159      * in <code>krb5.conf</code> (if such a file is found). The <code>krb5.conf</code> file is still consulted if values for items
160      * other than the default realm and KDC are needed. If no <code>krb5.conf</code> file is found,
161      * then the default values used for these items are implementation-specific.
162      * @param kerberosRealmSystemProperty system property to indicate realm.
163      * @see <a href="http://docs.oracle.com/javase/7/docs/technotes/guides/security/jgss/tutorials/KerberosReq.html">
164      *      Oracle documentation</a>
165      * @since 4.1.0
166      */
167     public final void setKerberosRealmSystemProperty(final String kerberosRealmSystemProperty) {
168         this.kerberosRealmSystemProperty = kerberosRealmSystemProperty;
169     }

172      * Typically, the default realm and the KDC for that realm are indicated in the Kerberos <code>krb5.conf</code> configuration file.
173      * However, if you like, you can instead specify the kdc value by setting this system property value.
174      * <p>If you set the realm property, you SHOULD also configure the {@link #setKerberosRealmSystemProperty(String)}.
175      * <p>Also note that if you set these properties, then no cross-realm authentication is possible unless
176      * a <code>krb5.conf</code> file is also provided from which the additional information required for cross-realm authentication
177      * may be obtained.
178      * <p>If you set values for these properties, then they override the default realm and KDC values specified
179      * in <code>krb5.conf</code> (if such a file is found). The <code>krb5.conf</code> file is still consulted if values for items
180      * other than the default realm and KDC are needed. If no <code>krb5.conf</code> file is found,

182      * @param kerberosKdcSystemProperty system property to indicate kdc

184      *      Oracle documentation</a>

188         this.kerberosKdcSystemProperty = kerberosKdcSystemProperty;

191     /**
192      * A simple JAAS CallbackHandler which accepts a Name String and Password
193      * String in the constructor. Only NameCallbacks and PasswordCallbacks are
194      * accepted in the callback array. This code based loosely on example given
195      * in Sun's javadoc for CallbackHandler interface.
196      */
197     protected static final class UsernamePasswordCallbackHandler implements CallbackHandler {
198 
199         /** The username of the principal we are trying to authenticate. */
200         private final String userName;
201 
202         /** The password of the principal we are trying to authenticate. */
203         private final String password;
204 
205         /**
206          * Constructor accepts name and password to be used for authentication.
207          *
208          * @param userName name to be used for authentication
209          * @param password Password to be used for authentication
210          */
211         protected UsernamePasswordCallbackHandler(final String userName,
212             final String password) {
213             this.userName = userName;
214             this.password = password;
215         }
216 
217         @Override
218         public void handle(final Callback[] callbacks)
219             throws UnsupportedCallbackException {
220             for (final Callback callback : callbacks) {
221                 if (callback.getClass().equals(NameCallback.class)) {
222                     ((NameCallback) callback).setName(this.userName);
223                 } else if (callback.getClass().equals(PasswordCallback.class)) {
224                     ((PasswordCallback) callback).setPassword(this.password
225                         .toCharArray());
226                 } else {
227                     throw new UnsupportedCallbackException(callback,
228                         "Unrecognized Callback");
229                 }
230             }
231         }
232     }
