19  package org.jasig.cas.web.flow;
20  
21  import org.jasig.cas.CentralAuthenticationService;
22  import org.jasig.cas.authentication.AuthenticationException;
23  import org.jasig.cas.authentication.Credential;
24  import org.jasig.cas.authentication.principal.DefaultPrincipalFactory;
25  import org.jasig.cas.authentication.principal.PrincipalFactory;
26  import org.jasig.cas.authentication.principal.Service;
27  import org.jasig.cas.ticket.TicketException;
28  import org.jasig.cas.ticket.ServiceTicket;
29  import org.jasig.cas.web.support.WebUtils;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  import org.springframework.util.StringUtils;
33  import org.springframework.webflow.action.AbstractAction;
34  import org.springframework.webflow.execution.Event;
35  import org.springframework.webflow.execution.RequestContext;
36  
37  import javax.validation.constraints.NotNull;
38  
39  /**
40   * Abstract class to handle the retrieval and authentication of non-interactive
41   * credential such as client certificates, NTLM, etc.
42   *
43   * @author Scott Battaglia
44  
45   * @since 3.0.0.4
46   */
47  public abstract class AbstractNonInteractiveCredentialsAction extends AbstractAction {
48  
49      /** The logger instance. */
50      protected final Logger logger = LoggerFactory.getLogger(this.getClass());
51  
52      /**
53       * The Principal factory.
54       */
55      protected PrincipalFactory principalFactory = new DefaultPrincipalFactory();
56  
57      /** Instance of CentralAuthenticationService. */
58      @NotNull
59      private CentralAuthenticationService centralAuthenticationService;
60  
61      /**
62       * Checks if is renew present.
63       *
64       * @param context the context
65       * @return true, if  renew present
66       */
67      protected final boolean isRenewPresent(final RequestContext context) {
68          return StringUtils.hasText(context.getRequestParameters().get("renew"));
69      }
70  
71      @Override
72      protected final Event doExecute(final RequestContext context) {
73          final Credential credential = constructCredentialsFromRequest(context);
74  
75          if (credential == null) {
76              return error();
77          }
78  
79          final String ticketGrantingTicketId = WebUtils.getTicketGrantingTicketId(context);
80          final Service service = WebUtils.getService(context);
81  
82          if (isRenewPresent(context)
83              && ticketGrantingTicketId != null
84              && service != null) {
85  
86              try {
87                  final ServiceTicket serviceTicketId = this.centralAuthenticationService
88                      .grantServiceTicket(ticketGrantingTicketId,
89                          service,
90                              credential);
91                  WebUtils.putServiceTicketInRequestScope(context, serviceTicketId);
92                  return result("warn");
93              } catch (final AuthenticationException e) {
94                  onError(context, credential);
95                  return error();
96              } catch (final TicketException e) {
97                  this.centralAuthenticationService.destroyTicketGrantingTicket(ticketGrantingTicketId);
98                  logger.debug("Attempted to generate a ServiceTicket using renew=true with different credential", e);
99              }
100         }
101 
102         try {
103             WebUtils.putTicketGrantingTicketInScopes(
104                 context,
105                 this.centralAuthenticationService
106                     .createTicketGrantingTicket(credential));
107             onSuccess(context, credential);
108             return success();
109         } catch (final Exception e) {
110             onError(context, credential);
111             return error();
112         }
113     }
114 
115     public final void setCentralAuthenticationService(
116         final CentralAuthenticationService centralAuthenticationService) {
117         this.centralAuthenticationService = centralAuthenticationService;
118     }
119 
120     /**
121      * Sets principal factory to create principal objects.
122      *
123      * @param principalFactory the principal factory
124      */
125     public void setPrincipalFactory(final PrincipalFactory principalFactory) {
126         this.principalFactory = principalFactory;
127     }
128 
129     /**
130      * Hook method to allow for additional processing of the response before
131      * returning an error event.
132      *
133      * @param context the context for this specific request.
134      * @param credential the credential for this request.
135      */
136     protected void onError(final RequestContext context,
137         final Credential credential) {
138         // default implementation does nothing
139     }
140 
141     /**
142      * Hook method to allow for additional processing of the response before
143      * returning a success event.
144      *
145      * @param context the context for this specific request.
146      * @param credential the credential for this request.
147      */
148     protected void onSuccess(final RequestContext context,
149         final Credential credential) {
150         // default implementation does nothing
151     }
152 
153     /**
154      * Abstract method to implement to construct the credential from the
155      * request object.
156      *
157      * @param context the context for this request.
158      * @return the constructed credential or null if none could be constructed
159      * from the request.
160      */
161     protected abstract Credential constructCredentialsFromRequest(
162         final RequestContext context);
163 }
