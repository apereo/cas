21  import java.io.Serializable;

23  /**
24   * Generates a unique consistent Id based on the principal.
25   *
26   * @author Scott Battaglia
27   * @since 3.1
28   */
29  public interface PersistentIdGenerator extends Serializable {
30  
31      /**
32       * Generates a PersistentId based on some algorithm plus the principal.
33       *
34       * @param principal the principal to generate the id for.
35       * @param service the service for which the id may be generated.
36       * @return the generated persistent id.
37       */
38      String generate(Principal principal, Service service);
39  }
