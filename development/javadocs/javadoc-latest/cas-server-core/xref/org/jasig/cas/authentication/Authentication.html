19  package org.jasig.cas.authentication;
20  
21  import java.io.Serializable;
22  import java.util.Date;
23  import java.util.List;
24  import java.util.Map;
25  
26  import org.jasig.cas.authentication.principal.Principal;
27  
28  /**
29   * <p>
30   * The Authentication object represents a successful authentication request. It
31   * contains the principal that the authentication request was made for as well
32   * as the additional meta information such as the authenticated date and a map
33   * of attributes.
34   * </p>
35   * <p>
36   * An Authentication object must be serializable to permit persistance and
37   * clustering.
38   * </p>
39   * <p>
40   * Implementing classes must take care to ensure that the Map returned by
41   * getAttributes is serializable by using a Serializable map such as HashMap.
42   * </p>
43   *
44   * @author Dmitriy Kopylenko
45   * @author Scott Battaglia
46   * @author Marvin S. Addison
47   * @since 3.0.0
48   */
49  public interface Authentication extends Serializable {
50  
51      /**
52       * Method to obtain the Principal.
53       *
54       * @return a Principal implementation
55       */
56      Principal getPrincipal();
57  
58      /**
59       * Method to retrieve the timestamp of when this Authentication object was
60       * created.
61       *
62       * @return the date/time the authentication occurred.
63       */
64      Date getAuthenticationDate();
65  
66      /**
67       * Attributes of the authentication (not the Principal).
68       *
69       * @return the map of attributes.
70       */
71      Map<String, Object> getAttributes();
72  
73      /**
74       * Gets a list of metadata about the credentials supplied at authentication time.
75       *
76       * @return Non-null list of supplied credentials represented as metadata that should be considered safe for
77       * long-term storage (e.g. serializable and secure with respect to credential disclosure). The order of items in
78       * the returned list SHOULD be the same as the order in which the source credentials were presented and subsequently
79       * processed.
80       */
81      List<CredentialMetaData> getCredentials();
82  
83      /**
84       * Gets a map describing successful authentications produced by {@link AuthenticationHandler} components.
85       *
86       * @return Map of handler names to successful authentication result produced by that handler.
87       */
88      Map<String, HandlerResult> getSuccesses();
89  
90      /**
91       * Gets a map describing failed authentications. By definition the failures here were not sufficient to prevent
92       * authentication.
93       *
94       * @return Map of authentication handler names to the authentication errors produced on attempted authentication.
95       */
96      Map<String, Class<? extends Exception>> getFailures();
97  }
