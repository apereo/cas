47  public class AuthenticationViaFormActionTests extends AbstractCentralAuthenticationServiceTest {
48  
49      private AuthenticationViaFormAction action;
50  
51      private CookieGenerator warnCookieGenerator;
52  
53      @Before
54      public void onSetUp() throws Exception {
55          this.action = new AuthenticationViaFormAction();
56  
57          this.warnCookieGenerator = new CookieGenerator();
58          this.warnCookieGenerator.setCookieName("WARN");
59          this.warnCookieGenerator.setCookieName("TGT");
60          this.warnCookieGenerator.setCookieDomain("/");
61          this.warnCookieGenerator.setCookiePath("/");
62  
63          this.action.setCentralAuthenticationService(getCentralAuthenticationService());
64          this.action.setWarnCookieGenerator(this.warnCookieGenerator);
65      }
66  
67      @Test
68      public void verifySuccessfulAuthenticationWithNoService() throws Exception {
69          final MockHttpServletRequest request = new MockHttpServletRequest();
70          final MockRequestContext context = new MockRequestContext();
71  
72          WebUtils.putLoginTicket(context, "LOGIN");
73          request.addParameter("lt", "LOGIN");
74          request.addParameter("username", "test");
75          request.addParameter("password", "test");
76  
77          context.setExternalContext(new ServletExternalContext(
78                  new MockServletContext(), request, new MockHttpServletResponse()));
79          final Credential c = TestUtils.getCredentialsWithSameUsernameAndPassword();
80          putCredentialInRequestScope(context, c);
81  
82          final MessageContext messageContext = mock(MessageContext.class);
83          assertEquals("success", this.action.submit(context, c, messageContext).getId());
84      }
85  
86      @Test
87      public void verifySuccessfulAuthenticationWithNoServiceAndWarn()
88          throws Exception {
89          final MockHttpServletRequest request = new MockHttpServletRequest();
90          final MockHttpServletResponse response = new MockHttpServletResponse();
91          final MockRequestContext context = new MockRequestContext();
92  
93          WebUtils.putLoginTicket(context, "LOGIN");
94          request.addParameter("lt", "LOGIN");
95  
96          request.addParameter("username", "test");
97          request.addParameter("password", "test");
98          request.addParameter("warn", "true");
99  
100         context.setExternalContext(new ServletExternalContext(
101                 new MockServletContext(), request, response));
102         final Credential c = TestUtils.getCredentialsWithSameUsernameAndPassword();
103         putCredentialInRequestScope(context, c);
104 
105         final MessageContext messageContext = mock(MessageContext.class);
106         assertEquals("success", this.action.submit(context, c, messageContext).getId());
107         assertNotNull(WebUtils.getTicketGrantingTicketId(context));
108         assertNotNull(response.getCookie(this.warnCookieGenerator.getCookieName()));
109     }

135     @Test

157     @Test
158     public void verifyRenewWithServiceAndSameCredentials() throws Exception {
159         final Credential c = TestUtils.getCredentialsWithSameUsernameAndPassword();
160         final TicketGrantingTicket ticketGrantingTicket = getCentralAuthenticationService().createTicketGrantingTicket(c);
161         final MockHttpServletRequest request = new MockHttpServletRequest();
162         final MockRequestContext context = new MockRequestContext();
163 
164         WebUtils.putTicketGrantingTicketInScopes(context, ticketGrantingTicket);
165         WebUtils.putLoginTicket(context, "LOGIN");
166         request.addParameter("lt", "LOGIN");
167 
168         request.addParameter("renew", "true");
169         request.addParameter("service", "test");
170         request.addParameter("username", "test");
171         request.addParameter("password", "test");
172 
173         context.setExternalContext(new ServletExternalContext(
174             new MockServletContext(), request, new MockHttpServletResponse()));
175         context.getFlowScope().put("service", TestUtils.getService());
176 
177         final MessageContext messageContext = mock(MessageContext.class);
178         assertEquals("warn", this.action.submit(context, c, messageContext).getId());
179     }

228     /**
229      * Put credentials in request scope.
230      *
231      * @param context the context
232      * @param c the credential
233      */
234     private static void putCredentialInRequestScope(
235             final RequestContext context, @NotNull final Credential c) {
236         context.getRequestScope().put("credentials", c);
237     }
