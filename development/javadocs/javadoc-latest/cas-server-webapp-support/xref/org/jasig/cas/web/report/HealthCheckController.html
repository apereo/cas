19  package org.jasig.cas.web.report;
20  
21  import java.util.Map;
22  import javax.servlet.http.HttpServletRequest;
23  import javax.servlet.http.HttpServletResponse;
24  import javax.validation.constraints.NotNull;
25  
26  import org.jasig.cas.monitor.HealthCheckMonitor;
27  import org.jasig.cas.monitor.HealthStatus;
28  import org.jasig.cas.monitor.Status;
29  import org.springframework.beans.factory.annotation.Autowired;
30  import org.springframework.stereotype.Controller;
31  import org.springframework.web.bind.annotation.RequestMapping;
32  import org.springframework.web.bind.annotation.RequestMethod;
33  import org.springframework.web.servlet.ModelAndView;
34  
35  
36  /**
37   * Reports overall CAS health based on the observations of the configured {@link HealthCheckMonitor} instance.
38   *
39   * @author Marvin S. Addison
40   * @since 3.5
41   */
42  @Controller("healthCheckController")
43  @RequestMapping("/status")
44  public class HealthCheckController {
45  
46      /** Prefix for custom response headers with health check details. */
47      private static final String HEADER_PREFIX = "X-CAS-";
48  
49      @NotNull
50      @Autowired
51      private HealthCheckMonitor healthCheckMonitor;
52  
53      /**
54       * Handle request.
55       *
56       * @param request the request
57       * @param response the response
58       * @return the model and view
59       * @throws Exception the exception
60       */
61      @RequestMapping(method = RequestMethod.GET)
62      protected ModelAndView handleRequestInternal(
63              final HttpServletRequest request, final HttpServletResponse response)
64              throws Exception {
65  
66          final HealthStatus healthStatus = this.healthCheckMonitor.observe();
67          final StringBuilder sb = new StringBuilder();
68          sb.append("Health: ").append(healthStatus.getCode());
69          String name;
70          Status status;
71          int i = 0;
72          for (final Map.Entry<String, Status> entry : healthStatus.getDetails().entrySet()) {
73              name = entry.getKey();
74              status = entry.getValue();
75              response.addHeader("X-CAS-" + name, String.format("%s;%s", status.getCode(), status.getDescription()));
76  
77              sb.append("\n\n\t").append(++i).append('.').append(name).append(": ");
78              sb.append(status.getCode());
79              if (status.getDescription() != null) {
80                  sb.append(" - ").append(status.getDescription());
81              }
82          }
83          response.setStatus(healthStatus.getCode().value());
84          response.setContentType("text/plain");
85          response.getOutputStream().write(sb.toString().getBytes(response.getCharacterEncoding()));
86  
87          // Return null to signal MVC framework that we handled response directly
88          return null;
89      }
90  }
