50  /**
51   * @author Scott Battaglia
52   * @since 3.3.5
53   */
54  @Controller("statisticsController")
55  @RequestMapping("/statistics")
56  public final class StatisticsController implements ServletContextAware {
57  
58      private static final int NUMBER_OF_MILLISECONDS_IN_A_DAY = 86400000;
59  
60      private static final int NUMBER_OF_MILLISECONDS_IN_AN_HOUR = 3600000;
61  
62      private static final int NUMBER_OF_MILLISECONDS_IN_A_MINUTE = 60000;
63  
64      private static final int NUMBER_OF_MILLISECONDS_IN_A_SECOND = 1000;
65  
66      private static final int NUMBER_OF_BYTES_IN_A_KILOBYTE = 1024;
67  
68      private final Logger logger = LoggerFactory.getLogger(this.getClass());
69  
70      private final Date upTimeStartDate = new Date();
71  
72      @Value("${host.name}")
73      private String casTicketSuffix;
74  
75      private String viewPath = "/WEB-INF/view/jsp/monitoring/viewStatistics.jsp";
76  
77      @Autowired
78      private CentralAuthenticationService centralAuthenticationService;
79  
80      @Autowired
81      @Qualifier("metrics")
82      private MetricRegistry metricsRegistry;
83  
84      @Autowired
85      @Qualifier("healthCheckMetrics")
86      private HealthCheckRegistry healthCheckRegistry;
87  
88      public void setViewPath(final String viewPath){
89          this.viewPath = viewPath;
90      }
91  
92      /**
93       * Handles the request.
94       *
95       * @param httpServletRequest the http servlet request
96       * @param httpServletResponse the http servlet response
97       * @return the model and view
98       * @throws Exception the exception
99       */
100     @RequestMapping(method = RequestMethod.GET)
101     protected ModelAndView handleRequestInternal(final HttpServletRequest httpServletRequest, final HttpServletResponse httpServletResponse)
102                 throws Exception {
103         final ModelAndView modelAndView = new ModelAndView(new InternalResourceView(viewPath));
104         modelAndView.addObject("startTime", this.upTimeStartDate);
105         final double difference = System.currentTimeMillis() - this.upTimeStartDate.getTime();
106 
107         modelAndView.addObject("upTime", calculateUptime(difference, new LinkedList<Integer>(
108                         Arrays.asList(NUMBER_OF_MILLISECONDS_IN_A_DAY, NUMBER_OF_MILLISECONDS_IN_AN_HOUR,
109                         NUMBER_OF_MILLISECONDS_IN_A_MINUTE, NUMBER_OF_MILLISECONDS_IN_A_SECOND, 1)),
110                         new LinkedList<String>(Arrays.asList("day", "hour", "minute", "second", "millisecond"))));
111 
112         modelAndView.addObject("totalMemory", convertToMegaBytes(Runtime.getRuntime().totalMemory()));
113         modelAndView.addObject("maxMemory", convertToMegaBytes(Runtime.getRuntime().maxMemory()));
114         modelAndView.addObject("freeMemory", convertToMegaBytes(Runtime.getRuntime().freeMemory()));
115         modelAndView.addObject("availableProcessors", Runtime.getRuntime().availableProcessors());
116         modelAndView.addObject("serverHostName", httpServletRequest.getServerName());
117         modelAndView.addObject("serverIpAddress", httpServletRequest.getLocalAddr());
118         modelAndView.addObject("casTicketSuffix", this.casTicketSuffix);
119 
120         int unexpiredTgts = 0;
121         int unexpiredSts = 0;
122         int expiredTgts = 0;
123         int expiredSts = 0;
124 
125         try {
126             final Collection<Ticket> tickets = this.centralAuthenticationService.getTickets(TruePredicate.getInstance());
127 
128             for (final Ticket ticket : tickets) {
129                 if (ticket instanceof ServiceTicket) {
130                     if (ticket.isExpired()) {
131                         expiredSts++;
132                     } else {
133                         unexpiredSts++;
134                     }
135                 } else {
136                     if (ticket.isExpired()) {
137                         expiredTgts++;
138                     } else {
139                         unexpiredTgts++;
140                     }
141                 }
142             }
143         } catch (final UnsupportedOperationException e) {
144             logger.trace("The ticket registry doesn't support this information.");
145         }
146 
147         modelAndView.addObject("unexpiredTgts", unexpiredTgts);
148         modelAndView.addObject("unexpiredSts", unexpiredSts);
149         modelAndView.addObject("expiredTgts", expiredTgts);
150         modelAndView.addObject("expiredSts", expiredSts);
151         modelAndView.addObject("pageTitle", modelAndView.getViewName());
152 
153         return modelAndView;
154     }
155 
156     /**
157      * Convert to megabytes from bytes.
158      * @param bytes the total number of bytes
159      * @return value converted to MB
160      */
161     private double convertToMegaBytes(final double bytes) {
162         return bytes / NUMBER_OF_BYTES_IN_A_KILOBYTE / NUMBER_OF_BYTES_IN_A_KILOBYTE;
163     }
164     /**
165      * Calculates the up time.
166      *
167      * @param difference the difference
168      * @param calculations the calculations
169      * @param labels the labels
170      * @return the uptime as a string.
171      */
172     protected String calculateUptime(final double difference, final Queue<Integer> calculations, final Queue<String> labels) {
173         if (calculations.isEmpty()) {
174             return "";
175         }
176 
177         final int value = calculations.remove();
178         final double time = Math.floor(difference / value);
179         final double newDifference = difference - time * value;
180         final String currentLabel = labels.remove();
181         final String label = time == 0 || time > 1 ? currentLabel + "s" : currentLabel;
182 
183         return Integer.toString((int) time) + " " + label + " " + calculateUptime(newDifference, calculations, labels);
184     }
185 
186     @Override
187     public void setServletContext(final ServletContext servletContext) {
188         servletContext.setAttribute(MetricsServlet.METRICS_REGISTRY, this.metricsRegistry);
189         servletContext.setAttribute(MetricsServlet.SHOW_SAMPLES, true);
190         servletContext.setAttribute(HealthCheckServlet.HEALTH_CHECK_REGISTRY, this.healthCheckRegistry);
191     }
192 }
