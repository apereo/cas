19  package org.jasig.cas.ticket.registry;
20  
21  import java.io.IOException;
22  import java.util.Arrays;
23  import java.util.Collection;
24  import java.util.concurrent.TimeUnit;
25  
26  import javax.validation.constraints.Min;
27  import javax.validation.constraints.NotNull;
28  
29  import net.spy.memcached.AddrUtil;
30  import net.spy.memcached.MemcachedClient;
31  import net.spy.memcached.MemcachedClientIF;
32  
33  import org.jasig.cas.ticket.ServiceTicket;
34  import org.jasig.cas.ticket.Ticket;
35  import org.jasig.cas.ticket.TicketGrantingTicket;
36  import org.springframework.beans.factory.DisposableBean;
37  
38  /**
39   * Key-value ticket registry implementation that stores tickets in memcached keyed on the ticket ID.
40   *
41   * @author Scott Battaglia
42   * @author Marvin S. Addison
43   * @since 3.3
44   */
45  public final class MemCacheTicketRegistry extends AbstractDistributedTicketRegistry implements DisposableBean {
46  
47      /** Memcached client. */
48      @NotNull
49      private final MemcachedClientIF client;
50  
51      /**
52       * TGT cache entry timeout in seconds.
53       */
54      @Min(0)
55      private final int tgtTimeout;
56  
57      /**
58       * ST cache entry timeout in seconds.
59       */
60      @Min(0)
61      private final int stTimeout;
62  
63  
64      /**
65       * Creates a new instance that stores tickets in the given memcached hosts.
66       *
67       * @param hostnames                   Array of memcached hosts where each element is of the form host:port.
68       * @param ticketGrantingTicketTimeOut TGT timeout in seconds.
69       * @param serviceTicketTimeOut        ST timeout in seconds.
70       */
71      public MemCacheTicketRegistry(final String[] hostnames, final int ticketGrantingTicketTimeOut,
72                                    final int serviceTicketTimeOut) {
73          try {
74              this.client = new MemcachedClient(AddrUtil.getAddresses(Arrays.asList(hostnames)));
75          } catch (final IOException e) {
76              throw new IllegalArgumentException("Invalid memcached host specification.", e);
77          }
78          this.tgtTimeout = ticketGrantingTicketTimeOut;
79          this.stTimeout = serviceTicketTimeOut;
80      }
81  
82      /**
83       * This alternative constructor takes time in milliseconds.
84       * It has the timeout parameters in order to create a unique method signature.
85       *
86       * @param ticketGrantingTicketTimeOut TGT timeout in milliseconds.
87       * @param serviceTicketTimeOut ST timeout in milliseconds.
88       * @param hostnames  Array of memcached hosts where each element is of the form host:port.
89       * @see MemCacheTicketRegistry#MemCacheTicketRegistry(String[], int, int)
90       * @deprecated This has been deprecated
91       */
92      @Deprecated
93      public MemCacheTicketRegistry(final long ticketGrantingTicketTimeOut, final long serviceTicketTimeOut,
94              final String[] hostnames) {
95          this(hostnames, (int) TimeUnit.MILLISECONDS.toSeconds(ticketGrantingTicketTimeOut),
96                  (int) TimeUnit.MILLISECONDS.toSeconds(serviceTicketTimeOut));
97      }
98  
99      /**
100      * Creates a new instance using the given memcached client instance, which is presumably configured via
101      * <code>net.spy.memcached.spring.MemcachedClientFactoryBean</code>.
102      *
103      * @param client                      Memcached client.
104      * @param ticketGrantingTicketTimeOut TGT timeout in seconds.
105      * @param serviceTicketTimeOut        ST timeout in seconds.
106      */
107     public MemCacheTicketRegistry(final MemcachedClientIF client, final int ticketGrantingTicketTimeOut,
108             final int serviceTicketTimeOut) {
109         this.tgtTimeout = ticketGrantingTicketTimeOut;
110         this.stTimeout = serviceTicketTimeOut;
111         this.client = client;
112     }
113 
114     @Override
115     protected void updateTicket(final Ticket ticket) {
116         logger.debug("Updating ticket {}", ticket);
117         try {
118             if (!this.client.replace(ticket.getId(), getTimeout(ticket), ticket).get()) {
119                 logger.error("Failed updating {}", ticket);
120             }
121         } catch (final InterruptedException e) {
122             logger.warn("Interrupted while waiting for response to async replace operation for ticket {}. "
123                         + "Cannot determine whether update was successful.", ticket);
124         } catch (final Exception e) {
125             logger.error("Failed updating {}", ticket, e);
126         }
127     }
128 
129     @Override
130     public void addTicket(final Ticket ticket) {
131         logger.debug("Adding ticket {}", ticket);
132         try {
133             if (!this.client.add(ticket.getId(), getTimeout(ticket), ticket).get()) {
134                 logger.error("Failed adding {}", ticket);
135             }
136         } catch (final InterruptedException e) {
137             logger.warn("Interrupted while waiting for response to async add operation for ticket {}."
138                     + "Cannot determine whether add was successful.", ticket);
139         } catch (final Exception e) {
140             logger.error("Failed adding {}", ticket, e);
141         }
142     }
143     @Override
144     public boolean deleteTicket(final String ticketId) {
145         logger.debug("Deleting ticket {}", ticketId);
146         try {
147             return this.client.delete(ticketId).get();
148         } catch (final Exception e) {
149             logger.error("Failed deleting {}", ticketId, e);
150         }
151         return false;
152     }
153     @Override
154     public Ticket getTicket(final String ticketId) {
155         try {
156             final Ticket t = (Ticket) this.client.get(ticketId);
157             if (t != null) {
158                 return getProxiedTicketInstance(t);
159             }
160         } catch (final Exception e) {
161             logger.error("Failed fetching {} ", ticketId, e);
162         }
163         return null;
164     }
165 
166     /**
167      * {@inheritDoc}
168      * This operation is not supported.
169      *
170      * @throws UnsupportedOperationException if you try and call this operation.
171      */
172     @Override
173     public Collection<Ticket> getTickets() {
174         throw new UnsupportedOperationException("GetTickets not supported.");
175     }
176 
177     /**
178      * Destroy the client and shut down.
179      *
180      * @throws Exception the exception
181      */
182     public void destroy() throws Exception {
183         this.client.shutdown();
184     }
185 
186     /**
187      * @param sync set to true, if updates to registry are to be synchronized
188      * @deprecated As of version 3.5, this operation has no effect since async writes can cause registry consistency issues.
189      */
190     @Deprecated
191     public void setSynchronizeUpdatesToRegistry(final boolean sync) {}
192 
193     @Override
194     protected boolean needsCallback() {
195         return true;
196     }
197 
198     /**
199      * Gets the timeout value for the ticket.
200      *
201      * @param t the t
202      * @return the timeout
203      */
204     private int getTimeout(final Ticket t) {
205         if (t instanceof TicketGrantingTicket) {
206             return this.tgtTimeout;
207         } else if (t instanceof ServiceTicket) {
208             return this.stTimeout;
209         }
210         throw new IllegalArgumentException("Invalid ticket type");
211     }
212 }
