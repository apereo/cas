74      /**
75       * Sign SAML response.
76       *
77       * @param samlResponse the SAML response
78       * @param privateKey the private key
79       * @param publicKey the public key
80       * @return the response
81       */
82      public static String signSamlResponse(final String samlResponse,
83              final PrivateKey privateKey, final PublicKey publicKey) {
84          final Document doc = constructDocumentFromXmlString(samlResponse);
85  
86          if (doc != null) {
87              final Element signedElement = signSamlElement(doc.getRootElement(),
88                      privateKey, publicKey);
89              doc.setRootElement((Element) signedElement.detach());
90              return new XMLOutputter().outputString(doc);
91          }
92          throw new RuntimeException("Error signing SAML Response: Null document");
93      }

95      /**
96       * Construct document from xml string.
97       *
98       * @param xmlString the xml string
99       * @return the document
100      */
101     public static Document constructDocumentFromXmlString(final String xmlString) {
102         try {
103             final SAXBuilder builder = new SAXBuilder();
104             builder.setFeature("http://xml.org/sax/features/external-general-entities", false);
105             builder.setFeature("http://apache.org/xml/features/disallow-doctype-decl", true);
106             return builder
107                     .build(new ByteArrayInputStream(xmlString.getBytes()));
108         } catch (final Exception e) {
109             return null;
110         }
111     }

113     /**
114      * Sign SAML element.
115      *
116      * @param element the element
117      * @param privKey the priv key
118      * @param pubKey the pub key
119      * @return the element
120      */
121     private static Element signSamlElement(final Element element, final PrivateKey privKey,
122             final PublicKey pubKey) {
123         try {
124             final String providerName = System.getProperty("jsr105Provider",
125                     JSR_105_PROVIDER);
126             final XMLSignatureFactory sigFactory = XMLSignatureFactory
127                     .getInstance("DOM", (Provider) Class.forName(providerName)
128                             .newInstance());
129 
130             final List<Transform> envelopedTransform = Collections
131                     .singletonList(sigFactory.newTransform(Transform.ENVELOPED,
132                             (TransformParameterSpec) null));
133 
134             final Reference ref = sigFactory.newReference("", sigFactory
135                     .newDigestMethod(DigestMethod.SHA1, null), envelopedTransform,
136                     null, null);
137 
138             // Create the SignatureMethod based on the type of key
139             SignatureMethod signatureMethod;
140             if (pubKey instanceof DSAPublicKey) {
141                 signatureMethod = sigFactory.newSignatureMethod(
142                         SignatureMethod.DSA_SHA1, null);
143             } else if (pubKey instanceof RSAPublicKey) {
144                 signatureMethod = sigFactory.newSignatureMethod(
145                         SignatureMethod.RSA_SHA1, null);
146             } else {
147                 throw new RuntimeException(
148                         "Error signing SAML element: Unsupported type of key");
149             }
150 
151             final CanonicalizationMethod canonicalizationMethod = sigFactory
152                     .newCanonicalizationMethod(
153                             CanonicalizationMethod.INCLUSIVE_WITH_COMMENTS,
154                             (C14NMethodParameterSpec) null);
155 
156             // Create the SignedInfo
157             final SignedInfo signedInfo = sigFactory.newSignedInfo(
158                     canonicalizationMethod, signatureMethod, Collections
159                     .singletonList(ref));
160 
161             // Create a KeyValue containing the DSA or RSA PublicKey
162             final KeyInfoFactory keyInfoFactory = sigFactory
163                     .getKeyInfoFactory();
164             final KeyValue keyValuePair = keyInfoFactory.newKeyValue(pubKey);
165 
166             // Create a KeyInfo and add the KeyValue to it
167             final KeyInfo keyInfo = keyInfoFactory.newKeyInfo(Collections
168                     .singletonList(keyValuePair));
169             // Convert the JDOM document to w3c (Java XML signature API requires
170             // w3c
171             // representation)
172             final org.w3c.dom.Element w3cElement = toDom(element);
173 
174             // Create a DOMSignContext and specify the DSA/RSA PrivateKey and
175             // location of the resulting XMLSignature's parent element
176             final DOMSignContext dsc = new DOMSignContext(privKey, w3cElement);
177 
178             final org.w3c.dom.Node xmlSigInsertionPoint = getXmlSignatureInsertLocation(w3cElement);
179             dsc.setNextSibling(xmlSigInsertionPoint);
180 
181             // Marshal, generate (and sign) the enveloped signature
182             final XMLSignature signature = sigFactory.newXMLSignature(signedInfo,
183                     keyInfo);
184             signature.sign(dsc);
185 
186             return toJdom(w3cElement);
187 
188         } catch (final Exception e) {
189             throw new RuntimeException("Error signing SAML element: "
190                     + e.getMessage(), e);
191         }
192     }
193 
194     /**
195      * Gets the xml signature insert location.
196      *
197      * @param elem the elem
198      * @return the xml signature insert location
199      */
200     private static Node getXmlSignatureInsertLocation(final org.w3c.dom.Element elem) {
201         org.w3c.dom.Node insertLocation = null;
202         org.w3c.dom.NodeList nodeList = elem.getElementsByTagNameNS(
203                 SAML_PROTOCOL_NS_URI_V20, "Extensions");
204         if (nodeList.getLength() != 0) {
205             insertLocation = nodeList.item(nodeList.getLength() - 1);
206         } else {
207             nodeList = elem.getElementsByTagNameNS(SAML_PROTOCOL_NS_URI_V20,
208                     "Status");
209             insertLocation = nodeList.item(nodeList.getLength() - 1);
210         }
211         return insertLocation;
212     }

214     /**
215      * Convert the received jdom element to an Element.
216      *
217      * @param element the element
218      * @return the org.w3c.dom. element
219      */
220     private static org.w3c.dom.Element toDom(final Element element) {
221         return toDom(element.getDocument()).getDocumentElement();
222     }
223 
224     /**
225      * Convert the received jdom doc to a Document element.
226      *
227      * @param doc the doc
228      * @return the org.w3c.dom. document
229      */
230     private static org.w3c.dom.Document toDom(final Document doc) {
231         try {
232             final XMLOutputter xmlOutputter = new XMLOutputter();
233             final StringWriter elemStrWriter = new StringWriter();
234             xmlOutputter.output(doc, elemStrWriter);
235             final byte[] xmlBytes = elemStrWriter.toString().getBytes();
236             final DocumentBuilderFactory dbf = DocumentBuilderFactory
237                     .newInstance();
238             dbf.setNamespaceAware(true);
239             return dbf.newDocumentBuilder().parse(
240                     new ByteArrayInputStream(xmlBytes));
241         } catch (final Exception e) {
242             return null;
243         }
244     }

246     /**
247      * Convert to a jdom element.
248      *
249      * @param e the e
250      * @return the element
251      */
252     private static Element toJdom(final org.w3c.dom.Element e) {
253         return  new DOMBuilder().build(e);
254     }
