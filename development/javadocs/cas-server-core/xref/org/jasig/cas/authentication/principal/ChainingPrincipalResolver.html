19  package org.jasig.cas.authentication.principal;
20  
21  import java.util.List;
22  
23  import javax.validation.constraints.NotNull;
24  import javax.validation.constraints.Size;
25  
26  import org.jasig.cas.authentication.Credential;
27  
28  /**
29   * Delegates to one or more principal resolves in series to resolve a principal. The input to first configured resolver
30   * is the authenticated credential; for every subsequent resolver, the input is a {@link Credential} whose ID is the
31   * resolved princpial ID of the previous resolver.
32   * <p>
33   * A common use case for this component is resolving a temporary principal ID from an X.509 credential followed by
34   * a search (e.g. LDAP, database) for the final principal based on the temporary ID.
35   *
36   * @author Marvin S. Addison
37   * @since 4.0.0
38   */
39  public class ChainingPrincipalResolver implements PrincipalResolver {
40  
41      /** The chain of delegate resolvers that are invoked in order. */
42      @NotNull
43      @Size(min = 1)
44      private List<PrincipalResolver> chain;
45  
46      /**
47       * Sets the resolver chain. The resolvers other than the first one MUST be capable of performing resolution
48       * on the basis of {@link org.jasig.cas.authentication.Credential#getId()} alone;
49       * {@link PersonDirectoryPrincipalResolver} notably meets that requirement.
50       *
51       * @param chain List of delegate resolvers that are invoked in a chain.
52       */
53      public void setChain(final List<PrincipalResolver> chain) {
54          this.chain = chain;
55      }
56  
57      /**
58       * Resolves a credential by delegating to each of the configured resolvers in sequence. Note that the
59       * {@link PrincipalResolver#supports(org.jasig.cas.authentication.Credential)} method is called only for the
60       * first configured resolver.
61       *
62       * @param credential Authenticated credential.
63       *
64       * @return The principal from the last configured resolver in the chain.
65       */
66      public Principal resolve(final Credential credential) {
67          Principal result = null;
68          Credential input = credential;
69          for (final PrincipalResolver resolver : this.chain) {
70              if (result != null) {
71                  input = new IdentifiableCredential(result.getId());
72              }
73              result = resolver.resolve(input);
74          }
75          return result;
76      }
77  
78      /**
79       * Determines whether the credential is supported by this component by delegating to the first configured
80       * resolver in the chain.
81       *
82       * @param credential The credential to check for support.
83       *
84       * @return True if the first configured resolver in the chain supports the credential, false otherwise.
85       */
86      public boolean supports(final Credential credential) {
87          return this.chain.get(0).supports(credential);
88      }
89  
90      /** Credential that stores only an ID. */
91      static class IdentifiableCredential implements Credential {
92          /** Credential identifier. */
93          private final String id;
94  
95          /**
96           * Creates a new instance with the given ID.
97           *
98           * @param id the credential id
99           */
100         public IdentifiableCredential(final String id) {
101             this.id = id;
102         }
103 
104         @Override
105         public String getId() {
106             return this.id;
107         }
108     }
109 }
