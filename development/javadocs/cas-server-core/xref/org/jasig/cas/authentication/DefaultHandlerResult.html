19  package org.jasig.cas.authentication;
20  
21  import java.util.Collections;
22  import java.util.List;
23  
24  import org.apache.commons.lang3.builder.EqualsBuilder;
25  import org.apache.commons.lang3.builder.HashCodeBuilder;
26  import org.jasig.cas.MessageDescriptor;
27  import org.jasig.cas.authentication.principal.Principal;
28  import org.springframework.util.Assert;
29  import org.springframework.util.StringUtils;
30  
31  /**
32   * Contains information about a successful authentication produced by an {@link AuthenticationHandler}.
33   * Handler results are naturally immutable since they contain sensitive information that should not be modified outside
34   * the {@link AuthenticationHandler} that produced it.
35   *
36   * @author Marvin S. Addison
37   * @since 4.0.0
38   */
39  public class DefaultHandlerResult implements HandlerResult {
40  
41      /** Serialization support. */
42      private static final long serialVersionUID = -3113998493287982485L;
43  
44      /** The name of the authentication handler that successfully authenticated a credential. */
45      private String handlerName;
46  
47      /** Credential meta data. */
48      private CredentialMetaData credentialMetaData;
49  
50      /** Resolved principal for authenticated credential. */
51      private Principal principal;
52  
53      /** List of warnings issued by the authentication source while authenticating the credential. */
54      private List<MessageDescriptor> warnings;
55  
56      /** No-arg constructor for serialization support. */
57      private DefaultHandlerResult() {}
58  
59      /**
60       * Instantiates a new handler result.
61       *
62       * @param source the source
63       * @param metaData the meta data
64       */
65      public DefaultHandlerResult(final AuthenticationHandler source, final CredentialMetaData metaData) {
66          this(source, metaData, null, null);
67      }
68  
69      /**
70       * Instantiates a new handler result.
71       *
72       * @param source the source
73       * @param metaData the meta data
74       * @param p the p
75       */
76      public DefaultHandlerResult(final AuthenticationHandler source, final CredentialMetaData metaData, final Principal p) {
77          this(source, metaData, p, null);
78      }
79  
80      /**
81       * Instantiates a new handler result.
82       *
83       * @param source the source
84       * @param metaData the meta data
85       * @param warnings the warnings
86       */
87      public DefaultHandlerResult(
88              final AuthenticationHandler source, final CredentialMetaData metaData, final List<MessageDescriptor> warnings) {
89          this(source, metaData, null, warnings);
90      }
91  
92      /**
93       * Instantiates a new handler result.
94       *
95       * @param source the source
96       * @param metaData the meta data
97       * @param p the p
98       * @param warnings the warnings
99       */
100     public DefaultHandlerResult(
101             final AuthenticationHandler source,
102             final CredentialMetaData metaData,
103             final Principal p,
104             final List<MessageDescriptor> warnings) {
105         Assert.notNull(source, "Source cannot be null.");
106         Assert.notNull(metaData, "Credential metadata cannot be null.");
107         this.handlerName = source.getName();
108         if (!StringUtils.hasText(this.handlerName)) {
109             this.handlerName = source.getClass().getSimpleName();
110         }
111         this.credentialMetaData = metaData;
112         this.principal = p;
113         this.warnings = warnings;
114     }
115 
116     @Override
117     public String getHandlerName() {
118         return this.handlerName;
119     }
120 
121     @Override
122     public CredentialMetaData getCredentialMetaData() {
123         return this.credentialMetaData;
124     }
125 
126     @Override
127     public Principal getPrincipal() {
128         return this.principal;
129     }
130 
131     @Override
132     public List<MessageDescriptor> getWarnings() {
133         return this.warnings == null
134                 ? Collections.<MessageDescriptor>emptyList()
135                 : Collections.unmodifiableList(this.warnings);
136     }
137 
138     @Override
139     public int hashCode() {
140         final HashCodeBuilder builder = new HashCodeBuilder(109, 31);
141         builder.append(this.handlerName);
142         builder.append(this.credentialMetaData);
143         builder.append(this.principal);
144         builder.append(this.warnings);
145         return builder.toHashCode();
146     }
147 
148     @Override
149     public boolean equals(final Object obj) {
150         if (!(obj instanceof DefaultHandlerResult)) {
151             return false;
152         }
153         if (obj == this) {
154             return true;
155         }
156         final DefaultHandlerResult other = (DefaultHandlerResult) obj;
157         final EqualsBuilder builder = new EqualsBuilder();
158         builder.append(this.handlerName, other.handlerName);
159         builder.append(this.credentialMetaData, other.credentialMetaData);
160         builder.append(this.principal, other.principal);
161         builder.append(this.warnings, other.warnings);
162         return builder.isEquals();
163     }
164 
165     @Override
166     public String toString() {
167         return this.handlerName + ':' + this.credentialMetaData;
168     }
169 }
