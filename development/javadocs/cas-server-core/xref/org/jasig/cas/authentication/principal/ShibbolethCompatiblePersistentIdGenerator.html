19  package org.jasig.cas.authentication.principal;
20  
21  import com.google.common.io.ByteSource;
22  import org.apache.commons.lang3.RandomStringUtils;
23  import org.apache.commons.lang3.builder.EqualsBuilder;
24  import org.apache.commons.lang3.builder.HashCodeBuilder;
25  import org.jasig.cas.util.CompressionUtils;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import javax.validation.constraints.NotNull;
30  import java.io.IOException;
31  import java.nio.charset.Charset;
32  import java.security.MessageDigest;
33  import java.security.NoSuchAlgorithmException;
34  
35  /**
36   * Generates PersistentIds based on the Shibboleth algorithm.
37   *
38   * @author Scott Battaglia
39   * @since 3.1
40   */
41  public final class ShibbolethCompatiblePersistentIdGenerator implements PersistentIdGenerator {
42  
43      private static final long serialVersionUID = 6182838799563190289L;
44  
45      /** Log instance. */
46      private static final Logger LOGGER = LoggerFactory.getLogger(ShibbolethCompatiblePersistentIdGenerator.class);
47  
48      private static final byte CONST_SEPARATOR = (byte) '!';
49  
50      private static final int CONST_DEFAULT_SALT_COUNT = 16;
51  
52      private byte[] salt;
53  
54      /**
55       * Instantiates a new shibboleth compatible persistent id generator.
56       * The salt is initialized to a random 16-digit alphanumeric string.
57       * The generated id is pseudo-anonymous which allows it to be continually uniquely
58       * identified by for a particular service.
59       */
60      public ShibbolethCompatiblePersistentIdGenerator() {
61          this.salt = RandomStringUtils.randomAlphanumeric(CONST_DEFAULT_SALT_COUNT).getBytes(Charset.defaultCharset());
62      }
63      
64      /**
65       * Instantiates a new shibboleth compatible persistent id generator.
66       *
67       * @param salt the the salt
68       */
69      public ShibbolethCompatiblePersistentIdGenerator(@NotNull final String salt) {
70          this.salt = salt.getBytes(Charset.defaultCharset());
71      }
72  
73      /**
74       * @deprecated As of 4.1.
75       * Sets salt.
76       *
77       * @param salt the salt
78       */
79      @Deprecated
80      public void setSalt(final String salt) {
81          this.salt = salt.getBytes(Charset.defaultCharset());
82          LOGGER.warn("setSalt() is deprecated and will be removed. Use the constructor instead.");
83      }
84  
85  
86      /**
87       * Get salt.
88       *
89       * @return the byte[] for the salt or null
90       */
91      public byte[] getSalt() {
92          try {
93              return ByteSource.wrap(this.salt).read();
94          } catch (final IOException e) {
95              LOGGER.warn("Salt cannot be read because the byte array from source could not be consumed");
96          }
97          return null;
98      }
99  
100     @Override
101     public String generate(final Principal principal, final Service service) {
102         try {
103             final MessageDigest md = MessageDigest.getInstance("SHA");
104             final Charset charset = Charset.defaultCharset();
105             md.update(service.getId().getBytes(charset));
106             md.update(CONST_SEPARATOR);
107             md.update(principal.getId().getBytes(charset));
108             md.update(CONST_SEPARATOR);
109 
110             final String result = CompressionUtils.encodeBase64(md.digest(this.salt));
111             return result.replaceAll(System.getProperty("line.separator"), "");
112         } catch (final NoSuchAlgorithmException e) {
113             throw new RuntimeException(e);
114         }
115     }
116 
117 
118     @Override
119     public boolean equals(final Object obj) {
120         if (obj == null) {
121             return false;
122         }
123         if (obj == this) {
124             return true;
125         }
126         if (obj.getClass() != getClass()) {
127             return false;
128         }
129         final ShibbolethCompatiblePersistentIdGenerator rhs = (ShibbolethCompatiblePersistentIdGenerator) obj;
130         return new EqualsBuilder()
131                 .append(this.salt, rhs.salt)
132                 .isEquals();
133     }
134 
135     @Override
136     public int hashCode() {
137         return new HashCodeBuilder()
138                 .append(this.salt)
139                 .toHashCode();
140     }
141 }
