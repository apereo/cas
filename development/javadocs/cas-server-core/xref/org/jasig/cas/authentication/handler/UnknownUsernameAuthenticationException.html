19  package org.jasig.cas.authentication.handler;
20  
21  /**
22   * The exception to throw when we explicitly don't know anything about the
23   * username.
24   *
25   * @author Scott Battaglia
26  
27   * @since 3.0.0
28   */
29  public class UnknownUsernameAuthenticationException extends
30      BadUsernameOrPasswordAuthenticationException {
31  
32      /** Static instance of UnknownUsernameAuthenticationException. */
33      public static final UnknownUsernameAuthenticationException ERROR = new UnknownUsernameAuthenticationException();
34  
35      /** Unique ID for serializing. */
36      private static final long serialVersionUID = 3977861752513837361L;
37  
38      /** The code description of this exception. */
39      private static final String CODE = "error.authentication.credentials.bad.usernameorpassword.username";
40  
41      /**
42       * Default constructor that does not allow the chaining of exceptions and
43       * uses the default code as the error code for this exception.
44       */
45      public UnknownUsernameAuthenticationException() {
46          super(CODE);
47      }
48  
49      /**
50       * Constructor that allows for the chaining of exceptions. Defaults to the
51       * default code provided for this exception.
52       *
53       * @param throwable the chained exception.
54       */
55      public UnknownUsernameAuthenticationException(final Throwable throwable) {
56          super(CODE, throwable);
57      }
58  
59      /**
60       * Constructor that allows for providing a custom error code for this class.
61       * Error codes are often used to resolve exceptions into messages. Providing
62       * a custom error code allows the use of a different message.
63       *
64       * @param code the custom code to use with this exception.
65       */
66      public UnknownUsernameAuthenticationException(final String code) {
67          super(code);
68      }
69  
70      /**
71       * Constructor that allows for chaining of exceptions and a custom error
72       * code.
73       *
74       * @param code the custom error code to use in message resolving.
75       * @param throwable the chained exception.
76       */
77      public UnknownUsernameAuthenticationException(final String code,
78          final Throwable throwable) {
79          super(code, throwable);
80      }
81  }
