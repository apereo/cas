19  package org.jasig.cas.authentication;
20  
21  /**
22   * Authentication policy that defines success as at least one authentication success and no authentication attempts
23   * that were prevented by system errors. This policy may be a desirable alternative to {@link AnyAuthenticationPolicy}
24   * for cases where deployers wish to fail closed for indeterminate security events.
25   *
26   * @author Marvin S. Addison
27   * @since 4.0.0
28   */
29  public class NotPreventedAuthenticationPolicy extends AnyAuthenticationPolicy {
30  
31      @Override
32      public boolean isSatisfiedBy(final Authentication authentication) {
33          for (final String handler : authentication.getFailures().keySet()) {
34              if (authentication.getFailures().get(handler).isAssignableFrom(PreventedException.class)) {
35                  return false;
36              }
37          }
38          return super.isSatisfiedBy(authentication);
39      }
40  }
