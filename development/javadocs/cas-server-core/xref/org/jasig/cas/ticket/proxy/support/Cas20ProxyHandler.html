19  package org.jasig.cas.ticket.proxy.support;
20  
21  import org.jasig.cas.authentication.Credential;
22  import org.jasig.cas.authentication.HttpBasedServiceCredential;
23  import org.jasig.cas.ticket.TicketGrantingTicket;
24  import org.jasig.cas.ticket.proxy.ProxyHandler;
25  import org.jasig.cas.util.DefaultUniqueTicketIdGenerator;
26  import org.jasig.cas.util.UniqueTicketIdGenerator;
27  import org.jasig.cas.util.http.HttpClient;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import javax.validation.constraints.NotNull;
32  import java.net.URL;
33  
34  /**
35   * Proxy Handler to handle the default callback functionality of CAS 2.0.
36   * <p>
37   * The default behavior as defined in the CAS 2 Specification is to callback the
38   * URL provided and give it a pgtIou and a pgtId.
39   * </p>
40   *
41   * @author Scott Battaglia
42   * @since 3.0.0
43   */
44  public final class Cas20ProxyHandler implements ProxyHandler {
45      private static final int BUFFER_LENGTH_ADDITIONAL_CHARGE = 15;
46  
47      /** The proxy granting ticket identifier parameter. */
48      private static final String PARAMETER_PROXY_GRANTING_TICKET_IOU = "pgtIou";
49  
50      /** The Constant proxy granting ticket parameter. */
51      private static final String PARAMETER_PROXY_GRANTING_TICKET_ID = "pgtId";
52  
53      /** The Commons Logging instance. */
54      private final Logger logger = LoggerFactory.getLogger(getClass());
55  
56      /** Generate unique ids. */
57      @NotNull
58      private UniqueTicketIdGenerator uniqueTicketIdGenerator = new DefaultUniqueTicketIdGenerator();
59  
60      /** Instance of Apache Commons HttpClient. */
61      @NotNull
62      private HttpClient httpClient;
63  
64      @Override
65      public String handle(final Credential credential, final TicketGrantingTicket proxyGrantingTicketId) {
66          final HttpBasedServiceCredential serviceCredentials = (HttpBasedServiceCredential) credential;
67          final String proxyIou = this.uniqueTicketIdGenerator.getNewTicketId(TicketGrantingTicket.PROXY_GRANTING_TICKET_IOU_PREFIX);
68  
69          final URL callbackUrl = serviceCredentials.getCallbackUrl();
70          final String serviceCredentialsAsString = callbackUrl.toExternalForm();
71          final int bufferLength = serviceCredentialsAsString.length() + proxyIou.length()
72                  + proxyGrantingTicketId.getId().length() + BUFFER_LENGTH_ADDITIONAL_CHARGE;
73          final StringBuilder stringBuffer = new StringBuilder(bufferLength);
74  
75          stringBuffer.append(serviceCredentialsAsString);
76  
77          if (callbackUrl.getQuery() != null) {
78              stringBuffer.append("&");
79          } else {
80              stringBuffer.append("?");
81          }
82  
83          stringBuffer.append(PARAMETER_PROXY_GRANTING_TICKET_IOU);
84          stringBuffer.append("=");
85          stringBuffer.append(proxyIou);
86          stringBuffer.append("&");
87          stringBuffer.append(PARAMETER_PROXY_GRANTING_TICKET_ID);
88          stringBuffer.append("=");
89          stringBuffer.append(proxyGrantingTicketId);
90  
91          if (this.httpClient.isValidEndPoint(stringBuffer.toString())) {
92              logger.debug("Sent ProxyIou of {} for service: {}", proxyIou, serviceCredentials);
93              return proxyIou;
94          }
95  
96          logger.debug("Failed to send ProxyIou of {} for service: {}", proxyIou, serviceCredentials);
97          return null;
98      }
99  
100     /**
101      * @param uniqueTicketIdGenerator The uniqueTicketIdGenerator to set.
102      */
103     public void setUniqueTicketIdGenerator(final UniqueTicketIdGenerator uniqueTicketIdGenerator) {
104         this.uniqueTicketIdGenerator = uniqueTicketIdGenerator;
105     }
106 
107     public void setHttpClient(final HttpClient httpClient) {
108         this.httpClient = httpClient;
109     }
110 
111     @Override
112     public boolean canHandle(final Credential credential) {
113         return true;
114     }
115 }
