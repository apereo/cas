31  /**
32   * Abstract Implementation that handles some of the commonalities between
33   * distributed ticket registries.
34   *
35   * @author Scott Battaglia
36   * @since 3.1
37   */
38  public abstract class AbstractDistributedTicketRegistry extends AbstractTicketRegistry {
39  
40      /**
41       * Update the received ticket.
42       *
43       * @param ticket the ticket
44       */
45      protected abstract void updateTicket(final Ticket ticket);
46  
47      /**
48       * Whether or not a callback to the TGT is required when checking for expiration.
49       *
50       * @return true, if successful
51       */
52      protected abstract boolean needsCallback();
53  
54      /**
55       * Gets the proxied ticket instance.
56       *
57       * @param ticket the ticket
58       * @return the proxied ticket instance
59       */
60      protected final Ticket getProxiedTicketInstance(final Ticket ticket) {
61          if (ticket == null) {
62              return null;
63          }
64  
65          if (ticket instanceof TicketGrantingTicket) {
66              return new TicketGrantingTicketDelegator(this, (TicketGrantingTicket) ticket, needsCallback());
67          }
68  
69          return new ServiceTicketDelegator(this, (ServiceTicket) ticket, needsCallback());
70      }
71  
72      private static class TicketDelegator<T extends Ticket> implements Ticket {
73  
74          private static final long serialVersionUID = 1780193477774123440L;
75  
76          private final AbstractDistributedTicketRegistry ticketRegistry;
77  
78          private final T ticket;
79  
80          private final boolean callback;
81  
82          /**
83           * Instantiates a new ticket delegator.
84           *
85           * @param ticketRegistry the ticket registry
86           * @param ticket the ticket
87           * @param callback the callback
88           */
89          protected TicketDelegator(final AbstractDistributedTicketRegistry ticketRegistry,
90                  final T ticket, final boolean callback) {
91              this.ticketRegistry = ticketRegistry;
92              this.ticket = ticket;
93              this.callback = callback;
94          }
95          
96          
97          /**
98           * Update ticket by the delegated registry.
99           */
100         protected void updateTicket() {
101             this.ticketRegistry.updateTicket(this.ticket);
102         }
103 
104         protected T getTicket() {
105             return this.ticket;
106         }
107 
108         public final String getId() {
109             return this.ticket.getId();
110         }
111 
112         @Override
113         public final boolean isExpired() {
114             if (!callback) {
115                 return this.ticket.isExpired();
116             }
117 
118             final TicketGrantingTicket t = getGrantingTicket();
119 
120             return this.ticket.isExpired() || (t != null && t.isExpired());
121         }
122 
123         @Override
124         public final TicketGrantingTicket getGrantingTicket() {
125             final TicketGrantingTicket old = this.ticket.getGrantingTicket();
126 
127             if (old == null || !callback) {
128                 return old;
129             }
130 
131             return this.ticketRegistry.getTicket(old.getId(), Ticket.class);
132         }
133 
134         public final long getCreationTime() {
135             return this.ticket.getCreationTime();
136         }
137 
138         public final int getCountOfUses() {
139             return this.ticket.getCountOfUses();
140         }
141 
142         @Override
143         public int hashCode() {
144             return this.ticket.hashCode();
145         }
146 
147         @Override
148         public boolean equals(final Object o) {
149             return this.ticket.equals(o);
150         }
151     }

153     private static final class ServiceTicketDelegator extends TicketDelegator<ServiceTicket>
154                            implements ServiceTicket {
155 
156         private static final long serialVersionUID = 8160636219307822967L;
157 
158         /**
159          * Instantiates a new service ticket delegator.
160          *
161          * @param ticketRegistry the ticket registry
162          * @param serviceTicket the service ticket
163          * @param callback the callback
164          */
165         protected ServiceTicketDelegator(final AbstractDistributedTicketRegistry ticketRegistry,
166                 final ServiceTicket serviceTicket, final boolean callback) {
167             super(ticketRegistry, serviceTicket, callback);
168         }
169 
170         @Override
171         public Service getService() {
172             return getTicket().getService();
173         }
174 
175         @Override
176         public boolean isFromNewLogin() {
177             return getTicket().isFromNewLogin();
178         }
179 
180         @Override
181         public boolean isValidFor(final Service service) {
182             final boolean b = this.getTicket().isValidFor(service);
183             updateTicket();
184             return b;
185         }
186 
187         @Override
188         public TicketGrantingTicket grantTicketGrantingTicket(final String id,
189                 final Authentication authentication, final ExpirationPolicy expirationPolicy) {
190             final TicketGrantingTicket t = this.getTicket().grantTicketGrantingTicket(id,
191                     authentication, expirationPolicy);
192             updateTicket();
193             return t;
194         }
195     }

199 
200         private static final long serialVersionUID = 5312560061970601497L;
201 
202         /**
203          * Instantiates a new ticket granting ticket delegator.
204          *
205          * @param ticketRegistry the ticket registry
206          * @param ticketGrantingTicket the ticket granting ticket
207          * @param callback the callback
208          */
209         protected TicketGrantingTicketDelegator(final AbstractDistributedTicketRegistry ticketRegistry,
210                 final TicketGrantingTicket ticketGrantingTicket, final boolean callback) {
211             super(ticketRegistry, ticketGrantingTicket, callback);
212         }
213 
214         @Override
215         public Authentication getAuthentication() {
216             return getTicket().getAuthentication();
217         }
218 
219         @Override
220         public Service getProxiedBy() {
221             return getTicket().getProxiedBy();
222         }
223 
224         @Override
225         public List<Authentication> getSupplementalAuthentications() {
226             return getTicket().getSupplementalAuthentications();
227         }
228 
229         @Override
230         public ServiceTicket grantServiceTicket(final String id, final Service service,
231                 final ExpirationPolicy expirationPolicy, final boolean credentialsProvided) {
232             final ServiceTicket t = this.getTicket().grantServiceTicket(id, service,
233                     expirationPolicy, credentialsProvided);
234             updateTicket();
235             return t;
236         }
237 
238         @Override
239         public void markTicketExpired() {
240             this.getTicket().markTicketExpired();
241             updateTicket();
242         }
243 
244         @Override
245         public boolean isRoot() {
246             return getTicket().isRoot();
247         }
248 
249         @Override
250         public TicketGrantingTicket getRoot() {
251             return getTicket().getRoot();
252         }
253 
254         @Override
255         public List<Authentication> getChainedAuthentications() {
256             return getTicket().getChainedAuthentications();
257         }
258 
259         @Override
260         public Map<String, Service> getServices() {
261             return this.getTicket().getServices();
262         }
263 
264         @Override
265         public void removeAllServices() {
266             this.getTicket().removeAllServices();
267         }
268     }
