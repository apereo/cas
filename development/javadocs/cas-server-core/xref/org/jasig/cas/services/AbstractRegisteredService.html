46  /**
47   * Base class for mutable, persistable registered services.
48   *
49   * @author Marvin S. Addison
50   * @author Scott Battaglia
51   * @author Misagh Moayyed
52   * @since 3.0.0
53   */
54  @Entity
55  @Inheritance
56  @DiscriminatorColumn(name = "expression_type", length = 15, discriminatorType = DiscriminatorType.STRING,
57                       columnDefinition = "VARCHAR(15) DEFAULT 'ant'")
58  @Table(name = "RegisteredServiceImpl")
59  public abstract class AbstractRegisteredService implements RegisteredService, Comparable<RegisteredService> {
60  
61      private static final long serialVersionUID = 7645279151115635245L;
62  
63      /** The logger instance. */
64      @Transient
65      protected final Logger logger = LoggerFactory.getLogger(this.getClass());
66  
67      /** The unique identifier for this service. */
68      @Column(length = 255, updatable = true, insertable = true, nullable = false)
69      protected String serviceId;
70  
71      @Column(length = 255, updatable = true, insertable = true, nullable = false)
72      private String name;
73  
74      @Column(length = 255, updatable = true, insertable = true, nullable = true)
75      private String theme;
76  
77      @Id
78      @GeneratedValue(strategy = GenerationType.AUTO)
79      private long id = RegisteredService.INITIAL_IDENTIFIER_VALUE;
80  
81      @Column(length = 255, updatable = true, insertable = true, nullable = false)
82      private String description;
83  
84      /**
85       * Proxy policy for the service.
86       * By default, the policy is {@link RefuseRegisteredServiceProxyPolicy}.
87       */
88      @Lob
89      @Column(name = "proxy_policy", nullable = true)
90      private RegisteredServiceProxyPolicy proxyPolicy = new RefuseRegisteredServiceProxyPolicy();
91  
92      @Column(name = "evaluation_order", nullable = false)
93      private int evaluationOrder;
94  
95      /**
96       * Resolve the username for this service.
97       * By default the resolver is {@link DefaultRegisteredServiceUsernameProvider}.
98       */
99      @Lob
100     @Column(name = "username_attr", nullable = true)
101     private RegisteredServiceUsernameAttributeProvider usernameAttributeProvider =
102         new DefaultRegisteredServiceUsernameProvider();
103 
104     /**
105      * The logout type of the service. 
106      * The default logout type is the back channel one.
107      */
108     @Column(name = "logout_type", nullable = true)
109     private LogoutType logoutType = LogoutType.BACK_CHANNEL;
110 
111     @Lob
112     @Column(name = "required_handlers")
113     private HashSet<String> requiredHandlers = new HashSet<>();
114 
115     /** The attribute filtering policy. */
116     @Lob
117     @Column(name = "attribute_release", nullable = true)
118     private AttributeReleasePolicy attributeReleasePolicy = new ReturnAllowedAttributeReleasePolicy();
119 
120     @Column(name = "logo")
121     private URL logo;
122 
123     @Column(name = "logout_url")
124     private URL logoutUrl;
125 
126     @Lob
127     @Column(name = "access_strategy", nullable = true)
128     private RegisteredServiceAccessStrategy accessStrategy =
129             new DefaultRegisteredServiceAccessStrategy();
130 
131     @Lob
132     @Column(name = "public_key", nullable = true)
133     private RegisteredServicePublicKey publicKey;

175     /**
176      * Initializes the registered service with default values
177      * for fields that are unspecified. Only triggered by JPA.
178      * @since 4.1
179      */
180     @PostLoad
181     public final void postLoad() {
182         if (this.proxyPolicy == null) {
183             this.proxyPolicy = new RefuseRegisteredServiceProxyPolicy();
184         }
185         if (this.usernameAttributeProvider == null) {
186             this.usernameAttributeProvider = new DefaultRegisteredServiceUsernameProvider();
187         }
188         if (this.logoutType == null) {
189             this.logoutType = LogoutType.BACK_CHANNEL;
190         }
191         if (this.requiredHandlers == null) {
192             this.requiredHandlers = new HashSet<>();
193         }
194         if (this.accessStrategy == null) {
195             this.accessStrategy = new DefaultRegisteredServiceAccessStrategy();
196         }
197         if (this.attributeReleasePolicy == null) {
198             this.attributeReleasePolicy = new ReturnAllowedAttributeReleasePolicy();
199         }
200     }

202     @Override
203     public boolean equals(final Object o) {
204         if (o == null) {
205             return false;
206         }
207 
208         if (this == o) {
209             return true;
210         }
211 
212         if (!(o instanceof AbstractRegisteredService)) {
213             return false;
214         }
215 
216         final AbstractRegisteredService that = (AbstractRegisteredService) o;
217 
218         return new EqualsBuilder()
219                 .append(this.proxyPolicy, that.proxyPolicy)
220                 .append(this.evaluationOrder, that.evaluationOrder)
221                 .append(this.description, that.description)
222                 .append(this.name, that.name)
223                 .append(this.serviceId, that.serviceId)
224                 .append(this.theme, that.theme)
225                 .append(this.usernameAttributeProvider, that.usernameAttributeProvider)
226                 .append(this.logoutType, that.logoutType)
227                 .append(this.attributeReleasePolicy, that.attributeReleasePolicy)
228                 .append(this.accessStrategy, that.accessStrategy)
229                 .append(this.logo, that.logo)
230                 .append(this.publicKey, that.publicKey)
231                 .isEquals();
232     }

237                 .append(this.description)
238                 .append(this.serviceId)
239                 .append(this.name)
240                 .append(this.theme)
241                 .append(this.evaluationOrder)
242                 .append(this.usernameAttributeProvider)
243                 .append(this.accessStrategy)
244                 .append(this.logoutType)
245                 .append(this.attributeReleasePolicy)
246                 .append(this.accessStrategy)
247                 .append(this.logo)
248                 .append(this.publicKey)
249                 .toHashCode();
250     }

333     /**
334      * Copies the properties of the source service into this instance.
335      *
336      * @param source Source service from which to copy properties.
337      */
338     public void copyFrom(final RegisteredService source) {
339         this.setId(source.getId());
340         this.setProxyPolicy(source.getProxyPolicy());
341         this.setDescription(source.getDescription());
342         this.setName(source.getName());
343         this.setServiceId(source.getServiceId());
344         this.setTheme(source.getTheme());
345         this.setEvaluationOrder(source.getEvaluationOrder());
346         this.setUsernameAttributeProvider(source.getUsernameAttributeProvider());
347         this.setLogoutType(source.getLogoutType());
348         this.setAttributeReleasePolicy(source.getAttributeReleasePolicy());
349         this.setAccessStrategy(source.getAccessStrategy());
350         this.setLogo(source.getLogo());
351         this.setPublicKey(source.getPublicKey());
352     }

354     /**
355      * {@inheritDoc}
356      * Compares this instance with the <code>other</code> registered service based on
357      * evaluation order, name. The name comparison is case insensitive.
358      *
359      * @see #getEvaluationOrder()
360      */
361     @Override
362     public int compareTo(final RegisteredService other) {
363         return new CompareToBuilder()
364                   .append(this.getEvaluationOrder(), other.getEvaluationOrder())
365                   .append(this.getName().toLowerCase(), other.getName().toLowerCase())
366                   .append(this.getServiceId(), other.getServiceId())
367                   .toComparison();
368     }

370     @Override
371     public String toString() {
372         final ToStringBuilder toStringBuilder = new ToStringBuilder(null, ToStringStyle.SHORT_PREFIX_STYLE);
373         toStringBuilder.append("id", this.id);
374         toStringBuilder.append("name", this.name);
375         toStringBuilder.append("description", this.description);
376         toStringBuilder.append("serviceId", this.serviceId);
377         toStringBuilder.append("usernameAttributeProvider", this.usernameAttributeProvider);
378         toStringBuilder.append("theme", this.theme);
379         toStringBuilder.append("evaluationOrder", this.evaluationOrder);
380         toStringBuilder.append("logoutType", this.logoutType);
381         toStringBuilder.append("attributeReleasePolicy", this.attributeReleasePolicy);
382         toStringBuilder.append("accessStrategy", this.accessStrategy);
383         toStringBuilder.append("publicKey", this.publicKey);
384         toStringBuilder.append("proxyPolicy", this.proxyPolicy);
385         toStringBuilder.append("logo", this.logo);
386 
387         return toStringBuilder.toString();
388     }

397     @Override

399         if (this.requiredHandlers == null) {
400             this.requiredHandlers = new HashSet<>();
401         }
402         return this.requiredHandlers;
403     }

405     /**
406      * Sets the required handlers for this service.
407      *
408      * @param handlers the new required handlers
409      */
410     public void setRequiredHandlers(final Set<String> handlers) {
411         getRequiredHandlers().clear();
412         if (handlers == null) {
413             return;
414         }
415         for (final String handler : handlers) {
416             getRequiredHandlers().add(handler);
417         }
418     }
