28  import org.slf4j.LoggerFactory;

30  import java.util.Collection;
31  import java.util.Collections;

36  /**
37   * This is {@link DefaultRegisteredServiceAccessStrategy}
38   * that allows the following rules:
39   *
40   * <ul>
41   *     <li>A service may be disallowed to use CAS for authentication</li>
42   *     <li>A service may be disallowed to take part in CAS single sign-on such that
43   *     presentation of credentials would always be required.</li>
44   *     <li>A service may be prohibited from receiving a service ticket
45   *     if the existing principal attributes don't contain the required attributes
46   *     that otherwise grant access to the service.</li>
47   * </ul>
48   *
49   * @author Misagh Moayyed mmoayyed@unicon.net
50   * @since 4.1
51   */
52  public class DefaultRegisteredServiceAccessStrategy implements RegisteredServiceAccessStrategy {
53  
54      private static final long serialVersionUID = 1245279151345635245L;
55  
56      private final Logger logger = LoggerFactory.getLogger(this.getClass());
57  
58      /** Is the service allowed at all? **/
59      private boolean enabled = true;
60  
61      /** Is the service allowed to use SSO? **/
62      private boolean ssoEnabled = true;
63  
64      /**
65       * Defines the attribute aggregation behavior when checking for required attributes.
66       * Default requires that all attributes be present and match the principal's.
67       */
68      private boolean requireAllAttributes = true;
69  
70      /**
71       * Collection of required attributes
72       * for this service to proceed.
73       */
74      private Map<String, Set<String>> requiredAttributes = new HashMap<>();
75  
76      /**
77       * Instantiates a new Default registered service authorization strategy.
78       * By default, rules indicate that services are both enabled
79       * and can participate in SSO.
80       */
81      public DefaultRegisteredServiceAccessStrategy() {
82          this(true, true);
83      }
84  
85      /**
86       * Instantiates a new Default registered service authorization strategy.
87       *
88       * @param enabled the enabled
89       * @param ssoEnabled the sso enabled
90       */
91      public DefaultRegisteredServiceAccessStrategy(final boolean enabled, final boolean ssoEnabled) {
92          this.enabled = enabled;
93          this.ssoEnabled = ssoEnabled;
94      }
95  
96      public final void setEnabled(final boolean enabled) {
97          this.enabled = enabled;
98      }
99  
100     /**
101      * Set to enable/authorize this service.
102      * @param ssoEnabled true to enable service
103      */
104     public final void setSsoEnabled(final boolean ssoEnabled) {
105         this.ssoEnabled = ssoEnabled;
106     }
107 
108     public boolean isEnabled() {
109         return this.enabled;
110     }
111 
112     public boolean isSsoEnabled() {
113         return this.ssoEnabled;
114     }
115 
116     /**
117      * Defines the attribute aggregation when checking for required attributes.
118      * Default requires that all attributes be present and match the principal's.
119      * @param requireAllAttributes the require all attributes
120      */
121     public final void setRequireAllAttributes(final boolean requireAllAttributes) {
122         this.requireAllAttributes = requireAllAttributes;
123     }
124 
125     public final boolean isRequireAllAttributes() {
126         return this.requireAllAttributes;
127     }
128 
129     public Map<String, Set<String>> getRequiredAttributes() {
130         return ImmutableMap.copyOf(this.requiredAttributes);
131     }
132 
133     /**
134      * Defines the required attribute names and values that
135      * must be available to the principal before the flow
136      * can proceed to the next step. Every attribute in
137      * the map can be linked to multiple values.
138      *
139      * @param requiredAttributes the required attributes
140      */
141     public final void setRequiredAttributes(final Map<String, Set<String>> requiredAttributes) {
142         this.requiredAttributes = requiredAttributes;
143     }
144 
145     /**
146      * {@inheritDoc}
147      *
148      * Verify presence of service required attributes.
149      * <ul>
150      *     <li>If no required attributes are specified, authz is granted.</li>
151      *     <li>If ALL required attributes must be present, and the principal contains all and there is
152      *     at least one attribute value that matches the required, authz is granted.</li>
153      *     <li>If ALL required attributes don't have to be present, and there is at least
154      *     one principal attribute present whose value matches the required, authz is granted.</li>
155      *     <li>Otherwise, access is denied</li>
156      * </ul>
157      * Note that comparison of principal/required attributes is case-sensitive. Exact matches are required
158      * for any individual attribute value.
159      */
160     @Override
161     public boolean doPrincipalAttributesAllowServiceAccess(final Map<String, Object> principalAttributes) {
162         if (this.requiredAttributes.isEmpty()) {
163             logger.debug("No required attributes are specified");
164             return true;
165         }
166         if (principalAttributes.isEmpty()) {
167             logger.debug("No principal attributes are found to satisfy attribute requirements");
168             return false;
169         }
170 
171         if (principalAttributes.size() < this.requiredAttributes.size()) {
172             logger.debug("The size of the principal attributes that are [{}] does not match requirements, "
173                     + "which means the principal is not carrying enough data to grant authorization",
174                     principalAttributes);
175             return false;
176         }
177 
178         final Map<String, Set<String>> requiredAttrs = this.getRequiredAttributes();
179         logger.debug("These required attributes [{}] are examined against [{}] before service can proceed.",
180                 requiredAttrs, principalAttributes);
181 
182         final Sets.SetView<String> difference = Sets.intersection(requiredAttrs.keySet(), principalAttributes.keySet());
183         final Set<String> copy = difference.immutableCopy();
184 
185         if (this.requireAllAttributes && copy.size() < this.requiredAttributes.size()) {
186             logger.debug("Not all required attributes are available to the principal");
187             return false;
188         }
189 
190         for (final String key : copy) {
191             final Set<?> requiredValues = this.requiredAttributes.get(key);
192             final Set<?> availableValues;
193 
194             final Object objVal = principalAttributes.get(key);
195             if (objVal instanceof Collection) {
196                 final Collection valCol = (Collection) objVal;
197                 availableValues = Sets.newHashSet(valCol.toArray());
198             } else {
199                 availableValues = Collections.singleton(objVal);
200             }
201 
202             final Sets.SetView<?> differenceInValues = Sets.intersection(availableValues, requiredValues);
203             if (differenceInValues.size() > 0) {
204                 logger.info("Principal is authorized to access the service");
205                 return true;
206             }
207         }
208         logger.info("Principal is denied access as the required attributes for the registered service are missing");
209         return false;
210     }
211 
212     @Override
213     public boolean isServiceAccessAllowedForSso() {
214         if (!this.ssoEnabled) {
215             logger.trace("Service is not authorized to participate in SSO.");
216         }
217         return this.ssoEnabled;
218     }
219 
220     @Override
221     public boolean isServiceAccessAllowed() {
222         if (!this.enabled) {
223             logger.trace("Service is not enabled in service registry.");
224         }
225         return this.enabled;
226     }
227 
228 
229     @Override
230     public boolean equals(final Object obj) {
231         if (obj == null) {
232             return false;
233         }
234         if (obj == this) {
235             return true;
236         }
237         if (obj.getClass() != getClass()) {
238             return false;
239         }
240         final DefaultRegisteredServiceAccessStrategy rhs = (DefaultRegisteredServiceAccessStrategy) obj;
241         return new EqualsBuilder()
242                 .append(this.enabled, rhs.enabled)
243                 .append(this.ssoEnabled, rhs.ssoEnabled)
244                 .append(this.requireAllAttributes, rhs.requireAllAttributes)
245                 .append(this.requiredAttributes, rhs.requiredAttributes)
246                 .isEquals();
247     }
248 
249     @Override
250     public int hashCode() {
251         return new HashCodeBuilder()
252                 .append(this.enabled)
253                 .append(this.ssoEnabled)
254                 .append(this.requireAllAttributes)
255                 .append(this.requiredAttributes)
256                 .toHashCode();
257     }
258 
259 
260     @Override
261     public String toString() {
262         return new ToStringBuilder(this)
263                 .append("enabled", enabled)
264                 .append("ssoEnabled", ssoEnabled)
265                 .append("requireAllAttributes", requireAllAttributes)
266                 .append("requiredAttributes", requiredAttributes)
267                 .toString();
268     }
