174     @Override
175     public SimpleHttpClient getObject() throws Exception {
176 
177         final CloseableHttpClient httpClient = buildHttpClient();
178 
179         final FutureRequestExecutionService requestExecutorService = buildRequestExecutorService(httpClient);
180 
181         return new SimpleHttpClient(this.acceptableCodes, httpClient, requestExecutorService);
182     }

194     /**
195      * Build a HTTP client based on the current properties.
196      *
197      * @return the built HTTP client
198      */
199     private CloseableHttpClient buildHttpClient() {
200         try {
201 
202             final ConnectionSocketFactory plainsf = PlainConnectionSocketFactory.getSocketFactory();
203             final LayeredConnectionSocketFactory sslsf = this.sslSocketFactory;
204 
205             final Registry<ConnectionSocketFactory> registry = RegistryBuilder.<ConnectionSocketFactory>create()
206                     .register("http", plainsf)
207                     .register("https", sslsf)
208                     .build();
209 
210             final PoolingHttpClientConnectionManager connMgmr = new PoolingHttpClientConnectionManager(registry);
211             connMgmr.setMaxTotal(this.maxPooledConnections);
212             connMgmr.setDefaultMaxPerRoute(this.maxConnectionsPerRoute);
213 
214             final HttpHost httpHost = new HttpHost(InetAddress.getLocalHost());
215             final HttpRoute httpRoute = new HttpRoute(httpHost);
216             connMgmr.setMaxPerRoute(httpRoute, MAX_CONNECTIONS_PER_ROUTE);
217     
218             final RequestConfig requestConfig = RequestConfig.custom()
219                     .setSocketTimeout(this.readTimeout)
220                     .setConnectTimeout(this.connectionTimeout)
221                     .setConnectionRequestTimeout(this.connectionTimeout)
222                     .setStaleConnectionCheckEnabled(true)
223                     .setCircularRedirectsAllowed(this.circularRedirectsAllowed)
224                     .setRedirectsEnabled(this.redirectsEnabled)
225                     .setAuthenticationEnabled(this.authenticationEnabled)
226                     .build();
227 
228 
229             final HttpClientBuilder builder = HttpClients.custom()
230                     .setConnectionManager(connMgmr)
231                     .setDefaultRequestConfig(requestConfig)
232                     .setSSLSocketFactory(sslsf)
233                     .setSSLHostnameVerifier(this.hostnameVerifier)
234                     .setRedirectStrategy(this.redirectionStrategy)
235                     .setDefaultCredentialsProvider(this.credentialsProvider)
236                     .setDefaultCookieStore(this.cookieStore)
237                     .setConnectionReuseStrategy(this.connectionReuseStrategy)
238                     .setConnectionBackoffStrategy(this.connectionBackoffStrategy)
239                     .setServiceUnavailableRetryStrategy(this.serviceUnavailableRetryStrategy)
240                     .setProxyAuthenticationStrategy(this.proxyAuthenticationStrategy)
241                     .setDefaultHeaders(this.defaultHeaders)
242                     .useSystemProperties();
243 
244             return builder.build();
245 
246         } catch (final Exception e) {
247             LOGGER.error(e.getMessage(), e);
248             throw new RuntimeException(e);
249         }
250     }

252     /**
253      * Build a {@link FutureRequestExecutionService} from the current properties and a HTTP client.
254      *
255      * @param httpClient the provided HTTP client
256      * @return the built request executor service
257      */
258     private FutureRequestExecutionService buildRequestExecutorService(final CloseableHttpClient httpClient) {
259 
260         final ExecutorService definedExecutorService;
261         // no executor service provided -> create a default one
262         if (this.executorService == null) {
263             definedExecutorService = new ThreadPoolExecutor(this.threadsNumber, this.threadsNumber,
264                     0L, TimeUnit.MILLISECONDS, new LinkedBlockingQueue<Runnable>(this.queueSize));
265         } else {
266             definedExecutorService = this.executorService;
267         }
268 
269         return new FutureRequestExecutionService(httpClient, definedExecutorService);
270     }
