19  package org.jasig.cas.logout;
20  
21  import org.jasig.cas.util.DefaultUniqueTicketIdGenerator;
22  import org.jasig.cas.util.ISOStandardDateFormat;
23  import org.jasig.cas.util.UniqueTicketIdGenerator;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  /**
28   * A builder that uses the saml standard's <code>LogoutRequest</code> template in order
29   * to build the logout request.
30   * @author Misagh Moayyed
31   * @see LogoutRequest
32   * @since 4.0.0
33   */
34  public final class SamlCompliantLogoutMessageCreator implements LogoutMessageCreator {
35  
36      /** The logger. */
37      private static final Logger LOGGER = LoggerFactory.getLogger(SamlCompliantLogoutMessageCreator.class);
38      
39      /** A ticket Id generator. */
40      private static final UniqueTicketIdGenerator GENERATOR = new DefaultUniqueTicketIdGenerator();
41  
42      /** The logout request template. */
43      private static final String LOGOUT_REQUEST_TEMPLATE =
44              "<samlp:LogoutRequest xmlns:samlp=\"urn:oasis:names:tc:SAML:2.0:protocol\" ID=\"%s\" Version=\"2.0\" "
45              + "IssueInstant=\"%s\"><saml:NameID xmlns:saml=\"urn:oasis:names:tc:SAML:2.0:assertion\">@NOT_USED@"
46              + "</saml:NameID><samlp:SessionIndex>%s</samlp:SessionIndex></samlp:LogoutRequest>";
47  
48      @Override
49      public String create(final LogoutRequest request) {
50          final String logoutRequest = String.format(LOGOUT_REQUEST_TEMPLATE, GENERATOR.getNewTicketId("LR"),
51                  new ISOStandardDateFormat().getCurrentDateAndTime(), request.getTicketId());
52          
53          LOGGER.debug("Generated logout message: [{}]", logoutRequest);
54          return logoutRequest;
55      }
56      
57  }
