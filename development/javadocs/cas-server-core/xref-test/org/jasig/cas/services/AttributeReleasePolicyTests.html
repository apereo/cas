39      @Test
40      public void testAttributeFilterMappedAttributes() {
41          final ReturnMappedAttributeReleasePolicy policy = new ReturnMappedAttributeReleasePolicy();
42          final Map<String, String> mappedAttr = new HashMap<String, String>();
43          mappedAttr.put("attr1", "newAttr1");
44          
45          policy.setAllowedAttributes(mappedAttr);
46                  
47          final Principal p = mock(Principal.class);
48          
49          final Map<String, Object> map = new HashMap<String, Object>();
50          map.put("attr1", "value1");
51          map.put("attr2", "value2");
52          map.put("attr3", Arrays.asList("v3", "v4"));
53          
54          when(p.getAttributes()).thenReturn(map);
55          when(p.getId()).thenReturn("principalId");
56          
57          final Map<String, Object> attr = policy.getAttributes(p);
58          assertEquals(attr.size(), 1);
59          assertTrue(attr.containsKey("newAttr1"));
60          
61          final byte[] data = SerializationUtils.serialize(policy);
62          final ReturnMappedAttributeReleasePolicy p2 = (ReturnMappedAttributeReleasePolicy) SerializationUtils.deserialize(data);
63          assertNotNull(p2);
64          assertEquals(p2.getAllowedAttributes(), policy.getAllowedAttributes());
65      }
