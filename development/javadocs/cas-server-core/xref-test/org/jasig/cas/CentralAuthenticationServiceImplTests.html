437     /**
438      * This test simulates :
439      * - a first authentication for a default service
440      * - a second authentication with the renew parameter and the same service (and same credentials)
441      * - a validation of the second ticket.
442      * 
443      * When supplemental authentications were returned with the chained authentications, the validation specification
444      * failed as it only expects one authentication. Thus supplemental authentications should not be returned in the
445      * chained authentications. Both concepts are orthogonal.
446      *  
447      * @throws org.jasig.cas.ticket.TicketException
448      * @throws AuthenticationException
449      */
450     @Test
451     public void authenticateTwiceWithRenew() throws TicketException, AuthenticationException {
452         final CentralAuthenticationService cas = getCentralAuthenticationService();
453         final Service svc = TestUtils.getService("testDefault");
454         final UsernamePasswordCredential goodCredential = TestUtils.getCredentialsWithSameUsernameAndPassword();
455         final TicketGrantingTicket tgtId = cas.createTicketGrantingTicket(goodCredential);
456         cas.grantServiceTicket(tgtId.getId(), svc);
457         // simulate renew with new good same credentials
458         final ServiceTicket st2Id = cas.grantServiceTicket(tgtId.getId(), svc, goodCredential);
459         final Assertion assertion = cas.validateServiceTicket(st2Id.getId(), svc);
460         final ValidationSpecification validationSpecification = new Cas20WithoutProxyingValidationSpecification();
461         assertTrue(validationSpecification.isSatisfiedBy(assertion));
462     }

464     /**
465      * This test checks that the TGT destruction happens properly for a remote registry.
466      * It previously failed when the deletion happens before the ticket was marked expired because an update was necessary for that.
467      *
468      * @throws AuthenticationException
469      * @throws org.jasig.cas.ticket.TicketException
470      */
471     @Test
472     public void verifyDestroyRemoteRegistry() throws TicketException, AuthenticationException {
473         final MockOnlyOneTicketRegistry registry = new MockOnlyOneTicketRegistry();
474         final TicketGrantingTicketImpl tgt = new TicketGrantingTicketImpl("TGT-1", mock(Authentication.class),
475                 mock(ExpirationPolicy.class));
476         final MockExpireUpdateTicketLogoutManager logoutManager = new MockExpireUpdateTicketLogoutManager(registry);
477         // consider authentication has happened and the TGT is in the registry
478         registry.addTicket(tgt);
479         // create a new CASimpl
480         final CentralAuthenticationServiceImpl cas = new CentralAuthenticationServiceImpl(registry,  null,  null, null, null, null, null,
481                 null, logoutManager);
482         // destroy to mark expired and then delete : the opposite would fail with a "No ticket to update" error from the registry
483         cas.destroyTicketGrantingTicket(tgt.getId());
484     }
