40  /**
41   * This is test cases for {@link DefaultCasAttributeEncoder}.
42   *
43   * @author Misagh Moayyed
44   * @since 4.1
45   */
46  @RunWith(SpringJUnit4ClassRunner.class)
47  @ContextConfiguration(locations= {"/core-context.xml", "/WEB-INF/cas-servlet.xml"})
48  public class DefaultCasAttributeEncoderTests {
49  
50      private Map<String, Object> attributes;
51  
52      @Autowired
53      private ServicesManager servicesManager;
54  
55      @Before
56      public void before() {
57          this.attributes = new HashMap<>();
58          for (int i = 0; i < 3; i++) {
59              this.attributes.put("attr" + i, newSingleAttribute("value" + i));
60          }
61          this.attributes.put(CasViewConstants.MODEL_ATTRIBUTE_NAME_PROXY_GRANTING_TICKET, newSingleAttribute("PGT-1234567"));
62          this.attributes.put(CasViewConstants.MODEL_ATTRIBUTE_NAME_PRINCIPAL_CREDENTIAL, newSingleAttribute("PrincipalPassword"));
63      }
64  
65      private Collection<String> newSingleAttribute(final String attr) {
66          return Collections.singleton(attr);
67      }
68  
69      @Test
70      public void checkNoPublicKeyDefined() {
71          final Service service = TestUtils.getService("testDefault");
72          final CasAttributeEncoder encoder = new DefaultCasAttributeEncoder(this.servicesManager);
73          final Map<String, Object> encoded = encoder.encodeAttributes(this.attributes, service);
74          assertEquals(encoded.size(), this.attributes.size() - 2);
75      }
76  
77      @Test
78      public void checkAttributesEncodedCorrectly() {
79          final Service service = TestUtils.getService("testencryption");
80          final CasAttributeEncoder encoder = new DefaultCasAttributeEncoder(this.servicesManager);
81          final Map<String, Object> encoded = encoder.encodeAttributes(this.attributes, service);
82          assertEquals(encoded.size(), this.attributes.size());
83          checkEncryptedValues(CasViewConstants.MODEL_ATTRIBUTE_NAME_PRINCIPAL_CREDENTIAL, encoded);
84          checkEncryptedValues(CasViewConstants.MODEL_ATTRIBUTE_NAME_PROXY_GRANTING_TICKET, encoded);
85      }
86  
87      private void checkEncryptedValues(final String name, final Map<String, Object> encoded) {
88          final String v1 = ((Collection<?>) this.attributes.get(
89                  name)).iterator().next().toString();
90          final String v2 = (String) encoded.get(name);
91          assertNotEquals(v1, v2);
92      }
93  }
