36  /**
37   * High-level MFA functionality tests that leverage registered service metadata
38   * ala {@link org.jasig.cas.authentication.RequiredHandlerAuthenticationPolicyFactory} to drive
39   * authentication policy.
40   *
41   * @author Marvin S. Addison
42   * @since 4.0
43   */
44  @RunWith(SpringJUnit4ClassRunner.class)
45  @ContextConfiguration(locations = { "/mfa-test-context.xml" })
46  public class MultifactorAuthenticationTests {
47  
48      @Autowired
49      private CentralAuthenticationService cas;
50  
51      @Test
52      public void testAllowsAccessToNormalSecurityServiceWithPassword() throws Exception {
53          final String tgt = cas.createTicketGrantingTicket(newUserPassCredentials("alice", "alice"));
54          assertNotNull(tgt);
55          final String st = cas.grantServiceTicket(tgt, newService("https://example.com/normal/"));
56          assertNotNull(st);
57      }
58  
59      @Test
60      public void testAllowsAccessToNormalSecurityServiceWithOTP() throws Exception {
61          final String tgt = cas.createTicketGrantingTicket(new OneTimePasswordCredential("alice", "31415"));
62          assertNotNull(tgt);
63          final String st = cas.grantServiceTicket(tgt, newService("https://example.com/normal/"));
64          assertNotNull(st);
65      }
66  
67      @Test(expected = UnsatisfiedAuthenticationPolicyException.class)
68      public void testDeniesAccessToHighSecurityServiceWithPassword() throws Exception {
69          final String tgt = cas.createTicketGrantingTicket(newUserPassCredentials("alice", "alice"));
70          assertNotNull(tgt);
71          cas.grantServiceTicket(tgt, newService("https://example.com/high/"));
72      }
73  
74      @Test(expected = UnsatisfiedAuthenticationPolicyException.class)
75      public void testDeniesAccessToHighSecurityServiceWithOTP() throws Exception {
76          final String tgt = cas.createTicketGrantingTicket(new OneTimePasswordCredential("alice", "31415"));
77          assertNotNull(tgt);
78          final String st = cas.grantServiceTicket(tgt, newService("https://example.com/high/"));
79          assertNotNull(st);
80      }
81  
82      @Test
83      public void testAllowsAccessToHighSecurityServiceWithPasswordAndOTP() throws Exception {
84          final String tgt = cas.createTicketGrantingTicket(
85                  newUserPassCredentials("alice", "alice"),
86                  new OneTimePasswordCredential("alice", "31415"));
87          assertNotNull(tgt);
88          final String st = cas.grantServiceTicket(tgt, newService("https://example.com/high/"));
89          assertNotNull(st);
90      }
91  
92      @Test
93      public void testAllowsAccessToHighSecurityServiceWithPasswordAndOTPViaRenew() throws Exception {
94          // Note the original credential used to start SSO session does not satisfy security policy
95          final String tgt = cas.createTicketGrantingTicket(newUserPassCredentials("alice", "alice"));
96          assertNotNull(tgt);
97          final Service service = newService("https://example.com/high/");
98          final String st = cas.grantServiceTicket(
99                  tgt,
100                 service,
101                 newUserPassCredentials("alice", "alice"),
102                 new OneTimePasswordCredential("alice", "31415"));
103         assertNotNull(st);
104         // Confirm the authentication in the assertion is the one that satisfies security policy
105         final Assertion assertion = cas.validateServiceTicket(st, service);
106         assertEquals(2, assertion.getPrimaryAuthentication().getSuccesses().size());
107         assertTrue(assertion.getPrimaryAuthentication().getSuccesses().containsKey("passwordHandler"));
108         assertTrue(assertion.getPrimaryAuthentication().getSuccesses().containsKey("oneTimePasswordHandler"));
109         assertTrue(assertion.getPrimaryAuthentication().getAttributes().containsKey(
110                 SuccessfulHandlerMetaDataPopulator.SUCCESSFUL_AUTHENTICATION_HANDLERS));
111     }
112 
113     private static UsernamePasswordCredential newUserPassCredentials(final String user, final String pass) {
114         final UsernamePasswordCredential userpass = new UsernamePasswordCredential();
115         userpass.setUsername(user);
116         userpass.setPassword(pass);
117         return userpass;
118     }
119 
120     private static Service newService(final String id) {
121         return new SimpleWebApplicationServiceImpl(id);
122     }
