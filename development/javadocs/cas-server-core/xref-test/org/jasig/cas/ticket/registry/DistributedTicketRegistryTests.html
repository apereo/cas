54      @Test
55      public void testProxiedInstancesEqual() {
56          final TicketGrantingTicket t = new TicketGrantingTicketImpl("test", TestUtils.getAuthentication(),
57                  new NeverExpiresExpirationPolicy());
58          this.ticketRegistry.addTicket(t);
59  
60          final TicketGrantingTicket returned = (TicketGrantingTicket) this.ticketRegistry.getTicket("test");
61          assertEquals(t, returned);
62          assertEquals(returned, t);
63  
64          assertEquals(t.getCreationTime(), returned.getCreationTime());
65          assertEquals(t.getAuthentication(), returned.getAuthentication());
66          assertEquals(t.getCountOfUses(), returned.getCountOfUses());
67          assertEquals(t.getGrantingTicket(), returned.getGrantingTicket());
68          assertEquals(t.getId(), returned.getId());
69          assertEquals(t.getChainedAuthentications(), returned.getChainedAuthentications());
70          assertEquals(t.isExpired(), returned.isExpired());
71          assertEquals(t.isRoot(), returned.isRoot());
72  
73          final ServiceTicket s = t.grantServiceTicket("stest", TestUtils.getService(),
74                  new NeverExpiresExpirationPolicy(), false);
75          this.ticketRegistry.addTicket(s);
76  
77          final ServiceTicket sreturned = (ServiceTicket) this.ticketRegistry.getTicket("stest");
78          assertEquals(s, sreturned);
79          assertEquals(sreturned, s);
80  
81          assertEquals(s.getCreationTime(), sreturned.getCreationTime());
82          assertEquals(s.getCountOfUses(), sreturned.getCountOfUses());
83          assertEquals(s.getGrantingTicket(), sreturned.getGrantingTicket());
84          assertEquals(s.getId(), sreturned.getId());
85          assertEquals(s.isExpired(), sreturned.isExpired());
86          assertEquals(s.getService(), sreturned.getService());
87          assertEquals(s.isFromNewLogin(), sreturned.isFromNewLogin());
88      }

90      @Test
91      public void testUpdateOfRegistry() {
92          final TicketGrantingTicket t = new TicketGrantingTicketImpl("test", TestUtils.getAuthentication(),
93                  new NeverExpiresExpirationPolicy());
94          this.ticketRegistry.addTicket(t);
95          final TicketGrantingTicket returned = (TicketGrantingTicket) this.ticketRegistry.getTicket("test");
96  
97          final ServiceTicket s = returned.grantServiceTicket("test2", TestUtils.getService(),
98                  new NeverExpiresExpirationPolicy(), true);
99  
100         this.ticketRegistry.addTicket(s);
101         final ServiceTicket s2 = (ServiceTicket) this.ticketRegistry.getTicket("test2");
102         assertNotNull(s2.grantTicketGrantingTicket("ff", TestUtils.getAuthentication(),
103                 new NeverExpiresExpirationPolicy()));
104 
105         assertTrue(s2.isValidFor(TestUtils.getService()));
106         assertTrue(this.wasTicketUpdated);
107 
108         returned.markTicketExpired();
109         assertTrue(t.isExpired());
110     }

117     protected class TestDistributedTicketRegistry extends AbstractDistributedTicketRegistry {

119         private Map<String, Ticket> tickets = new HashMap<String, Ticket>();

125         public void addTicket(final Ticket ticket) {
126             this.tickets.put(ticket.getId(), ticket);
127         }
128 
129         public boolean deleteTicket(final String ticketId) {
130             return this.tickets.remove(ticketId) != null;
131         }
132 
133         public Ticket getTicket(final String ticketId) {
134             return getProxiedTicketInstance(this.tickets.get(ticketId));
135         }
136
