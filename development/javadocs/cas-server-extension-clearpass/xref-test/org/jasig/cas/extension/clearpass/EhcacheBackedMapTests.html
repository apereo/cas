20  package org.jasig.cas.extension.clearpass;
21  
22  import net.sf.ehcache.Cache;
23  import net.sf.ehcache.CacheManager;
24  import org.apache.commons.lang3.ArrayUtils;
25  import org.junit.After;
26  import org.junit.Before;
27  import org.junit.Test;
28  
29  import java.util.Map;
30  
31  import static org.junit.Assert.*;
32  
33  /**
34   * This is tests for
35   * {@link org.jasig.cas.extension.clearpass.EhcacheBackedMap}.
36   * @author Misagh Moayyed mmoayyed@unicon.net
37   * @since 3.5.0
38   */
39  public class EhcacheBackedMapTests {
40  
41      private CacheManager manager;
42      private Cache cache;
43      private EhcacheBackedMap map;
44  
45      @Before
46      public void prep() {
47          this.manager = CacheManager.create();
48          this.manager.addCache("cascache");
49          this.cache = this.manager.getCache("cascache");
50          this.map = new EhcacheBackedMap(this.cache);
51      }
52  
53      @After
54      public void shutdown() {
55          this.manager.shutdown();
56      }
57  
58      @Test
59      public void verifyEmptyMapSize() {
60          assertEquals(map.size(), 0);
61          assertTrue(map.isEmpty());
62      }
63  
64      @Test
65      public void verifyGetPutOps() {
66          this.map.put("key", "value");
67          assertNotNull(this.map.get("key"));
68  
69          this.map.remove("key");
70          assertTrue(map.isEmpty());
71      }
72  
73      @Test
74      public void verifyPutClear() {
75          final String[][] arrayItems = {{"key0", "Item0"}, {"key1", "Item1"}, {"key2", "Item2"}};
76          final Map mapItems = ArrayUtils.toMap(arrayItems);
77  
78          this.map.putAll(mapItems);
79          assertEquals(map.size(), 3);
80  
81          this.map.clear();
82          assertTrue(map.isEmpty());
83      }
84  
85      @Test
86      public void verifyKeysValues() {
87          final String[][] arrayItems = {{"key0", "Item0"}, {"key1", "Item1"}, {"key2", "Item2"}};
88          final Map mapItems = ArrayUtils.toMap(arrayItems);
89  
90          this.map.putAll(mapItems);
91          assertEquals(map.keySet().size(), 3);
92          assertEquals(map.values().size(), 3);
93      }
94  
95      @Test
96      public void verifyContains() {
97          final String[][] arrayItems = {{"key0", "Item0"}, {"key1", "Item1"}, {"key2", "Item2"}};
98          final Map mapItems = ArrayUtils.toMap(arrayItems);
99  
100         this.map.putAll(mapItems);
101         assertTrue(map.containsKey("key2"));
102         assertTrue(map.containsValue("Item1"));
103     }
104 }
