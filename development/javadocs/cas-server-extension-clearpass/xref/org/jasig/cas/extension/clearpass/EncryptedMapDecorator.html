253     /**
254      * Construct hashed key.
255      *
256      * @param key the key
257      * @return the string
258      */
259     protected String constructHashedKey(final String key) {
260         if (key == null) {
261             return null;
262         }
263 
264         final MessageDigest messageDigest = getMessageDigest();
265         messageDigest.update(this.salt);
266         messageDigest.update(key.toLowerCase().getBytes());
267         final String hash = getFormattedText(messageDigest.digest());
268 
269         logger.debug(String.format("Generated hash of value [%s] for key [%s].", hash, key));
270         return hash;
271     }

273     /**
274      * Decrypt the value.
275      *
276      * @param value the value
277      * @param hashedKey the hashed key
278      * @return the string
279      */
280     protected String decrypt(final String value, final String hashedKey) {
281         if (value == null) {
282             return null;
283         }
284 
285         try {
286             final Cipher cipher = getCipherObject();
287             final byte[] ivCiphertext = decode(value.getBytes());
288             final int ivSize = byte2int(Arrays.copyOfRange(ivCiphertext, 0, INTEGER_LEN));
289             final byte[] ivValue = Arrays.copyOfRange(ivCiphertext, INTEGER_LEN, (INTEGER_LEN + ivSize));
290             final byte[] ciphertext = Arrays.copyOfRange(ivCiphertext, INTEGER_LEN + ivSize, ivCiphertext.length);
291             final IvParameterSpec ivSpec = new IvParameterSpec(ivValue);
292             
293             cipher.init(Cipher.DECRYPT_MODE, this.key, ivSpec);
294 
295             final byte[] plaintext = cipher.doFinal(ciphertext);
296 
297             return new String(plaintext);
298         } catch (final Exception e) {
299             throw new RuntimeException(e);
300         }
301     }

357     /**
358      * Encrypt.
359      *
360      * @param value the value
361      * @param hashedKey the hashed key
362      * @return the string
363      */
364     protected String encrypt(final String value, final String hashedKey) {
365         if (value == null) {
366             return null;
367         }
368 
369         try {
370             final Cipher cipher = getCipherObject();
371             final byte[] ivValue = generateIV(this.ivSize);
372             final IvParameterSpec ivSpec = new IvParameterSpec(ivValue);
373 
374             cipher.init(Cipher.ENCRYPT_MODE, this.key, ivSpec);
375             
376             final byte[] ciphertext = cipher.doFinal(value.getBytes());
377             final byte[] ivCiphertext = new byte[INTEGER_LEN + this.ivSize + ciphertext.length];
378 
379             System.arraycopy(int2byte(this.ivSize), 0, ivCiphertext, 0, INTEGER_LEN);
380             System.arraycopy(ivValue, 0, ivCiphertext, INTEGER_LEN, this.ivSize);
381             System.arraycopy(ciphertext, 0, ivCiphertext, INTEGER_LEN + this.ivSize, ciphertext.length);
382 
383             return new String(encode(ivCiphertext));
384         } catch(final Exception e) {
385             throw new RuntimeException(e);
386         }
387     }

433     /**
434      * Tries to clone the {@link MessageDigest} that was created during construction. If the clone fails
435      * that is remembered and from that point on new {@link MessageDigest} instances will be created on
436      * every call.
437      * <p>
438      * Adopted from the Spring EhCache Annotations project.
439      *
440      * @return Generates a {@link MessageDigest} to use
441      */
442     protected MessageDigest getMessageDigest() {
443         if (this.cloneNotSupported) {
444             final String algorithm = this.messageDigest.getAlgorithm();
445             try {
446                 return MessageDigest.getInstance(algorithm);
447             } catch (final NoSuchAlgorithmException e) {
448                 throw new IllegalStateException("MessageDigest algorithm '" + algorithm + "' was supported when "
449                         + this.getClass().getSimpleName()
450                         + " was created but is not now. This should not be possible.", e);
451             }
452         }
453 
454         try {
455             return (MessageDigest) this.messageDigest.clone();
456         } catch (final CloneNotSupportedException e) {
457             this.cloneNotSupported = true;
458             final String msg = String.format("Could not clone MessageDigest using algorithm '%s'. "
459                         + "MessageDigest.getInstance will be used from now on which will be much more expensive.",
460                         this.messageDigest.getAlgorithm());
461             logger.warn(msg, e);
462             return this.getMessageDigest();
463         }
464     }

466     /**
467      * Takes the raw bytes from the digest and formats them correct.
468      *
469      * @param bytes the raw bytes from the digest.
470      * @return the formatted bytes.
471      */
472     private static String getFormattedText(final byte[] bytes) {
473         final StringBuilder buf = new StringBuilder(bytes.length * 2);
474 
475         for (byte b : bytes) {
476             buf.append(HEX_DIGITS[b >> 4 & 0x0f]);
477             buf.append(HEX_DIGITS[b & 0x0f]);
478         }
479         return buf.toString();
480     }

495     /**
496      * Gets the secret key.
497      *
498      * @param secretKeyAlgorithm the secret key algorithm
499      * @param secretKey the secret key
500      * @param salt the salt
501      * @return the secret key
502      * @throws Exception the exception
503      */
504     private static Key getSecretKey(final String secretKeyAlgorithm, final String secretKey,
505             final String salt) throws Exception {
506 
507         final SecretKeyFactory factory = SecretKeyFactory.getInstance(SECRET_KEY_FACTORY_ALGORITHM);
508         final KeySpec spec = new PBEKeySpec(secretKey.toCharArray(), char2byte(salt), 65536, 128);
509         final SecretKey tmp = factory.generateSecret(spec);
510         final SecretKey secret = new SecretKeySpec(tmp.getEncoded(), secretKeyAlgorithm);
511 
512         return secret;
513     }
