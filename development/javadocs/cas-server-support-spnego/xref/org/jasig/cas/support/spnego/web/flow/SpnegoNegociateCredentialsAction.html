19  package org.jasig.cas.support.spnego.web.flow;
20  
21  import java.util.ArrayList;
22  import java.util.List;
23  
24  import javax.servlet.http.HttpServletRequest;
25  import javax.servlet.http.HttpServletResponse;
26  
27  import org.jasig.cas.support.spnego.util.SpnegoConstants;
28  import org.jasig.cas.web.support.WebUtils;
29  
30  import org.springframework.util.StringUtils;
31  import org.springframework.webflow.action.AbstractAction;
32  import org.springframework.webflow.execution.Event;
33  import org.springframework.webflow.execution.RequestContext;
34  
35  /**
36   * First action of a SPNEGO flow : negotiation.<br/> The server checks if the
37   * negotiation string is in the request header and this is a supported browser:
38   * <ul>
39   * <li>If found do nothing and return <code>success()</code></li>
40   * <li>else add a WWW-Authenticate response header and a 401 response status,
41   * then return <code>success()</code></li>
42   * </ul>
43   *
44   * @see <a href="http://ietfreport.isoc.org/idref/rfc4559/#page-2">RFC 4559</a>
45   * @author Arnaud Lesueur
46   * @author Marc-Antoine Garrigue
47   * @author Scott Battaglia
48   * @author John Gasper
49   * @since 3.1
50   */
51  public final class SpnegoNegociateCredentialsAction extends AbstractAction {
52  
53      /** Whether this is using the NTLM protocol or not. */
54      private boolean ntlm = false;
55  
56      private boolean mixedModeAuthentication = false;
57  
58      private List<String> supportedBrowser;
59  
60      private String messageBeginPrefix = constructMessagePrefix();
61  
62      @Override
63      protected Event doExecute(final RequestContext context) {
64          final HttpServletRequest request = WebUtils
65                  .getHttpServletRequest(context);
66          final HttpServletResponse response = WebUtils
67                  .getHttpServletResponse(context);
68          final String authorizationHeader = request
69                  .getHeader(SpnegoConstants.HEADER_AUTHORIZATION);
70          final String userAgent = request
71                  .getHeader(SpnegoConstants.HEADER_USER_AGENT);
72  
73          if (StringUtils.hasText(userAgent) && isSupportedBrowser(userAgent)) {
74              if (!StringUtils.hasText(authorizationHeader)
75                      || !authorizationHeader.startsWith(this.messageBeginPrefix)
76                      || authorizationHeader.length() <= this.messageBeginPrefix
77                      .length()) {
78                  if (logger.isDebugEnabled()) {
79                      logger
80                      .debug("Authorization header not found. Sending WWW-Authenticate header");
81                  }
82                  response.setHeader(SpnegoConstants.HEADER_AUTHENTICATE,
83                          this.ntlm ? SpnegoConstants.NTLM
84                                  : SpnegoConstants.NEGOTIATE);
85                  response.setStatus(HttpServletResponse.SC_UNAUTHORIZED);
86                  // The responseComplete flag tells the pausing view-state not to render the response
87                  // because another object has taken care of it. If mixed mode authentication is allowed
88                  // then responseComplete should not be called so that webflow will display the login page.
89                  if (!this.mixedModeAuthentication) {
90                      context.getExternalContext().recordResponseComplete();
91                  }
92              }
93          }
94          return success();
95      }
96  
97      /**
98       * Sets the ntlm. Generates the message prefix as well.
99       *
100      * @param ntlm the new ntlm
101      */
102     public void setNtlm(final boolean ntlm) {
103         this.ntlm = ntlm;
104         this.messageBeginPrefix = constructMessagePrefix();
105     }
106 
107     public void setSupportedBrowser(final List<String> supportedBrowser) {
108         this.supportedBrowser = supportedBrowser;
109     }
110 
111    /**
112     * Sets whether mixed mode authentication should be enabled. If it is
113     * enabled then control is allowed to pass back to the Spring Webflow
114     * instead of immediately terminating the page after issuing the
115     * unauthorized (401) header. This has the effect of displaying the login
116     * page on unsupported/configured browsers.
117     * <p>
118     * If this is set to false then the page is immediately closed after the
119     * unauthorized header is sent. This is ideal in environments that only
120     * want to use Windows Integrated Auth/SPNEGO and not forms auth.
121     *
122     * @param enabled should mixed mode authentication be allowed. Default is false.
123     */
124     public void setMixedModeAuthentication(final boolean enabled) {
125         this.mixedModeAuthentication = enabled;
126     }
127 
128     @Override
129     public void afterPropertiesSet() throws Exception {
130         if (this.supportedBrowser == null) {
131             this.supportedBrowser = new ArrayList<String>();
132             this.supportedBrowser.add("MSIE");
133             this.supportedBrowser.add("Firefox");
134             this.supportedBrowser.add("AppleWebKit");
135         }
136     }
137 
138     /**
139      * Construct message prefix.
140      *
141      * @return the string
142      */
143     protected String constructMessagePrefix() {
144         return (this.ntlm ? SpnegoConstants.NTLM : SpnegoConstants.NEGOTIATE)
145                 + " ";
146     }
147 
148     /**
149      * Checks if is supported browser.
150      *
151      * @param userAgent the user agent
152      * @return true, if  supported browser
153      */
154     protected boolean isSupportedBrowser(final String userAgent) {
155         for (final String supportedBrowser : this.supportedBrowser) {
156             if (userAgent.contains(supportedBrowser)) {
157                 return true;
158             }
159         }
160         return false;
161     }
162 }
